33static int header_do(void *dummy, const char *key, const char *value)
34{
35    request_rec *r = (request_rec *) dummy;
36    const char *payload;
37
38    payload = apr_table_get(r->headers_in, key);
39    if (payload) {
40        apr_table_setn(r->headers_out, value, payload);
41    }
42
43    return 1;
44}

46static int reflector_handler(request_rec * r)
47{
48    apr_bucket_brigade *bbin, *bbout;
49    reflector_cfg *conf;
50    apr_status_t status;
51
52    if (strcmp(r->handler, "reflector")) {
53        return DECLINED;
54    }
55
56    conf = (reflector_cfg *) ap_get_module_config(r->per_dir_config,
57                                                  &reflector_module);
58
59    ap_allow_methods(r, 1, "POST", "OPTIONS", NULL);
60
61    if (r->method_number == M_OPTIONS) {
62        return ap_send_http_options(r);
63    }
64
65    else if (r->method_number == M_POST) {
66        const char *content_length, *content_type;
67        int seen_eos;
68
69        /*
70         * Sometimes we'll get in a state where the input handling has
71         * detected an error where we want to drop the connection, so if
72         * that's the case, don't read the data as that is what we're trying
73         * to avoid.
74         *
75         * This function is also a no-op on a subrequest.
76         */ view japanese
77        if (r->main || r->connection->keepalive == AP_CONN_CLOSE ||
78            ap_status_drops_connection(r->status)) {
79            return OK;
80        }
81
82        /* copy headers from in to out if configured */ view japanese
83        apr_table_do(header_do, r, conf->headers, NULL);
84
85        /* last modified defaults to now, unless otherwise set on the way in */ view japanese
86        if (!apr_table_get(r->headers_out, "Last-Modified")) {
87            ap_update_mtime(r, apr_time_now());
88            ap_set_last_modified(r);
89        }
90        ap_set_accept_ranges(r);
91
92        /* reflect the content length, if present */ view japanese
93        if ((content_length = apr_table_get(r->headers_in, "Content-Length"))) {
94            apr_off_t clen;
95
96            if (!ap_parse_strict_length(&clen, content_length)) {
97                ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(10243)
98                              "reflector_handler: invalid content-length '%s'",
99                              content_length);
100                return HTTP_BAD_REQUEST;
101            }
102
103            ap_set_content_length(r, clen);
104        }
105
106        /* reflect the content type, if present */ view japanese
107        if ((content_type = apr_table_get(r->headers_in, "Content-Type"))) {
108
109            ap_set_content_type(r, content_type);
110
111        }
112
113        bbin = apr_brigade_create(r->pool, r->connection->bucket_alloc);
114        bbout = apr_brigade_create(r->pool, r->connection->bucket_alloc);
115
116        seen_eos = 0;
117        do {
118            apr_bucket *bucket;
119
120            status = ap_get_brigade(r->input_filters, bbin, AP_MODE_READBYTES,
121                                    APR_BLOCK_READ, HUGE_STRING_LEN);
122
123            if (status != APR_SUCCESS) {
124                apr_brigade_destroy(bbin);
125                return ap_map_http_request_error(status, HTTP_BAD_REQUEST);
126            }
127
128            for (bucket = APR_BRIGADE_FIRST(bbin);
129                 bucket != APR_BRIGADE_SENTINEL(bbin);
130                 bucket = APR_BUCKET_NEXT(bucket)) {
131                const char *data;
132                apr_size_t len;
133
134                if (APR_BUCKET_IS_EOS(bucket)) {
135                    seen_eos = 1;
136                    break;
137                }
138
139                /* These are metadata buckets. */ view japanese
140                if (bucket->length == 0) {
141                    continue;
142                }
143
144                /*
145                 * We MUST read because in case we have an unknown-length
146                 * bucket or one that morphs, we want to exhaust it.
147                 */ view japanese
148                status = apr_bucket_read(bucket, &data, &len, APR_BLOCK_READ);
149                if (status != APR_SUCCESS) {
150                    apr_brigade_destroy(bbin);
151                    return HTTP_BAD_REQUEST;
152                }
153
154                apr_brigade_write(bbout, NULL, NULL, data, len);
155
156                status = ap_pass_brigade(r->output_filters, bbout);
157                if (status != APR_SUCCESS) {
158                    /* no way to know what type of error occurred */ view japanese
159                    ap_log_rerror(APLOG_MARK, APLOG_DEBUG, status, r, APLOGNO(01410)
160                             "reflector_handler: ap_pass_brigade returned %i",
161                                  status);
162                    return AP_FILTER_ERROR;
163                }
164
165            }
166
167            apr_brigade_cleanup(bbin);
168
169        } while (!seen_eos);
170
171        return OK;
172
173    }
174
175    else {
176        return HTTP_METHOD_NOT_ALLOWED;
177    }
178
179}

181static void *create_reflector_dir_config(apr_pool_t * p, char *d)
182{
183    reflector_cfg *conf = apr_pcalloc(p, sizeof(reflector_cfg));
184
185    conf->headers = apr_table_make(p, 8);
186
187    return conf;
188}
189
190static void *merge_reflector_dir_config(apr_pool_t * p, void *basev, void *addv)
191{
192    reflector_cfg *new = (reflector_cfg *) apr_pcalloc(p,
193            sizeof(reflector_cfg));
194    reflector_cfg *add = (reflector_cfg *) addv;
195    reflector_cfg *base = (reflector_cfg *) basev;
196
197    new->headers = apr_table_overlay(p, add->headers, base->headers);
198
199    return new;
200}
201
202static const char *reflector_header(cmd_parms * cmd, void *dummy, const char *in,
203        const char *out)
204{
205    reflector_cfg *cfg = (reflector_cfg *) dummy;
206
207    apr_table_addn(cfg->headers, in, out ? out : in);
208
209    return NULL;
210}
211
212static void reflector_hooks(apr_pool_t * p)
213{
214    ap_hook_handler(reflector_handler, NULL, NULL, APR_HOOK_MIDDLE);
215}

220    {NULL}
