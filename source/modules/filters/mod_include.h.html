58#define SSI_FLAG_SIZE_IN_BYTES    (1<<2)  /* Sizes displayed in bytes.   */ view japanese
59#define SSI_FLAG_NO_EXEC          (1<<3)  /* No Exec in current context. */ view japanese
60
61#define SSI_FLAG_SIZE_ABBREV      (~(SSI_FLAG_SIZE_IN_BYTES))

67 * The public SSI context structure

69typedef struct {
70    /* permanent pool, use this for creating bucket data */ view japanese
71    apr_pool_t  *pool;

73    /* temp pool; will be cleared after the execution of every directive */ view japanese
74    apr_pool_t  *dpool;
75
76    /* See the SSI_FLAG_XXXXX definitions. */ view japanese
77    int          flags;
78
79    /* nesting of *invisible* ifs */ view japanese
80    int          if_nesting_level;
81
82    /* if true, the current buffer will be passed down the filter chain before
83     * continuing with next input bucket and the variable will be reset to
84     * false.
85     */ view japanese
86    int          flush_now;
87
88    /* argument counter (of the current directive) */ view japanese
89    unsigned     argc;
90
91    /* currently configured error string */ view japanese
92    const char  *error_str;
93
94    /* currently configured time format */ view japanese
95    const char  *time_str;
96
97    /* the current request */ view japanese
98    request_rec  *r;
99
100    /* pointer to internal (non-public) data, don't touch */ view japanese
101    struct ssi_internal_ctx *intern;
102
103} include_ctx_t;
