119/* Derive a code page ID give a language name or ID */ view japanese
120static char* derive_codepage_from_lang (apr_pool_t *p, char *language)
121{
122    char *charset;
123
124    if (!language)          /* our default codepage */ view japanese
125        return apr_pstrdup(p, "ISO-8859-1");
126
127    charset = (char*) apr_hash_get(charset_conversions, language, APR_HASH_KEY_STRING);
128
129    /*
130     * Test if language values like 'en-US' return a match from the charset
131     * conversion map when shortened to 'en'.
132     */ view japanese
133    if (!charset && strlen(language) > 3 && language[2] == '-') {
134        char *language_short = apr_pstrndup(p, language, 2);
135        charset = (char*) apr_hash_get(charset_conversions, language_short, APR_HASH_KEY_STRING);
136    }
137
138    if (charset) {
139        charset = apr_pstrdup(p, charset);
140    }
141
142    return charset;
143}

145static apr_xlate_t* get_conv_set (request_rec *r)
146{
147    char *lang_line = (char*)apr_table_get(r->headers_in, "accept-language");
148    char *lang;
149    apr_xlate_t *convset;
150
151    if (lang_line) {
152        lang_line = apr_pstrdup(r->pool, lang_line);
153        for (lang = lang_line;*lang;lang++) {
154            if ((*lang == ',') || (*lang == ';')) {
155                *lang = '\0';
156                break;
157            }
158        }
159        lang = derive_codepage_from_lang(r->pool, lang_line);
160
161        if (lang && (apr_xlate_open(&convset, to_charset, lang, r->pool) == APR_SUCCESS)) {
162            return convset;
163        }
164    }
165
166    return NULL;
167}

170static const char* authn_ldap_xlate_password(request_rec *r,
171                                             const char* sent_password)
172{
173    apr_xlate_t *convset = NULL;
174    apr_size_t inbytes;
175    apr_size_t outbytes;
176    char *outbuf;
177
178    if (charset_conversions && (convset = get_conv_set(r)) ) {
179        inbytes = strlen(sent_password);
180        outbytes = (inbytes+1)*3;
181        outbuf = apr_pcalloc(r->pool, outbytes);
182
183        /* Convert the password to UTF-8. */ view japanese
184        if (apr_xlate_conv_buffer(convset, sent_password, &inbytes, outbuf,
185                                  &outbytes) == APR_SUCCESS)
186            return outbuf;
187    }
188
189    return sent_password;
190}

193/*
194 * Build the search filter, or at least as much of the search filter that
195 * will fit in the buffer, and return APR_EGENERAL if it won't fit, otherwise
196 * APR_SUCCESS.
197 *
198 * The search filter consists of the filter provided with the URL,
199 * combined with a filter made up of the attribute provided with the URL,
200 * and the actual username passed by the HTTP client. For example, assume
201 * that the LDAP URL is
202 *
203 *   ldap://ldap.airius.com/ou=People, o=Airius?uid??(posixid=*)
204 *
205 * Further, assume that the userid passed by the client was `userj'.  The
206 * search filter will be (&(posixid=*)(uid=userj)).
207 */ view japanese
208#define FILTER_LENGTH MAX_STRING_LEN
209static apr_status_t authn_ldap_build_filter(char filtbuf[FILTER_LENGTH],
210                             request_rec *r,
211                             const char *user,
212                             const char *filter,
213                             authn_ldap_config_t *sec)
214{
215    char *q;
216    const char *p, *filtbuf_end;
217    apr_xlate_t *convset = NULL;
218    apr_size_t inbytes;
219    apr_size_t outbytes;
220    char *outbuf;
221    int nofilter = 0, len;
222    apr_status_t rv = APR_SUCCESS;
223
224    if (!filter) {
225        filter = sec->filter;
226    }
227
228    if (charset_conversions) {
229        convset = get_conv_set(r);
230    }
231
232    if (convset) {
233        inbytes = strlen(user);
234        outbytes = (inbytes+1)*3;
235        outbuf = apr_pcalloc(r->pool, outbytes);
236
237        /* Convert the user name to UTF-8.  This is only valid for LDAP v3 */ view japanese
238        if (apr_xlate_conv_buffer(convset, user, &inbytes, outbuf, &outbytes) == APR_SUCCESS) {
239            user = outbuf;
240        }
241    }
242
243    /*
244     * Create the first part of the filter, which consists of the
245     * config-supplied portions.
246     */ view japanese
247
248    if ((nofilter = (!filter || !*filter || !strcasecmp(filter, "none")))) { 
249        len = apr_snprintf(filtbuf, FILTER_LENGTH, "(%s=", sec->attribute);
250    }
251    else { 
252        len = apr_snprintf(filtbuf, FILTER_LENGTH, "(&(%s)(%s=", filter, sec->attribute);
253    }
254
255    /*
256     * Now add the client-supplied username to the filter, ensuring that any
257     * LDAP filter metachars are escaped.
258     */ view japanese
259    filtbuf_end = filtbuf + FILTER_LENGTH - 1;
260    for (p = user, q = filtbuf + len; *p; ) {
261        if (strchr("*()\\", *p) != NULL) {
262#if APR_HAS_MICROSOFT_LDAPSDK
263            if (q + 3 >= filtbuf_end) { /* accounts for final \0 */
264                rv = APR_EGENERAL;
265                goto out;
266            }
267            *q++ = '\\';
268            switch ( *p++ )
269            {
270              case '*':
271                *q++ = '2';
272                *q++ = 'a';
273                break;
274              case '(':
275                *q++ = '2';
276                *q++ = '8';
277                break;
278              case ')':
279                *q++ = '2';
280                *q++ = '9';
281                break;
282              case '\\':
283                *q++ = '5';
284                *q++ = 'c';
285                break;
286            }
287#else
288            if (q + 2 >= filtbuf_end) { /* accounts for final \0 */
289                rv = APR_EGENERAL;
290                goto out;
291            }
292            *q++ = '\\';
293            *q++ = *p++;
294#endif
295        }
296        else {
297            if (q + 1 >= filtbuf_end) { /* accounts for final \0 */
298                rv = APR_EGENERAL;
299                goto out;
300            }
301            *q++ = *p++;
302        }
303    }
304
305    /*
306     * Append the closing parens of the filter, unless doing so would
307     * overrun the buffer.
308     */ view japanese
309
310    if (nofilter) { 
311        if (q + 1 >= filtbuf_end) { /* accounts for final \0 */
312            rv = APR_EGENERAL;
313            goto out;
314        }
315        *q++ = ')';
316    } 
317    else { 
318        if (q + 2 >= filtbuf_end) { /* accounts for final \0 */ view japanese
319            rv = APR_EGENERAL;
320            goto out;
321        }
322        *q++ = ')';
323        *q++ = ')';
324    }
325
326out:
327    *q = '\0';
328    return rv;
329}

376static int set_request_vars(request_rec *r, enum auth_ldap_phase phase, const char **vals) {
377    char *prefix = NULL;
378    int prefix_len;
379    int remote_user_attribute_set = 0;
380    authn_ldap_config_t *sec =
381        (authn_ldap_config_t *)ap_get_module_config(r->per_dir_config, &authnz_ldap_module);
382
383    prefix = (phase == LDAP_AUTHN) ? AUTHN_PREFIX : sec->authz_prefix;
384    prefix_len = strlen(prefix);
385
386    if (sec->attributes && vals) {
387        apr_table_t *e = r->subprocess_env;
388        int i = 0;
389        while (sec->attributes[i]) {
390            char *str = apr_pstrcat(r->pool, prefix, sec->attributes[i], NULL);
391            int j = prefix_len;
392            while (str[j]) {
393                str[j] = apr_toupper(str[j]);
394                j++;
395            }
396            apr_table_setn(e, str, vals[i] ? vals[i] : "");
397
398            /* handle remote_user_attribute, if set */ view japanese
399            if ((phase == LDAP_AUTHN) &&
400                sec->remote_user_attribute &&
401                !strcmp(sec->remote_user_attribute, sec->attributes[i])) {
402                r->user = (char *)apr_pstrdup(r->pool, vals[i]);
403                remote_user_attribute_set = 1;
404            }
405            i++;
406        }
407    }
408    return remote_user_attribute_set;
409}

411static const char *ldap_determine_binddn(request_rec *r, const char *user) {
412    authn_ldap_config_t *sec =
413        (authn_ldap_config_t *)ap_get_module_config(r->per_dir_config, &authnz_ldap_module);
414    const char *result = user;
415    ap_regmatch_t regm[AP_MAX_REG_MATCH];
416
417    if (NULL == user || NULL == sec || !sec->bind_regex || !sec->bind_subst) {
418        return result;
419    }
420
421    if (!ap_regexec(sec->bind_regex, user, AP_MAX_REG_MATCH, regm, 0)) {
422        char *substituted = ap_pregsub(r->pool, sec->bind_subst, user, AP_MAX_REG_MATCH, regm);
423        if (NULL != substituted) {
424            result = substituted;
425        }
426    }
427
428    apr_table_set(r->subprocess_env, "LDAP_BINDASUSER", result);
429
430    return result;
431}

453    }
454
455    return util_ldap_connection_find(r, sec->host, sec->port,
456                                     binddn, bindpw,
457                                     sec->deref, sec->secure);

459/*
460 * Authentication Phase
461 * --------------------
462 *
463 * This phase authenticates the credentials the user has sent with
464 * the request (ie the username and password are checked). This is done
465 * by making an attempt to bind to the LDAP server using this user's
466 * DN and the supplied password.
467 *
468 */ view japanese
469static authn_status authn_ldap_check_password(request_rec *r, const char *user,
470                                              const char *password)
471{
472    char filtbuf[FILTER_LENGTH];
473    authn_ldap_config_t *sec =
474        (authn_ldap_config_t *)ap_get_module_config(r->per_dir_config, &authnz_ldap_module);
475
476    util_ldap_connection_t *ldc = NULL;
477    int result = 0;
478    int remote_user_attribute_set = 0;
479    const char *dn = NULL;
480    const char *utfpassword;
481
482    authn_ldap_request_t *req =
483        (authn_ldap_request_t *)apr_pcalloc(r->pool, sizeof(authn_ldap_request_t));
484    ap_set_module_config(r->request_config, &authnz_ldap_module, req);
485
486/*
487    if (!sec->enabled) {
488        return AUTH_USER_NOT_FOUND;
489    }
490*/ view japanese
491
492    /*
493     * Basic sanity checks before any LDAP operations even happen.
494     */ view japanese
495    if (!sec->have_ldap_url) {
496        ap_log_rerror(APLOG_MARK, APLOG_WARNING, 0, r, APLOGNO(02558) 
497                      "no AuthLDAPURL");
498
499        return AUTH_GENERAL_ERROR;
500    }
501
502    /* Get the password that the client sent */ view japanese
503    if (password == NULL) {
504        ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(01692)
505                      "auth_ldap authenticate: no password specified");
506        return AUTH_GENERAL_ERROR;
507    }
508
509    if (user == NULL) {
510        ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(01693)
511                      "auth_ldap authenticate: no user specified");
512        return AUTH_GENERAL_ERROR;
513    }
514
515    /*
516     * A bind to the server with an empty password always succeeds, so
517     * we check to ensure that the password is not empty. This implies
518     * that users who actually do have empty passwords will never be
519     * able to authenticate with this module. I don't see this as a big
520     * problem.
521     */ view japanese
522    if (!(*password)) {
523        ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(10263)
524                      "auth_ldap authenticate: empty password specified");
525        return AUTH_DENIED;
526    }
527
528    /* There is a good AuthLDAPURL, right? */ view japanese
529    if (sec->host) {
530        const char *binddn = sec->binddn;
531        const char *bindpw = sec->bindpw;
532        if (sec->initial_bind_as_user) {
533            bindpw = password;
534            binddn = ldap_determine_binddn(r, user);
535        }
536
537        ldc = util_ldap_connection_find(r, sec->host, sec->port,
538                                       binddn, bindpw,
539                                       sec->deref, sec->secure);
540    }
541    else {
542        ap_log_rerror(APLOG_MARK, APLOG_WARNING, 0, r, APLOGNO(01690)
543                      "auth_ldap authenticate: no sec->host - weird...?");
544        return AUTH_GENERAL_ERROR;
545    }
546
547    ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(01691)
548                  "auth_ldap authenticate: using URL %s", sec->url);
549
550    /* build the username filter */ view japanese
551    if (APR_SUCCESS != authn_ldap_build_filter(filtbuf, r, user, NULL, sec)) {
552        ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02622)
553                      "auth_ldap authenticate: ldap filter too long (>%d): %s",
554                      FILTER_LENGTH, filtbuf);
555        util_ldap_connection_close(ldc);
556        return AUTH_GENERAL_ERROR;
557    }
558
559    ap_log_rerror(APLOG_MARK, APLOG_TRACE1, 0, r,
560                      "auth_ldap authenticate: final authn filter is %s", filtbuf);
561
562    /* convert password to utf-8 */ view japanese
563    utfpassword = authn_ldap_xlate_password(r, password);
564
565    /* do the user search */ view japanese
566    result = util_ldap_cache_checkuserid(r, ldc, sec->url, sec->basedn, sec->scope,
567                                         sec->attributes, filtbuf, utfpassword,
568                                         &dn, &(req->vals));
569    util_ldap_connection_close(ldc);
570
571    /* handle bind failure */ view japanese
572    if (result != LDAP_SUCCESS) {
573        if (!sec->bind_authoritative) {
574           ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(01694)
575                      "auth_ldap authenticate: user %s authentication failed; "
576                      "URI %s [%s][%s] (not authoritative)",
577                      user, r->uri, ldc->reason, ldap_err2string(result));
578           return AUTH_USER_NOT_FOUND;
579        }
580
581        ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, r, APLOGNO(01695)
582                      "auth_ldap authenticate: "
583                      "user %s authentication failed; URI %s [%s][%s]",
584                      user, r->uri, ldc->reason, ldap_err2string(result));
585
586        /* talking to a primitive LDAP server (like RACF-over-LDAP) that doesn't return specific errors */ view japanese
587        if (!strcasecmp(sec->filter, "none") && LDAP_OTHER == result) { 
588            return AUTH_USER_NOT_FOUND;
589        }
590

599#endif
600#ifdef LDAP_INSUFFICIENT_RIGHTS

602#endif
603#endif
604#ifdef LDAP_CONSTRAINT_VIOLATION
605    /* At least Sun Directory Server sends this if a user is
606     * locked. This is not covered by LDAP_SECURITY_ERROR.
607     */ view japanese
608                 : (LDAP_CONSTRAINT_VIOLATION == result) ? AUTH_DENIED
609#endif
610                 : AUTH_GENERAL_ERROR;
611    }
612
613    /* mark the user and DN */ view japanese
614    req->dn = dn;
615    req->user = user;
616    req->password = password;
617    if (sec->user_is_dn) {
618        r->user = (char *)req->dn;
619    }
620
621    /* add environment variables */ view japanese
622    remote_user_attribute_set = set_request_vars(r, LDAP_AUTHN, req->vals);
623
624    /* sanity check */ view japanese
625    if (sec->remote_user_attribute && !remote_user_attribute_set) {
626        ap_log_rerror(APLOG_MARK, APLOG_WARNING, 0, r, APLOGNO(01696)

628                  "REMOTE_USER was to be set with attribute '%s', "
629                  "but this attribute was not requested for in the "
630                  "LDAP query for the user. REMOTE_USER will fall "
631                  "back to username or DN as appropriate.",
632                  sec->remote_user_attribute);
633    }
634
635    ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(01697)
636                  "auth_ldap authenticate: accepting %s", user);
637
638    return AUTH_GRANTED;
