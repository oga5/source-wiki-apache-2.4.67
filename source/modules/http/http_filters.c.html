60typedef struct http_filter_ctx
61{
62    apr_off_t remaining;
63    apr_off_t limit;
64    apr_off_t limit_used;
65    apr_int32_t chunk_used;
66    apr_int32_t chunk_bws;
67    apr_int32_t chunkbits;
68    enum
69    {
70        BODY_NONE, /* streamed data */ view japanese
71        BODY_LENGTH, /* data constrained by content length */ view japanese
72        BODY_CHUNK, /* chunk expected */ view japanese
73        BODY_CHUNK_PART, /* chunk digits */ view japanese
74        BODY_CHUNK_EXT, /* chunk extension */ view japanese
75        BODY_CHUNK_CR, /* got space(s) after digits, expect [CR]LF or ext */ view japanese
76        BODY_CHUNK_LF, /* got CR after digits or ext, expect LF */ view japanese
77        BODY_CHUNK_DATA, /* data constrained by chunked encoding */ view japanese
78        BODY_CHUNK_END, /* chunked data terminating CRLF */ view japanese
79        BODY_CHUNK_END_LF, /* got CR after data, expect LF */ view japanese
80        BODY_CHUNK_TRAILER /* trailers */ view japanese
81    } state;
82    unsigned int eos_sent :1,
83                 seen_data:1;
84    apr_bucket_brigade *bb;
85} http_ctx_t;

87/* bail out if some error in the HTTP input filter happens */ view japanese
88static apr_status_t bail_out_on_error(http_ctx_t *ctx,
89                                      ap_filter_t *f,
90                                      int http_error)
91{
92    apr_bucket *e;
93    apr_bucket_brigade *bb = ctx->bb;
94
95    apr_brigade_cleanup(bb);
96
97    if (f->r->proxyreq == PROXYREQ_RESPONSE) {
98        switch (http_error) {
99        case HTTP_REQUEST_ENTITY_TOO_LARGE:
100            return APR_ENOSPC;
101
102        case HTTP_REQUEST_TIME_OUT:
103            return APR_INCOMPLETE;
104
105        case HTTP_NOT_IMPLEMENTED:
106            return APR_ENOTIMPL;
107
108        default:
109            return APR_EGENERAL;
110        }
111    }
112
113    e = ap_bucket_error_create(http_error,
114                               NULL, f->r->pool,
115                               f->c->bucket_alloc);
116    APR_BRIGADE_INSERT_TAIL(bb, e);
117    e = apr_bucket_eos_create(f->c->bucket_alloc);
118    APR_BRIGADE_INSERT_TAIL(bb, e);
119    ctx->eos_sent = 1;
120    /* If chunked encoding / content-length are corrupt, we may treat parts
121     * of this request's body as the next one's headers.
122     * To be safe, disable keep-alive.
123     */ view japanese
124    f->r->connection->keepalive = AP_CONN_CLOSE;
125    return ap_pass_brigade(f->r->output_filters, bb);
126}

128/**
129 * Parse a chunk line with optional extension, detect overflow.
130 * There are several error cases:
131 *  1) If the chunk link is misformatted, APR_EINVAL is returned.
132 *  2) If the conversion would require too many bits, APR_EGENERAL is returned.
133 *  3) If the conversion used the correct number of bits, but an overflow
134 *     caused only the sign bit to flip, then APR_ENOSPC is returned.
135 * A negative chunk length always indicates an overflow error.
136 */ view japanese
137static apr_status_t parse_chunk_size(http_ctx_t *ctx, const char *buffer,
138                                     apr_size_t len, int linelimit, int strict)
139{
140    apr_size_t i = 0;
141
142    while (i < len) {
143        char c = buffer[i];
144
145        ap_xlate_proto_from_ascii(&c, 1);
146
147        /* handle CRLF after the chunk */ view japanese
148        if (ctx->state == BODY_CHUNK_END
149                || ctx->state == BODY_CHUNK_END_LF) {
150            if (c == LF) {
151                if (strict && (ctx->state != BODY_CHUNK_END_LF)) {
152                    /*
153                     * CR missing before LF.
154                     */ view japanese
155                    return APR_EINVAL;
156                }
157                ctx->state = BODY_CHUNK;
158            }
159            else if (c == CR && ctx->state == BODY_CHUNK_END) {
160                ctx->state = BODY_CHUNK_END_LF;
161            }
162            else {
163                /*
164                 * CRLF expected.
165                 */ view japanese
166                return APR_EINVAL;
167            }
168            i++;
169            continue;
170        }
171
172        /* handle start of the chunk */ view japanese
173        if (ctx->state == BODY_CHUNK) {
174            if (!apr_isxdigit(c)) {
175                /*
176                 * Detect invalid character at beginning. This also works for
177                 * empty chunk size lines.
178                 */ view japanese
179                return APR_EINVAL;
180            }
181            else {
182                ctx->state = BODY_CHUNK_PART;
183            }
184            ctx->remaining = 0;
185            ctx->chunkbits = sizeof(apr_off_t) * 8;
186            ctx->chunk_used = 0;
187            ctx->chunk_bws = 0;
188        }
189
190        if (c == LF) {
191            if (strict && (ctx->state != BODY_CHUNK_LF)) {
192                /*
193                 * CR missing before LF.
194                 */ view japanese
195                return APR_EINVAL;
196            }
197            if (ctx->remaining) {
198                ctx->state = BODY_CHUNK_DATA;
199            }
200            else {
201                ctx->state = BODY_CHUNK_TRAILER;
202            }
203        }
204        else if (ctx->state == BODY_CHUNK_LF) {
205            /*
206             * LF expected.
207             */ view japanese
208            return APR_EINVAL;
209        }
210        else if (c == CR) {
211            ctx->state = BODY_CHUNK_LF;
212        }
213        else if (c == ';') {
214            ctx->state = BODY_CHUNK_EXT;
215        }
216        else if (ctx->state == BODY_CHUNK_EXT) {
217            /*
218             * Control chars (excluding tabs) are invalid.
219             * TODO: more precisely limit input
220             */ view japanese
221            if (c != '\t' && apr_iscntrl(c)) {
222                return APR_EINVAL;
223            }
224        }
225        else if (c == ' ' || c == '\t') {
226            /* Be lenient up to 10 implied *LWS, a legacy of RFC 2616,
227             * and noted as errata to RFC7230;
228             * https://www.rfc-editor.org/errata_search.php?rfc=7230&eid=4667
229             */ view japanese
230            ctx->state = BODY_CHUNK_CR;
231            if (++ctx->chunk_bws > 10) {
232                return APR_EINVAL;
233            }
234        }
235        else if (ctx->state == BODY_CHUNK_CR) {
236            /*
237             * ';', CR or LF expected.
238             */ view japanese
239            return APR_EINVAL;
240        }
241        else if (ctx->state == BODY_CHUNK_PART) {
242            int xvalue;
243
244            /* ignore leading zeros */ view japanese
245            if (!ctx->remaining && c == '0') {
246                i++;
247                continue;
248            }
249
250            ctx->chunkbits -= 4;
251            if (ctx->chunkbits < 0) {
252                /* overflow */
253                return APR_ENOSPC;
254            }
255
256            if (c >= '0' && c <= '9') {
257                xvalue = c - '0';
258            }
259            else if (c >= 'A' && c <= 'F') {
260                xvalue = c - 'A' + 0xa;
261            }
262            else if (c >= 'a' && c <= 'f') {
263                xvalue = c - 'a' + 0xa;
264            }
265            else {
266                /* bogus character */ view japanese
267                return APR_EINVAL;
268            }
269
270            ctx->remaining = (ctx->remaining << 4) | xvalue;
271            if (ctx->remaining < 0) {
272                /* overflow */ view japanese
273                return APR_ENOSPC;
274            }
275        }
276        else {
277            /* Should not happen */
278            return APR_EGENERAL;
279        }
280
281        i++;
282    }
283
284    /* sanity check */ view japanese
285    ctx->chunk_used += len;
286    if (ctx->chunk_used < 0 || ctx->chunk_used > linelimit) {
287        return APR_ENOSPC;
288    }
289
290    return APR_SUCCESS;
291}

293static apr_status_t read_chunked_trailers(http_ctx_t *ctx, ap_filter_t *f,
294                                          apr_bucket_brigade *b, int merge)
295{
296    int rv;
297    apr_bucket *e;
298    request_rec *r = f->r;
299    apr_table_t *saved_headers_in = r->headers_in;
300    int saved_status = r->status;
301
302    r->status = HTTP_OK;
303    r->headers_in = r->trailers_in;
304    apr_table_clear(r->headers_in);
305    ap_get_mime_headers(r);
306
307    if(r->status == HTTP_OK) {
308        r->status = saved_status;
309        e = apr_bucket_eos_create(f->c->bucket_alloc);
310        APR_BRIGADE_INSERT_TAIL(b, e);
311        ctx->eos_sent = 1;
312        rv = APR_SUCCESS;
313    }
314    else {
315        const char *error_notes = apr_table_get(r->notes,
316                                                "error-notes");
317        ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, r, APLOGNO(02656)
318                      "Error while reading HTTP trailer: %i%s%s",
319                      r->status, error_notes ? ": " : "",
320                      error_notes ? error_notes : "");
321        rv = APR_EINVAL;
322    }
323
324    if(!merge) {
325        r->headers_in = saved_headers_in;
326    }
327    else {
328        r->headers_in = apr_table_overlay(r->pool, saved_headers_in,
329                r->trailers_in);
330    }
331
332    return rv;
333}

335/* This is the HTTP_INPUT filter for HTTP requests and responses from
336 * proxied servers (mod_proxy).  It handles chunked and content-length
337 * bodies.  This can only be inserted/used after the headers
338 * are successfully parsed.
339 */ view japanese
340apr_status_t ap_http_filter(ap_filter_t *f, apr_bucket_brigade *b,
341                            ap_input_mode_t mode, apr_read_type_e block,
342                            apr_off_t readbytes)
343{
344    core_server_config *conf =
345        (core_server_config *) ap_get_module_config(f->r->server->module_config,
346                                                    &core_module);
347    int strict = (conf->http_conformance != AP_HTTP_CONFORMANCE_UNSAFE);
348    apr_bucket *e;
349    http_ctx_t *ctx = f->ctx;
350    apr_status_t rv;
351    int http_error = HTTP_REQUEST_ENTITY_TOO_LARGE;
352    int again;
353
354    /* just get out of the way of things we don't want. */ view japanese
355    if (mode != AP_MODE_READBYTES && mode != AP_MODE_GETLINE) {
356        return ap_get_brigade(f->next, b, mode, block, readbytes);
357    }
358
359    if (!ctx) {
360        const char *tenc, *lenp;
361        f->ctx = ctx = apr_pcalloc(f->r->pool, sizeof(*ctx));
362        ctx->state = BODY_NONE;
363        ctx->bb = apr_brigade_create(f->r->pool, f->c->bucket_alloc);
364
365        /* LimitRequestBody does not apply to proxied responses.
366         * Consider implementing this check in its own filter.
367         * Would adding a directive to limit the size of proxied
368         * responses be useful?
369         */ view japanese
370        if (!f->r->proxyreq) {
371            ctx->limit = ap_get_limit_req_body(f->r);
372        }
373        else {
374            ctx->limit = 0;
375        }
376
377        tenc = apr_table_get(f->r->headers_in, "Transfer-Encoding");
378        lenp = apr_table_get(f->r->headers_in, "Content-Length");
379
380        if (tenc) {
381            if (ap_is_chunked(f->r->pool, tenc)) {
382                ctx->state = BODY_CHUNK;
383            }
384            else if (f->r->proxyreq == PROXYREQ_RESPONSE) {
385                /* http://tools.ietf.org/html/draft-ietf-httpbis-p1-messaging-23
386                 * Section 3.3.3.3: "If a Transfer-Encoding header field is
387                 * present in a response and the chunked transfer coding is not
388                 * the final encoding, the message body length is determined by
389                 * reading the connection until it is closed by the server."
390                 */ view japanese
391                ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, f->r, APLOGNO(02555)
392                              "Unknown Transfer-Encoding: %s; "
393                              "using read-until-close", tenc);
394                tenc = NULL;
395            }
396            else {
397                /* Something that isn't a HTTP request, unless some future
398                 * edition defines new transfer encodings, is unsupported.
399                 */ view japanese
400                ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, f->r, APLOGNO(01585)
401                              "Unknown Transfer-Encoding: %s", tenc);
402                return bail_out_on_error(ctx, f, HTTP_BAD_REQUEST);
403            }
404            lenp = NULL;
405        }
406        if (lenp) {
407            ctx->state = BODY_LENGTH;
408
409            /* Protects against over/underflow, non-digit chars in the
410             * string, leading plus/minus signs, trailing characters and
411             * a negative number.
412             */ view japanese
413            if (!ap_parse_strict_length(&ctx->remaining, lenp)) {
414                ctx->remaining = 0;
415                ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, f->r, APLOGNO(01587)
416                              "Invalid Content-Length");
417
418                return bail_out_on_error(ctx, f, HTTP_BAD_REQUEST);
419            }
420
421            /* If we have a limit in effect and we know the C-L ahead of
422             * time, stop it here if it is invalid.
423             */ view japanese
424            if (ctx->limit && ctx->limit < ctx->remaining) {
425                ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, f->r, APLOGNO(01588)
426                          "Requested content-length of %" APR_OFF_T_FMT
427                          " is larger than the configured limit"
428                          " of %" APR_OFF_T_FMT, ctx->remaining, ctx->limit);
429                return bail_out_on_error(ctx, f, HTTP_REQUEST_ENTITY_TOO_LARGE);
430            }
431        }
432
433        /* If we don't have a request entity indicated by the headers, EOS.
434         * (BODY_NONE is a valid intermediate state due to trailers,
435         *  but it isn't a valid starting state.)
436         *
437         * RFC 2616 Section 4.4 note 5 states that connection-close
438         * is invalid for a request entity - request bodies must be
439         * denoted by C-L or T-E: chunked.
440         *
441         * Note that since the proxy uses this filter to handle the
442         * proxied *response*, proxy responses MUST be exempt.
443         */ view japanese
444        if (ctx->state == BODY_NONE && f->r->proxyreq != PROXYREQ_RESPONSE) {
445            e = apr_bucket_eos_create(f->c->bucket_alloc);
446            APR_BRIGADE_INSERT_TAIL(b, e);
447            ctx->eos_sent = 1;
448            return APR_SUCCESS;
449        }
450    }
451
452    /* Since we're about to read data, send 100-Continue if needed.
453     * Only valid on chunked and C-L bodies where the C-L is > 0.
454     *
455     * If the read is to be nonblocking though, the caller may not want to
456     * handle this just now (e.g. mod_proxy_http), and is prepared to read
457     * nothing if the client really waits for 100 continue, so we don't
458     * send it now and wait for later blocking read.
459     *
460     * In any case, even if r->expecting remains set at the end of the
461     * request handling, ap_set_keepalive() will finally do the right
462     * thing (i.e. "Connection: close" the connection).
463     */ view japanese
464    if (block == APR_BLOCK_READ
465            && (ctx->state == BODY_CHUNK
466                || (ctx->state == BODY_LENGTH && ctx->remaining > 0))
467            && f->r->expecting_100 && f->r->proto_num >= HTTP_VERSION(1,1)
468            && !(ctx->eos_sent || f->r->eos_sent || f->r->bytes_sent)) {
469        if (!ap_is_HTTP_SUCCESS(f->r->status)) {
470            ctx->state = BODY_NONE;
471            ctx->eos_sent = 1; /* send EOS below */ view japanese
472        }
473        else if (!ctx->seen_data) {
474            int saved_status = f->r->status;
475            const char *saved_status_line = f->r->status_line;
476            f->r->status = HTTP_CONTINUE;
477            f->r->status_line = NULL;
478            ap_send_interim_response(f->r, 0);
479            AP_DEBUG_ASSERT(!f->r->expecting_100);
480            f->r->status_line = saved_status_line;
481            f->r->status = saved_status;
482        }
483        else {
484            /* https://tools.ietf.org/html/rfc7231#section-5.1.1
485             *   A server MAY omit sending a 100 (Continue) response if it
486             *   has already received some or all of the message body for
487             *   the corresponding request [...]
488             */ view japanese
489            f->r->expecting_100 = 0;
490        }
491    }
492
493    /* sanity check in case we're read twice */ view japanese
494    if (ctx->eos_sent) {
495        e = apr_bucket_eos_create(f->c->bucket_alloc);
496        APR_BRIGADE_INSERT_TAIL(b, e);
497        return APR_SUCCESS;
498    }
499
500    do {
501        apr_brigade_cleanup(b);
502        again = 0; /* until further notice */ view japanese
503
504        /* read and handle the brigade */ view japanese
505        switch (ctx->state) {
506        case BODY_CHUNK:
507        case BODY_CHUNK_PART:
508        case BODY_CHUNK_EXT:
509        case BODY_CHUNK_CR:
510        case BODY_CHUNK_LF:
511        case BODY_CHUNK_END:
512        case BODY_CHUNK_END_LF: {
513
514            rv = ap_get_brigade(f->next, b, AP_MODE_GETLINE, block, 0);
515
516            /* for timeout */
517            if (block == APR_NONBLOCK_READ
518                    && ((rv == APR_SUCCESS && APR_BRIGADE_EMPTY(b))
519                            || (APR_STATUS_IS_EAGAIN(rv)))) {
520                return APR_EAGAIN;
521            }
522
523            if (rv == APR_EOF) {
524                return APR_INCOMPLETE;
525            }
526
527            if (rv != APR_SUCCESS) {
528                return rv;
529            }
530
531            e = APR_BRIGADE_FIRST(b);
532            while (e != APR_BRIGADE_SENTINEL(b)) {
533                const char *buffer;
534                apr_size_t len;
535
536                if (!APR_BUCKET_IS_METADATA(e)) {
537                    int parsing = 0;
538
539                    rv = apr_bucket_read(e, &buffer, &len, APR_BLOCK_READ);
540                    if (rv == APR_SUCCESS) {
541                        parsing = 1;
542                        if (len > 0) {
543                            ctx->seen_data = 1;
544                        }
545                        rv = parse_chunk_size(ctx, buffer, len,
546                                f->r->server->limit_req_fieldsize, strict);
547                    }
548                    if (rv != APR_SUCCESS) {
549                        ap_log_rerror(APLOG_MARK, APLOG_INFO, rv, f->r, APLOGNO(01590)
550                                      "Error reading/parsing chunk %s ",
551                                      (APR_ENOSPC == rv) ? "(overflow)" : "");
552                        if (parsing) {
553                            if (rv != APR_ENOSPC) {
554                                http_error = HTTP_BAD_REQUEST;
555                            }
556                            return bail_out_on_error(ctx, f, http_error);
557                        }
558                        return rv;
559                    }
560                }
561
562                apr_bucket_delete(e);
563                e = APR_BRIGADE_FIRST(b);
564            }
565            again = 1; /* come around again */ view japanese
566
567            if (ctx->state == BODY_CHUNK_TRAILER) {
568                /* Treat UNSET as DISABLE - trailers aren't merged by default */ view japanese
569                return read_chunked_trailers(ctx, f, b,
570                            conf->merge_trailers == AP_MERGE_TRAILERS_ENABLE);
571            }
572
573            break;
574        }
575        case BODY_NONE:
576        case BODY_LENGTH:
577        case BODY_CHUNK_DATA: {
578
579            /* Ensure that the caller can not go over our boundary point. */ view japanese
580            if (ctx->state != BODY_NONE && ctx->remaining < readbytes) {
581                readbytes = ctx->remaining;
582            }
583            if (readbytes > 0) {
584                apr_off_t totalread;
585
586                rv = ap_get_brigade(f->next, b, mode, block, readbytes);
587
588                /* for timeout */
589                if (block == APR_NONBLOCK_READ
590                        && ((rv == APR_SUCCESS && APR_BRIGADE_EMPTY(b))
591                                || (APR_STATUS_IS_EAGAIN(rv)))) {
592                    return APR_EAGAIN;
593                }
594
595                if (rv == APR_EOF && ctx->state != BODY_NONE
596                        && ctx->remaining > 0) {
597                    return APR_INCOMPLETE;
598                }
599
600                if (rv != APR_SUCCESS) {
601                    return rv;
602                }
603
604                /* How many bytes did we just read? */ view japanese
605                apr_brigade_length(b, 0, &totalread);
606                if (totalread > 0) {
607                    ctx->seen_data = 1;
608                }
609
610                /* If this happens, we have a bucket of unknown length.  Die because
611                 * it means our assumptions have changed. */ view japanese
612                AP_DEBUG_ASSERT(totalread >= 0);
613
614                if (ctx->state != BODY_NONE) {
615                    ctx->remaining -= totalread;
616                    if (ctx->remaining > 0) {
617                        e = APR_BRIGADE_LAST(b);
618                        if (APR_BUCKET_IS_EOS(e)) {
619                            apr_bucket_delete(e);
620                            return APR_INCOMPLETE;
621                        }
622                    }
623                    else if (ctx->state == BODY_CHUNK_DATA) {
624                        /* next chunk please */ view japanese
625                        ctx->state = BODY_CHUNK_END;
626                        ctx->chunk_used = 0;
627                    }
628                }
629
630                /* We have a limit in effect. */ view japanese
631                if (ctx->limit) {
632                    /* FIXME: Note that we might get slightly confused on
633                     * chunked inputs as we'd need to compensate for the chunk
634                     * lengths which may not really count.  This seems to be up
635                     * for interpretation.
636                     */ view japanese
637                    ctx->limit_used += totalread;
638                    if (ctx->limit < ctx->limit_used) {
639                        ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, f->r,
640                                      APLOGNO(01591) "Read content length of "

645                                                 HTTP_REQUEST_ENTITY_TOO_LARGE);
646                    }
647                }
648            }
649
650            /* If we have no more bytes remaining on a C-L request,
651             * save the caller a round trip to discover EOS.
652             */ view japanese
653            if (ctx->state == BODY_LENGTH && ctx->remaining == 0) {
654                e = apr_bucket_eos_create(f->c->bucket_alloc);
655                APR_BRIGADE_INSERT_TAIL(b, e);
656                ctx->eos_sent = 1;
657            }
658
659            break;
660        }
661        case BODY_CHUNK_TRAILER: {
662
663            rv = ap_get_brigade(f->next, b, mode, block, readbytes);
664
665            /* for timeout */ view japanese
666            if (block == APR_NONBLOCK_READ
667                    && ((rv == APR_SUCCESS && APR_BRIGADE_EMPTY(b))
668                            || (APR_STATUS_IS_EAGAIN(rv)))) {
669                return APR_EAGAIN;
670            }
671
672            if (rv != APR_SUCCESS) {
673                return rv;
674            }
675
676            break;
677        }
678        default: {
679            /* Should not happen */ view japanese
680            ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, f->r, APLOGNO(02901)
681                          "Unexpected body state (%i)", (int)ctx->state);
682            return APR_EGENERAL;
683        }
684        }
685
686    } while (again);

688    return APR_SUCCESS;
689}

696/* check a single header, to be used with apr_table_do() */ view japanese
697static int check_header(struct check_header_ctx *ctx,
698                        const char *name, const char **val)
699{
700    const char *pos, *end;
701    char *dst = NULL;
702
703    if (name[0] == '\0') {
704        ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, ctx->r, APLOGNO(02428)
705                      "Empty response header name, aborting request");
706        return 0;
707    }
708
709    if (ctx->strict) { 
710        end = ap_scan_http_token(name);
711    }
712    else {
713        end = ap_scan_vchar_obstext(name);
714    }
715    if (*end) {
716        ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, ctx->r, APLOGNO(02429)
717                      "Response header name '%s' contains invalid "
718                      "characters, aborting request",
719                      name);
720        return 0;
721    }
722
723    for (pos = *val; *pos; pos = end) {
724        end = ap_scan_http_field_content(pos);
725        if (*end) {
726            if (end[0] != CR || end[1] != LF || (end[2] != ' ' &&
727                                                 end[2] != '\t')) {
728                ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, ctx->r, APLOGNO(02430)
729                              "Response header '%s' value of '%s' contains "
730                              "invalid characters, aborting request",
731                              name, pos);
732                return 0;
733            }
734            if (!dst) {
735                *val = dst = apr_palloc(ctx->r->pool, strlen(*val) + 1);
736            }
737        }
738        if (dst) {
739            memcpy(dst, pos, end - pos);
740            dst += end - pos;
741            if (*end) {
742                /* skip folding and replace with a single space */ view japanese
743                end += 3 + strspn(end + 3, "\t ");
744                *dst++ = ' ';
745            }
746        }
747    }
748    if (dst) {
749        *dst = '\0';
750    }
751    return 1;
752}

851/* This routine is called by apr_table_do and merges all instances of
852 * the passed field values into a single array that will be further
853 * processed by some later routine.  Originally intended to help split

990/* Confirm that the status line is well-formed and matches r->status.
991 * If they don't match, a filter may have negated the status line set by a
992 * handler.
993 * Zap r->status_line if bad.
994 */ view japanese

996{
997    char *end;
998
999    if (r->status_line) {

1002            || apr_strtoi64(r->status_line, &end, 10) != r->status

1009         * is equivalent to only a 3 digit numeric http status.
1010         * RFC2616 mandates a trailing space, let's add it.

1012        if (len == 3) {

1017        if (len > 4 && *ap_scan_http_field_content(r->status_line + 4)) {

1024}

1168}

1308static void merge_response_headers(request_rec *r, const char **protocol)
1309{
1310    const char *ctype = NULL;
1311    const char *clheader = NULL;
1312
1313    /*
1314     * Now that we are ready to send a response, we need to combine the two
1315     * header field tables into a single table.  If we don't do this, our
1316     * later attempts to set or unset a given fieldname might be bypassed.
1317     */ view japanese
1318    if (!apr_is_empty_table(r->err_headers_out)) {
1319        r->headers_out = apr_table_overlay(r->pool, r->err_headers_out,
1320                                           r->headers_out);
1321        apr_table_clear(r->err_headers_out);
1322    }
1323
1324    /*
1325     * Remove the 'Vary' header field if the client can't handle it.
1326     * Since this will have nasty effects on HTTP/1.1 caches, force
1327     * the response into HTTP/1.0 mode.
1328     *
1329     * Note: the force-response-1.0 should come before the call to
1330     *       basic_http_header_check()
1331     */ view japanese
1332    if (apr_table_get(r->subprocess_env, "force-no-vary") != NULL) {
1333        apr_table_unset(r->headers_out, "Vary");
1334        r->proto_num = HTTP_VERSION(1,0);
1335        apr_table_setn(r->subprocess_env, "force-response-1.0", "1");
1336    }
1337    else {
1338        fixup_vary(r);
1339    }
1340
1341    /* determine the protocol and whether we should use keepalives. */ view japanese
1342    basic_http_header_check(r, protocol);
1343    ap_set_keepalive(r);
1344
1345    /*
1346     * Control cachability for non-cacheable responses if not already set by
1347     * some other part of the server configuration.
1348     */ view japanese
1349    if (r->no_cache && !apr_table_get(r->headers_out, "Expires")) {
1350        char *date = apr_palloc(r->pool, APR_RFC822_DATE_LEN);
1351        ap_recent_rfc822_date(date, r->request_time);
1352        apr_table_addn(r->headers_out, "Expires", date);
1353    }
1354
1355    /*
1356     * Now remove any ETag response header field if earlier processing
1357     * says so (such as a 'FileETag None' directive).
1358     */ view japanese
1359    if (apr_table_get(r->notes, "no-etag") != NULL) {
1360        apr_table_unset(r->headers_out, "ETag");
1361    }
1362
1363    /* 204/304 responses don't have content related headers */ view japanese
1364    if (AP_STATUS_IS_HEADER_ONLY(r->status)) {
1365        apr_table_unset(r->headers_out, "Transfer-Encoding");
1366        apr_table_unset(r->headers_out, "Content-Length");
1367        r->content_type = r->content_encoding = NULL;
1368        r->content_languages = NULL;
1369        r->clength = r->chunked = 0;
1370    }
1371    else if (r->chunked) {
1372        apr_table_mergen(r->headers_out, "Transfer-Encoding", "chunked");
1373        apr_table_unset(r->headers_out, "Content-Length");
1374    }
1375
1376    ctype = ap_make_content_type(r, r->content_type);
1377    if (ctype) {
1378        apr_table_setn(r->headers_out, "Content-Type", ctype);
1379    }
1380
1381    if (r->content_encoding) {
1382        apr_table_setn(r->headers_out, "Content-Encoding",
1383                       r->content_encoding);
1384    }
1385
1386    if (!apr_is_empty_array(r->content_languages)) {
1387        int i;
1388        char *token;
1389        char **languages = (char **)(r->content_languages->elts);
1390        const char *field = apr_table_get(r->headers_out, "Content-Language");
1391
1392        while (field && (token = ap_get_list_item(r->pool, &field)) != NULL) {
1393            for (i = 0; i < r->content_languages->nelts; ++i) {
1394                if (!ap_cstr_casecmp(token, languages[i]))
1395                    break;
1396            }
1397            if (i == r->content_languages->nelts) {
1398                *((char **) apr_array_push(r->content_languages)) = token;
1399            }
1400        }
1401
1402        field = apr_array_pstrcat(r->pool, r->content_languages, ',');
1403        apr_table_setn(r->headers_out, "Content-Language", field);
1404    }
1405
1406    /* This is a hack, but I can't find anyway around it.  The idea is that
1407     * we don't want to send out 0 Content-Lengths if it is a head request.
1408     * This happens when modules try to outsmart the server, and return
1409     * if they see a HEAD request.  Apache 1.3 handlers were supposed to
1410     * just return in that situation, and the core handled the HEAD.  In
1411     * 2.0, if a handler returns, then the core sends an EOS bucket down
1412     * the filter stack, and the content-length filter computes a C-L of
1413     * zero and that gets put in the headers, and we end up sending a
1414     * zero C-L to the client.  We can't just remove the C-L filter,
1415     * because well behaved 2.0 handlers will send their data down the stack,
1416     * and we will compute a real C-L for the head request. RBB
1417     */ view japanese
1418    if (r->header_only
1419        && (clheader = apr_table_get(r->headers_out, "Content-Length"))
1420        && !strcmp(clheader, "0")) {
1421        apr_table_unset(r->headers_out, "Content-Length");
1422    }
1423}

1425AP_CORE_DECLARE_NONSTD(apr_status_t) ap_http_header_filter(ap_filter_t *f,
1426                                                           apr_bucket_brigade *b)
1427{
1428    request_rec *r = f->r;
1429    conn_rec *c = r->connection;
1430    int header_only = (r->header_only || AP_STATUS_IS_HEADER_ONLY(r->status));
1431    apr_bucket *e;
1432    apr_bucket_brigade *b2;
1433    header_struct h;
1434    header_filter_ctx *ctx = f->ctx;
1435    ap_bucket_error *eb = NULL;
1436    apr_status_t rv = APR_SUCCESS;
1437    int recursive_error = 0;
1438    const char *protocol;
1439
1440    AP_DEBUG_ASSERT(!r->main);
1441
1442    if (!ctx) {
1443        ctx = f->ctx = apr_pcalloc(r->pool, sizeof(header_filter_ctx));
1444    }
1445    else if (ctx->headers_sent) {
1446        /* Eat body if response must not have one. */ view japanese
1447        if (header_only) {
1448            /* Still next filters may be waiting for EOS, so pass it (alone)
1449             * when encountered and be done with this filter.
1450             */ view japanese
1451            e = APR_BRIGADE_LAST(b);
1452            if (e != APR_BRIGADE_SENTINEL(b) && APR_BUCKET_IS_EOS(e)) {
1453                APR_BUCKET_REMOVE(e);
1454                apr_brigade_cleanup(b);
1455                APR_BRIGADE_INSERT_HEAD(b, e);
1456                ap_remove_output_filter(f);
1457                rv = ap_pass_brigade(f->next, b);
1458            }
1459            apr_brigade_cleanup(b);
1460            return rv;
1461        }
1462    }
1463
1464    for (e = APR_BRIGADE_FIRST(b);
1465         e != APR_BRIGADE_SENTINEL(b);
1466         e = APR_BUCKET_NEXT(e))
1467    {
1468        if (AP_BUCKET_IS_ERROR(e) && !eb) {
1469            eb = e->data;
1470            continue;
1471        }
1472        /*
1473         * If we see an EOC bucket it is a signal that we should get out
1474         * of the way doing nothing.
1475         */ view japanese
1476        if (AP_BUCKET_IS_EOC(e)) {
1477            ap_remove_output_filter(f);
1478            return ap_pass_brigade(f->next, b);
1479        }
1480    }
1481
1482    if (!ctx->headers_sent) {
1483        merge_response_headers(r, &protocol);
1484        if (!check_headers(r)) {
1485            /* We may come back here from ap_die() below,
1486             * so clear anything from this response.
1487             */ view japanese
1488            apr_table_clear(r->headers_out);
1489            apr_table_clear(r->err_headers_out);
1490            r->content_type = r->content_encoding = NULL;
1491            r->content_languages = NULL;
1492            r->clength = r->chunked = 0;
1493            apr_brigade_cleanup(b);
1494
1495            /* Don't recall ap_die() if we come back here (from its own internal
1496             * redirect or error response), otherwise we can end up in infinite
1497             * recursion; better fall through with 500, minimal headers and an
1498             * empty body (EOS only).
1499             */ view japanese
1500            if (!check_headers_recursion(r)) {
1501                ap_die(HTTP_INTERNAL_SERVER_ERROR, r);
1502                return AP_FILTER_ERROR;
1503            }
1504            r->status = HTTP_INTERNAL_SERVER_ERROR;
1505            e = ap_bucket_eoc_create(c->bucket_alloc);
1506            APR_BRIGADE_INSERT_TAIL(b, e);
1507            e = apr_bucket_eos_create(c->bucket_alloc);
1508            APR_BRIGADE_INSERT_TAIL(b, e);
1509            ap_set_content_length(r, 0);
1510            recursive_error = 1;
1511        }
1512        else if (eb) {
1513            int status;
1514            status = eb->status;
1515            apr_brigade_cleanup(b);
1516            ap_die(status, r);
1517            return AP_FILTER_ERROR;
1518        }
1519    }
1520
1521    if (r->assbackwards) {
1522        r->sent_bodyct = 1;
1523        ap_remove_output_filter(f);
1524        rv = ap_pass_brigade(f->next, b);
1525        goto out;
1526    }
1527
1528    if (!ctx->headers_sent) {
1529        b2 = apr_brigade_create(r->pool, c->bucket_alloc);
1530        basic_http_header(r, b2, protocol);
1531
1532        h.pool = r->pool;
1533        h.bb = b2;
1534
1535        send_all_header_fields(&h, r);
1536
1537        terminate_header(b2);
1538
1539        if (header_only) {
1540            e = APR_BRIGADE_LAST(b);
1541            if (e != APR_BRIGADE_SENTINEL(b) && APR_BUCKET_IS_EOS(e)) {
1542                APR_BUCKET_REMOVE(e);
1543                APR_BRIGADE_INSERT_TAIL(b2, e);
1544                ap_remove_output_filter(f);
1545            }
1546            apr_brigade_cleanup(b);
1547        }
1548
1549        rv = ap_pass_brigade(f->next, b2);
1550        apr_brigade_cleanup(b2);
1551        ctx->headers_sent = 1;
1552    }
1553
1554    if (rv != APR_SUCCESS || header_only) {
1555        goto out;
1556    }
1557
1558    r->sent_bodyct = 1;         /* Whatever follows is real body stuff... */ view japanese
1559
1560    if (r->chunked) {
1561        /* We can't add this filter until we have already sent the headers.
1562         * If we add it before this point, then the headers will be chunked
1563         * as well, and that is just wrong.
1564         */ view japanese
1565        ap_add_output_filter("CHUNK", NULL, r, r->connection);
1566    }
1567
1568    /* Don't remove this filter until after we have added the CHUNK filter.
1569     * Otherwise, f->next won't be the CHUNK filter and thus the first
1570     * brigade won't be chunked properly.
1571     */ view japanese
1572    ap_remove_output_filter(f);
1573    rv = ap_pass_brigade(f->next, b);
1574out:
1575    if (recursive_error) {
1576        return AP_FILTER_ERROR;
1577    }
1578    return rv;
1579}

1614/* In HTTP/1.1, any method can have a body.  However, most GET handlers
1615 * wouldn't know what to do with a request body if they received one.
1616 * This helper routine tests for and reads any message body in the request,
1617 * simply discarding whatever it receives.  We need to do this because
1618 * failing to read the request body would cause it to be interpreted
1619 * as the next request on a persistent connection.
1620 *
1621 * Since we return an error status if the request is malformed, this
1622 * routine should be called at the beginning of a no-body handler, e.g.,
1623 *
1624 *    if ((retval = ap_discard_request_body(r)) != OK) {
1625 *        return retval;
1626 *    }
1627 */ view japanese
1628AP_DECLARE(int) ap_discard_request_body(request_rec *r)
1629{
1630    int rc = OK;
1631    conn_rec *c = r->connection;
1632    apr_bucket_brigade *bb;
1633
1634    /* Sometimes we'll get in a state where the input handling has
1635     * detected an error where we want to drop the connection, so if
1636     * that's the case, don't read the data as that is what we're trying
1637     * to avoid.
1638     *
1639     * This function is also a no-op on a subrequest.
1640     */ view japanese
1641    if (r->main || c->keepalive == AP_CONN_CLOSE) {
1642        return OK;
1643    }
1644    if (ap_status_drops_connection(r->status)) {
1645        c->keepalive = AP_CONN_CLOSE;
1646        return OK;
1647    }
1648
1649    bb = apr_brigade_create(r->pool, r->connection->bucket_alloc);
1650    for (;;) {
1651        apr_status_t rv;
1652
1653        rv = ap_get_brigade(r->input_filters, bb, AP_MODE_READBYTES,
1654                            APR_BLOCK_READ, HUGE_STRING_LEN);
1655        if (rv != APR_SUCCESS) {
1656            rc = ap_map_http_request_error(rv, HTTP_BAD_REQUEST);
1657            goto cleanup;
1658        }
1659
1660        while (!APR_BRIGADE_EMPTY(bb)) {
1661            apr_bucket *b = APR_BRIGADE_FIRST(bb);
1662
1663            if (APR_BUCKET_IS_EOS(b)) {
1664                goto cleanup;
1665            }
1666
1667            /* There is no need to read empty or metadata buckets or
1668             * buckets of known length, but we MUST read buckets of
1669             * unknown length in order to exhaust them.
1670             */ view japanese
1671            if (b->length == (apr_size_t)-1) {
1672                apr_size_t len;
1673                const char *data;
1674
1675                rv = apr_bucket_read(b, &data, &len, APR_BLOCK_READ);
1676                if (rv != APR_SUCCESS) {
1677                    rc = HTTP_BAD_REQUEST;
1678                    goto cleanup;
1679                }
1680            }
1681
1682            apr_bucket_delete(b);
1683        }
1684    }
1685
1686cleanup:
1687    apr_brigade_cleanup(bb);
1688    if (rc != OK) {
1689        c->keepalive = AP_CONN_CLOSE;
1690    }
1691    return rc;
1692}

1694/* Here we deal with getting the request message body from the client.
1695 * Whether or not the request contains a body is signaled by the presence
1696 * of a non-zero Content-Length or by a Transfer-Encoding: chunked.
1697 *
1698 * Note that this is more complicated than it was in Apache 1.1 and prior
1699 * versions, because chunked support means that the module does less.
1700 *
1701 * The proper procedure is this:
1702 *
1703 * 1. Call ap_setup_client_block() near the beginning of the request
1704 *    handler. This will set up all the necessary properties, and will
1705 *    return either OK, or an error code. If the latter, the module should
1706 *    return that error code. The second parameter selects the policy to
1707 *    apply if the request message indicates a body, and how a chunked
1708 *    transfer-coding should be interpreted. Choose one of
1709 *
1710 *    REQUEST_NO_BODY          Send 413 error if message has any body
1711 *    REQUEST_CHUNKED_ERROR    Send 411 error if body without Content-Length
1712 *    REQUEST_CHUNKED_DECHUNK  If chunked, remove the chunks for me.
1713 *    REQUEST_CHUNKED_PASS     If chunked, pass the chunk headers with body.
1714 *
1715 *    In order to use the last two options, the caller MUST provide a buffer
1716 *    large enough to hold a chunk-size line, including any extensions.
1717 *
1718 * 2. When you are ready to read a body (if any), call ap_should_client_block().
1719 *    This will tell the module whether or not to read input. If it is 0,
1720 *    the module should assume that there is no message body to read.
1721 *
1722 * 3. Finally, call ap_get_client_block in a loop. Pass it a buffer and its size.
1723 *    It will put data into the buffer (not necessarily a full buffer), and
1724 *    return the length of the input block. When it is done reading, it will
1725 *    return 0 if EOF, or -1 if there was an error.
1726 *    If an error occurs on input, we force an end to keepalive.
1727 *
1728 *    This step also sends a 100 Continue response to HTTP/1.1 clients if appropriate.
1729 */ view japanese
1730
1731AP_DECLARE(int) ap_setup_client_block(request_rec *r, int read_policy)
1732{
1733    const char *tenc = apr_table_get(r->headers_in, "Transfer-Encoding");
1734    const char *lenp = apr_table_get(r->headers_in, "Content-Length");
1735    apr_off_t limit_req_body = ap_get_limit_req_body(r);
1736
1737    r->read_body = read_policy;
1738    r->read_chunked = 0;
1739    r->remaining = 0;
1740
1741    if (tenc) {
1742        if (ap_cstr_casecmp(tenc, "chunked")) {
1743            ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, r, APLOGNO(01592)
1744                          "Unknown Transfer-Encoding %s", tenc);
1745            return HTTP_NOT_IMPLEMENTED;
1746        }
1747        if (r->read_body == REQUEST_CHUNKED_ERROR) {
1748            ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, r, APLOGNO(01593)
1749                          "chunked Transfer-Encoding forbidden: %s", r->uri);
1750            return (lenp) ? HTTP_BAD_REQUEST : HTTP_LENGTH_REQUIRED;
1751        }
1752
1753        r->read_chunked = 1;
1754    }
1755    else if (lenp) {
1756        if (!ap_parse_strict_length(&r->remaining, lenp)) {
1757            r->remaining = 0;
1758            ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, r, APLOGNO(01594)
1759                          "Invalid Content-Length '%s'", lenp);
1760            return HTTP_BAD_REQUEST;
1761        }
1762    }
1763
1764    if ((r->read_body == REQUEST_NO_BODY)
1765        && (r->read_chunked || (r->remaining > 0))) {
1766        ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, r, APLOGNO(01595)
1767                      "%s with body is not allowed for %s", r->method, r->uri);
1768        return HTTP_REQUEST_ENTITY_TOO_LARGE;
1769    }
1770
1771    if (limit_req_body > 0 && (r->remaining > limit_req_body)) {
1772        /* will be logged when the body is discarded */ view japanese
1773        return HTTP_REQUEST_ENTITY_TOO_LARGE;
1774    }
1775
1776#ifdef AP_DEBUG
1777    {
1778        /* Make sure ap_getline() didn't leave any droppings. */ view japanese
1779        core_request_config *req_cfg =
1780            (core_request_config *)ap_get_core_module_config(r->request_config);
1781        AP_DEBUG_ASSERT(APR_BRIGADE_EMPTY(req_cfg->bb));
1782    }
1783#endif
1784
1785    return OK;
1786}

1799/* get_client_block is called in a loop to get the request message body.
1800 * This is quite simple if the client includes a content-length
1801 * (the normal case), but gets messy if the body is chunked. Note that
1802 * r->remaining is used to maintain state across calls and that
1803 * r->read_length is the total number of bytes given to the caller
1804 * across all invocations.  It is messy because we have to be careful not
1805 * to read past the data provided by the client, since these reads block.
1806 * Returns 0 on End-of-body, -1 on error or premature chunk end.
1807 *
1808 */ view japanese
1809AP_DECLARE(long) ap_get_client_block(request_rec *r, char *buffer,
1810                                     apr_size_t bufsiz)
1811{
1812    apr_status_t rv;
1813    apr_bucket_brigade *bb;
1814
1815    if (r->remaining < 0 || (!r->read_chunked && r->remaining == 0)) {
1816        return 0;
1817    }
1818
1819    bb = apr_brigade_create(r->pool, r->connection->bucket_alloc);
1820    if (bb == NULL) {
1821        r->connection->keepalive = AP_CONN_CLOSE;
1822        return -1;
1823    }
1824
1825    rv = ap_get_brigade(r->input_filters, bb, AP_MODE_READBYTES,
1826                        APR_BLOCK_READ, bufsiz);
1827
1828    /* We lose the failure code here.  This is why ap_get_client_block should
1829     * not be used.
1830     */ view japanese
1831    if (rv == AP_FILTER_ERROR) {
1832        /* AP_FILTER_ERROR means a filter has responded already,
1833         * we are DONE.
1834         */ view japanese
1835        apr_brigade_destroy(bb);
