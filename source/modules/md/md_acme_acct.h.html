26/** 
27 * An ACME account at an ACME server.
28 */ view japanese
29typedef struct md_acme_acct_t md_acme_acct_t;
30
31typedef enum {
32    MD_ACME_ACCT_ST_UNKNOWN,
33    MD_ACME_ACCT_ST_VALID,
34    MD_ACME_ACCT_ST_DEACTIVATED,
35    MD_ACME_ACCT_ST_REVOKED,
36} md_acme_acct_st;
37
38struct md_acme_acct_t {
39    const char *id;                 /* short, unique id for the account */ view japanese
40    const char *url;                /* url of the account, once registered */ view japanese
41    const char *ca_url;             /* url of the ACME protocol endpoint */ view japanese
42    md_acme_acct_st status;         /* status of this account */ view japanese
43    apr_array_header_t *contacts;   /* list of contact uris, e.g. mailto:xxx */ view japanese
44    const char *tos_required;       /* terms of service asked for by CA */ view japanese
45    const char *agreement;          /* terms of service agreed to by user */ view japanese
46    const char *orders;             /* URL where certificate orders are found (ACMEv2) */ view japanese
47    const char *eab_kid;            /* external account binding keyid used or NULL */ view japanese
48    const char *eab_hmac;           /* external account binding hmac used or NULL */ view japanese
49    struct md_json_t *registration; /* data from server registration */ view japanese
50};

84/**
85 * Confirm with the server that the current account agrees to the Terms-of-Service
86 * given in the agreement url.
87 * If the known agreement is equal to this, nothing is done.
88 * If it differs, the account is re-validated in the hope that the server
89 * announces the Tos URL it wants. If this is equal to the agreement specified,
90 * the server is notified of this. If the server requires a ToS that the account
91 * thinks it has already given, it is resend.
92 *
93 * If an agreement is required, different from the current one, APR_INCOMPLETE is
94 * returned and the agreement url is returned in the parameter.
95 */ view japanese
96apr_status_t md_acme_check_agreement(md_acme_t *acme, apr_pool_t *p, 
97                                     const char *agreement, const char **prequired);
