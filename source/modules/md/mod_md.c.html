78static int log_is_level(void *baton, apr_pool_t *p, md_log_level_t level)
79{
80    (void)baton;
81    (void)p;
82    if (log_server) {
83        return APLOG_IS_LEVEL(log_server, (int)level);
84    }
85    return level <= MD_LOG_INFO;
86}

88#define LOG_BUF_LEN 16*1024
89
90static void log_print(const char *file, int line, md_log_level_t level,
91                      apr_status_t rv, void *baton, apr_pool_t *p, const char *fmt, va_list ap)
92{
93    if (log_is_level(baton, p, level)) {
94        char buffer[LOG_BUF_LEN];
95
96        memset(buffer, 0, sizeof(buffer));
97        apr_vsnprintf(buffer, LOG_BUF_LEN-1, fmt, ap);
98        buffer[LOG_BUF_LEN-1] = '\0';
99
100        if (log_server) {
101            ap_log_error(file, line, APLOG_MODULE_INDEX, (int)level, rv, log_server, "%s", buffer);
102        }
103        else {
104            ap_log_perror(file, line, APLOG_MODULE_INDEX, (int)level, rv, p, "%s", buffer);
105        }
106    }
107}

136typedef struct {
137    const char *reason;         /* what the notification is about */ view japanese
138    apr_time_t min_interim;     /* minimum time between notifying for this reason */ view japanese
139} notify_rate;
140
141static notify_rate notify_rates[] = {
142    { "renewing", apr_time_from_sec(MD_SECS_PER_HOUR) }, /* once per hour */
143    { "renewed", apr_time_from_sec(MD_SECS_PER_DAY) }, /* once per day */
144    { "installed", apr_time_from_sec(MD_SECS_PER_DAY) }, /* once per day */
145    { "expiring", apr_time_from_sec(MD_SECS_PER_DAY) },     /* once per day */
146    { "errored", apr_time_from_sec(MD_SECS_PER_HOUR) },     /* once per hour */
147    { "ocsp-renewed", apr_time_from_sec(MD_SECS_PER_DAY) }, /* once per day */ view japanese
148    { "ocsp-errored", apr_time_from_sec(MD_SECS_PER_HOUR) }, /* once per hour */ view japanese
149};

151static apr_status_t notify(md_job_t *job, const char *reason,
152                           md_result_t *result, apr_pool_t *p, void *baton)
153{
154    md_mod_conf_t *mc = baton;
155    const char * const *argv;
156    const char *cmdline;
157    int exit_code;
158    apr_status_t rv = APR_SUCCESS;
159    apr_time_t min_interim = 0;
160    md_timeperiod_t since_last;
161    const char *log_msg_reason;
162    int i;
163
164    log_msg_reason = apr_psprintf(p, "message-%s", reason);
165    for (i = 0; i < (int)(sizeof(notify_rates)/sizeof(notify_rates[0])); ++i) {
166        if (!strcmp(reason, notify_rates[i].reason)) {
167            min_interim = notify_rates[i].min_interim;
168        }
169    }
170    if (min_interim > 0) {
171        since_last.start = md_job_log_get_time_of_latest(job, log_msg_reason);
172        since_last.end = apr_time_now();
173        if (since_last.start > 0 && md_timeperiod_length(&since_last) < min_interim) {
174            /* not enough time has passed since we sent the last notification
175             * for this reason. */ view japanese
176            md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, 0, p, APLOGNO(10267)
177                "%s: rate limiting notification about '%s'", job->mdomain, reason);
178            return APR_SUCCESS;
179        }
180    }
181
182    if (!strcmp("renewed", reason)) {
183        if (mc->notify_cmd) {
184            cmdline = apr_psprintf(p, "%s %s", mc->notify_cmd, job->mdomain);
185            apr_tokenize_to_argv(cmdline, (char***)&argv, p);
186            rv = md_util_exec(p, argv[0], argv, &exit_code);
187
188            if (APR_SUCCESS == rv && exit_code) rv = APR_EGENERAL;
189            if (APR_SUCCESS != rv) {
190                md_result_problem_printf(result, rv, MD_RESULT_LOG_ID(APLOGNO(10108)),
191                                         "MDNotifyCmd %s failed with exit code %d.",
192                                         mc->notify_cmd, exit_code);
193                md_result_log(result, MD_LOG_ERR);
194                md_job_log_append(job, "notify-error", result->problem, result->detail);
195                return rv;
196            }
197        }
198        md_log_perror(MD_LOG_MARK, MD_LOG_NOTICE, 0, p, APLOGNO(10059)
199                     "The Managed Domain %s has been setup and changes "
200                     "will be activated on next (graceful) server restart.", job->mdomain);
201    }
202    if (mc->message_cmd) {
203        cmdline = apr_psprintf(p, "%s %s %s", mc->message_cmd, reason, job->mdomain);
204        apr_tokenize_to_argv(cmdline, (char***)&argv, p);
205        rv = md_util_exec(p, argv[0], argv, &exit_code);
206
207        if (APR_SUCCESS == rv && exit_code) rv = APR_EGENERAL;
208        if (APR_SUCCESS != rv) {
209            md_result_problem_printf(result, rv, MD_RESULT_LOG_ID(APLOGNO(10109)),
210                                     "MDMessageCmd %s failed with exit code %d.",
211                                     mc->message_cmd, exit_code);
212            md_result_log(result, MD_LOG_ERR);
213            md_job_log_append(job, "message-error", reason, result->detail);
214            return rv;
215        }
216    }
217
218    md_job_log_append(job, log_msg_reason, NULL, NULL);
219    return APR_SUCCESS;
220}

232static apr_status_t store_file_ev(void *baton, struct md_store_t *store,
233                                    md_store_fs_ev_t ev, unsigned int group,
234                                    const char *fname, apr_filetype_e ftype,
235                                    apr_pool_t *p)
236{
237    server_rec *s = baton;
238    apr_status_t rv;
239
240    (void)store;
241    ap_log_error(APLOG_MARK, APLOG_TRACE3, 0, s, "store event=%d on %s %s (group %d)",
242                 ev, (ftype == APR_DIR)? "dir" : "file", fname, group);
243
244    /* Directories in group CHALLENGES, STAGING and OCSP are written to
245     * under a different user. Give her ownership.
246     */ view japanese
247    if (ftype == APR_DIR) {
248        switch (group) {
249            case MD_SG_CHALLENGES:
250            case MD_SG_STAGING:
251            case MD_SG_OCSP:
252                rv = md_make_worker_accessible(fname, p);
253                if (APR_ENOTIMPL != rv) {
254                    return rv;
255                }
256                break;
257            default:
258                break;
259        }
260    }
261    return APR_SUCCESS;
262}

264static apr_status_t check_group_dir(md_store_t *store, md_store_group_t group,
265                                    apr_pool_t *p, server_rec *s)
266{
267    const char *dir;
268    apr_status_t rv;
269
270    if (APR_SUCCESS == (rv = md_store_get_fname(&dir, store, group, NULL, NULL, p))
271        && APR_SUCCESS == (rv = apr_dir_make_recursive(dir, MD_FPROT_D_UALL_GREAD, p))) {
272        rv = store_file_ev(s, store, MD_S_FS_EV_CREATED, group, dir, APR_DIR, p);
273    }
274    return rv;
275}

277static apr_status_t setup_store(md_store_t **pstore, md_mod_conf_t *mc,
278                                apr_pool_t *p, server_rec *s)
279{
280    const char *base_dir;
281    apr_status_t rv;
282
283    base_dir = ap_server_root_relative(p, mc->base_dir);
284
285    if (APR_SUCCESS != (rv = md_store_fs_init(pstore, p, base_dir))) {
286        ap_log_error(APLOG_MARK, APLOG_ERR, rv, s, APLOGNO(10046)"setup store for %s", base_dir);
287        goto leave;
288    }
289
290    md_store_fs_set_event_cb(*pstore, store_file_ev, s);
291    if (APR_SUCCESS != (rv = check_group_dir(*pstore, MD_SG_CHALLENGES, p, s))
292        || APR_SUCCESS != (rv = check_group_dir(*pstore, MD_SG_STAGING, p, s))
293        || APR_SUCCESS != (rv = check_group_dir(*pstore, MD_SG_ACCOUNTS, p, s))
294        || APR_SUCCESS != (rv = check_group_dir(*pstore, MD_SG_OCSP, p, s))
295        ) {
296        ap_log_error(APLOG_MARK, APLOG_ERR, rv, s, APLOGNO(10047)
297                     "setup challenges directory");
298        goto leave;
299    }
300
301leave:
302    return rv;
303}

321    }
322    if (!md->ca_agreement) {
323        md->ca_agreement = md_config_gets(md->sc, MD_CONFIG_CA_AGREEMENT);
324    }
325    contact = md_config_gets(md->sc, MD_CONFIG_CA_CONTACT);
326    if (md->contacts && md->contacts->nelts > 0) {
327        /* set explicitly */
328    }
329    else if (contact && contact[0]) {

338        md_util_schemify(p, md->sc->s->server_admin, "mailto");

362        md->must_staple = md_config_geti(md->sc, MD_CONFIG_MUST_STAPLE);
363    }
364    if (md->stapling < 0) {
365        md->stapling = md_config_geti(md->sc, MD_CONFIG_STAPLING);
366    }
367    if (!md->profile) {
368        md->profile = md_config_gets(md->sc, MD_CONFIG_CA_PROFILE);
369    }
370    if (md->profile_mandatory < 0) {
371        md->profile_mandatory = md_config_geti(md->sc, MD_CONFIG_CA_PROFILE_MANDATORY);
372    }
373    if (md->ari_renewals < 0) {
374        md->ari_renewals = md_config_geti(md->sc, MD_CONFIG_ARI_RENEWALS);
375    }

378static apr_status_t check_coverage(md_t *md, const char *domain, server_rec *s,
379                                   int *pupdates, apr_pool_t *p)
380{
381    if (md_contains(md, domain, 0)) {
382        return APR_SUCCESS;
383    }
384    else if (md->transitive) {
385        APR_ARRAY_PUSH(md->domains, const char*) = apr_pstrdup(p, domain);
386        *pupdates |= MD_UPD_DOMAINS;
387        return APR_SUCCESS;
388    }
389    else {
390        ap_log_error(APLOG_MARK, APLOG_WARNING, 0, s, APLOGNO(10040)
391                     "Virtual Host %s:%d matches Managed Domain '%s', but the "
392                     "name/alias %s itself is not managed. A requested MD certificate "
393                     "will not match ServerName.",
394                     s->server_hostname, s->port, md->name, domain);
395        return APR_SUCCESS;
396    }
397}

399static apr_status_t md_cover_server(md_t *md, server_rec *s, int *pupdates, apr_pool_t *p)
400{
401    apr_status_t rv;
402    const char *name;
403    int i;
404
405    if (APR_SUCCESS == (rv = check_coverage(md, s->server_hostname, s, pupdates, p))) {
406        ap_log_error(APLOG_MARK, APLOG_TRACE1, 0, s,
407                     "md[%s]: auto add, covers name %s", md->name, s->server_hostname);
408        for (i = 0; s->names && i < s->names->nelts; ++i) {
409            name = APR_ARRAY_IDX(s->names, i, const char*);
410            if (APR_SUCCESS != (rv = check_coverage(md, name, s, pupdates, p))) {
411                break;
412            }
413            ap_log_error(APLOG_MARK, APLOG_TRACE1, 0, s,
414                         "md[%s]: auto add, covers alias %s", md->name, name);
415        }
416    }
417    return rv;
418}

420static int uses_port(server_rec *s, int port)
421{
422    server_addr_rec *sa;
423    int match = 0;
424    for (sa = s->addrs; sa; sa = sa->next) {
425        if (sa->host_port == port) {
426            /* host_addr might be general (0.0.0.0) or specific, we count this as match */ view japanese
427            match = 1;
428        }
429        else {
430            /* uses other port/wildcard */ view japanese
431            return 0;
432        }
433    }
434    return match;
435}

437static apr_status_t detect_supported_protocols(md_mod_conf_t *mc, server_rec *s,
438                                               apr_pool_t *p, int log_level)
439{
440    ap_listen_rec *lr;
441    apr_sockaddr_t *sa;
442    int can_http, can_https;
443
444    if (mc->can_http >= 0 && mc->can_https >= 0) goto set_and_leave;
445
446    can_http = can_https = 0;
447    for (lr = ap_listeners; lr; lr = lr->next) {
448        for (sa = lr->bind_addr; sa; sa = sa->next) {
449            if  (sa->port == mc->local_80
450                 && (!lr->protocol || !strncmp("http", lr->protocol, 4))) {
451                can_http = 1;
452            }
453            else if (sa->port == mc->local_443
454                     && (!lr->protocol || !strncmp("http", lr->protocol, 4))) {
455                can_https = 1;
456            }
457        }
458    }
459    if (mc->can_http < 0) mc->can_http = can_http;
460    if (mc->can_https < 0) mc->can_https = can_https;
461    ap_log_error(APLOG_MARK, log_level, 0, s, APLOGNO(10037)
462                 "server seems%s reachable via http: and%s reachable via https:",
463                 mc->can_http? "" : " not", mc->can_https? "" : " not");
464set_and_leave:
465    return md_reg_set_props(mc->reg, p, mc->can_http, mc->can_https);
466}

468static server_rec *get_public_https_server(md_t *md, const char *domain, server_rec *base_server)
469{
470    md_srv_conf_t *sc;
471    md_mod_conf_t *mc;
472    server_rec *s;
473    server_rec *res = NULL;
474    request_rec r;
475    int i;
476    int check_port = 1;
477
478    sc = md_config_get(base_server);
479    mc = sc->mc;
480    memset(&r, 0, sizeof(r));
481
482    if (md->ca_challenges && md->ca_challenges->nelts > 0) {
483        /* skip the port check if "tls-alpn-01" is pre-configured */ view japanese
484        check_port = !(md_array_str_index(md->ca_challenges, MD_AUTHZ_TYPE_TLSALPN01, 0, 0) >= 0);
485    }
486
487    if (check_port && !mc->can_https) return NULL;
488
489    /* find an ssl server matching domain from MD */ view japanese
490    for (s = base_server; s; s = s->next) {
491        sc = md_config_get(s);
492        if (!sc || !sc->is_ssl || !sc->assigned) continue;
493        if (base_server == s && !mc->manage_base_server) continue;
494        if (base_server != s && check_port && mc->local_443 > 0 && !uses_port(s, mc->local_443)) continue;
495        for (i = 0; i < sc->assigned->nelts; ++i) {
496            if (md == APR_ARRAY_IDX(sc->assigned, i, md_t*)) {
497                r.server = s;
498                if (ap_matches_request_vhost(&r, domain, s->port)) {
499                    if (check_port) {
500                        return s;
501                    }
502                    else {
503                        /* there may be multiple matching servers because we ignore the port.
504                           if possible, choose a server that supports the acme-tls/1 protocol */ view japanese
505                        if (ap_is_allowed_protocol(NULL, NULL, s, PROTO_ACME_TLS_1)) {
506                            return s;
507                        }
508                        res = s;
509                    }
510                }
511            }
512        }
513    }
514    return res;
515}

517static apr_status_t auto_add_domains(md_t *md, server_rec *base_server, apr_pool_t *p)
518{
519    md_srv_conf_t *sc;
520    server_rec *s;
521    apr_status_t rv = APR_SUCCESS;
522    int updates;
523
524    /* Ad all domain names used in SSL VirtualHosts, if not already there */ view japanese
525    ap_log_error(APLOG_MARK, APLOG_TRACE1, 0, base_server,
526                 "md[%s]: auto add domains", md->name);
527    updates = 0;
528    for (s = base_server; s; s = s->next) {
529        sc = md_config_get(s);
530        if (!sc || !sc->is_ssl || !sc->assigned || sc->assigned->nelts != 1) continue;
531        if (md != APR_ARRAY_IDX(sc->assigned, 0, md_t*)) continue;
532        if (APR_SUCCESS != (rv = md_cover_server(md, s, &updates, p))) {
533            return rv;
534        }
535    }
536    return rv;
537}

539static void init_acme_tls_1_domains(md_t *md, server_rec *base_server)
540{
541    md_srv_conf_t *sc;
542    md_mod_conf_t *mc;
543    server_rec *s;
544    int i;
545    const char *domain;
546
547    /* Collect those domains that support the "acme-tls/1" protocol. This
548     * is part of the MD (and not tested dynamically), since challenge selection
549     * may be done outside the server, e.g. in the a2md command. */ view japanese
550    sc = md_config_get(base_server);
551    mc = sc->mc;
552    apr_array_clear(md->acme_tls_1_domains);
553    for (i = 0; i < md->domains->nelts; ++i) {
554        domain = APR_ARRAY_IDX(md->domains, i, const char*);
555        s = get_public_https_server(md, domain, base_server);
556        /* If we did not find a specific virtualhost for md and manage
557         * the base_server, that one is inspected */ view japanese
558        if (NULL == s && mc->manage_base_server) s = base_server;
559        if (NULL == s) {
560            ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, base_server, APLOGNO(10168)
561                         "%s: no https server_rec found for %s", md->name, domain);
562            continue;
563        }
564        if (!ap_is_allowed_protocol(NULL, NULL, s, PROTO_ACME_TLS_1)) {
565            ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, base_server, APLOGNO(10169)
566                         "%s: https server_rec for %s does not have protocol %s enabled",
567                         md->name, domain, PROTO_ACME_TLS_1);
568            continue;
569        }
570        APR_ARRAY_PUSH(md->acme_tls_1_domains, const char*) = domain;
571    }

574static apr_status_t link_md_to_servers(md_mod_conf_t *mc, md_t *md, server_rec *base_server,
575                                       apr_pool_t *p)
576{
577    server_rec *s;
578    request_rec r;
579    md_srv_conf_t *sc;
580    int i;
581    const char *domain, *uri;
582
583    sc = md_config_get(base_server);
584
585    /* Assign the MD to all server_rec configs that it matches. If there already
586     * is an assigned MD not equal this one, the configuration is in error.
587     */ view japanese
588    memset(&r, 0, sizeof(r));
589    for (s = base_server; s; s = s->next) {
590        if (!mc->manage_base_server && s == base_server) {
591            /* we shall not assign ourselves to the base server */ view japanese
592            continue;
593        }
594
595        r.server = s;
596        for (i = 0; i < md->domains->nelts; ++i) {
597            domain = APR_ARRAY_IDX(md->domains, i, const char*);
598
599            if ((mc->match_mode == MD_MATCH_ALL &&
600                 ap_matches_request_vhost(&r, domain, s->port))
601                || (((mc->match_mode == MD_MATCH_SERVERNAMES) || md_dns_is_wildcard(p, domain)) &&
602                    md_dns_matches(domain, s->server_hostname))) {
603                /* Create a unique md_srv_conf_t record for this server, if there is none yet */ view japanese
604                sc = md_config_get_unique(s, p);
605                if (!sc->assigned) sc->assigned = apr_array_make(p, 2, sizeof(md_t*));
606                if (sc->assigned->nelts == 1 && mc->match_mode == MD_MATCH_SERVERNAMES) {
607                    /* there is already an MD assigned for this server. But in
608                     * this match mode, wildcard matches are pre-empted by non-wildcards */ view japanese
609                    int existing_wild = md_is_wild_match(
610                          APR_ARRAY_IDX(sc->assigned, 0, const md_t*)->domains,
611                          s->server_hostname);
612                    if (!existing_wild && md_dns_is_wildcard(p, domain))
613                        continue;  /* do not add */ view japanese
614                    if (existing_wild && !md_dns_is_wildcard(p, domain))
615                        sc->assigned->nelts = 0;  /* overwrite existing */ view japanese
616                }
617                APR_ARRAY_PUSH(sc->assigned, md_t*) = md;
618                ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, base_server, APLOGNO(10041)
619                             "Server %s:%d matches md %s (config %s, match-mode=%d) "
620                             "for domain %s, has now %d MDs",
621                             s->server_hostname, s->port, md->name, sc->name,
622                             mc->match_mode, domain, (int)sc->assigned->nelts);
623
624                if (md->contacts && md->contacts->nelts > 0) {
625                    /* set explicitly */ view japanese
626                }
627                else if (sc->ca_contact && sc->ca_contact[0]) {
628                    uri = md_util_schemify(p, sc->ca_contact, "mailto");
629                    if (md_array_str_index(md->contacts, uri, 0, 0) < 0) {
630                        APR_ARRAY_PUSH(md->contacts, const char *) = uri;
631                        ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, base_server, APLOGNO(10044)
632                                     "%s: added contact %s", md->name, uri);
633                    }
634                }
635                else if (s->server_admin && strcmp(DEFAULT_ADMIN, s->server_admin)) {
636                    uri = md_util_schemify(p, s->server_admin, "mailto");
637                    if (md_array_str_index(md->contacts, uri, 0, 0) < 0) {
638                        APR_ARRAY_PUSH(md->contacts, const char *) = uri;
639                        ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, base_server, APLOGNO(10237)
640                                     "%s: added contact %s", md->name, uri);
641                    }
642                }
643                break;
644            }
645        }
646    }
647    return APR_SUCCESS;
648}

667static apr_status_t merge_mds_with_conf(md_mod_conf_t *mc, apr_pool_t *p,
668                                        server_rec *base_server, int log_level)
669{
670    md_srv_conf_t *base_conf;
671    md_t *md, *omd;
672    const char *domain;
673    md_timeslice_t *ts;
674    apr_status_t rv = APR_SUCCESS;
675    int i, j;
676
677    /* The global module configuration 'mc' keeps a list of all configured MDomains
678     * in the server. This list is collected during configuration processing and,
679     * in the post config phase, get updated from all merged server configurations
680     * before the server starts processing.
681     */ view japanese
682    base_conf = md_config_get(base_server);
683    md_config_get_timespan(&ts, base_conf, MD_CONFIG_RENEW_WINDOW);
684    if (ts) md_reg_set_renew_window_default(mc->reg, ts);
685    md_config_get_timespan(&ts, base_conf, MD_CONFIG_WARN_WINDOW);
686    if (ts) md_reg_set_warn_window_default(mc->reg, ts);
687
688    /* Complete the properties of the MDs, now that we have the complete, merged
689     * server configurations.
690     */ view japanese
691    for (i = 0; i < mc->mds->nelts; ++i) {
692        md = APR_ARRAY_IDX(mc->mds, i, md_t*);
693        merge_srv_config(md, base_conf, p);
694
695        if (mc->match_mode == MD_MATCH_ALL) {
696          /* Check that we have no overlap with the MDs already completed */ view japanese
697          for (j = 0; j < i; ++j) {
698              omd = APR_ARRAY_IDX(mc->mds, j, md_t*);
699              if ((domain = md_common_name(md, omd)) != NULL) {
700                  ap_log_error(APLOG_MARK, APLOG_ERR, 0, base_server, APLOGNO(10038)
701                               "two Managed Domains have an overlap in domain '%s'"
702                               ", first definition in %s(line %d), second in %s(line %d)",
703                               domain, md->defn_name, md->defn_line_number,
704                               omd->defn_name, omd->defn_line_number);
705                  return APR_EINVAL;
706              }
707          }
708        }
709
710        if (md->cert_files && md->cert_files->nelts) {
711            if (!md->pkey_files || (md->cert_files->nelts != md->pkey_files->nelts)) {
712                ap_log_error(APLOG_MARK, APLOG_ERR, 0, base_server, APLOGNO(10170)
713                             "The Managed Domain '%s' "
714                             "needs one MDCertificateKeyFile for each MDCertificateFile.",
715                             md->name);
716                return APR_EINVAL;
717            }
718        }
719        else if (md->pkey_files && md->pkey_files->nelts 
720            && (!md->cert_files || !md->cert_files->nelts)) {
721            ap_log_error(APLOG_MARK, APLOG_ERR, 0, base_server, APLOGNO(10171)
722                         "The Managed Domain '%s' "
723                         "has MDCertificateKeyFile(s) but no MDCertificateFile.",
724                         md->name);
725            return APR_EINVAL;
726        }
727
728        if (APLOG_IS_LEVEL(base_server, log_level)) {
729            ap_log_error(APLOG_MARK, log_level, 0, base_server, APLOGNO(10039)
730                         "Completed MD[%s, CA=%s, Proto=%s, Agreement=%s, renew-mode=%d "
731                         "renew_window=%s, warn_window=%s",
732                         md->name, md->ca_effective, md->ca_proto, md->ca_agreement, md->renew_mode,
733                         md->renew_window? md_timeslice_format(md->renew_window, p) : "unset",
734                         md->warn_window? md_timeslice_format(md->warn_window, p) : "unset");
735        }
736    }
737    return rv;
738}

740static apr_status_t check_invalid_duplicates(server_rec *base_server)
741{
742    server_rec *s;
743    md_srv_conf_t *sc;
744
745    ap_log_error( APLOG_MARK, APLOG_TRACE1, 0, base_server,
746                 "checking duplicate ssl assignments");
747    for (s = base_server; s; s = s->next) {
748        sc = md_config_get(s);
749        if (!sc || !sc->assigned) continue;
750
751        if (sc->assigned->nelts > 1 && sc->is_ssl) {
752            /* duplicate assignment to SSL VirtualHost, not allowed */ view japanese
753            ap_log_error(APLOG_MARK, APLOG_ERR, 0, base_server, APLOGNO(10042)
754                         "conflict: %d MDs match to SSL VirtualHost %s, there can at most be one.",
755                         (int)sc->assigned->nelts, s->server_hostname);
756            return APR_EINVAL;
757        }
758    }
759    return APR_SUCCESS;
760}

762static apr_status_t check_usage(md_mod_conf_t *mc, md_t *md, server_rec *base_server,
763                                apr_pool_t *p, apr_pool_t *ptemp)
764{
765    server_rec *s;
766    md_srv_conf_t *sc;
767    apr_status_t rv = APR_SUCCESS;
768    int i, has_ssl;
769    apr_array_header_t *servers;
770
771    (void)p;
772    servers = apr_array_make(ptemp, 5, sizeof(server_rec*));
773    has_ssl = 0;
774    for (s = base_server; s; s = s->next) {
775        sc = md_config_get(s);
776        if (!sc || !sc->assigned) continue;
777        for (i = 0; i < sc->assigned->nelts; ++i) {
778            if (md == APR_ARRAY_IDX(sc->assigned, i, md_t*)) {
779                APR_ARRAY_PUSH(servers, server_rec*) = s;
780                if (sc->is_ssl) has_ssl = 1;
781            }
782        }
783    }
784
785    if (!has_ssl && md->require_https > MD_REQUIRE_OFF) {
786        /* We require https for this MD, but do we have a SSL vhost? */ view japanese
787        ap_log_error(APLOG_MARK, APLOG_WARNING, 0, base_server, APLOGNO(10105)
788                     "MD %s does not match any VirtualHost with 'SSLEngine on', "
789                     "but is configured to require https. This cannot work.", md->name);
790    }
791    if (apr_is_empty_array(servers)) {
792        if (md->renew_mode != MD_RENEW_ALWAYS) {
793            /* Not an error, but looks suspicious */ view japanese
794            ap_log_error(APLOG_MARK, APLOG_WARNING, 0, base_server, APLOGNO(10045)
795                         "No VirtualHost matches Managed Domain %s", md->name);
796            APR_ARRAY_PUSH(mc->unused_names, const char*)  = md->name;
797        }
798    }
799    return rv;
800}

802static int init_cert_watch_status(md_mod_conf_t *mc, apr_pool_t *p, apr_pool_t *ptemp, server_rec *s)
803{
804    md_t *md;
805    md_result_t *result;
806    int i, count;
807
808    /* Calculate the list of MD names which we need to watch:
809     * - all MDs that are used somewhere
810     * - all MDs in drive mode 'AUTO' that are not in 'unused_names'
811     */ view japanese
812    count = 0;
813    result = md_result_make(ptemp, APR_SUCCESS);
814    for (i = 0; i < mc->mds->nelts; ++i) {
815        md = APR_ARRAY_IDX(mc->mds, i, md_t*);
816        md_result_set(result, APR_SUCCESS, NULL);
817        md->watched = 0;
818        if (md->state == MD_S_ERROR) {
819            md_result_set(result, APR_EGENERAL,
820                          "in error state, unable to drive forward. This "
821                          "indicates an incomplete or inconsistent configuration. "
822                          "Please check the log for warnings in this regard.");
823            continue;
824        }
825
826        if (md->renew_mode == MD_RENEW_AUTO
827            && md_array_str_index(mc->unused_names, md->name, 0, 0) >= 0) {
828            /* This MD is not used in any virtualhost, do not watch */ view japanese
829            continue;
830        }
831
832        if (md_will_renew_cert(md)) {
833            /* make a test init to detect early errors. */ view japanese
834            md_reg_test_init(mc->reg, md, mc->env, result, p);
835            if (APR_SUCCESS != result->status && result->detail) {
836                apr_hash_set(mc->init_errors, md->name, APR_HASH_KEY_STRING, apr_pstrdup(p, result->detail));
837                ap_log_error(APLOG_MARK, APLOG_ERR, 0, s, APLOGNO(10173)
838                             "md[%s]: %s", md->name, result->detail);
839            }
840        }
841
842        md->watched = 1;
843        ++count;
844    }
845    return count;
846}

848static apr_status_t md_post_config_before_ssl(apr_pool_t *p, apr_pool_t *plog,
849                                              apr_pool_t *ptemp, server_rec *s)
850{
851    void *data = NULL;
852    const char *mod_md_init_key = "mod_md_init_counter";
853    md_srv_conf_t *sc;
854    md_mod_conf_t *mc;
855    apr_status_t rv = APR_SUCCESS;
856    int dry_run = 0, log_level = APLOG_DEBUG;
857    md_store_t *store;
858
859    apr_pool_userdata_get(&data, mod_md_init_key, s->process->pool);
860    if (data == NULL) {
861        /* At the first start, httpd makes a config check dry run. It
862         * runs all config hooks to check if it can. If so, it does
863         * this all again and starts serving requests.
864         *
865         * On a dry run, we therefore do all the cheap config things we
866         * need to do to find out if the settings are ok. More expensive
867         * things we delay to the real run.
868         */ view japanese
869        dry_run = 1;
870        log_level = APLOG_TRACE1;
871        ap_log_error( APLOG_MARK, log_level, 0, s, APLOGNO(10070)
872                     "initializing post config dry run");
873        apr_pool_userdata_set((const void *)1, mod_md_init_key,
874                              apr_pool_cleanup_null, s->process->pool);
875    }
876    else {
877        ap_log_error( APLOG_MARK, APLOG_INFO, 0, s, APLOGNO(10071)
878                     "mod_md (v%s), initializing...", MOD_MD_VERSION);
879    }
880
881    (void)plog;
882    init_setups(p, s);
883    md_log_set(log_is_level, log_print, NULL);
884
885    md_config_post_config(s, p);
886    sc = md_config_get(s);
887    mc = sc->mc;
888    mc->dry_run = dry_run;
889
890    md_event_init(p);
891    md_event_subscribe(on_event, mc);
892
893    rv = setup_store(&store, mc, p, s);
894    if (APR_SUCCESS != rv) goto leave;
895
896    rv = md_reg_create(&mc->reg, p, store, mc->proxy_url, mc->ca_certs,
897                       mc->min_delay, mc->retry_failover,
898                       mc->use_store_locks, mc->lock_wait_timeout);
899    if (APR_SUCCESS != rv) {
900        ap_log_error(APLOG_MARK, APLOG_ERR, rv, s, APLOGNO(10072) "setup md registry");
901        goto leave;
902    }
903
904    /* renew on 30% remaining /*/ view japanese
905    rv = md_ocsp_reg_make(&mc->ocsp, p, store, mc->ocsp_renew_window,
906                          AP_SERVER_BASEVERSION, mc->proxy_url,
907                          mc->min_delay);
908    if (APR_SUCCESS != rv) {
909        ap_log_error(APLOG_MARK, APLOG_ERR, rv, s, APLOGNO(10196) "setup ocsp registry");
910        goto leave;
911    }
912
913    init_ssl();
914
915    /* How to bootstrap this module:
916     * 1. find out if we know if http: and/or https: requests will arrive
917     * 2. apply the now complete configuration settings to the MDs
918     * 3. Link MDs to the server_recs they are used in. Detect unused MDs.
919     * 4. Update the store with the MDs. Change domain names, create new MDs, etc.
920     *    Basically all MD properties that are configured directly.
921     *    WARNING: this may change the name of an MD. If an MD loses the first
922     *    of its domain names, it first gets the new first one as name. The
923     *    store will find the old settings and "recover" the previous name.
924     * 5. Load any staged data from previous driving.
925     * 6. on a dry run, this is all we do
926     * 7. Read back the MD properties that reflect the existence and aspect of
927     *    credentials that are in the store (or missing there).
928     *    Expiry times, MD state, etc.
929     * 8. Determine the list of MDs that need driving/supervision.
930     * 9. Cleanup any left-overs in registry/store that are no longer needed for
931     *    the list of MDs as we know it now.
932     * 10. If this list is non-empty, setup a watchdog to run.
933     */ view japanese
934    /*1*/
935    if (APR_SUCCESS != (rv = detect_supported_protocols(mc, s, p, log_level))) goto leave;
936    /*2*/
937    if (APR_SUCCESS != (rv = merge_mds_with_conf(mc, p, s, log_level))) goto leave;
938    /*3*/
939    if (APR_SUCCESS != (rv = link_mds_to_servers(mc, s, p))) goto leave;
940    /*4*/
941    if (APR_SUCCESS != (rv = md_reg_lock_global(mc->reg, ptemp))) {
942        ap_log_error(APLOG_MARK, APLOG_ERR, rv, s, APLOGNO(10398)
943                     "unable to obtain global registry lock, "
944                     "renewed certificates may remain inactive on "
945                     "this httpd instance!");
946        /* FIXME: or should we fail the server start/reload here? */ view japanese
947        rv = APR_SUCCESS;
948        goto leave;
949    }
950    if (APR_SUCCESS != (rv = md_reg_sync_start(mc->reg, mc->mds, ptemp))) {
951        ap_log_error(APLOG_MARK, APLOG_ERR, rv, s, APLOGNO(10073)
952                     "syncing %d mds to registry", mc->mds->nelts);
953        goto leave;
954    }
955    /*5*/
956    md_reg_load_stagings(mc->reg, mc->mds, mc->env, p);
957leave:
958    if (mc->reg)
959        md_reg_unlock_global(mc->reg, ptemp);
960    return rv;
961}

963static apr_status_t md_post_config_after_ssl(apr_pool_t *p, apr_pool_t *plog,
964                                             apr_pool_t *ptemp, server_rec *s)
965{
966    md_srv_conf_t *sc;
967    apr_status_t rv = APR_SUCCESS;
968    md_mod_conf_t *mc;
969    int watched, i;
970    md_t *md;
971
972    (void)ptemp;
973    (void)plog;
974    sc = md_config_get(s);
975
976    /*6*/
977    if (!sc || !sc->mc || sc->mc->dry_run) goto leave;
978    mc = sc->mc;
979
980    /*7*/
981    if (APR_SUCCESS != (rv = check_invalid_duplicates(s))) {
982        goto leave;
983    }
984    apr_array_clear(mc->unused_names);
985    for (i = 0; i < mc->mds->nelts; ++i) {
986        md = APR_ARRAY_IDX(mc->mds, i, md_t *);
987
988        ap_log_error( APLOG_MARK, APLOG_TRACE2, rv, s, "md{%s}: auto_add", md->name);
989        if (APR_SUCCESS != (rv = auto_add_domains(md, s, p))) {
990            goto leave;
991        }
992        init_acme_tls_1_domains(md, s);
993        ap_log_error( APLOG_MARK, APLOG_TRACE2, rv, s, "md{%s}: check_usage", md->name);
994        if (APR_SUCCESS != (rv = check_usage(mc, md, s, p, ptemp))) {
995            goto leave;
996        }
997        ap_log_error( APLOG_MARK, APLOG_TRACE2, rv, s, "md{%s}: sync_finish", md->name);
998        if (APR_SUCCESS != (rv = md_reg_sync_finish(mc->reg, md, p, ptemp))) {
999            ap_log_error( APLOG_MARK, APLOG_ERR, rv, s, APLOGNO(10172)
1000                         "md[%s]: error syncing to store", md->name);
1001            goto leave;
1002        }
1003    }
1004    /*8*/
1005    ap_log_error( APLOG_MARK, APLOG_TRACE2, rv, s, "init_cert_watch");
1006    watched = init_cert_watch_status(mc, p, ptemp, s);
1007    /*9*/
1008    ap_log_error( APLOG_MARK, APLOG_TRACE2, rv, s, "cleanup challenges");
1009    md_reg_cleanup_challenges(mc->reg, p, ptemp, mc->mds);
1010
1011    /* From here on, the domains in the registry are readonly
1012     * and only staging/challenges may be manipulated */ view japanese
1013    md_reg_freeze_domains(mc->reg, mc->mds);
1014
1015    if (watched) {
1016        /*10*/
1017        ap_log_error(APLOG_MARK, APLOG_DEBUG, rv, s, APLOGNO(10074)
1018                     "%d out of %d mds need watching", watched, mc->mds->nelts);
1019
1020        md_http_use_implementation(md_curl_get_impl(p));
1021        rv = md_renew_start_watching(mc, s, p);
1022    }
1023    else {
1024        ap_log_error( APLOG_MARK, APLOG_DEBUG, 0, s, APLOGNO(10075) "no mds to supervise");
1025    }
1026
1027    if (!mc->ocsp || md_ocsp_count(mc->ocsp) == 0) {
1028        ap_log_error( APLOG_MARK, APLOG_TRACE1, 0, s, "no ocsp to manage");
1029        goto leave;
1030    }
1031
1032    md_http_use_implementation(md_curl_get_impl(p));
1033    rv = md_ocsp_start_watching(mc, s, p);
1034
1035leave:
1036    ap_log_error( APLOG_MARK, APLOG_TRACE2, rv, s, "post_config done");
1037    return rv;
1038}

1225                                 md->name, md_pkey_spec_name(spec), s->server_hostname);

1239        *pkey_files = key_files;
1240        *pcert_files = chain_files;
1241    }
1242    else if (APR_SUCCESS == rv) {
1243        rv = APR_ENOENT;
1244    }
1245    return rv;

1344static int md_answer_challenge(conn_rec *c, const char *servername,
1345                               const char **pcert_pem, const char **pkey_pem)
1346{
1347    const char *protocol;
1348    int hook_rv = DECLINED;
1349    md_srv_conf_t *sc;
1350
1351    *pcert_pem = *pkey_pem = NULL;
1352
1353    if (!servername
1354        || !(protocol = md_protocol_get(c))
1355        || strcmp(PROTO_ACME_TLS_1, protocol)) {
1356        goto cleanup;
1357    }
1358    sc = md_config_get(c->base_server);
1359    if (!sc || !sc->mc->reg) goto cleanup;
1360
1361    ap_log_cerror(APLOG_MARK, APLOG_TRACE1, 0, c,
1362                  "Answer challenge[tls-alpn-01] for %s", servername);
1363
1364    /* A challenge for TLS-ALPN-01 used to have a single certificate,
1365     * overriding the single fallback certificate already installed in
1366     * the connections SSL* instance.
1367     * Since the addition of `MDPrivateKeys`, there can be more than one,
1368     * but the server API for a challenge cert can return only one. This
1369     * is a short coming of the API.
1370     * This means we cannot override all fallback certificates present, just
1371     * a single one. If there is an RSA cert in fallback, we need to override
1372     * that, because the ACME server has a preference for that (at least LE
1373     * has). So we look for an RSA challenge cert first.
1374     * The fallback is an EC cert and that works since without RSA challenges,
1375     * there should also be no RSA fallbacks.
1376     * Bit of a mess. */ view japanese
1377    hook_rv = md_get_challenge_cert(c, servername, sc, MD_PKEY_TYPE_DEFAULT,
1378                                    pcert_pem, pkey_pem);
1379    if (hook_rv == DECLINED)
1380      hook_rv = md_get_challenge_cert(c, servername, sc, MD_PKEY_TYPE_RSA,
1381                                      pcert_pem, pkey_pem);
1382    if (hook_rv == DECLINED)
1383      hook_rv = md_get_challenge_cert(c, servername, sc, MD_PKEY_TYPE_EC,
1384                                      pcert_pem, pkey_pem);
1385
1386    if (DECLINED == hook_rv) {
1387        ap_log_cerror(APLOG_MARK, APLOG_INFO, 0, c, APLOGNO(10080)
1388                      "%s: unknown tls-alpn-01 challenge host", servername);
1389    }
1390
1391cleanup:
1392    return hook_rv;
1393}

1491     * https: redirects or HSTS header additions.

1495        || !strncmp(WELL_KNOWN_PREFIX, r->parsed_uri.path, sizeof(WELL_KNOWN_PREFIX)-1)) {

1501    if (!md) goto declined;

1505         * if 'permanent' and no one else set a HSTS header already, do it */ view japanese

1507            && sc->mc->hsts_header && !apr_table_get(r->headers_out, MD_HSTS_HEADER)) {
1508            apr_table_setn(r->headers_out, MD_HSTS_HEADER, sc->mc->hsts_header);
1509        }
1510    }
1511    else {

1515                /* safe to use the old-fashioned codes */ view japanese
1516                status = ((MD_REQUIRE_PERMANENT == md->require_https)?
1517                          HTTP_MOVED_PERMANENTLY : HTTP_MOVED_TEMPORARILY);
1518            }
1519            else {
1520                /* these should keep the method unchanged on retry */ view japanese
1521                status = ((MD_REQUIRE_PERMANENT == md->require_https)?
1522                          HTTP_PERMANENT_REDIRECT : HTTP_TEMPORARY_REDIRECT);
1523            }
1524
1525            s = ap_construct_url(r->pool, r->uri, r);
1526            if (APR_SUCCESS == apr_uri_parse(r->pool, s, &uri)) {
1527                uri.scheme = (char*)"https";
1528                uri.port = 443;
1529                uri.port_str = (char*)"443";
1530                uri.query = r->parsed_uri.query;
1531                uri.fragment = r->parsed_uri.fragment;
1532                s = apr_uri_unparse(r->pool, &uri, APR_URI_UNP_OMITUSERINFO);
1533                if (s && *s) {
1534                    apr_table_setn(r->headers_out, "Location", s);
1535                    return status;
1536                }
1537            }
1538        }
1539    }
1540declined:
1541    return DECLINED;
1542}
