35int md_contains(const md_t *md, const char *domain, int case_sensitive)
36{
37    if (md_array_str_index(md->domains, domain, 0, case_sensitive) >= 0) {
38        return 1;
39    }
40    return md_dns_domains_match(md->domains, domain);
41}
42
43const char *md_common_name(const md_t *md1, const md_t *md2)
44{
45    int i;
46    
47    if (md1 == NULL || md1->domains == NULL
48        || md2 == NULL || md2->domains == NULL) {
49        return NULL;
50    }
51    
52    for (i = 0; i < md1->domains->nelts; ++i) {
53        const char *name1 = APR_ARRAY_IDX(md1->domains, i, const char*);
54        if (md_contains(md2, name1, 0)) {
55            return name1;
56        }
57    }
58    return NULL;
59}
60
61int md_domains_overlap(const md_t *md1, const md_t *md2)
62{
63    return md_common_name(md1, md2) != NULL;
64}

66apr_size_t md_common_name_count(const md_t *md1, const md_t *md2)
67{
68    int i;
69    apr_size_t hits;
70    
71    if (md1 == NULL || md1->domains == NULL
72        || md2 == NULL || md2->domains == NULL) {
73        return 0;
74    }
75    
76    hits = 0;
77    for (i = 0; i < md1->domains->nelts; ++i) {
78        const char *name1 = APR_ARRAY_IDX(md1->domains, i, const char*);
79        if (md_contains(md2, name1, 0)) {
80            ++hits;
81        }
82    }
83    return hits;
84}

86int md_is_covered_by_alt_names(const md_t *md, const struct apr_array_header_t* alt_names)
87{
88    const char *name;
89    int i;
90    
91    if (alt_names) {
92        for (i = 0; i < md->domains->nelts; ++i) {
93            name = APR_ARRAY_IDX(md->domains, i, const char *);
94            if (!md_dns_domains_match(alt_names, name)) {
95                return 0;
96            }
97        }
98        return 1;
99    }
100    return 0;
101}

103md_t *md_create_empty(apr_pool_t *p)
104{
105    md_t *md = apr_pcalloc(p, sizeof(*md));
106    if (md) {
107        md->domains = apr_array_make(p, 5, sizeof(const char *));
108        md->contacts = apr_array_make(p, 5, sizeof(const char *));
109        md->renew_mode = MD_RENEW_DEFAULT;
110        md->require_https = MD_REQUIRE_UNSET;
111        md->must_staple = -1;
112        md->transitive = -1;
113        md->acme_tls_1_domains = apr_array_make(p, 5, sizeof(const char *));
114        md->stapling = -1;
115        md->profile_mandatory = -1;
116        md->ari_renewals = -1;
117        md->defn_name = "unknown";
118        md->defn_line_number = 0;
119    }
120    return md;
121}

123int md_equal_domains(const md_t *md1, const md_t *md2, int case_sensitive)
124{
125    int i;
126    if (md1->domains->nelts == md2->domains->nelts) {
127        for (i = 0; i < md1->domains->nelts; ++i) {
128            const char *name1 = APR_ARRAY_IDX(md1->domains, i, const char*);
129            if (!md_contains(md2, name1, case_sensitive)) {
130                return 0;
131            }
132        }
133        return 1;
134    }
135    return 0;
136}
137
138int md_contains_domains(const md_t *md1, const md_t *md2)
139{
140    int i;
141    if (md1->domains->nelts >= md2->domains->nelts) {
142        for (i = 0; i < md2->domains->nelts; ++i) {
143            const char *name2 = APR_ARRAY_IDX(md2->domains, i, const char*);
144            if (!md_contains(md1, name2, 0)) {
145                return 0;
146            }
147        }
148        return 1;
149    }
150    return 0;
151}

153md_t *md_get_by_name(struct apr_array_header_t *mds, const char *name)
154{
155    int i;
156    for (i = 0; i < mds->nelts; ++i) {
157        md_t *md = APR_ARRAY_IDX(mds, i, md_t *);
158        if (!strcmp(name, md->name)) {
159            return md;
160        }
161    }
162    return NULL;
163}

165md_t *md_get_by_domain(struct apr_array_header_t *mds, const char *domain)
166{
167    int i;
168    for (i = 0; i < mds->nelts; ++i) {
169        md_t *md = APR_ARRAY_IDX(mds, i, md_t *);
170        if (md_contains(md, domain, 0)) {
171            return md;
172        }
173    }
174    return NULL;
175}

177md_t *md_get_by_dns_overlap(struct apr_array_header_t *mds, const md_t *md)
178{
179    int i;
180    for (i = 0; i < mds->nelts; ++i) {
181        md_t *o = APR_ARRAY_IDX(mds, i, md_t *);
182        if (strcmp(o->name, md->name) && md_common_name(o, md)) {
183            return o;
184        }
185    }
186    return NULL;
187}

189int md_cert_count(const md_t *md)
190{
191    /* cert are defined as a list of static files or a list of private key specs */
192    if (md->cert_files && md->cert_files->nelts) {
193        return md->cert_files->nelts;
194    }
195    return md_pkeys_spec_count(md->pks);
196}

198md_t *md_create(apr_pool_t *p, apr_array_header_t *domains)
199{
200    md_t *md;
201    
202    md = md_create_empty(p);
203    md->domains = md_array_str_compact(p, domains, 0);
204    md->name = APR_ARRAY_IDX(md->domains, 0, const char *);
205    
206    return md;
207}

212md_t *md_copy(apr_pool_t *p, const md_t *src)
213{
214    md_t *md;
215    
216    md = apr_pcalloc(p, sizeof(*md));
217    if (md) {
218        memcpy(md, src, sizeof(*md));
219        md->domains = apr_array_copy(p, src->domains);
220        md->contacts = apr_array_copy(p, src->contacts);
221        if (src->ca_challenges) {
222            md->ca_challenges = apr_array_copy(p, src->ca_challenges);
223        }
224        md->acme_tls_1_domains = apr_array_copy(p, src->acme_tls_1_domains);
225        md->pks = md_pkeys_spec_clone(p, src->pks);
226    }    
227    return md;   
228}

230md_t *md_clone(apr_pool_t *p, const md_t *src)
231{
232    md_t *md;
233    
234    md = apr_pcalloc(p, sizeof(*md));
235    if (md) {
236        md->state = src->state;
237        md->name = apr_pstrdup(p, src->name);
238        md->require_https = src->require_https;
239        md->must_staple = src->must_staple;
240        md->renew_mode = src->renew_mode;
241        md->domains = md_array_str_compact(p, src->domains, 0);
242        md->pks = md_pkeys_spec_clone(p, src->pks);
243        md->renew_window = src->renew_window;
244        md->warn_window = src->warn_window;
245        md->contacts = md_array_str_clone(p, src->contacts);
246        if (src->ca_proto) md->ca_proto = apr_pstrdup(p, src->ca_proto);
247        if (src->ca_urls) {
248            md->ca_urls = md_array_str_clone(p, src->ca_urls);
249        }
250        if (src->ca_effective) md->ca_effective = apr_pstrdup(p, src->ca_effective);
251        if (src->ca_account) md->ca_account = apr_pstrdup(p, src->ca_account);
252        if (src->ca_agreement) md->ca_agreement = apr_pstrdup(p, src->ca_agreement);
253        if (src->defn_name) md->defn_name = apr_pstrdup(p, src->defn_name);
254        md->defn_line_number = src->defn_line_number;
255        if (src->ca_challenges) {
256            md->ca_challenges = md_array_str_clone(p, src->ca_challenges);
257        }
258        md->acme_tls_1_domains = md_array_str_compact(p, src->acme_tls_1_domains, 0);
259        md->stapling = src->stapling;
260        if (src->dns01_cmd) md->dns01_cmd = apr_pstrdup(p, src->dns01_cmd);
261        if (src->cert_files) md->cert_files = md_array_str_clone(p, src->cert_files);
262        if (src->pkey_files) md->pkey_files = md_array_str_clone(p, src->pkey_files);
263    }    
264    return md;   
265}

270md_json_t *md_to_json(const md_t *md, apr_pool_t *p)
271{
272    md_json_t *json = md_json_create(p);
273    if (json) {
274        apr_array_header_t *domains = md_array_str_compact(p, md->domains, 0);
275        md_json_sets(md->name, json, MD_KEY_NAME, NULL);
276        md_json_setsa(domains, json, MD_KEY_DOMAINS, NULL);
277        md_json_setsa(md->contacts, json, MD_KEY_CONTACTS, NULL);
278        md_json_setl(md->transitive, json, MD_KEY_TRANSITIVE, NULL);
279        md_json_sets(md->ca_account, json, MD_KEY_CA, MD_KEY_ACCOUNT, NULL);
280        md_json_sets(md->ca_proto, json, MD_KEY_CA, MD_KEY_PROTO, NULL);
281        md_json_sets(md->ca_effective, json, MD_KEY_CA, MD_KEY_URL, NULL);
282        if (md->ca_urls && !apr_is_empty_array(md->ca_urls)) {
283            md_json_setsa(md->ca_urls, json, MD_KEY_CA, MD_KEY_URLS, NULL);
284        }
285        md_json_sets(md->ca_agreement, json, MD_KEY_CA, MD_KEY_AGREEMENT, NULL);
286        if (!md_pkeys_spec_is_empty(md->pks)) {
287            md_json_setj(md_pkeys_spec_to_json(md->pks, p), json, MD_KEY_PKEY, NULL);
288        }
289        md_json_setl(md->state, json, MD_KEY_STATE, NULL);
290        if (md->state_descr)
291            md_json_sets(md->state_descr, json, MD_KEY_STATE_DESCR, NULL);
292        md_json_setl(md->renew_mode, json, MD_KEY_RENEW_MODE, NULL);
293        if (md->renew_window)
294            md_json_sets(md_timeslice_format(md->renew_window, p), json, MD_KEY_RENEW_WINDOW, NULL);
295        if (md->warn_window)
296            md_json_sets(md_timeslice_format(md->warn_window, p), json, MD_KEY_WARN_WINDOW, NULL);
297        if (md->ca_challenges && md->ca_challenges->nelts > 0) {
298            apr_array_header_t *na;
299            na = md_array_str_compact(p, md->ca_challenges, 0);
300            md_json_setsa(na, json, MD_KEY_CA, MD_KEY_CHALLENGES, NULL);
301        }
302        switch (md->require_https) {
303            case MD_REQUIRE_TEMPORARY:
304                md_json_sets(MD_KEY_TEMPORARY, json, MD_KEY_REQUIRE_HTTPS, NULL);
305                break;
306            case MD_REQUIRE_PERMANENT:
307                md_json_sets(MD_KEY_PERMANENT, json, MD_KEY_REQUIRE_HTTPS, NULL);
308                break;
309            default:
310                break;
311        }
312        md_json_setb(md->must_staple > 0, json, MD_KEY_MUST_STAPLE, NULL);
313        md_json_setsa(md->acme_tls_1_domains, json, MD_KEY_PROTO, MD_KEY_ACME_TLS_1, NULL);
314        if (md->cert_files) md_json_setsa(md->cert_files, json, MD_KEY_CERT_FILES, NULL);
315        if (md->pkey_files) md_json_setsa(md->pkey_files, json, MD_KEY_PKEY_FILES, NULL);
316        md_json_setb(md->stapling > 0, json, MD_KEY_STAPLING, NULL);
317        if (md->dns01_cmd) md_json_sets(md->dns01_cmd, json, MD_KEY_CMD_DNS01, NULL);
318        if (md->ca_eab_kid && strcmp("none", md->ca_eab_kid)) {
319            md_json_sets(md->ca_eab_kid, json, MD_KEY_EAB, MD_KEY_KID, NULL);
320            if (md->ca_eab_hmac) md_json_sets(md->ca_eab_hmac, json, MD_KEY_EAB, MD_KEY_HMAC, NULL);
321        }
322        if (md->profile) md_json_sets(md->profile, json, MD_KEY_PROFILE, NULL);
323        md_json_setb(md->profile_mandatory > 0, json, MD_KEY_PROFILE_MANDATORY, NULL);
324        md_json_setb(md->ari_renewals > 0, json, MD_KEY_ARI_RENEWALS, NULL);
325        return json;
326    }
327    return NULL;
328}

330md_t *md_from_json(md_json_t *json, apr_pool_t *p)
331{
332    const char *s;
333    md_t *md = md_create_empty(p);
334    if (md) {
335        md->name = md_json_dups(p, json, MD_KEY_NAME, NULL);            
336        md_json_dupsa(md->domains, p, json, MD_KEY_DOMAINS, NULL);
337        md_json_dupsa(md->contacts, p, json, MD_KEY_CONTACTS, NULL);
338        md->ca_account = md_json_dups(p, json, MD_KEY_CA, MD_KEY_ACCOUNT, NULL);
339        md->ca_proto = md_json_dups(p, json, MD_KEY_CA, MD_KEY_PROTO, NULL);
340        md->ca_effective = md_json_dups(p, json, MD_KEY_CA, MD_KEY_URL, NULL);
341        if (md_json_has_key(json, MD_KEY_CA, MD_KEY_URLS, NULL)) {
342            md->ca_urls = apr_array_make(p, 5, sizeof(const char*));
343            md_json_dupsa(md->ca_urls, p, json, MD_KEY_CA, MD_KEY_URLS, NULL);
344        }
345        else if (md->ca_effective) {
346            /* compat for old format where we had only a single url */
347            md->ca_urls = apr_array_make(p, 5, sizeof(const char*));
348            APR_ARRAY_PUSH(md->ca_urls, const char*) = md->ca_effective;
349        }
350        md->ca_agreement = md_json_dups(p, json, MD_KEY_CA, MD_KEY_AGREEMENT, NULL);
351        if (md_json_has_key(json, MD_KEY_PKEY, NULL)) {
352            md->pks = md_pkeys_spec_from_json(md_json_getj(json, MD_KEY_PKEY, NULL), p);
353        }
354        md->state = (md_state_t)md_json_getl(json, MD_KEY_STATE, NULL);
355        md->state_descr = md_json_dups(p, json, MD_KEY_STATE_DESCR, NULL);
356        if (MD_S_EXPIRED_DEPRECATED == md->state) md->state = MD_S_COMPLETE;
357        md->renew_mode = (int)md_json_getl(json, MD_KEY_RENEW_MODE, NULL);
358        md->domains = md_array_str_compact(p, md->domains, 0);
359        md->transitive = (int)md_json_getl(json, MD_KEY_TRANSITIVE, NULL);
360        s = md_json_gets(json, MD_KEY_RENEW_WINDOW, NULL);
361        md_timeslice_parse(&md->renew_window, p, s, MD_TIME_LIFE_NORM);
362        s = md_json_gets(json, MD_KEY_WARN_WINDOW, NULL);
363        md_timeslice_parse(&md->warn_window, p, s, MD_TIME_LIFE_NORM);
364        if (md_json_has_key(json, MD_KEY_CA, MD_KEY_CHALLENGES, NULL)) {
365            md->ca_challenges = apr_array_make(p, 5, sizeof(const char*));
366            md_json_dupsa(md->ca_challenges, p, json, MD_KEY_CA, MD_KEY_CHALLENGES, NULL);
367        }
368        md->require_https = MD_REQUIRE_OFF;
369        s = md_json_gets(json, MD_KEY_REQUIRE_HTTPS, NULL);
370        if (s && !strcmp(MD_KEY_TEMPORARY, s)) {
371            md->require_https = MD_REQUIRE_TEMPORARY;
372        }
373        else if (s && !strcmp(MD_KEY_PERMANENT, s)) {
374            md->require_https = MD_REQUIRE_PERMANENT;
375        }
376        md->must_staple = (int)md_json_getb(json, MD_KEY_MUST_STAPLE, NULL);
377        md_json_dupsa(md->acme_tls_1_domains, p, json, MD_KEY_PROTO, MD_KEY_ACME_TLS_1, NULL);
378            
379        if (md_json_has_key(json, MD_KEY_CERT_FILES, NULL)) {
380            md->cert_files = apr_array_make(p, 3, sizeof(char*));
381            md->pkey_files = apr_array_make(p, 3, sizeof(char*));
382            md_json_dupsa(md->cert_files, p, json, MD_KEY_CERT_FILES, NULL);
383            md_json_dupsa(md->pkey_files, p, json, MD_KEY_PKEY_FILES, NULL);
384        }
385        md->stapling = (int)md_json_getb(json, MD_KEY_STAPLING, NULL);
386        md->dns01_cmd = md_json_dups(p, json, MD_KEY_CMD_DNS01, NULL);
387        if (md_json_has_key(json, MD_KEY_EAB, NULL)) {
388            md->ca_eab_kid = md_json_dups(p, json, MD_KEY_EAB, MD_KEY_KID, NULL);
389            md->ca_eab_hmac = md_json_dups(p, json, MD_KEY_EAB, MD_KEY_HMAC, NULL);
390        }
391
392        md->profile_mandatory = (int)md_json_getb(json, MD_KEY_PROFILE_MANDATORY, NULL);
393        if (md_json_has_key(json, MD_KEY_PROFILE, NULL))
394            md->profile = md_json_dups(p, json, MD_KEY_PROFILE, NULL);
395        md->ari_renewals = (int)md_json_getb(json, MD_KEY_ARI_RENEWALS, NULL);
396        return md;
397    }
398    return NULL;
399}

401md_json_t *md_to_public_json(const md_t *md, apr_pool_t *p)
402{
403    md_json_t *json = md_to_json(md, p);
404    if (md_json_has_key(json, MD_KEY_EAB, MD_KEY_HMAC, NULL)) {
405        md_json_sets("***", json, MD_KEY_EAB, MD_KEY_HMAC, NULL);
406    }
407    return json;
408}

427const char *md_get_ca_name_from_url(apr_pool_t *p, const char *url)
428{
429    apr_uri_t uri_parsed;
430    unsigned int i;
431
432    for (i = 0; i < sizeof(KNOWN_CAs)/sizeof(KNOWN_CAs[0]); ++i) {
433        if (!apr_cstr_casecmp(KNOWN_CAs[i].url, url)) {
434            return KNOWN_CAs[i].name;
435        }
436    }
437    if (APR_SUCCESS == apr_uri_parse(p, url, &uri_parsed)) {
438        return uri_parsed.hostname;
439    }
440    return apr_pstrdup(p, url);
441}

443apr_status_t md_get_ca_url_from_name(const char **purl, apr_pool_t *p, const char *name)
444{
445    const char *err;
446    unsigned int i;
447    apr_status_t rv = APR_SUCCESS;
448
449    *purl = NULL;
450    for (i = 0; i < sizeof(KNOWN_CAs)/sizeof(KNOWN_CAs[0]); ++i) {
451        if (!apr_cstr_casecmp(KNOWN_CAs[i].name, name)) {
452            *purl = KNOWN_CAs[i].url;
453            goto leave;
454        }
455    }
456    *purl = name;
457    rv = md_util_abs_uri_check(p, name, &err);
458    if (APR_SUCCESS != rv) {
459        apr_array_header_t *names;
460
461        names = apr_array_make(p, 10, sizeof(const char*));
462        for (i = 0; i < sizeof(KNOWN_CAs)/sizeof(KNOWN_CAs[0]); ++i) {
463            APR_ARRAY_PUSH(names, const char *) = KNOWN_CAs[i].name;
464        }
465        *purl = apr_psprintf(p,
466            "The CA name '%s' is not known and it is not a URL either (%s). "
467            "Known CA names are: %s.",
468            name, err, apr_array_pstrcat(p, names, ' '));
469    }
470leave:
471    return rv;
472}
