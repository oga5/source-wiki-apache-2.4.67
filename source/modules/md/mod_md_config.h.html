52typedef struct md_mod_conf_t md_mod_conf_t;
53struct md_mod_conf_t {
54    apr_array_header_t *mds;           /* all md_t* defined in the config, shared */
55    const char *base_dir;              /* base dir for store */
56    const char *proxy_url;             /* proxy url to use (or NULL) */
57    struct md_reg_t *reg;              /* md registry instance */
58    struct md_ocsp_reg_t *ocsp;        /* ocsp status registry */
59
60    int local_80;                      /* On which port http:80 arrives */
61    int local_443;                     /* On which port https:443 arrives */
62    int can_http;                      /* Does someone listen to the local port 80 equivalent? */
63    int can_https;                     /* Does someone listen to the local port 443 equivalent? */
64    int manage_base_server;            /* If base server outside vhost may be managed */
65    int hsts_max_age;                  /* max-age of HSTS (rfc6797) header */
66    const char *hsts_header;           /* computed HTST header to use or NULL */
67    apr_array_header_t *unused_names;  /* post config, names of all MDs not assigned to a vhost */
68    struct apr_hash_t *init_errors;    /* init errors reported with MD name as key */
69
70    const char *notify_cmd;            /* notification command to execute on signup/renew */
71    const char *message_cmd;           /* message command to execute on signup/renew/warnings */
72    struct apr_table_t *env;           /* environment for operation */
73    int dry_run;                       /* != 0 iff config dry run */
74    int server_status_enabled;         /* if module should add to server-status handler */
75    int certificate_status_enabled;    /* if module should expose /.httpd/certificate-status */
76    md_timeslice_t *ocsp_keep_window;  /* time that we keep ocsp responses around */
77    md_timeslice_t *ocsp_renew_window; /* time before exp. that we start renewing ocsp resp. */
78    const char *cert_check_name;       /* name of the linked certificate check site */
79    const char *cert_check_url;        /* url "template for" checking a certificate */
80    const char *ca_certs;              /* root certificates to use for connections */
81    apr_time_t initial_delay;          /* how long to delay the first cert renewal check */
82    apr_time_t check_interval;         /* duration between cert renewal checks */
83    apr_time_t min_delay;              /* minimum delay for retries */
84    int retry_failover;                /* number of errors to trigger CA failover */
85    int use_store_locks;               /* use locks when updating store */
86    apr_time_t lock_wait_timeout;      /* fail after this time when unable to obtain lock */
87    md_match_mode_t match_mode;        /* how dns names are match to vhosts */
88};
89
90typedef struct md_srv_conf_t {
91    const char *name;
92    const server_rec *s;               /* server this config belongs to */
93    md_mod_conf_t *mc;                 /* global config settings */
94    
95    int transitive;                    /* != 0 iff VirtualHost names/aliases are auto-added */
96    md_require_t require_https;        /* If MDs require https: access */
97    int renew_mode;                    /* mode of obtaining credentials */
98    int must_staple;                   /* certificates should set the OCSP Must Staple extension */
99    struct md_pkeys_spec_t *pks;       /* specification for private keys */
100    md_timeslice_t *renew_window;      /* time before expiration that starts renewal */
101    md_timeslice_t *warn_window;       /* time before expiration that warning are sent out */
102    
103    struct apr_array_header_t *ca_urls; /* urls of CAs */
104    const char *ca_contact;            /* contact email registered to account */
105    const char *ca_proto;              /* protocol used vs CA (e.g. ACME) */
106    const char *ca_agreement;          /* accepted agreement uri between CA and user */ 
107    struct apr_array_header_t *ca_challenges; /* challenge types configured */
108    const char *ca_eab_kid;            /* != NULL, external account binding keyid */
109    const char *ca_eab_hmac;           /* != NULL, external account binding hmac */
110    const char *profile;               /* != NULL, ACME order profile */
111    int profile_mandatory;             /* if ACME profile, when set, is mandatory */
112
113    int stapling;                      /* OCSP stapling enabled */
114    int staple_others;                 /* Provide OCSP stapling for non-MD certificates */
115    int ari_renewals;                  /* ACME ARI extension enabled */
116
117    const char *dns01_cmd;             /* DNS challenge command, override global command */
118
119    md_t *current;                     /* md currently defined in <MDomainSet xxx> section */
120    struct apr_array_header_t *assigned; /* post_config: MDs that apply to this server */
121    int is_ssl;                        /* SSLEngine is enabled here */
122} md_srv_conf_t;
