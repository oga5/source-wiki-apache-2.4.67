35static apr_status_t curl_status(unsigned int curl_code)
36{
37    switch (curl_code) {
38        case CURLE_OK:                   return APR_SUCCESS;
39        case CURLE_UNSUPPORTED_PROTOCOL: return APR_ENOTIMPL; 
40        case CURLE_NOT_BUILT_IN:         return APR_ENOTIMPL; 
41        case CURLE_URL_MALFORMAT:        return APR_EINVAL;
42        case CURLE_COULDNT_RESOLVE_PROXY:return APR_ECONNREFUSED;
43        case CURLE_COULDNT_RESOLVE_HOST: return APR_ECONNREFUSED;
44        case CURLE_COULDNT_CONNECT:      return APR_ECONNREFUSED;
45        case CURLE_REMOTE_ACCESS_DENIED: return APR_EACCES;
46        case CURLE_OUT_OF_MEMORY:        return APR_ENOMEM;
47        case CURLE_OPERATION_TIMEDOUT:   return APR_TIMEUP;
48        case CURLE_SSL_CONNECT_ERROR:    return APR_ECONNABORTED;
49        case CURLE_AGAIN:                return APR_EAGAIN;
50        default:                         return APR_EGENERAL;
51    }
52}

63static size_t req_data_cb(void *data, size_t len, size_t nmemb, void *baton)
64{
65    apr_bucket_brigade *body = baton;
66    size_t blen, read_len = 0, max_len = len * nmemb;
67    const char *bdata;
68    char *rdata = data;
69    apr_bucket *b;
70    apr_status_t rv;
71    
72    while (body && !APR_BRIGADE_EMPTY(body) && max_len > 0) {
73        b = APR_BRIGADE_FIRST(body);
74        if (APR_BUCKET_IS_METADATA(b)) {
75            if (APR_BUCKET_IS_EOS(b)) {
76                body = NULL;
77            }
78        }
79        else {
80            rv = apr_bucket_read(b, &bdata, &blen, APR_BLOCK_READ);
81            if (rv == APR_SUCCESS) {
82                if (blen > max_len) {
83                    apr_bucket_split(b, max_len);
84                    blen = max_len;
85                }
86                memcpy(rdata, bdata, blen);
87                read_len += blen;
88                max_len -= blen;
89                rdata += blen;
90            }
91            else {
92                body = NULL;
93                if (!APR_STATUS_IS_EOF(rv)) {
94                    /* everything beside EOF is an error */ view japanese
95                    read_len = CURL_READFUNC_ABORT;
96                }
97            }
98            
99        }
100        apr_bucket_delete(b);
101    }
102    
103    return read_len;
104}

106static size_t resp_data_cb(void *data, size_t len, size_t nmemb, void *baton)
107{
108    md_curl_internals_t *internals = baton;
109    md_http_response_t *res = internals->response;
110    size_t blen = len * nmemb;
111    apr_status_t rv;
112    
113    if (res->body) {
114        if (res->req->resp_limit) {
115            apr_off_t body_len = 0;
116            apr_brigade_length(res->body, 0, &body_len);
117            if (body_len + (apr_off_t)blen > res->req->resp_limit) {
118                return 0; /* signal curl failure */ view japanese
119            }
120        }
121        rv = apr_brigade_write(res->body, NULL, NULL, (const char *)data, blen);
122        if (rv != APR_SUCCESS) {
123            /* returning anything != blen will make CURL fail this */ view japanese
124            return 0;
125        }
126    }
127    return blen;
128}

130static size_t header_cb(void *buffer, size_t elen, size_t nmemb, void *baton)
131{
132    md_curl_internals_t *internals = baton;
133    md_http_response_t *res = internals->response;
134    size_t len, clen = elen * nmemb;
135    const char *name = NULL, *value = "", *b = buffer;
136    apr_size_t i;
137    
138    len = (clen && b[clen-1] == '\n')? clen-1 : clen;
139    len = (len && b[len-1] == '\r')? len-1 : len;
140    for (i = 0; i < len; ++i) {
141        if (b[i] == ':') {
142            name = apr_pstrndup(res->req->pool, b, i);
143            ++i;
144            while (i < len && b[i] == ' ') {
145                ++i;
146            }
147            if (i < len) {
148                value = apr_pstrndup(res->req->pool, b+i, len - i);
149            }
150            break;
151        }
152    }
153    
154    if (name != NULL) {
155        apr_table_add(res->headers, name, value);
156    }
157    return clen;
158}

160typedef struct {
161    md_http_request_t *req;
162    struct curl_slist *hdrs;
163    apr_status_t rv;
164} curlify_hdrs_ctx;

166static int curlify_headers(void *baton, const char *key, const char *value)
167{
168    curlify_hdrs_ctx *ctx = baton;
169    const char *s;
170    
171    if (strchr(key, '\r') || strchr(key, '\n')
172        || strchr(value, '\r') || strchr(value, '\n')) {
173        ctx->rv = APR_EINVAL;
174        return 0;
175    }
176    s = apr_psprintf(ctx->req->pool, "%s: %s", key, value);
177    ctx->hdrs = curl_slist_append(ctx->hdrs, s);
178    return 1;
179}

181/* Convert timeout values for curl. Since curl uses 0 to disable
182 * timeout, return at least 1 if the apr_time_t value is non-zero. */ view japanese
183static long timeout_msec(apr_time_t timeout)
184{
185    long ms = (long)apr_time_as_msec(timeout);
186    return ms? ms : (timeout? 1 : 0);
187}
188
189static long timeout_sec(apr_time_t timeout)
190{
191    long s = (long)apr_time_sec(timeout);
192    return s? s : (timeout? 1 : 0);
193}

195static int curl_debug_log(CURL *curl, curl_infotype type, char *data, size_t size, void *baton)
196{
197    md_http_request_t *req = baton;
198    
199    (void)curl;
200    switch (type) {
201        case CURLINFO_TEXT:
202            md_log_perror(MD_LOG_MARK, MD_LOG_TRACE4, 0, req->pool, 
203                          "req[%d]: info %s", req->id, apr_pstrndup(req->pool, data, size));
204            break;
205        case CURLINFO_HEADER_OUT:
206            md_log_perror(MD_LOG_MARK, MD_LOG_TRACE4, 0, req->pool, 
207                          "req[%d]: header --> %s", req->id, apr_pstrndup(req->pool, data, size));
208            break;
209        case CURLINFO_HEADER_IN:
210            md_log_perror(MD_LOG_MARK, MD_LOG_TRACE4, 0, req->pool, 
211                          "req[%d]: header <-- %s", req->id, apr_pstrndup(req->pool, data, size));
212            break;
213        case CURLINFO_DATA_OUT:
214            md_log_perror(MD_LOG_MARK, MD_LOG_TRACE4, 0, req->pool, 
215                          "req[%d]: data --> %ld bytes", req->id, (long)size);
216            if (md_log_is_level(req->pool, MD_LOG_TRACE5)) {
217                md_data_t d;
218                const char *s;
219                md_data_init(&d, data, size);
220                md_data_to_hex(&s, 0, req->pool, &d);
221                md_log_perror(MD_LOG_MARK, MD_LOG_TRACE5, 0, req->pool, 
222                              "req[%d]: data(hex) -->  %s", req->id, s);
223            }
224            break;
225        case CURLINFO_DATA_IN:
226            md_log_perror(MD_LOG_MARK, MD_LOG_TRACE4, 0, req->pool, 
227                          "req[%d]: data <-- %ld bytes", req->id, (long)size);
228            if (md_log_is_level(req->pool, MD_LOG_TRACE5)) {
229                md_data_t d;
230                const char *s;
231                md_data_init(&d, data, size);
232                md_data_to_hex(&s, 0, req->pool, &d);
233                md_log_perror(MD_LOG_MARK, MD_LOG_TRACE5, 0, req->pool, 
234                              "req[%d]: data(hex) <-- %s", req->id, s);
235            }
236            break;
237        default:
238            break;
239    }
240    return 0;
241}

243static apr_status_t internals_setup(md_http_request_t *req)
244{
245    md_curl_internals_t *internals;
246    CURL *curl;
247    apr_status_t rv = APR_SUCCESS;
248    long ssl_options = 0;
249
250    curl = md_http_get_impl_data(req->http);
251    if (!curl) {
252        md_log_perror(MD_LOG_MARK, MD_LOG_TRACE3, 0, req->pool, "creating curl instance");
253        curl = curl_easy_init();
254        if (!curl) {
255            rv = APR_EGENERAL;
256            goto leave;
257        }
258    }
259    else {
260        md_log_perror(MD_LOG_MARK, MD_LOG_TRACE3, 0, req->pool, "reusing curl instance from http");
261        curl_easy_reset(curl);
262    }
263
264    curl_easy_setopt(curl, CURLOPT_HEADERFUNCTION, header_cb);
265    curl_easy_setopt(curl, CURLOPT_HEADERDATA, NULL);
266    curl_easy_setopt(curl, CURLOPT_READFUNCTION, req_data_cb);
267    curl_easy_setopt(curl, CURLOPT_READDATA, NULL);
268    curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, resp_data_cb);
269    curl_easy_setopt(curl, CURLOPT_WRITEDATA, NULL);
270
271    internals = apr_pcalloc(req->pool, sizeof(*internals));
272    internals->curl = curl;
273        
274    internals->response = apr_pcalloc(req->pool, sizeof(md_http_response_t));
275    internals->response->req = req;
276    internals->response->status = 400;
277    internals->response->headers = apr_table_make(req->pool, 5);
278    internals->response->body = apr_brigade_create(req->pool, req->bucket_alloc);
279    
280    curl_easy_setopt(curl, CURLOPT_URL, req->url);
281    if (!apr_cstr_casecmp("GET", req->method)) {
282        /* nop */ view japanese
283    }
284    else if (!apr_cstr_casecmp("HEAD", req->method)) {
285        curl_easy_setopt(curl, CURLOPT_NOBODY, 1L);
286    }
287    else if (!apr_cstr_casecmp("POST", req->method)) {
288        curl_easy_setopt(curl, CURLOPT_POST, 1L);
289    }
290    else {
291        curl_easy_setopt(curl, CURLOPT_CUSTOMREQUEST, req->method);
292    }
293    curl_easy_setopt(curl, CURLOPT_HEADERDATA, internals);
294    curl_easy_setopt(curl, CURLOPT_READDATA, req->body);
295    curl_easy_setopt(curl, CURLOPT_WRITEDATA, internals);
296    
297    if (req->timeout.overall > 0) {
298        curl_easy_setopt(curl, CURLOPT_TIMEOUT_MS, timeout_msec(req->timeout.overall));
299    }
300    if (req->timeout.connect > 0) {
301        curl_easy_setopt(curl, CURLOPT_CONNECTTIMEOUT_MS, timeout_msec(req->timeout.connect));
302    }
303    if (req->timeout.stalled > 0) {
304        curl_easy_setopt(curl, CURLOPT_LOW_SPEED_LIMIT, req->timeout.stall_bytes_per_sec);
305        curl_easy_setopt(curl, CURLOPT_LOW_SPEED_TIME, timeout_sec(req->timeout.stalled));
306    }
307    if (req->ca_file) {
308        curl_easy_setopt(curl, CURLOPT_CAINFO, req->ca_file);
309        /* for a custom CA, allow certificates checking to ignore the
310         * Schannel error CRYPT_E_NO_REVOCATION_CHECK (could be a missing OCSP
311         * responder URL in the certs???). See issue #361 */ view japanese
312#ifdef CURLSSLOPT_NO_REVOKE
313        ssl_options |= CURLSSLOPT_NO_REVOKE;
314#endif
315    }
316    if (req->unix_socket_path) {
317        curl_easy_setopt(curl, CURLOPT_UNIX_SOCKET_PATH, req->unix_socket_path);
318    }
319
320    if (req->body_len >= 0) {
321        /* set the Content-Length */ view japanese
322        curl_easy_setopt(curl, CURLOPT_INFILESIZE_LARGE, (curl_off_t)req->body_len);
323        curl_easy_setopt(curl, CURLOPT_POSTFIELDSIZE_LARGE, (curl_off_t)req->body_len);
324    }
325    
326    if (req->user_agent) {
327        curl_easy_setopt(curl, CURLOPT_USERAGENT, req->user_agent);
328    }
329    if (req->proxy_url) {
330        curl_easy_setopt(curl, CURLOPT_PROXY, req->proxy_url);
331    }
332    if (!apr_is_empty_table(req->headers)) {
333        curlify_hdrs_ctx ctx;
334        
335        ctx.req = req;
336        ctx.hdrs = NULL;
337        ctx.rv = APR_SUCCESS;
338        apr_table_do(curlify_headers, &ctx, req->headers, NULL);
339        internals->req_hdrs = ctx.hdrs;
340        if (ctx.rv == APR_SUCCESS) {
341            curl_easy_setopt(curl, CURLOPT_HTTPHEADER, internals->req_hdrs);
342        }
343    }
344    
345    md_log_perror(MD_LOG_MARK, MD_LOG_TRACE3, 0, req->pool, 
346                  "req[%d]: %s %s", req->id, req->method, req->url);
347    
348    if (md_log_is_level(req->pool, MD_LOG_TRACE4)) {
349        curl_easy_setopt(curl, CURLOPT_VERBOSE, 1L);
350        curl_easy_setopt(curl, CURLOPT_DEBUGFUNCTION, curl_debug_log);
351        curl_easy_setopt(curl, CURLOPT_DEBUGDATA, req);
352    }
353
354    if (ssl_options)
355        curl_easy_setopt(curl, CURLOPT_SSL_OPTIONS, ssl_options);
356
357leave:
358    req->internals = (APR_SUCCESS == rv)? internals : NULL;
359    return rv;
360}

362static apr_status_t update_status(md_http_request_t *req)
363{
364    md_curl_internals_t *internals = req->internals;
365    long l;
366    apr_status_t rv = APR_SUCCESS;
367
368    if (internals) {
369        rv = curl_status(curl_easy_getinfo(internals->curl, CURLINFO_RESPONSE_CODE, &l));
370        if (APR_SUCCESS == rv) {
371            internals->response->status = (int)l;
372            md_log_perror(MD_LOG_MARK, MD_LOG_TRACE3, rv, req->pool,
373                          "req[%d]: http status is %d",
374                          req->id, internals->response->status);
375        }
376    }
377    return rv;
378}

380static void fire_status(md_http_request_t *req, apr_status_t rv)
381{
382    md_curl_internals_t *internals = req->internals;
383        
384    if (internals && !internals->status_fired) {
385        internals->status_fired = 1;
386        
387        md_log_perror(MD_LOG_MARK, MD_LOG_TRACE3, rv, req->pool, 
388                      "req[%d] fire callbacks", req->id);
389        if ((APR_SUCCESS == rv) && req->cb.on_response) {
390            rv = req->cb.on_response(internals->response, req->cb.on_response_data);
391        }
392    
393        internals->rv = rv;
394        if (req->cb.on_status) {
395            req->cb.on_status(req, rv, req->cb.on_status_data);
396        }
397    }
398}

400static apr_status_t md_curl_perform(md_http_request_t *req)
401{
402    apr_status_t rv = APR_SUCCESS;
403    CURLcode curle;
404    md_curl_internals_t *internals;
405    long l;
406
407    if (APR_SUCCESS != (rv = internals_setup(req))) goto leave;
408    internals = req->internals;
409    
410    curle = curl_easy_perform(internals->curl);
411    
412    rv = curl_status(curle);
413    if (APR_SUCCESS != rv) {
414        md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, rv, req->pool, 
415                      "request failed(%d): %s", curle, curl_easy_strerror(curle));
416        goto leave;
417    }
418    
419    rv = curl_status(curl_easy_getinfo(internals->curl, CURLINFO_RESPONSE_CODE, &l));
420    if (APR_SUCCESS == rv) {
421        internals->response->status = (int)l;
422    }
423    md_log_perror(MD_LOG_MARK, MD_LOG_TRACE1, rv, req->pool, "request <-- %d", 
424                  internals->response->status);
425    
426    if (req->cb.on_response) {
427        rv = req->cb.on_response(internals->response, req->cb.on_response_data);
428        req->cb.on_response = NULL;
429    }
430    
431leave:
432    fire_status(req, rv);
433    md_http_req_destroy(req);
434    return rv;
435}

437static md_http_request_t *find_curl_request(apr_array_header_t *requests, CURL *curl)
438{
439    md_http_request_t *req;
440    md_curl_internals_t *internals;
441    int i;
442    
443    for (i = 0; i < requests->nelts; ++i) {
444        req = APR_ARRAY_IDX(requests, i, md_http_request_t*);
445        internals = req->internals;
446        if (internals && internals->curl == curl) {
447            return req;
448        }
449    }
450    return NULL;
451}

453static void add_to_curlm(md_http_request_t *req, CURLM *curlm)
454{
455    md_curl_internals_t *internals = req->internals;
456    
457    assert(curlm);
458    assert(internals);
459    if (internals->curlm == NULL) {
460        internals->curlm = curlm;
461    }
462    assert(internals->curlm == curlm);
463    curl_multi_add_handle(curlm, internals->curl);
464}

478static apr_status_t md_curl_multi_perform(md_http_t *http, apr_pool_t *p,
479                                          md_http_next_req *nextreq, void *baton)
480{
481    md_http_t *sub_http;
482    md_http_request_t *req;
483    CURLM *curlm = NULL;
484    CURLMcode mc;
485    struct CURLMsg *curlmsg;
486    apr_array_header_t *http_spares;
487    apr_array_header_t *requests;
488    int i, running, numfds, slowdown, msgcount;
489    apr_status_t rv;
490    
491    http_spares = apr_array_make(p, 10, sizeof(md_http_t*));
492    requests = apr_array_make(p, 10, sizeof(md_http_request_t*));
493    curlm = curl_multi_init();
494    if (!curlm) {
495        rv = APR_ENOMEM;
496        goto leave;
497    }
498    
499    running = 1;
500    slowdown = 0;
501    while(1) {
502        while (1) {
503            /* fetch as many requests as nextreq gives us */ view japanese
504            if (http_spares->nelts > 0) {
505                sub_http = *(md_http_t **)(apr_array_pop(http_spares));
506            }
507            else {
508                rv = md_http_clone(&sub_http, p, http);
509                if (APR_SUCCESS != rv) {
510                    md_log_perror(MD_LOG_MARK, MD_LOG_ERR, rv, p,
511                                  "multi_perform[%d reqs]: setup failed", requests->nelts);
512                    goto leave;
513                }
514            }
515
516            rv = nextreq(&req, baton, sub_http, requests->nelts);
517            if (APR_STATUS_IS_ENOENT(rv)) {
518                md_log_perror(MD_LOG_MARK, MD_LOG_TRACE3, 0, p,
519                              "multi_perform[%d reqs]: no more requests", requests->nelts);
520                if (!requests->nelts) {
521                    goto leave;
522                }
523                break;
524            }
525            else if (APR_SUCCESS != rv) {
526                md_log_perror(MD_LOG_MARK, MD_LOG_TRACE3, rv, p,
527                              "multi_perform[%d reqs]: nextreq() failed", requests->nelts);
528                APR_ARRAY_PUSH(http_spares, md_http_t*) = sub_http;
529                goto leave;
530            }
531
532            if (APR_SUCCESS != (rv = internals_setup(req))) {
533                if (req->cb.on_status) req->cb.on_status(req, rv, req->cb.on_status_data);
534                md_log_perror(MD_LOG_MARK, MD_LOG_TRACE3, rv, p,
535                              "multi_perform[%d reqs]: setup failed", requests->nelts);
536                APR_ARRAY_PUSH(http_spares, md_http_t*) = sub_http;
537                goto leave;
538            }
539
540            APR_ARRAY_PUSH(requests, md_http_request_t*) = req;
541            add_to_curlm(req, curlm);
542            md_log_perror(MD_LOG_MARK, MD_LOG_TRACE3, rv, p,
543                          "multi_perform[%d reqs]: added request", requests->nelts);
544        }
545    
546        mc = curl_multi_perform(curlm, &running);
547        if (CURLM_OK == mc) {
548            mc = curl_multi_wait(curlm, NULL, 0, 1000, &numfds);
549            if (numfds) slowdown = 0;
550        }
551        if (CURLM_OK != mc) {
552            rv = APR_ECONNABORTED;
553            md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, rv, p, 
554                          "multi_perform[%d reqs] failed(%d): %s", 
555                          requests->nelts, mc, curl_multi_strerror(mc));
556            goto leave;
557        }
558        if (!numfds) {
559            /* no activity on any connection, timeout */ view japanese
560            md_log_perror(MD_LOG_MARK, MD_LOG_TRACE3, 0, p, 
561                          "multi_perform[%d reqs]: slowdown %d", requests->nelts, slowdown);
562            if (slowdown) apr_sleep(apr_time_from_msec(100));
563            ++slowdown;
564        }
565
566        /* process status messages, e.g. that a request is done */ view japanese
567        while (running < requests->nelts) {
568            curlmsg = curl_multi_info_read(curlm, &msgcount);
569            if (!curlmsg) break;
570            if (curlmsg->msg == CURLMSG_DONE) {
571                req = find_curl_request(requests, curlmsg->easy_handle);
572                if (req) {
573                    md_log_perror(MD_LOG_MARK, MD_LOG_TRACE2, 0, p,
574                                  "multi_perform[%d reqs]: req[%d] done", 
575                                  requests->nelts, req->id);
576                    update_status(req);
577                    fire_status(req, curl_status(curlmsg->data.result));
578                    md_array_remove(requests, req);
579                    sub_http = req->http;
580                    APR_ARRAY_PUSH(http_spares, md_http_t*) = sub_http;
