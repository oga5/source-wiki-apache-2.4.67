29/**
30 * A store for domain related data.
31 *
32 * The Key for a piece of data is the set of 3 items
33 *   <group> + <domain> + <aspect>
34 *
35 * Examples:
36 * "domains" + "greenbytes.de" + "pubcert.pem"
37 * "ocsp" + "greenbytes.de" + "ocsp-XXXXX.json"
38 *
39 * Storage groups are pre-defined, domain and aspect names can be freely chosen.
40 *
41 * Groups reflect use cases and come with security restrictions. The groups 
42 * DOMAINS, ARCHIVE and NONE are only accessible during the startup 
43 * phase of httpd.
44 *
45 * Private key are stored unencrypted only in restricted groups. Meaning that certificate
46 * keys in group DOMAINS are not encrypted, but only readable at httpd start/reload.
47 * Keys in unrestricted groups are encrypted using a pass phrase generated once and stored
48 * in NONE.
49 */
50
51/** Value types handled by a store */
52typedef enum {
53    MD_SV_TEXT,         /* plain text, value is (char*) */
54    MD_SV_JSON,         /* JSON serialization, value is (md_json_t*) */
55    MD_SV_CERT,         /* PEM x509 certificate, value is (md_cert_t*) */
56    MD_SV_PKEY,         /* PEM private key, value is (md_pkey_t*) */
57    MD_SV_CHAIN,        /* list of PEM x509 certificates, value is 
58                           (apr_array_header_t*) of (md_cert*) */
59} md_store_vtype_t;
60
61/** Store storage groups */
62typedef enum {
63    MD_SG_NONE,         /* top level of store, name MUST be NULL in calls */
64    MD_SG_ACCOUNTS,     /* ACME accounts */
65    MD_SG_CHALLENGES,   /* challenge response data for a domain */ 
66    MD_SG_DOMAINS,      /* live certificates and settings for a domain */
67    MD_SG_STAGING,      /* staged set of certificate and settings, maybe incomplete */
68    MD_SG_ARCHIVE,      /* Archived live sets of a domain */
69    MD_SG_TMP,          /* temporary domain storage */
70    MD_SG_OCSP,         /* OCSP stapling related domain data */
71    MD_SG_COUNT,        /* number of storage groups, used in setups */
72} md_store_group_t;

90/**
91 * Load the JSON value at key "group/name/aspect", allocated from pool p.
92 * @return APR_ENOENT if there is no such value
93 */
94apr_status_t md_store_load_json(md_store_t *store, md_store_group_t group, 
95                                const char *name, const char *aspect, 
96                                struct md_json_t **pdata, apr_pool_t *p);
97/**
98 * Save the JSON value at key "group/name/aspect". If create != 0, fail if there
99 * already is a value for this key.
100 */
101apr_status_t md_store_save_json(md_store_t *store, apr_pool_t *p, md_store_group_t group, 
102                                const char *name, const char *aspect, 
103                                struct md_json_t *data, int create);
104
105/**
106 * Load the value of type at key "group/name/aspect", allocated from pool p. Usually, the
107 * type is expected to be the same as used in saving the value. Some conversions will work,
108 * others will fail the format.
109 * @return APR_ENOENT if there is no such value
110 */
111apr_status_t md_store_load(md_store_t *store, md_store_group_t group, 
112                           const char *name, const char *aspect, 
113                           md_store_vtype_t vtype, void **pdata, 
114                           apr_pool_t *p);
115/**
116 * Save the JSON value at key "group/name/aspect". If create != 0, fail if there
117 * already is a value for this key. The provided data MUST be of the correct type.
118 */
119apr_status_t md_store_save(md_store_t *store, apr_pool_t *p, md_store_group_t group, 
120                           const char *name, const char *aspect, 
121                           md_store_vtype_t vtype, void *data, 
122                           int create);
123
124/**
125 * Remove the value stored at key "group/name/aspect". Unless force != 0, a missing
126 * value will cause the call to fail with APR_ENOENT.
127 */ 
128apr_status_t md_store_remove(md_store_t *store, md_store_group_t group, 
129                             const char *name, const char *aspect, 
130                             apr_pool_t *p, int force);
131/**
132 * Remove everything matching key "group/name".
133 */ 
134apr_status_t md_store_purge(md_store_t *store, apr_pool_t *p, 
135                            md_store_group_t group, const char *name);
136
137/**
138 * Remove all items matching the name/aspect patterns that have not been
139 * modified since the given timestamp.
140 */
141apr_status_t md_store_remove_not_modified_since(md_store_t *store, apr_pool_t *p, 
142                                                apr_time_t modified,
143                                                md_store_group_t group, 
144                                                const char *name, 
145                                                const char *aspect);
146
147/**
148 * inspect callback function. Invoked for each matched value. Values allocated from
149 * ptemp may disappear any time after the call returned. If this function returns
150 * 0, the iteration is aborted. 
151 */
152typedef int md_store_inspect(void *baton, const char *name, const char *aspect, 
153                             md_store_vtype_t vtype, void *value, apr_pool_t *ptemp);
154
155/**
156 * Iterator over all existing values matching the name pattern. Patterns are evaluated
157 * using apr_fnmatch() without flags.
158 */
159apr_status_t md_store_iter(md_store_inspect *inspect, void *baton, md_store_t *store, 
160                           apr_pool_t *p, md_store_group_t group, const char *pattern, 
161                           const char *aspect, md_store_vtype_t vtype);
162
163/**
164 * Move everything matching key "from/name" from one group to another. If archive != 0,
165 * move any existing "to/name" into a new "archive/new_name" location.
166 */
167apr_status_t md_store_move(md_store_t *store, apr_pool_t *p,
168                           md_store_group_t from, md_store_group_t to,
169                           const char *name, int archive);
170
171/**
172 * Rename a group member.
173 */
174apr_status_t md_store_rename(md_store_t *store, apr_pool_t *p,
175                             md_store_group_t group, const char *name, const char *to);
176
177/**
178 * Get the filename of an item stored in "group/name/aspect". The item does
179 * not have to exist.
180 */
181apr_status_t md_store_get_fname(const char **pfname, 
182                                md_store_t *store, md_store_group_t group, 
183                                const char *name, const char *aspect, 
184                                apr_pool_t *p);
185
186/**
187 * Make a compare on the modification time of "group1/name/aspect" vs. "group2/name/aspect".
188 */
189int md_store_is_newer(md_store_t *store, md_store_group_t group1, md_store_group_t group2,  
190                      const char *name, const char *aspect, apr_pool_t *p);
191
192/**
193 * Iterate over all names that exist in a group, e.g. there are items matching
194 * "group/pattern". The inspect function is called with the name and NULL aspect
195 * and value.
196 */
197apr_status_t md_store_iter_names(md_store_inspect *inspect, void *baton, md_store_t *store, 
198                                 apr_pool_t *p, md_store_group_t group, const char *pattern);
199
200/**
201 * Get the modification time of the item store under "group/name/aspect".
202 * @return modification time or 0 if the item does not exist.
203 */
204apr_time_t md_store_get_modified(md_store_t *store, md_store_group_t group,  
205                                 const char *name, const char *aspect, apr_pool_t *p);
206
207/**
208 * Acquire a cooperative, global lock on store modifications.
209
210 * This will only prevent other children/processes/cluster nodes from
211 * doing the same and does not protect individual store functions from
212 * being called without it.
213 * @param store the store
214 * @param p memory pool to use
215 * @param max_wait maximum time to wait in order to acquire
216 * @return APR_SUCCESS when lock was obtained
217 */
218apr_status_t md_store_lock_global(md_store_t *store, apr_pool_t *p, apr_time_t max_wait);
219
220/**
221 * Realease the global store lock. Will do nothing if there is no lock.
222 */
223void md_store_unlock_global(md_store_t *store, apr_pool_t *p);
