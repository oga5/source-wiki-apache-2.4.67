20struct apr_hash_t;

28#include "md_store.h"

30/**
31 * A registry for managed domains with a md_store_t as persistence.
32 *
33 */
34typedef struct md_reg_t md_reg_t;
35
36/**
37 * Create the MD registry, using the pool and store.
38 * @param preg on APR_SUCCESS, the create md_reg_t
39 * @param pm memory pool to use for creation
40 * @param store the store to base on
41 * @param proxy_url optional URL of a proxy to use for requests
42 * @param ca_file  optioinal CA trust anchor file to use
43 * @param min_delay minimum delay between renewal attempts for a domain
44 * @param retry_failover numer of failed renewals attempt to fail over to alternate ACME ca
45 */
46apr_status_t md_reg_create(md_reg_t **preg, apr_pool_t *pm, md_store_t *store,
47                           const char *proxy_url, const char *ca_file,
48                           apr_time_t min_delay, int retry_failover,
49                           int use_store_locks, apr_time_t lock_wait_timeout);
50
51md_store_t *md_reg_store_get(md_reg_t *reg);
52
53apr_status_t md_reg_set_props(md_reg_t *reg, apr_pool_t *p, int can_http, int can_https);
54
55/**
56 * Add a new md to the registry. This will check the name for uniqueness and
57 * that domain names do not overlap with already existing mds.
58 */
59apr_status_t md_reg_add(md_reg_t *reg, md_t *md, apr_pool_t *p);
60
61/**
62 * Find the md, if any, that contains the given domain name. 
63 * NULL if none found.
64 */
65md_t *md_reg_find(md_reg_t *reg, const char *domain, apr_pool_t *p);
66
67/**
68 * Find one md, which domain names overlap with the given md and that has a different
69 * name. There may be more than one existing md that overlaps. It is not defined
70 * which one will be returned. 
71 */
72md_t *md_reg_find_overlap(md_reg_t *reg, const md_t *md, const char **pdomain, apr_pool_t *p);
73
74/**
75 * Get the md with the given unique name. NULL if it does not exist.
76 * Will update the md->state.
77 */
78md_t *md_reg_get(md_reg_t *reg, const char *name, apr_pool_t *p);
79
80/**
81 * Callback invoked for every md in the registry. If 0 is returned, iteration stops.
82 */
83typedef int md_reg_do_cb(void *baton, md_reg_t *reg, md_t *md);
84
85/**
86 * Invoke callback for all mds in this registry. Order is not guaranteed.
87 * If the callback returns 0, iteration stops. Returns 0 if iteration was
88 * aborted.
89 */
90int md_reg_do(md_reg_do_cb *cb, void *baton, md_reg_t *reg, apr_pool_t *p);
91

93 * Bitmask for fields that are updated.
94 */
95#define MD_UPD_DOMAINS       0x00001
96#define MD_UPD_CA_URL        0x00002
97#define MD_UPD_CA_PROTO      0x00004

100#define MD_UPD_AGREEMENT     0x00020
101#define MD_UPD_DRIVE_MODE    0x00080

104#define MD_UPD_PKEY_SPEC     0x00400
105#define MD_UPD_REQUIRE_HTTPS 0x00800
106#define MD_UPD_TRANSITIVE    0x01000
107#define MD_UPD_MUST_STAPLE   0x02000
108#define MD_UPD_PROTO         0x04000

112
113/**
114 * Update the given fields for the managed domain. Take the new
115 * values from the given md, all other values remain unchanged.
116 */
117apr_status_t md_reg_update(md_reg_t *reg, apr_pool_t *p, 
118                           const char *name, const md_t *md, 
119                           int fields, int check_consistency);
120
121/**
122 * Get the chain of public certificates of the managed domain md, starting with the cert
123 * of the domain and going up the issuers. Returns APR_ENOENT when not available. 
124 */
125apr_status_t md_reg_get_pubcert(const md_pubcert_t **ppubcert, md_reg_t *reg, 
126                                const md_t *md, int i, apr_pool_t *p);
127
128/**
129 * Get the filenames of private key and pubcert of the MD - if they exist.
130 * @return APR_ENOENT if one or both do not exist.
131 */
132apr_status_t md_reg_get_cred_files(const char **pkeyfile, const char **pcertfile,
133                                   md_reg_t *reg, md_store_group_t group, 
134                                   const md_t *md, struct md_pkey_spec_t *spec, apr_pool_t *p);
135
136/**
137 * Synchronize the given master mds with the store.
138 */
139apr_status_t md_reg_sync_start(md_reg_t *reg, apr_array_header_t *master_mds, apr_pool_t *p);
140
141/**
142 * Re-compute the state of the MD, given current store contents.
143 */
144apr_status_t md_reg_sync_finish(md_reg_t *reg, md_t *md, apr_pool_t *p, apr_pool_t *ptemp);
145
146
147apr_status_t md_reg_remove(md_reg_t *reg, apr_pool_t *p, const char *name, int archive);
148
149/**
150 * Delete the account from the local store.
151 */
152apr_status_t md_reg_delete_acct(md_reg_t *reg, apr_pool_t *p, const char *acct_id);
153
154
155/**
156 * Cleanup any challenges that are no longer in use.
157 * 
158 * @param reg   the registry
159 * @param p     pool for permanent storage
160 * @param ptemp pool for temporary storage
161 * @param mds   the list of configured MDs
162 */
163apr_status_t md_reg_cleanup_challenges(md_reg_t *reg, apr_pool_t *p, apr_pool_t *ptemp, 
164                                       apr_array_header_t *mds);
165
166/**
167 * Mark all information from group MD_SG_DOMAINS as readonly, deny future modifications 
168 * (MD_SG_STAGING and MD_SG_CHALLENGES remain writeable). For the given MDs, cache
169 * the public information (MDs themselves and their pubcerts or lack of).
170 */
171apr_status_t md_reg_freeze_domains(md_reg_t *reg, apr_array_header_t *mds);
172
173/**
174 * Return if the certificate of the MD should be renewed. This includes reaching
175 * the renewal window of an otherwise valid certificate. It return also !0 iff
176 * no certificate has been obtained yet.
177 */
178int md_reg_should_renew(md_reg_t *reg, const md_t *md, apr_pool_t *p);
179
180/**
181 * Return the timestamp when the certificate should be renewed. A value of 0
182 * indicates that that renewal is not configured (see renew_mode).
183 */
184apr_time_t md_reg_renew_at(md_reg_t *reg, const md_t *md, apr_pool_t *p);
185
186/* Check ACME Renewal Info, if available, for the CA recommended time
187 * (and optional reason URL) for certificate renewal.
188 * Returns 0 if not availble.
189 */
190apr_time_t md_reg_ari_renew_at(const char **purl, md_reg_t *reg,
191                               const md_t *md, struct apr_table_t *env,
192                               struct md_result_t *result, apr_pool_t *p);
193
194/**
195 * Return the timestamp up to which *all* certificates for the MD can be used.
196 * A value of 0 indicates that there is no certificate.
197 */
198apr_time_t md_reg_valid_until(md_reg_t *reg, const md_t *md, apr_pool_t *p);
199
200/**
201 * Return if a warning should be issued about the certificate expiration. 
202 * This applies the configured warn window to the remaining lifetime of the 
203 * current certiciate. If no certificate is present, this returns 0.
204 */
205int md_reg_should_warn(md_reg_t *reg, const md_t *md, apr_pool_t *p);

207/**************************************************************************************************/
208/* protocol drivers */
209
210typedef struct md_proto_t md_proto_t;
211
212typedef struct md_proto_driver_t md_proto_driver_t;
213
214/** 
215 * Operating environment for a protocol driver. This is valid only for the
216 * duration of one run (init + renew, init + preload).
217 */
218struct md_proto_driver_t {
219    const md_proto_t *proto;
220    apr_pool_t *p;
221    void *baton;
222    struct apr_table_t *env;
223
224    md_reg_t *reg;
225    md_store_t *store;
226    const char *proxy_url;
227    const char *ca_file;
228    const md_t *md;
229
230    int can_http;
231    int can_https;
232    int reset;
233    int attempt;
234    int retry_failover;
235    apr_interval_time_t activation_delay;
236};
237
238typedef apr_status_t md_proto_init_cb(md_proto_driver_t *driver, struct md_result_t *result);
239typedef apr_status_t md_proto_renew_cb(md_proto_driver_t *driver, struct md_result_t *result);
240typedef apr_status_t md_proto_init_preload_cb(md_proto_driver_t *driver, struct md_result_t *result);
241typedef apr_status_t md_proto_preload_cb(md_proto_driver_t *driver, 
242                                         md_store_group_t group, struct md_result_t *result);
243typedef apr_status_t md_proto_complete_md_cb(md_t *md, apr_pool_t *p);
244typedef apr_status_t md_proto_get_ari(md_proto_driver_t *driver, struct md_result_t *result,
245                                      apr_time_t *prenew_at, const char **purl);
246
247struct md_proto_t {
248    const char *protocol;
249    md_proto_init_cb *init;
250    md_proto_renew_cb *renew;
251    md_proto_init_preload_cb *init_preload;
252    md_proto_preload_cb *preload;
253    md_proto_complete_md_cb *complete_md;
254    md_proto_get_ari *get_ari;
255};

257/**
258 * Run a test initialization of the renew protocol for the given MD. This verifies
259 * basic parameter settings and is expected to return a description of encountered
260 * problems in <pmessage> when != APR_SUCCESS.
261 * A message return is allocated fromt the given pool.
262 */
263apr_status_t md_reg_test_init(md_reg_t *reg, const md_t *md, struct apr_table_t *env, 
264                              struct md_result_t *result, apr_pool_t *p);
265
266/**
267 * Obtain new credentials for the given managed domain in STAGING.
268 * @param reg the registry instance
269 * @param md the mdomain to renew
270 * @param env global environment of settings
271 * @param reset != 0 if any previous, partial information should be wiped
272 * @param attempt the number of attempts made this far (for this md)
273 * @param result for reporting results of the renewal
274 * @param p the memory pool to use
275 * @return APR_SUCCESS if new credentials have been staged successfully
276 */
277apr_status_t md_reg_renew(md_reg_t *reg, const md_t *md, 
278                          struct apr_table_t *env, int reset, int attempt,
279                          struct md_result_t *result, apr_pool_t *p);
280
281/**
282 * Load a new set of credentials for the managed domain from STAGING - if it exists. 
283 * This will archive any existing credential data and make the staged set the new one
284 * in DOMAINS.
285 * If staging is incomplete or missing, the load will fail and all credentials remain
286 * as they are.
287 *
288 * @return APR_SUCCESS on loading new data, APR_ENOENT when nothing is staged, error otherwise.
289 */
290apr_status_t md_reg_load_staging(md_reg_t *reg, const md_t *md, struct apr_table_t *env, 
291                                 struct md_result_t *result, apr_pool_t *p);
292
293/**
294 * Check given MDomains for new data in staging areas and, if it exists, load
295 * the new credentials. On encountering errors, leave the credentails as
296 * they are.
297 */
298apr_status_t md_reg_load_stagings(md_reg_t *reg, apr_array_header_t *mds,
299                                  apr_table_t *env, apr_pool_t *p);
300
301void md_reg_set_renew_window_default(md_reg_t *reg, md_timeslice_t *renew_window);
302void md_reg_set_warn_window_default(md_reg_t *reg, md_timeslice_t *warn_window);
303
304struct md_job_t *md_reg_job_make(md_reg_t *reg, const char *mdomain, apr_pool_t *p);
305
306/**
307 * Acquire a cooperative, global lock on registry modifications. Will
308 * do nothing if locking is not configured.
309 *
310 * This will only prevent other children/processes/cluster nodes from
311 * doing the same and does not protect individual store functions from
312 * being called without it.
313 * @param reg the registy
314 * @param p memory pool to use
315 * @param max_wait maximum time to wait in order to acquire
316 * @return APR_SUCCESS when lock was obtained
317 */
318apr_status_t md_reg_lock_global(md_reg_t *reg, apr_pool_t *p);
319
320/**
321 * Realease the global registry lock. Will do nothing if there is no lock.
322 */
323void md_reg_unlock_global(md_reg_t *reg, apr_pool_t *p);
324
325/**
326 * @return != 0 iff `md` has any certificates known to be REVOKED.
327 */
328int md_reg_has_revoked_certs(md_reg_t *reg, struct md_ocsp_reg_t *ocsp,
329                             const md_t *md, apr_pool_t *p);
