50/**
51 * Either we have an order stored in the STAGING area, or we need to create a 
52 * new one at the ACME server.
53 */ view japanese
54static apr_status_t ad_setup_order(md_proto_driver_t *d, md_result_t *result, int *pis_new)
55{
56    md_acme_driver_t *ad = d->baton;
57    apr_status_t rv;
58    md_t *md = ad->md;
59    const char *profile = NULL;
60    const char *ari_cert_id = NULL;
61
62    assert(ad->md);
63    assert(ad->acme);
64
65    /* For each domain in MD: AUTHZ setup
66     * if an AUTHZ resource is known, check if it is still valid
67     * if known AUTHZ resource is not valid, remove, goto 4.1.1
68     * if no AUTHZ available, create a new one for the domain, store it
69     */ view japanese
70    if (pis_new) *pis_new = 0;
71    rv = md_acme_order_load(d->store, MD_SG_STAGING, md->name, &ad->order, d->p);
72    if (APR_SUCCESS == rv) {
73        md_result_activity_setn(result, "Loaded order from staging");
74        goto leave;
75    }
76    else if (!APR_STATUS_IS_ENOENT(rv)) {
77        md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, rv, d->p, "%s: loading order", md->name);
78        md_acme_order_purge(d->store, d->p, MD_SG_STAGING, md, d->env);
79    }
80    
81    if (ad->cred->spec && ad->md->ca_account && ad->md->ari_renewals &&
82        ad->acme->api.v2.renewal_info) {
83        /* are we replacing a previous certificate on the same account? */ view japanese
84        int i;
85        for (i = 0; i < md_pkeys_spec_count(d->md->pks); ++i) {
86            md_pkey_spec_t *spec = md_pkeys_spec_get(d->md->pks, i);
87            const md_pubcert_t *pubcert;
88            const md_cert_t *cert;
89            if (md_pkey_spec_eq(ad->cred->spec, spec)) {
90                rv = md_reg_get_pubcert(&pubcert, d->reg, d->md, i, d->p);
91                if (rv == APR_SUCCESS) {
92                    cert = APR_ARRAY_IDX(pubcert->certs, 0, const md_cert_t*);
93                    if (cert) {
94                        md_cert_get_ari_cert_id(&ari_cert_id, cert, d->p);
95                        break;
96                    }
97                }
98            }
99        }
100    }
101
102    md_result_activity_printf(result, "Creating new order, key-spec=%s, "
103                              "profile=%s, replacing-cert=%s",
104                              ad->cred->spec? md_pkey_spec_to_str(ad->cred->spec, d->p) : "default",
105                              ad->profile? ad->profile : "none",
106                              ari_cert_id? ari_cert_id : "none");
107
108    if (ad->profile) {
109        if (ad->acme->api.v2.profiles) {
110            int i;
111            for (i = 0; !profile && i < ad->acme->api.v2.profiles->nelts; ++i) {
112                const char *s = APR_ARRAY_IDX(ad->acme->api.v2.profiles, i, const char*);
113                if (!apr_cstr_casecmp(s, ad->profile))
114                   profile = s;
115            }
116        }
117        if (profile)
118            md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, rv, d->p,
119                          "%s: ordering ACME profile '%s'", md->name, profile);
120        else if (ad->profile_mandatory) {
121            md_log_perror(MD_LOG_MARK, MD_LOG_ERR, rv, d->p,
122                          "%s: mandatory ACME profile '%s' is not offered by CA",
123                          md->name, ad->profile);
124            rv = APR_EINVAL;
125            goto leave;
126        }
127        else {
128            md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, rv, d->p,
129                          "%s: ACME profile '%s' is not offered by CA, continuing without",
130                          md->name, ad->profile);
131        }
132    }
133
134    rv = md_acme_order_register(&ad->order, ad->acme, d->p, d->md->name,
135                                ad->domains, profile, ari_cert_id);
136    if (APR_SUCCESS !=rv) goto leave;
137    rv = md_acme_order_save(d->store, d->p, MD_SG_STAGING, d->md->name, ad->order, 0);
138    if (APR_SUCCESS != rv) {
139        md_result_set(result, rv, "saving order in staging");
140    }
141    if (pis_new) *pis_new = 1;
142
143leave:
144    md_acme_report_result(ad->acme, rv, result);
145    return rv;
146}

151apr_status_t md_acmev2_drive_renew(md_acme_driver_t *ad, md_proto_driver_t *d, md_result_t *result)
152{
153    apr_status_t rv = APR_SUCCESS;
154    int is_new_order = 0;
155
156    md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, 0, d->p, "%s: (ACMEv2) need certificate", d->md->name);
157    
158    /* Chose (or create) and ACME account to use */ view japanese
159    rv = md_acme_drive_set_acct(d, result);
160    if (APR_SUCCESS != rv) goto leave;
161
162    if (!md_array_is_empty(ad->cred->chain)) goto leave;
163        
164    /* ACMEv2 strategy:
165     * 1. load an md_acme_order_t from STAGING, if present
166     * 2. if no order found, register a new order at ACME server
167     * 3. update the order from the server
168     * 4. Switch order state:
169     *   * PENDING: process authz challenges
170     *   * READY: finalize the order
171     *   * PROCESSING: wait and re-assses later
172     *   * VALID: retrieve certificate
173     *   * COMPLETE: all done, return success
174     *   * INVALID and otherwise: fail renewal, delete local order
175     */ view japanese
176    if (APR_SUCCESS != (rv = ad_setup_order(d, result, &is_new_order))) {
177        goto leave;
178    }
179    
180    rv = md_acme_order_update(ad->order, ad->acme, result, d->p);
181    if (APR_STATUS_IS_ENOENT(rv)
182        || APR_STATUS_IS_EACCES(rv)
183        || MD_ACME_ORDER_ST_INVALID == ad->order->status) {
184        /* order is invalid or no longer known at the ACME server */ view japanese
185        ad->order = NULL;
186        md_acme_order_purge(d->store, d->p, MD_SG_STAGING, d->md, d->env);
187    }
188    else if (APR_SUCCESS != rv) {
189        goto leave;
190    }
191
192retry:
193    if (!ad->order) {
194        rv = ad_setup_order(d, result, &is_new_order);
195        if (APR_SUCCESS != rv) goto leave;
196    }
197    
198    rv = md_acme_order_start_challenges(ad->order, ad->acme, ad->ca_challenges,
199                                        d->store, d->md, d->env, result, d->p);
200    if (!is_new_order && APR_STATUS_IS_EINVAL(rv)) {
201        /* found 'invalid' domains in previous order, need to start over */ view japanese
202        ad->order = NULL;
203        md_acme_order_purge(d->store, d->p, MD_SG_STAGING, d->md, d->env);
204        goto retry;
205    }
206    if (APR_SUCCESS != rv) goto leave;
207    
208    rv = md_acme_order_monitor_authzs(ad->order, ad->acme, d->md,
209                                      ad->authz_monitor_timeout, result, d->p);
210    if (APR_SUCCESS != rv) {
211      md_result_set(result, rv, "Error waiting on domain names to be validated");
212      goto leave;
213    }
214
215    rv = md_acme_order_await_ready(ad->order, ad->acme, d->md,
216                                   ad->authz_monitor_timeout, result, d->p);
217    if (APR_SUCCESS != rv) {
218      md_result_set(result, rv, "Error waiting for order to become ready");
219      goto leave;
220    }
221
222    if (MD_ACME_ORDER_ST_READY == ad->order->status) {
223        rv = md_acme_drive_setup_cred_chain(d, result);
224        if (APR_SUCCESS != rv) goto leave;
225        md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, 0, d->p, "%s: finalized order", d->md->name);
226    }
227
228    rv = md_acme_order_await_valid(ad->order, ad->acme, d->md, 
229                                   ad->authz_monitor_timeout, result, d->p);
230    if (APR_SUCCESS != rv) {
231      md_result_set(result, rv, "Error waiting for order to become valid.");
232      goto leave;
233    }
234    
235    if (!ad->order->certificate) {
236        md_result_set(result, APR_EINVAL, "Order valid, but certificate url is missing.");
237        goto leave;
238    }
239    md_result_set(result, APR_SUCCESS, NULL);
240
241leave:    
242    md_result_log(result, MD_LOG_DEBUG);
243    return result->status;
244}
