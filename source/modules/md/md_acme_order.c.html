44md_acme_order_t *md_acme_order_create(apr_pool_t *p)
45{
46    md_acme_order_t *order;
47    
48    order = apr_pcalloc(p, sizeof(*order));
49    order->p = p;
50    order->authz_urls = apr_array_make(p, 5, sizeof(const char *));
51    order->challenge_setups = apr_array_make(p, 5, sizeof(const char *));
52    
53    return order;
54}

61static md_acme_order_st order_st_from_str(const char *s) 
62{
63    if (s) {
64        if (!strcmp("valid", s)) {
65            return MD_ACME_ORDER_ST_VALID;
66        }
67        else if (!strcmp("invalid", s)) {
68            return MD_ACME_ORDER_ST_INVALID;
69        }
70        else if (!strcmp("ready", s)) {
71            return MD_ACME_ORDER_ST_READY;
72        }
73        else if (!strcmp("pending", s)) {
74            return MD_ACME_ORDER_ST_PENDING;
75        }
76        else if (!strcmp("processing", s)) {
77            return MD_ACME_ORDER_ST_PROCESSING;
78        }
79    }
80    return MD_ACME_ORDER_ST_PENDING;
81}
82
83static const char *order_st_to_str(md_acme_order_st status) 
84{
85    switch (status) {
86        case MD_ACME_ORDER_ST_PENDING:
87            return "pending";
88        case MD_ACME_ORDER_ST_READY:
89            return "ready";
90        case MD_ACME_ORDER_ST_PROCESSING:
91            return "processing";
92        case MD_ACME_ORDER_ST_VALID:
93            return "valid";
94        case MD_ACME_ORDER_ST_INVALID:
95            return "invalid";
96        default:
97            return "invalid";
98    }
99}

101md_json_t *md_acme_order_to_json(md_acme_order_t *order, apr_pool_t *p)
102{
103    md_json_t *json = md_json_create(p);
104
105    if (order->url) {
106        md_json_sets(order->url, json, MD_KEY_URL, NULL);
107    }
108    md_json_sets(order_st_to_str(order->status), json, MD_KEY_STATUS, NULL);
109    md_json_setsa(order->authz_urls, json, MD_KEY_AUTHORIZATIONS, NULL);
110    md_json_setsa(order->challenge_setups, json, MD_KEY_CHALLENGE_SETUPS, NULL);
111    if (order->finalize) {
112        md_json_sets(order->finalize, json, MD_KEY_FINALIZE, NULL);
113    }
114    if (order->certificate) {
115        md_json_sets(order->certificate, json, MD_KEY_CERTIFICATE, NULL);
116    }
117    return json;
118}
119
120static void order_update_from_json(md_acme_order_t *order, md_json_t *json, apr_pool_t *p)
121{
122    if (!order->url && md_json_has_key(json, MD_KEY_URL, NULL)) {
123        order->url = md_json_dups(p, json, MD_KEY_URL, NULL);
124    }
125    order->status = order_st_from_str(md_json_gets(json, MD_KEY_STATUS, NULL));
126    if (md_json_has_key(json, MD_KEY_AUTHORIZATIONS, NULL)) {
127        md_json_dupsa(order->authz_urls, p, json, MD_KEY_AUTHORIZATIONS, NULL);
128    }
129    if (md_json_has_key(json, MD_KEY_CHALLENGE_SETUPS, NULL)) {
130        md_json_dupsa(order->challenge_setups, p, json, MD_KEY_CHALLENGE_SETUPS, NULL);
131    }
132    if (md_json_has_key(json, MD_KEY_FINALIZE, NULL)) {
133        order->finalize = md_json_dups(p, json, MD_KEY_FINALIZE, NULL);
134    }
135    if (md_json_has_key(json, MD_KEY_CERTIFICATE, NULL)) {
136        order->certificate = md_json_dups(p, json, MD_KEY_CERTIFICATE, NULL);
137    }
138}

140md_acme_order_t *md_acme_order_from_json(md_json_t *json, apr_pool_t *p)
141{
142    md_acme_order_t *order = md_acme_order_create(p);
143
144    order_update_from_json(order, json, p);
145    return order;
146}

148apr_status_t md_acme_order_add(md_acme_order_t *order, const char *authz_url)
149{
150    assert(authz_url);
151    if (md_array_str_index(order->authz_urls, authz_url, 0, 1) < 0) {
152        APR_ARRAY_PUSH(order->authz_urls, const char*) = apr_pstrdup(order->p, authz_url);
153    }
154    return APR_SUCCESS;
155}
156
157apr_status_t md_acme_order_remove(md_acme_order_t *order, const char *authz_url)
158{
159    int i;
160    
161    assert(authz_url);
162    i = md_array_str_index(order->authz_urls, authz_url, 0, 1);
163    if (i >= 0) {
164        order->authz_urls = md_array_str_remove(order->p, order->authz_urls, authz_url, 1);
165        return APR_SUCCESS;
166    }
167    return APR_ENOENT;
168}

170static apr_status_t add_setup_token(md_acme_order_t *order, const char *token)
171{
172    if (md_array_str_index(order->challenge_setups, token, 0, 1) < 0) {
173        APR_ARRAY_PUSH(order->challenge_setups, const char*) = apr_pstrdup(order->p, token);
174    }
175    return APR_SUCCESS;
176}

181apr_status_t md_acme_order_load(struct md_store_t *store, md_store_group_t group, 
182                                    const char *md_name, md_acme_order_t **pauthz_set, 
183                                    apr_pool_t *p)
184{
185    apr_status_t rv;
186    md_json_t *json;
187    md_acme_order_t *authz_set;
188    
189    rv = md_store_load_json(store, group, md_name, MD_FN_ORDER, &json, p);
190    if (APR_SUCCESS == rv) {
191        authz_set = md_acme_order_from_json(json, p);
192    }
193    *pauthz_set = (APR_SUCCESS == rv)? authz_set : NULL;
194    return rv;  
195}

224static apr_status_t p_purge(void *baton, apr_pool_t *p, apr_pool_t *ptemp, va_list ap)
225{
226    md_store_t *store = baton;
227    md_acme_order_t *order;
228    md_store_group_t group;
229    const md_t *md;
230    const char *setup_token;
231    apr_table_t *env;
232    int i;
233
234    group = (md_store_group_t)va_arg(ap, int);
235    md = va_arg(ap, const md_t *);
236    env = va_arg(ap, apr_table_t *);
237
238    if (APR_SUCCESS == md_acme_order_load(store, group, md->name, &order, p)) {
239        md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, 0, p, "order loaded for %s", md->name);
240        for (i = 0; i < order->challenge_setups->nelts; ++i) {
241            setup_token = APR_ARRAY_IDX(order->challenge_setups, i, const char*);
242            if (setup_token) {
243                md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, 0, p, 
244                              "order teardown setup %s", setup_token);
245                md_acme_authz_teardown(store, setup_token, md, env, p);
246            }
247        }
248    }
249    return md_store_remove(store, group, md->name, MD_FN_ORDER, ptemp, 1);
250}
251
252apr_status_t md_acme_order_purge(md_store_t *store, apr_pool_t *p, md_store_group_t group,
253                                 const md_t *md, apr_table_t *env)
254{
255    return md_util_pool_vdo(p_purge, store, p, group, md, env, NULL);
256}

261typedef struct {
262    apr_pool_t *p;

272#define ORDER_CTX_INIT(ctx, p, o, a, n, d, pf, cid, r) \
273    (ctx)->p = (p); (ctx)->order = (o); (ctx)->acme = (a); \
274    (ctx)->name = (n); (ctx)->domains = d; (ctx)->profile = pf; \
275    (ctx)->ari_cert_id = cid;(ctx)->result = r

277static apr_status_t identifier_to_json(void *value, md_json_t *json, apr_pool_t *p, void *baton)
278{
279    md_json_t *jid;
280    
281    (void)baton;
282    jid = md_json_create(p);
283    md_json_sets("dns", jid, "type", NULL);
284    md_json_sets(value, jid, "value", NULL);
285    return md_json_setj(jid, json, NULL);
286}

288static apr_status_t on_init_order_register(md_acme_req_t *req, void *baton)
289{
290    order_ctx_t *ctx = baton;
291    md_json_t *jpayload;
292
293    jpayload = md_json_create(req->p);
294    md_json_seta(ctx->domains, identifier_to_json, NULL, jpayload, "identifiers", NULL);
295    if (ctx->profile)
296        md_json_sets(ctx->profile, jpayload, "profile", NULL);
297    if (ctx->ari_cert_id)
298        md_json_sets(ctx->ari_cert_id, jpayload, "replaces", NULL);
299
300    return md_acme_req_body_init(req, jpayload);
301}

303static apr_status_t on_order_upd(md_acme_t *acme, apr_pool_t *p, const apr_table_t *hdrs, 
304                                 md_json_t *body, void *baton)
305{
306    order_ctx_t *ctx = baton;
307    const char *location = apr_table_get(hdrs, "location");
308    apr_status_t rv = APR_SUCCESS;
309    
310    (void)acme;
311    (void)p;
312    if (!ctx->order) {
313        if (location) {
314            ctx->order = md_acme_order_create(ctx->p);
315            ctx->order->url = apr_pstrdup(ctx->p, location);
316            md_log_perror(MD_LOG_MARK, MD_LOG_TRACE1, rv, ctx->p, "new order at %s", location);
317        }
318        else {
319            rv = APR_EINVAL;
320            md_log_perror(MD_LOG_MARK, MD_LOG_WARNING, rv, ctx->p, "new order, no location header");
321            goto out;
322        }
323    }
324    
325    order_update_from_json(ctx->order, body, ctx->p);
326out:
327    return rv;
328}

330apr_status_t md_acme_order_register(md_acme_order_t **porder, md_acme_t *acme, apr_pool_t *p, 
331                                    const char *name, apr_array_header_t *domains,
332                                    const char *profile, const char *ari_cert_id)
333{
334    order_ctx_t ctx;
335    apr_status_t rv;
336    
337    assert(MD_ACME_VERSION_MAJOR(acme->version) > 1);
338    ORDER_CTX_INIT(&ctx, p, NULL, acme, name, domains, profile, ari_cert_id, NULL);
339    rv = md_acme_POST(acme, acme->api.v2.new_order, on_init_order_register, on_order_upd, NULL, NULL, &ctx);
340    *porder = (APR_SUCCESS == rv)? ctx.order : NULL;
341    return rv;
342}

344apr_status_t md_acme_order_update(md_acme_order_t *order, md_acme_t *acme, 
345                                  md_result_t *result, apr_pool_t *p)
346{
347    order_ctx_t ctx;
348    apr_status_t rv;
349    
350    assert(MD_ACME_VERSION_MAJOR(acme->version) > 1);
351    ORDER_CTX_INIT(&ctx, p, order, acme, NULL, NULL, NULL, NULL, result);
352    rv = md_acme_GET(acme, order->url, NULL, on_order_upd, NULL, NULL, 1, &ctx);
353    if (APR_SUCCESS != rv && APR_SUCCESS != acme->last->status) {
354        md_result_dup(result, acme->last);
355    }
356    return rv;
357}

359static apr_status_t await_ready(void *baton, int attempt)
360{
361    order_ctx_t *ctx = baton;
362    apr_status_t rv = APR_SUCCESS;
363    
364    (void)attempt;
365    if (APR_SUCCESS != (rv = md_acme_order_update(ctx->order, ctx->acme,
366                                                  ctx->result, ctx->p))) goto out;
367    switch (ctx->order->status) {
368        case MD_ACME_ORDER_ST_READY:
369        case MD_ACME_ORDER_ST_PROCESSING:
370        case MD_ACME_ORDER_ST_VALID:
371            break;
372        case MD_ACME_ORDER_ST_PENDING:
373            rv = APR_EAGAIN;
374            break;
375        default:
376            rv = APR_EINVAL;
377            break;
378    }
379out:    
380    return rv;
381}

383apr_status_t md_acme_order_await_ready(md_acme_order_t *order, md_acme_t *acme, 
384                                       const md_t *md, apr_interval_time_t timeout, 
385                                       md_result_t *result, apr_pool_t *p)
386{
387    order_ctx_t ctx;
388    apr_status_t rv;
389    
390    assert(MD_ACME_VERSION_MAJOR(acme->version) > 1);
391    ORDER_CTX_INIT(&ctx, p, order, acme, md->name, NULL, NULL, NULL, result);
392
393    md_result_activity_setn(result, "Waiting for order to become ready");
394    rv = md_util_try(await_ready, &ctx, 0, timeout, 0, 0, 1);
395    md_result_log(result, MD_LOG_DEBUG);
396    return rv;
397}

399static apr_status_t await_valid(void *baton, int attempt)
400{
401    order_ctx_t *ctx = baton;
402    apr_status_t rv = APR_SUCCESS;
403
404    (void)attempt;
405    if (APR_SUCCESS != (rv = md_acme_order_update(ctx->order, ctx->acme, 
406                                                  ctx->result, ctx->p))) goto out;
407    switch (ctx->order->status) {
408        case MD_ACME_ORDER_ST_VALID:
409            md_result_set(ctx->result, APR_EINVAL, "ACME server order status is 'valid'.");
410            break;
411        case MD_ACME_ORDER_ST_PROCESSING:
412            rv = APR_EAGAIN;
413            break;
414        case MD_ACME_ORDER_ST_INVALID:
415            md_result_set(ctx->result, APR_EINVAL, "ACME server order status is 'invalid'.");
416            rv = APR_EINVAL;
417            break;
418        default:
419            rv = APR_EINVAL;
420            break;
421    }
422out:    
423    return rv;
424}

445apr_status_t md_acme_order_start_challenges(md_acme_order_t *order, md_acme_t *acme, 
446                                            apr_array_header_t *challenge_types,
447                                            md_store_t *store, const md_t *md, 
448                                            apr_table_t *env, md_result_t *result, 
449                                            apr_pool_t *p)
450{
451    apr_status_t rv = APR_SUCCESS;
452    md_acme_authz_t *authz;
453    const char *url, *setup_token;
454    int i;
455    
456    md_result_activity_printf(result, "Starting challenges for domains");
457    for (i = 0; i < order->authz_urls->nelts; ++i) {
458        url = APR_ARRAY_IDX(order->authz_urls, i, const char*);
459        md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, rv, p, "%s: check AUTHZ at %s", md->name, url);
460        
461        if (APR_SUCCESS != (rv = md_acme_authz_retrieve(acme, p, url, &authz))) {
462            md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, rv, p, "%s: check authz for %s",
463                          md->name, authz->domain);
464            goto leave;
465        }
466
467        switch (authz->state) {
468            case MD_ACME_AUTHZ_S_VALID:
469                break;
470                
471            case MD_ACME_AUTHZ_S_PENDING:
472                md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, rv, p,
473                              "%s: authorization pending for %s",
474                              md->name, authz->domain);
475                rv = md_acme_authz_respond(authz, acme, store, challenge_types,
476                                           md->pks,
477                                           md->acme_tls_1_domains, md,
478                                           env, p, &setup_token, result);
479                if (APR_SUCCESS != rv) {
480                    goto leave;
481                }
482                add_setup_token(order, setup_token);
483                md_acme_order_save(store, p, MD_SG_STAGING, md->name, order, 0);
484                break;
485                
486            case MD_ACME_AUTHZ_S_INVALID:
487                rv = APR_EINVAL;
488                if (authz->error_type) {
489                    md_result_problem_set(result, rv, authz->error_type, authz->error_detail, NULL);
490                    goto leave;
491                }
492                /* fall through */ view japanese
493            default:
494                rv = APR_EINVAL;
495                md_result_printf(result, rv, "unexpected AUTHZ state %d for domain %s", 
496                                 authz->state, authz->domain);
497                md_result_log(result, MD_LOG_ERR);
498                goto leave;
499        }
500    }
501leave:    
502    return rv;
503}

505static apr_status_t check_challenges(void *baton, int attempt)
506{
507    order_ctx_t *ctx = baton;
508    const char *url;
509    md_acme_authz_t *authz;
510    apr_status_t rv = APR_SUCCESS;
511    int i;
512    
513    for (i = 0; i < ctx->order->authz_urls->nelts; ++i) {
514        url = APR_ARRAY_IDX(ctx->order->authz_urls, i, const char*);
515        md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, rv, ctx->p, "%s: check AUTHZ at %s (attempt %d)", 
516                      ctx->name, url, attempt);
517        
518        rv = md_acme_authz_retrieve(ctx->acme, ctx->p, url, &authz);
519        if (APR_SUCCESS == rv) {
520            switch (authz->state) {
521                case MD_ACME_AUTHZ_S_VALID:
522                    md_result_printf(ctx->result, rv, 
523                                     "domain authorization for %s is valid", authz->domain);
524                    break;
525                case MD_ACME_AUTHZ_S_PENDING:
526                    rv = APR_EAGAIN;
527                    md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, rv, ctx->p, 
528                                  "%s: status pending at %s", authz->domain, authz->url);
529                    goto leave;
530                case MD_ACME_AUTHZ_S_INVALID:
531                    rv = APR_EINVAL;
532                    md_result_printf(ctx->result, rv,
533                                     "domain authorization for %s failed, CA considers "
534                                     "answer to challenge invalid%s.",
535                                     authz->domain, authz->error_type? "" : ", no error given");
536                    md_result_log(ctx->result, MD_LOG_ERR);
537                    goto leave;
538                default:
539                    rv = APR_EINVAL;
540                    md_result_printf(ctx->result, rv, 
541                                     "domain authorization for %s failed with state %d", 
542                                     authz->domain, authz->state);
543                    md_result_log(ctx->result, MD_LOG_ERR);
544                    goto leave;
545            }
546        }
547        else {
548            md_result_printf(ctx->result, rv, "authorization retrieval failed for %s on <%s>",
549                             ctx->name, url);
550        }
551    }
552leave:
553    return rv;
554}

556apr_status_t md_acme_order_monitor_authzs(md_acme_order_t *order, md_acme_t *acme, 
557                                          const md_t *md, apr_interval_time_t timeout, 
558                                          md_result_t *result, apr_pool_t *p)
559{
560    order_ctx_t ctx;
561    apr_status_t rv;
562    
563    ORDER_CTX_INIT(&ctx, p, order, acme, md->name, NULL, NULL, NULL, result);
564
