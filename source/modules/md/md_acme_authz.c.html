82apr_status_t md_acme_authz_retrieve(md_acme_t *acme, apr_pool_t *p, const char *url, 
83                                    md_acme_authz_t **pauthz)
84{
85    md_acme_authz_t *authz;
86    apr_status_t rv;
87    
88    authz = apr_pcalloc(p, sizeof(*authz));
89    authz->url = apr_pstrdup(p, url);
90    rv = md_acme_authz_update(authz, acme, p);
91    
92    *pauthz = (APR_SUCCESS == rv)? authz : NULL;
93    return rv;
94}

101static int copy_challenge_error(void *baton, size_t index, md_json_t *json)
102{
103    error_ctx_t *ctx = baton;
104    
105    (void)index;
106    if (md_json_has_key(json, MD_KEY_ERROR, NULL)) {
107        ctx->authz->error_type = md_json_dups(ctx->p, json, MD_KEY_ERROR, MD_KEY_TYPE, NULL);
108        ctx->authz->error_detail = md_json_dups(ctx->p, json, MD_KEY_ERROR, MD_KEY_DETAIL, NULL);
109        ctx->authz->error_subproblems = md_json_dupj(ctx->p, json, MD_KEY_ERROR, MD_KEY_SUBPROBLEMS, NULL);
110    }
111    return 1;
112}

114apr_status_t md_acme_authz_update(md_acme_authz_t *authz, md_acme_t *acme, apr_pool_t *p)
115{
116    md_json_t *json;
117    const char *s, *err;
118    md_log_level_t log_level;
119    apr_status_t rv;
120    error_ctx_t ctx;
121    
122    assert(acme);
123    assert(acme->http);
124    assert(authz);
125    assert(authz->url);
126
127    authz->state = MD_ACME_AUTHZ_S_UNKNOWN;
128    json = NULL;
129    authz->error_type = authz->error_detail = NULL;
130    authz->error_subproblems = NULL;
131    err = "unable to parse response";
132    log_level = MD_LOG_ERR;
133    
134    if (APR_SUCCESS == (rv = md_acme_get_json(&json, acme, authz->url, 1, p))
135        && (s = md_json_gets(json, MD_KEY_STATUS, NULL))) {
136            
137        authz->domain = md_json_gets(json, MD_KEY_IDENTIFIER, MD_KEY_VALUE, NULL); 
138        authz->resource = json;
139        if (!strcmp(s, "pending")) {
140            authz->state = MD_ACME_AUTHZ_S_PENDING;
141            err = "challenge 'pending'";
142            log_level = MD_LOG_DEBUG;
143        }
144        else if (!strcmp(s, "valid")) {
145            authz->state = MD_ACME_AUTHZ_S_VALID;
146            err = "challenge 'valid'";
147            log_level = MD_LOG_DEBUG;
148        }
149        else if (!strcmp(s, "invalid")) {
150            ctx.p = p;
151            ctx.authz = authz;
152            authz->state = MD_ACME_AUTHZ_S_INVALID;
153            md_json_itera(copy_challenge_error, &ctx, json, MD_KEY_CHALLENGES, NULL);
154            err = "challenge 'invalid'";
155        }
156    }
157
158    if (json && authz->state == MD_ACME_AUTHZ_S_UNKNOWN) {
159        err = "unable to understand response";
160        rv = APR_EINVAL;
161    }
162    
163    if (md_log_is_level(p, log_level)) {
164        md_log_perror(MD_LOG_MARK, log_level, rv, p, "ACME server authz: %s for %s at %s. "
165                      "Exact response was: %s", err, authz->domain, authz->url,
166                      json? md_json_writep(json, p, MD_JSON_FMT_COMPACT) : "not available");
167    }
168    
169    return rv;
170}

175static md_acme_authz_cha_t *cha_from_json(apr_pool_t *p, size_t index, md_json_t *json)
176{
177    md_acme_authz_cha_t * cha;
178    
179    cha = apr_pcalloc(p, sizeof(*cha));
180    cha->index = index;
181    cha->type = md_json_dups(p, json, MD_KEY_TYPE, NULL);
182    if (md_json_has_key(json, MD_KEY_URL, NULL)) { /* ACMEv2 */ view japanese
183        cha->uri = md_json_dups(p, json, MD_KEY_URL, NULL);
184    }
185    else {                                         /* ACMEv1 */ view japanese
186        cha->uri = md_json_dups(p, json, MD_KEY_URI, NULL);
187    }
188    cha->token = md_json_dups(p, json, MD_KEY_TOKEN, NULL);
189    cha->key_authz = md_json_dups(p, json, MD_KEY_KEYAUTHZ, NULL);
190
191    return cha;
192}

216static apr_status_t setup_key_authz(md_acme_authz_cha_t *cha, md_acme_authz_t *authz,
217                                    md_acme_t *acme, apr_pool_t *p, int *pchanged)
218{
219    const char *thumb64, *key_authz;
220    apr_status_t rv;
221    
222    (void)authz;
223    assert(cha);
224    assert(cha->token);
225    
226    *pchanged = 0;
227    if (APR_SUCCESS == (rv = md_jws_pkey_thumb(&thumb64, p, acme->acct_key))) {
228        key_authz = apr_psprintf(p, "%s.%s", cha->token, thumb64);
229        if (cha->key_authz) {
230            if (strcmp(key_authz, cha->key_authz)) {
231                /* Hu? Did the account change key? */ view japanese
232                cha->key_authz = NULL;
233            }
234        }
235        if (!cha->key_authz) {
236            cha->key_authz = key_authz;
237            *pchanged = 1;
238        }
239    }
240    return rv;
241}

243static apr_status_t cha_http_01_setup(md_acme_authz_cha_t *cha, md_acme_authz_t *authz,
244                                      md_acme_t *acme, md_store_t *store, 
245                                      md_pkeys_spec_t *key_specs,
246                                      apr_array_header_t *acme_tls_1_domains, const md_t *md,
247                                      apr_table_t *env, md_result_t *result,
248                                      const char **psetup_token, apr_pool_t *p)
249{
250    const char *data;
251    apr_status_t rv;
252    int notify_server;
253    
254    (void)key_specs;
255    (void)env;
256    (void)acme_tls_1_domains;
257    (void)md;
258
259    if (APR_SUCCESS != (rv = setup_key_authz(cha, authz, acme, p, &notify_server))) {
260        goto out;
261    }
262    
263    rv = md_store_load(store, MD_SG_CHALLENGES, authz->domain, MD_FN_HTTP01,
264                       MD_SV_TEXT, (void**)&data, p);
265    if ((APR_SUCCESS == rv && strcmp(cha->key_authz, data)) || APR_STATUS_IS_ENOENT(rv)) {
266        rv = md_store_save(store, p, MD_SG_CHALLENGES, authz->domain, MD_FN_HTTP01,
267                           MD_SV_TEXT, (void*)cha->key_authz, 0);
268        notify_server = 1;
269    }
270    
271    if (APR_SUCCESS == rv && notify_server) {
272        authz_req_ctx ctx;
273        const char *event;
274
275        /* Raise event that challenge data has been set up before we tell the
276           ACME server. Clusters might want to distribute it. */
277        event = apr_psprintf(p, "challenge-setup:%s:%s", MD_AUTHZ_TYPE_HTTP01, authz->domain);
278        rv = md_result_raise(result, event, p);
279        if (APR_SUCCESS != rv) {
280            md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, rv, p,
281                          "%s: event '%s' failed. aborting challenge setup",
282                          authz->domain, event);
283            goto out;
284        }
285        /* challenge is setup or was changed from previous data, tell ACME server
286         * so it may (re)try verification */        
287        authz_req_ctx_init(&ctx, acme, NULL, authz, p);
288        ctx.challenge = cha;
289        rv = md_acme_POST(acme, cha->uri, on_init_authz_resp, authz_http_set, NULL, NULL, &ctx);
290    }
291out:
292    *psetup_token = (APR_SUCCESS == rv)?
293        apr_psprintf(p, "%s:%s", MD_AUTHZ_TYPE_HTTP01, authz->domain) : NULL;
294    return rv;
295}

297void tls_alpn01_fnames(apr_pool_t *p, md_pkey_spec_t *kspec, char **keyfn, char **certfn )
298{
299    *keyfn  = apr_pstrcat(p, "acme-tls-alpn-01-", md_pkey_filename(kspec, p), NULL);
300    *certfn = apr_pstrcat(p, "acme-tls-alpn-01-", md_chain_filename(kspec, p), NULL);
301}

303static apr_status_t cha_tls_alpn_01_setup(md_acme_authz_cha_t *cha, md_acme_authz_t *authz, 
304                                          md_acme_t *acme, md_store_t *store, 
305                                          md_pkeys_spec_t *key_specs,
306                                          apr_array_header_t *acme_tls_1_domains, const md_t *md,
307                                          apr_table_t *env, md_result_t *result,
308                                          const char **psetup_token, apr_pool_t *p)
309{
310    const char *acme_id, *token;
311    apr_status_t rv;
312    int notify_server;
313    md_data_t data;
314    int i;
315
316    (void)env;
317    (void)md;
318    if (md_array_str_index(acme_tls_1_domains, authz->domain, 0, 0) < 0) {
319        rv = APR_ENOTIMPL;
320        if (acme_tls_1_domains->nelts) {
321            md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, 0, p,
322                          "%s: protocol 'acme-tls/1' seems not enabled for this domain, "
323                          "but is enabled for other associated domains. "
324                          "Continuing with fingers crossed.", authz->domain);
325        }
326        else {
327            md_log_perror(MD_LOG_MARK, MD_LOG_INFO, 0, p,
328                          "%s: protocol 'acme-tls/1' seems not enabled for this or "
329                          "any other associated domain. Not attempting challenge "
330                          "type tls-alpn-01.", authz->domain);
331            goto out;
332        }
333    }
334    if (APR_SUCCESS != (rv = setup_key_authz(cha, authz, acme, p, &notify_server))) {
335        goto out;
336    }
337
338    /* Create a "tls-alpn-01" certificate for the domain we want to authenticate.
339     * The server will need to answer a TLS connection with SNI == authz->domain
340     * and ALPN protocol "acme-tls/1" with this certificate.
341     */ view japanese
342    md_data_init_str(&data, cha->key_authz);
343    rv = md_crypt_sha256_digest_hex(&token, p, &data);
344    if (APR_SUCCESS != rv) {
345        md_log_perror(MD_LOG_MARK, MD_LOG_ERR, rv, p, "%s: create tls-alpn-01 validation token",
346                      authz->domain);
347        goto out;
348    }
349    acme_id = apr_psprintf(p, "critical,DER:04:20:%s", token);
350
351    /* Each configured key type must be generated to ensure:
352     * that any fallback certs already given to mod_ssl are replaced.
353     * We expect that the validation client (at the CA) can deal with at
354     * least one of them.
355     */ view japanese
356
357    for (i = 0; i < md_pkeys_spec_count(key_specs); ++i) {
358        char *kfn, *cfn;
359        md_cert_t *cha_cert;
360        md_pkey_t *cha_key;
361        md_pkey_spec_t *key_spec;
362
363        key_spec = md_pkeys_spec_get(key_specs, i);
364        tls_alpn01_fnames(p, key_spec, &kfn, &cfn);
365
366        rv = md_store_load(store, MD_SG_CHALLENGES, authz->domain, cfn,
367                           MD_SV_CERT, (void**)&cha_cert, p);
368        if ((APR_SUCCESS == rv && !md_cert_covers_domain(cha_cert, authz->domain))
369            || APR_STATUS_IS_ENOENT(rv)) {
370            if (APR_SUCCESS != (rv = md_pkey_gen(&cha_key, p, key_spec))) {
371                md_log_perror(MD_LOG_MARK, MD_LOG_ERR, rv, p, "%s: create tls-alpn-01 %s challenge key",
372                              authz->domain, md_pkey_spec_name(key_spec));
373                goto out;
374            }
375
376            if (APR_SUCCESS != (rv = md_cert_make_tls_alpn_01(&cha_cert, authz->domain, acme_id, cha_key,
377                                                              apr_time_from_sec(7 * MD_SECS_PER_DAY), p))) {
378                md_log_perror(MD_LOG_MARK, MD_LOG_ERR, rv, p, "%s: create tls-alpn-01 %s challenge cert",
379                              authz->domain, md_pkey_spec_name(key_spec));
380                goto out;
381            }
382        
383            if (APR_SUCCESS == (rv = md_store_save(store, p, MD_SG_CHALLENGES, authz->domain, kfn,
384                                                   MD_SV_PKEY, (void*)cha_key, 0))) {
385                rv = md_store_save(store, p, MD_SG_CHALLENGES, authz->domain, cfn,
386                                   MD_SV_CERT, (void*)cha_cert, 0);
387            }
388            ++notify_server;
389        }
390    }
391    
392    if (APR_SUCCESS == rv && notify_server) {
393        authz_req_ctx ctx;
394        const char *event;
395
396        /* Raise event that challenge data has been set up before we tell the
397           ACME server. Clusters might want to distribute it. */ view japanese
398        event = apr_psprintf(p, "challenge-setup:%s:%s", MD_AUTHZ_TYPE_TLSALPN01, authz->domain);
399        rv = md_result_raise(result, event, p);
400        if (APR_SUCCESS != rv) {
401            md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, rv, p,
402                          "%s: event '%s' failed. aborting challenge setup",
403                          authz->domain, event);
404            goto out;
405        }
406        /* challenge is setup or was changed from previous data, tell ACME server
407         * so it may (re)try verification */         view japanese
408        authz_req_ctx_init(&ctx, acme, NULL, authz, p);
409        ctx.challenge = cha;
410        rv = md_acme_POST(acme, cha->uri, on_init_authz_resp, authz_http_set, NULL, NULL, &ctx);
411    }
412out:    
413    *psetup_token = (APR_SUCCESS == rv)?
414        apr_psprintf(p, "%s:%s", MD_AUTHZ_TYPE_TLSALPN01, authz->domain) : NULL;
415    return rv;
416}

418static apr_status_t cha_dns_01_setup(md_acme_authz_cha_t *cha, md_acme_authz_t *authz, 
419                                     md_acme_t *acme, md_store_t *store, 
420                                     md_pkeys_spec_t *key_specs,
421                                     apr_array_header_t *acme_tls_1_domains, const md_t *md,
422                                     apr_table_t *env, md_result_t *result,
423                                     const char **psetup_token, apr_pool_t *p)
424{
425    const char *token;
426    const char * const *argv;
427    const char *cmdline, *dns01_cmd;
428    apr_status_t rv;
429    int exit_code, notify_server;
430    authz_req_ctx ctx;
431    md_data_t data;
432    const char *event;
433
434    (void)store;
435    (void)key_specs;
436    (void)acme_tls_1_domains;
437
438    dns01_cmd = md->dns01_cmd;
439    if (!dns01_cmd)
440      dns01_cmd = apr_table_get(env, MD_KEY_CMD_DNS01);
441    if (!dns01_cmd) {
442        rv = APR_ENOTIMPL;
443        md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, rv, p, "%s: dns-01 command not set", 
444                      authz->domain);
445        goto out;
446    }
447    
448    if (APR_SUCCESS != (rv = setup_key_authz(cha, authz, acme, p, &notify_server))) {
449        goto out;
450    }
451    
452    md_data_init_str(&data, cha->key_authz);
453    rv = md_crypt_sha256_digest64(&token, p, &data);
454    if (APR_SUCCESS != rv) {
455        md_log_perror(MD_LOG_MARK, MD_LOG_ERR, rv, p, "%s: create dns-01 token for %s",
456                      md->name, authz->domain);
457        goto out;
458    }
459
460    cmdline = apr_psprintf(p, "%s setup %s %s", dns01_cmd, authz->domain, token); 
461    md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, 0, p, 
462                  "%s: dns-01 setup command: %s", authz->domain, cmdline);
463
464    apr_tokenize_to_argv(cmdline, (char***)&argv, p);
465    if (APR_SUCCESS != (rv = md_util_exec(p, argv[0], argv, &exit_code))) {
466        md_log_perror(MD_LOG_MARK, MD_LOG_WARNING, rv, p, 
467                      "%s: dns-01 setup command failed to execute for %s", md->name, authz->domain);
468        goto out;
469    }
470    if (exit_code) {
471        rv = APR_EGENERAL;
472        md_log_perror(MD_LOG_MARK, MD_LOG_INFO, rv, p, 
473                      "%s: dns-01 setup command returns %d for %s", md->name, exit_code, authz->domain);
474        goto out;
475    }
476    
477    /* Raise event that challenge data has been set up before we tell the
478       ACME server. Clusters might want to distribute it. */ view japanese
479    event = apr_psprintf(p, "challenge-setup:%s:%s", MD_AUTHZ_TYPE_DNS01, authz->domain);
480    rv = md_result_raise(result, event, p);
481    if (APR_SUCCESS != rv) {
482        md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, rv, p,
483                      "%s: event '%s' failed. aborting challenge setup",
484                      authz->domain, event);
485        goto out;
486    }
487    /* challenge is setup, tell ACME server so it may (re)try verification */ view japanese
488    md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, rv, p, "%s: dns-01 setup succeeded for %s",
489       md->name, authz->domain);
490    authz_req_ctx_init(&ctx, acme, NULL, authz, p);
491    ctx.challenge = cha;
492    rv = md_acme_POST(acme, cha->uri, on_init_authz_resp, authz_http_set, NULL, NULL, &ctx);
493    
494out:    
495    *psetup_token = (APR_SUCCESS == rv)?
496        apr_psprintf(p, "%s:%s %s", MD_AUTHZ_TYPE_DNS01, authz->domain, token) : NULL;
497    return rv;
498}

500static apr_status_t cha_dns_01_teardown(md_store_t *store, const char *domain, const md_t *md,
501                                        apr_table_t *env, apr_pool_t *p)
502{
503    const char * const *argv;
504    const char *cmdline, *dns01_cmd, *dns01v;
505    char *tmp, *s;
506    apr_status_t rv;
507    int exit_code;
508    
509    (void)store;
510
511    dns01_cmd = md->dns01_cmd;
512    if (!dns01_cmd)
513      dns01_cmd = apr_table_get(env, MD_KEY_CMD_DNS01);
514    if (!dns01_cmd) {
515        rv = APR_ENOTIMPL;
516        md_log_perror(MD_LOG_MARK, MD_LOG_DEBUG, 0, p, "%s: dns-01 command not set for %s",
517            md->name, domain);
518        goto out;
519    }
520    dns01v = apr_table_get(env, MD_KEY_DNS01_VERSION);
521    if (!dns01v || strcmp(dns01v, "2")) {
522        /* use older version of teardown args with only domain, remove token */ view japanese
523        tmp = apr_pstrdup(p, domain);
524        s = strchr(tmp, ' ');
525        if (s) {
526            *s = '\0';
527            domain = tmp;
528        }
529    }
530
531    cmdline = apr_psprintf(p, "%s teardown %s", dns01_cmd, domain); 
532    apr_tokenize_to_argv(cmdline, (char***)&argv, p);
533    if (APR_SUCCESS != (rv = md_util_exec(p, argv[0], argv, &exit_code)) || exit_code) {
534        md_log_perror(MD_LOG_MARK, MD_LOG_WARNING, rv, p,

604apr_status_t md_acme_authz_respond(md_acme_authz_t *authz, md_acme_t *acme, md_store_t *store, 
605                                   apr_array_header_t *challenges, md_pkeys_spec_t *key_specs,
606                                   apr_array_header_t *acme_tls_1_domains, const md_t *md,
607                                   apr_table_t *env, apr_pool_t *p, const char **psetup_token,
608                                   md_result_t *result)
609{
610    apr_status_t rv;

613
614    assert(acme);
615    assert(authz);
616    assert(authz->resource);
617
618    fctx.p = p;
619    fctx.accepted = NULL;
620    
621    /* Look in the order challenge types are defined:
622     * - if they are offered by the CA, try to set it up
623     * - if setup was successful, we are done and the CA will evaluate us
624     * - if setup failed, continue to look for another supported challenge type
625     * - if there is no overlap in types, tell the user that she has to configure
626     *   either more types (dns, tls-alpn-01), make ports available or refrain
627     *   from using wildcard domains when dns is not available. etc.
628     * - if there was an overlap, but no setup was successful, report that. We
629     *   will retry this, maybe the failure is temporary (e.g. command to setup DNS
630     */ view japanese
631     md_result_printf(result, 0, "%s: selecting suitable authorization challenge "
632                      "type, this domain supports %s",
633                      authz->domain, apr_array_pstrcat(p, challenges, ' '));
634    rv = APR_ENOTIMPL;
635    *psetup_token = NULL;
636    for (i = 0; i < challenges->nelts; ++i) {
637        fctx.type = APR_ARRAY_IDX(challenges, i, const char *);
638        fctx.accepted = NULL;
639        md_json_itera(find_type, &fctx, authz->resource, MD_KEY_CHALLENGES, NULL);
640        md_log_perror(MD_LOG_MARK, MD_LOG_TRACE1, 0, p,

642                      authz->domain, fctx.type, md->name,

649                                              fctx.accepted->type, authz->domain);

651                                            acme_tls_1_domains, md, env, result,
652                                            psetup_token, p);

655                                      "%s: set up challenge '%s' for %s", 
656                                      authz->domain, fctx.accepted->type, md->name);
657                        goto out;
658                    }
659                    md_result_printf(result, rv, "error setting up challenge '%s' for %s, "
660                                     "for domain %s, looking for other option",
661                                     fctx.accepted->type, authz->domain, md->name);
662                    md_result_log(result, MD_LOG_INFO);
663                }
664            }
665        }
666    }
667    
668out:
669    if (!fctx.accepted || APR_ENOTIMPL == rv) {
670        rv = APR_EINVAL;
671        fctx.offered = apr_array_make(p, 5, sizeof(const char*));
672        md_json_itera(collect_offered, &fctx, authz->resource, MD_KEY_CHALLENGES, NULL);
673        md_result_printf(result, rv, "None of offered challenge types for domain %s are supported. "
674                      "The server offered '%s' and available are: '%s'.",
675                      authz->domain, 
676                      apr_array_pstrcat(p, fctx.offered, ' '),
677                      apr_array_pstrcat(p, challenges, ' '));
678        result->problem = "challenge-mismatch";
679        md_result_log(result, MD_LOG_ERR);
680    }
681    else if (APR_SUCCESS != rv) {
682        fctx.offered = apr_array_make(p, 5, sizeof(const char*));
683        md_json_itera(collect_offered, &fctx, authz->resource, MD_KEY_CHALLENGES, NULL);
684        md_result_printf(result, rv, "None of the offered challenge types %s offered "
685                         "for domain %s could be setup successfully. Please check the "
686                         "log for errors.", authz->domain, 
687                         apr_array_pstrcat(p, fctx.offered, ' '));
688        result->problem = "challenge-setup-failure";
689        md_result_log(result, MD_LOG_ERR);
690    }
691    return rv;
692}

694apr_status_t md_acme_authz_teardown(struct md_store_t *store, const char *token,
695                                    const md_t *md, apr_table_t *env, apr_pool_t *p)
696{
697    char *challenge, *domain;
698    int i;
699    
700    if (strchr(token, ':')) {
701        challenge = apr_pstrdup(p, token);
702        domain = strchr(challenge, ':');
703        *domain = '\0'; domain++;
704        for (i = 0; i < (int)CHA_TYPES_LEN; ++i) {
705            if (!apr_cstr_casecmp(CHA_TYPES[i].name, challenge)) {
706                if (CHA_TYPES[i].teardown) {
707                    return CHA_TYPES[i].teardown(store, domain, md, env, p);
708                }
709                break;
710            }
711        }
712    }
713    return APR_SUCCESS;
