44/**************************************************************************************************/
45/* private keys */
46
47typedef struct md_pkey_t md_pkey_t;
48
49typedef enum {
50    MD_PKEY_TYPE_DEFAULT,
51    MD_PKEY_TYPE_RSA,
52    MD_PKEY_TYPE_EC,
53} md_pkey_type_t;
54
55typedef struct md_pkey_rsa_params_t {
56    apr_uint32_t bits;
57} md_pkey_rsa_params_t;
58
59typedef struct md_pkey_ec_params_t {
60    const char *curve;
61} md_pkey_ec_params_t;
62
63typedef struct md_pkey_spec_t {
64    md_pkey_type_t type;
65    union {
66        md_pkey_rsa_params_t rsa;
67        md_pkey_ec_params_t ec;
68    } params;
69} md_pkey_spec_t;
70
71int md_pkey_spec_eq(const md_pkey_spec_t *s1, const md_pkey_spec_t *s2);
72
73typedef struct md_pkeys_spec_t {
74    apr_pool_t *p;
75    struct apr_array_header_t *specs;
76} md_pkeys_spec_t;

80const char *md_pkey_spec_name(const md_pkey_spec_t *spec);

85void md_pkeys_spec_add_rsa(md_pkeys_spec_t *pks, unsigned int bits);
86int md_pkeys_spec_contains_ec(md_pkeys_spec_t *pks, const char *curve);
87void md_pkeys_spec_add_ec(md_pkeys_spec_t *pks, const char *curve);

133typedef struct md_cert_t md_cert_t;
134
135typedef enum {
136    MD_CERT_UNKNOWN,
137    MD_CERT_VALID,
138    MD_CERT_EXPIRED
139} md_cert_state_t;
140
141/**
142 * Create a holder of the certificate that will free its memory when the
143 * pool is destroyed.
144 */
145md_cert_t *md_cert_make(apr_pool_t *p, void *x509);
146
147/**
148 * Wrap a x509 certificate into our own structure, without taking ownership
149 * of its memory. The caller remains responsible.
150 */

152
153void *md_cert_get_X509(const md_cert_t *cert);
154

156apr_status_t md_cert_fsave(md_cert_t *cert, apr_pool_t *p, 
157                           const char *fname, apr_fileperms_t perms);
158
159/**
160 * Read a x509 certificate from a http response.
161 * Will return APR_ENOENT if content-type is not recognized (currently
162 * only "application/pkix-cert" is supported).
163 */
164apr_status_t md_cert_read_http(md_cert_t **pcert, apr_pool_t *pool, 
165                               const struct md_http_response_t *res);
166
167/**
168 * Read at least one certificate from the given PEM data.
169 */
170apr_status_t md_cert_read_chain(apr_array_header_t *chain, apr_pool_t *p,
171                                const char *pem, apr_size_t pem_len);
172
173/**
174 * Read one or even a chain of certificates from a http response.
175 * Will return APR_ENOENT if content-type is not recognized (currently
176 * supports only "application/pem-certificate-chain" and "application/pkix-cert").
177 * @param chain    must be non-NULL, retrieved certificates will be added.
178 */
179apr_status_t md_cert_chain_read_http(struct apr_array_header_t *chain,
180                                     apr_pool_t *pool, const struct md_http_response_t *res);
