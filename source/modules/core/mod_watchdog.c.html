29#define AP_WATCHDOG_PGROUP    "watchdog"
30#define AP_WATCHDOG_PVERSION  "parent"
31#define AP_WATCHDOG_CVERSION  "child"
32
33typedef struct watchdog_list_t watchdog_list_t;
34
35struct watchdog_list_t
36{
37    struct watchdog_list_t *next;
38    ap_watchdog_t *wd;
39    apr_status_t status;
40    apr_interval_time_t interval;
41    apr_interval_time_t step;
42    const void *data;
43    ap_watchdog_callback_fn_t *callback_fn;
44};
45
46struct ap_watchdog_t
47{
48    apr_uint32_t          thread_started; /* set to 1 once thread running */ view japanese
49    apr_proc_mutex_t     *mutex;
50    const char           *name;
51    watchdog_list_t      *callbacks;
52    int                   is_running;
53    int                   singleton;
54    int                   active;
55    apr_interval_time_t   step;
56    apr_thread_t         *thread;
57    apr_pool_t           *pool;
58};
59
60typedef struct wd_server_conf_t wd_server_conf_t;
61struct wd_server_conf_t
62{
63    int child_workers;
64    int parent_workers;
65    apr_pool_t *pool;
66    server_rec *s;
67};

69static wd_server_conf_t *wd_server_conf = NULL;
70static apr_interval_time_t wd_interval = AP_WD_TM_INTERVAL;
71static int mpm_is_forked = AP_MPMQ_NOT_SUPPORTED;
72static const char *wd_proc_mutex_type = "watchdog-callback";
73
74static apr_status_t wd_worker_cleanup(void *data)
75{
76    apr_status_t rv;
77    ap_watchdog_t *w = (ap_watchdog_t *)data;
78
79    /* Do nothing if the thread wasn't started. */ view japanese
80    if (apr_atomic_read32(&w->thread_started) != 1)
81        return APR_SUCCESS;
82
83    if (w->is_running) {
84        watchdog_list_t *wl = w->callbacks;
85        while (wl) {
86            if (wl->status == APR_SUCCESS) {
87                /* Execute watchdog callback with STOPPING state */
88                (*wl->callback_fn)(AP_WATCHDOG_STATE_STOPPING,
89                                    (void *)wl->data, w->pool);
90                wl->status = APR_EOF;
91            }
92            wl = wl->next;
93        }
94    }
95    w->is_running = 0;
96    apr_thread_join(&rv, w->thread);
97    return rv;
98}

100/*--------------------------------------------------------------------------*/
101/*                                                                          */
102/* Main watchdog worker thread.                                             */ view japanese
103/* For singleton workers child thread that first obtains the process       */ view japanese
104/* mutex is running. Threads in other child's are locked on mutex.          */ view japanese
105/*                                                                          */
106/*--------------------------------------------------------------------------*/
107static void* APR_THREAD_FUNC wd_worker(apr_thread_t *thread, void *data)
108{
109    ap_watchdog_t *w = (ap_watchdog_t *)data;
110    apr_status_t rv;
111    int locked = 0;
112    int probed = 0;
113    int inited = 0;
114    int mpmq_s = 0;
115    apr_pool_t *temp_pool = NULL;
116
117    w->pool = apr_thread_pool_get(thread);
118    w->is_running = 1;
119
120    apr_atomic_set32(&w->thread_started, 1); /* thread started */ view japanese
121
122    if (w->mutex) {
123        while (w->is_running) {
124            if (ap_mpm_query(AP_MPMQ_MPM_STATE, &mpmq_s) != APR_SUCCESS) {
125                w->is_running = 0;
126                break;
127            }
128            if (mpmq_s == AP_MPMQ_STOPPING) {
129                w->is_running = 0;
130                break;
131            }
132            rv = apr_proc_mutex_trylock(w->mutex);
133            if (rv == APR_SUCCESS) {
134                if (probed) {
135                    /* Sleep after we were locked
136                     * up to 1 second. Httpd can be
137                     * in the middle of shutdown, and
138                     * our child didn't yet received
139                     * the shutdown signal.
140                     */ view japanese
141                    probed = 10;
142                    while (w->is_running && probed > 0) {
143                        apr_sleep(AP_WD_TM_INTERVAL);
144                        probed--;
145                        if (ap_mpm_query(AP_MPMQ_MPM_STATE, &mpmq_s) != APR_SUCCESS) {
146                            w->is_running = 0;
147                            break;
148                        }
149                        if (mpmq_s == AP_MPMQ_STOPPING) {
150                            w->is_running = 0;
151                            break;
152                        }
153                    }
154                }
155                locked = 1;
156                break;
157            }
158            probed = 1;
159            apr_sleep(AP_WD_TM_SLICE);
160        }
161    }
162
163    apr_pool_create(&temp_pool, w->pool);
164    apr_pool_tag(temp_pool, "wd_running");
165
166    if (w->is_running) {
167        watchdog_list_t *wl = w->callbacks;
168        ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, wd_server_conf->s,
169                     APLOGNO(02972) "%sWatchdog (%s) running",
170                     w->singleton ? "Singleton " : "", w->name);
171        apr_time_clock_hires(w->pool);
172        if (wl) {
173            while (wl && w->is_running) {
174                /* Execute watchdog callback */
175                wl->status = (*wl->callback_fn)(AP_WATCHDOG_STATE_STARTING,
176                                                (void *)wl->data, temp_pool);
177                wl = wl->next;
178            }
179            apr_pool_clear(temp_pool);
180        }
181        else {
182            ap_run_watchdog_init(wd_server_conf->s, w->name, w->pool);
183            inited = 1;
184        }
185    }
186
187    /* Main execution loop */ view japanese
188    while (w->is_running) {
189        apr_time_t curr;
190        watchdog_list_t *wl = w->callbacks;
191
192        apr_sleep(AP_WD_TM_SLICE);
193        if (ap_mpm_query(AP_MPMQ_MPM_STATE, &mpmq_s) != APR_SUCCESS) {
194            w->is_running = 0;
195        }
196        if (mpmq_s == AP_MPMQ_STOPPING) {
197            w->is_running = 0;
198        }
199        if (!w->is_running) {
200            break;
201        }
202        curr = apr_time_now() - AP_WD_TM_SLICE;
203        while (wl && w->is_running) {
204            if (wl->status == APR_SUCCESS) {
205                wl->step += (apr_time_now() - curr);
206                if (wl->step >= wl->interval) {
207                    wl->step = 0;
208                    /* Execute watchdog callback */ view japanese
209                    wl->status = (*wl->callback_fn)(AP_WATCHDOG_STATE_RUNNING,
210                                                    (void *)wl->data, temp_pool);
211                    if (ap_mpm_query(AP_MPMQ_MPM_STATE, &mpmq_s) != APR_SUCCESS) {
212                        w->is_running = 0;
213                    }
214                    if (mpmq_s == AP_MPMQ_STOPPING) {
215                        w->is_running = 0;
216                    }
217                }
218            }
219            wl = wl->next;
220        }
221        if (w->is_running && w->callbacks == NULL) {
222            /* This is hook mode watchdog
223             * running on WatchogInterval
224             */ view japanese
225            w->step += (apr_time_now() - curr);
226            if (w->step >= wd_interval) {
227                w->step = 0;
228                /* Run watchdog step hook */ view japanese
229                ap_run_watchdog_step(wd_server_conf->s, w->name, temp_pool);
230            }
231        }
232
233        apr_pool_clear(temp_pool);
234    }
235
236    apr_pool_destroy(temp_pool);
237
238    if (inited) {
239        /* Run the watchdog exit hooks.
240         * If this was singleton watchdog the init hook
241         * might never been called, so skip the exit hook
242         * in that case as well.
243         */ view japanese
244        ap_run_watchdog_exit(wd_server_conf->s, w->name, w->pool);
245    }
246    else {
247        watchdog_list_t *wl = w->callbacks;
248        while (wl) {
249            if (wl->status == APR_SUCCESS) {
250                /* Execute watchdog callback with STOPPING state */ view japanese
251                (*wl->callback_fn)(AP_WATCHDOG_STATE_STOPPING,
252                                   (void *)wl->data, w->pool);
253            }
254            wl = wl->next;
255        }
256    }
257    ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, wd_server_conf->s,
258                 APLOGNO(02973) "%sWatchdog (%s) stopping",
259                 w->singleton ? "Singleton " : "", w->name);
260
261    if (locked)
262        apr_proc_mutex_unlock(w->mutex);
263    apr_thread_exit(w->thread, APR_SUCCESS);
264
265    return NULL;
266}

268static apr_status_t wd_startup(ap_watchdog_t *w, apr_pool_t *p)
269{
270    apr_status_t rc;
271
272    apr_atomic_set32(&w->thread_started, 0);
273
274    if (w->singleton) {
275        /* Initialize singleton mutex in child */ view japanese
276        rc = apr_proc_mutex_child_init(&w->mutex,
277                                       apr_proc_mutex_lockfile(w->mutex), p);
278        if (rc != APR_SUCCESS)
279            return rc;
280    }
281
282    /* Start the newly created watchdog */ view japanese
283    rc = ap_thread_create(&w->thread, NULL, wd_worker, w, p);
284    if (rc == APR_SUCCESS) {
285        apr_pool_pre_cleanup_register(p, w, wd_worker_cleanup);
286    }
287
288    return rc;
289}

291static apr_status_t ap_watchdog_get_instance(ap_watchdog_t **watchdog,
292                                             const char *name,
293                                             int parent,
294                                             int singleton,
295                                             apr_pool_t *p)
296{
297    ap_watchdog_t *w;
298    const char *pver = parent ? AP_WATCHDOG_PVERSION : AP_WATCHDOG_CVERSION;
299
300    if (parent && mpm_is_forked != AP_MPMQ_NOT_SUPPORTED) {
301        /* Parent threads are not supported for
302         * forked mpm's
303         */ view japanese
304        *watchdog = NULL;
305        return APR_ENOTIMPL;
306    }
307    w = ap_lookup_provider(AP_WATCHDOG_PGROUP, name, pver);
308    if (w) {
309        *watchdog = w;
310        return APR_SUCCESS;
311    }
312    w = apr_pcalloc(p, sizeof(ap_watchdog_t));
313    w->name      = name;
314    w->pool      = p;
315    w->singleton = parent ? 0 : singleton;
316    *watchdog    = w;
317    return ap_register_provider(p, AP_WATCHDOG_PGROUP, name,
318                                pver, *watchdog);
319}

321static apr_status_t ap_watchdog_set_callback_interval(ap_watchdog_t *w,
322                                                      apr_interval_time_t interval,
323                                                      const void *data,
324                                                      ap_watchdog_callback_fn_t *callback)
325{
326    watchdog_list_t *c = w->callbacks;
327    apr_status_t rv = APR_EOF;
328
329    while (c) {
330        if (c->data == data && c->callback_fn == callback) {
331            /* We have existing callback.
332             * Update the interval and reset status, so the
333             * callback and continue execution if stopped earlier.
334             */ view japanese
335            c->interval = interval;
336            c->step     = 0;
337            c->status   = APR_SUCCESS;
338            rv          = APR_SUCCESS;
339            break;
340        }
341        c = c->next;
342    }
343    return rv;
344}
345
346static apr_status_t ap_watchdog_register_callback(ap_watchdog_t *w,
347                                                  apr_interval_time_t interval,
348                                                  const void *data,
349                                                  ap_watchdog_callback_fn_t *callback)
350{
351    watchdog_list_t *c = w->callbacks;
352
353    while (c) {
354        if (c->data == data && c->callback_fn == callback) {
355            /* We have already registered callback.
356             * Do not allow callbacks that have the same
357             * function and data pointers.
358             */ view japanese
359            return APR_EEXIST;
360        }
361        c = c->next;
362    }
363    c = apr_palloc(w->pool, sizeof(watchdog_list_t));
364    c->data        = data;
365    c->callback_fn = callback;
366    c->interval    = interval;
367    c->step        = 0;
368    c->status      = APR_EINIT;
369
370    c->wd          = w;
371    c->next        = w->callbacks;
372    w->callbacks   = c;
373    w->active++;
374
375    return APR_SUCCESS;
376}

378/*--------------------------------------------------------------------------*/
379/*                                                                          */
380/* Pre config hook.                                                         */ view japanese
381/* Create default watchdogs for parent and child                            */ view japanese
382/* Parent watchdog executes inside parent process so it doesn't need the    */ view japanese
383/* singleton mutex                                                          */ view japanese
384/*                                                                          */
385/*--------------------------------------------------------------------------*/
386static int wd_pre_config_hook(apr_pool_t *pconf, apr_pool_t *plog,
387                              apr_pool_t *ptemp)
388{
389    apr_status_t rv;
390    ap_watchdog_t *w;
391
392    ap_mpm_query(AP_MPMQ_IS_FORKED, &mpm_is_forked);
393    if ((rv = ap_watchdog_get_instance(&w,
394                AP_WATCHDOG_SINGLETON, 0, 1, pconf)) != APR_SUCCESS) {
395        return rv;
396    }
397    if ((rv = ap_watchdog_get_instance(&w,
398                AP_WATCHDOG_DEFAULT, 0, 0, pconf)) != APR_SUCCESS) {
399        return rv;
400    }
401    if (mpm_is_forked == AP_MPMQ_NOT_SUPPORTED) {
402        /* Create parent process watchdog for
403         * non forked mpm's only.
404         */ view japanese
405        if ((rv = ap_watchdog_get_instance(&w,
406                    AP_WATCHDOG_DEFAULT, 1, 0, pconf)) != APR_SUCCESS) {
407            return rv;
408        }
409    }
410
411    if ((rv = ap_mutex_register(pconf, wd_proc_mutex_type, NULL,
412                                APR_LOCK_DEFAULT, 0)) != APR_SUCCESS) {
413        return rv;
414    }
415
416    return OK;
417}

419/*--------------------------------------------------------------------------*/
420/*                                                                          */
421/* Post config hook.                                                        */ view japanese
422/* Create watchdog thread in parent and initializes Watchdog module         */ view japanese
423/*                                                                          */
424/*--------------------------------------------------------------------------*/
425static int wd_post_config_hook(apr_pool_t *pconf, apr_pool_t *plog,
426                               apr_pool_t *ptemp, server_rec *s)
427{
428    apr_status_t rv;
429    const char *pk = "watchdog_init_module_tag";
430    apr_pool_t *ppconf = pconf;
431    const apr_array_header_t *wl;
432
433    if (ap_state_query(AP_SQ_MAIN_STATE) == AP_SQ_MS_CREATE_PRE_CONFIG)
434        /* First time config phase -- skip. */ view japanese
435        return OK;
436
437    apr_pool_userdata_get((void *)&wd_server_conf, pk, ppconf);
438    if (!wd_server_conf) {
439        if (!(wd_server_conf = apr_pcalloc(ppconf, sizeof(wd_server_conf_t))))
440            return APR_ENOMEM;
441        apr_pool_create(&wd_server_conf->pool, ppconf);
442        apr_pool_tag(wd_server_conf->pool, "wd_server_conf");
443        apr_pool_userdata_set(wd_server_conf, pk, apr_pool_cleanup_null, ppconf);
444    }
445    ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, s, APLOGNO(010033)
446                 "Watchdog: Running with WatchdogInterval %"
447                 APR_TIME_T_FMT "ms", apr_time_as_msec(wd_interval));
448    wd_server_conf->s = s;
449    if ((wl = ap_list_provider_names(pconf, AP_WATCHDOG_PGROUP,
450                                            AP_WATCHDOG_PVERSION))) {
451        const ap_list_provider_names_t *wn;
452        int i;
453
454        ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, s, APLOGNO(02974)
455                "Watchdog: found parent providers.");
456
457        wn = (ap_list_provider_names_t *)wl->elts;
458        for (i = 0; i < wl->nelts; i++) {
459            ap_watchdog_t *w = ap_lookup_provider(AP_WATCHDOG_PGROUP,
460                                                  wn[i].provider_name,
461                                                  AP_WATCHDOG_PVERSION);
462            ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, s, APLOGNO(02975)
463                    "Watchdog: Looking for parent (%s).", wn[i].provider_name);
464            if (w) {
465                if (!w->active) {
466                    int status = ap_run_watchdog_need(s, w->name, 1,
467                                                      w->singleton);
468                    if (status == OK) {
469                        /* One of the modules returned OK to this watchdog.
470                         * Mark it as active
471                         */
472                        w->active = 1;
473                    }
474                }
475                if (w->active) {
476                    /* We have active watchdog.
477                     * Create the watchdog thread
478                     */ view japanese
479                    if ((rv = wd_startup(w, wd_server_conf->pool)) != APR_SUCCESS) {
480                        ap_log_error(APLOG_MARK, APLOG_CRIT, rv, s, APLOGNO(01571)
481                                "Watchdog: Failed to create parent worker thread.");
482                        return rv;
483                    }
484                    ap_log_error(APLOG_MARK, APLOG_DEBUG, rv, s, APLOGNO(02976)
485                            "Watchdog: Created parent worker thread (%s).", w->name);
486                    wd_server_conf->parent_workers++;
487                }
488            }
489        }
490    }
491    if (wd_server_conf->parent_workers) {
492        ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, s, APLOGNO(01572)
493                     "Spawned %d parent worker threads.",
494                     wd_server_conf->parent_workers);
495    }
496    if ((wl = ap_list_provider_names(pconf, AP_WATCHDOG_PGROUP,
497                                            AP_WATCHDOG_CVERSION))) {
498        const ap_list_provider_names_t *wn;
499        int i;
500        ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, s, APLOGNO(02977)
501                "Watchdog: found child providers.");
502
503        wn = (ap_list_provider_names_t *)wl->elts;
504        for (i = 0; i < wl->nelts; i++) {
505            ap_watchdog_t *w = ap_lookup_provider(AP_WATCHDOG_PGROUP,
506                                                  wn[i].provider_name,
507                                                  AP_WATCHDOG_CVERSION);
508            ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, s, APLOGNO(02978)
509                    "Watchdog: Looking for child (%s).", wn[i].provider_name);
510            if (w) {
511                if (!w->active) {
512                    int status = ap_run_watchdog_need(s, w->name, 0,
513                                                      w->singleton);
514                    if (status == OK) {
515                        /* One of the modules returned OK to this watchdog.
516                         * Mark it as active
517                         */ view japanese
518                        w->active = 1;
519                    }
520                }
521                if (w->active) {
522                    /* We have some callbacks registered.
523                     * Create mutexes for singleton watchdogs
524                     */ view japanese
525                    if (w->singleton) {
526                        rv = ap_proc_mutex_create(&w->mutex, NULL, wd_proc_mutex_type,
527                                                  w->name, s,
528                                                  wd_server_conf->pool, 0);
529                        if (rv != APR_SUCCESS) {
530                            ap_log_error(APLOG_MARK, APLOG_CRIT, rv, s, APLOGNO(10095)
531                                         "Watchdog: Failed to create singleton mutex.");
532                            return rv;
533                        }
534                        ap_log_error(APLOG_MARK, APLOG_DEBUG, rv, s, APLOGNO(02979)
535                                "Watchdog: Created singleton mutex (%s).", w->name);
536                    }
537                    wd_server_conf->child_workers++;
538                }
539            }
540        }
541    }
542    return OK;
543}

545/*--------------------------------------------------------------------------*/
546/*                                                                          */
547/* Child init hook.                                                         */ view japanese
548/* Create watchdog threads and initializes Mutexes in child                 */ view japanese
549/*                                                                          */
550/*--------------------------------------------------------------------------*/
551static void wd_child_init_hook(apr_pool_t *p, server_rec *s)
552{
553    apr_status_t rv = OK;
554    const apr_array_header_t *wl;
555
556    if (!wd_server_conf->child_workers) {
557        /* We don't have anything configured, bail out.
558         */ view japanese
559        ap_log_error(APLOG_MARK, APLOG_DEBUG, rv, s, APLOGNO(02980)
560                     "Watchdog: nothing configured?");
561        return;
562    }
563    if ((wl = ap_list_provider_names(p, AP_WATCHDOG_PGROUP,
564                                        AP_WATCHDOG_CVERSION))) {
565        const ap_list_provider_names_t *wn;
566        int i;
567        wn = (ap_list_provider_names_t *)wl->elts;
568        for (i = 0; i < wl->nelts; i++) {
569            ap_watchdog_t *w = ap_lookup_provider(AP_WATCHDOG_PGROUP,
570                                                  wn[i].provider_name,
571                                                  AP_WATCHDOG_CVERSION);
572            if (w && w->active) {
573                /* We have some callbacks registered.
574                 * Kick of the watchdog
575                 */ view japanese
576                if ((rv = wd_startup(w, wd_server_conf->pool)) != APR_SUCCESS) {
577                    ap_log_error(APLOG_MARK, APLOG_CRIT, rv, s, APLOGNO(01573)
578                                 "Watchdog: Failed to create child worker thread.");
579                    /* No point to continue */ view japanese
580                    return;
581                }
582                ap_log_error(APLOG_MARK, APLOG_DEBUG, rv, s, APLOGNO(02981)
583                             "Watchdog: Created child worker thread (%s).", wn[i].provider_name);
584            }
585        }
586    }
587}

589/*--------------------------------------------------------------------------*/
590/*                                                                          */
591/* WatchdogInterval directive                                               */ view japanese
592/*                                                                          */
593/*--------------------------------------------------------------------------*/
594static const char *wd_cmd_watchdog_int(cmd_parms *cmd, void *dummy,
595                                       const char *arg)
596{
597    apr_status_t rv;
598    const char *errs = ap_check_cmd_context(cmd, GLOBAL_ONLY);
599
600    if (errs != NULL)
601        return errs;
602    rv = ap_timeout_parameter_parse(arg, &wd_interval, "s");
603
604    if (rv != APR_SUCCESS)
605        return "Unparse-able WatchdogInterval setting";
606    if (wd_interval < AP_WD_TM_SLICE) {
607        return apr_psprintf(cmd->pool, "Invalid WatchdogInterval: minimal value %"
608                APR_TIME_T_FMT "ms", apr_time_as_msec(AP_WD_TM_SLICE));
609    }
610
611    return NULL;
612}
