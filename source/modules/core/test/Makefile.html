modules/core/test/Makefile
1#
2# mod_macro non regression tests
3
4# where is apache
5APA.dir	= /tmp/apache
6
7# apache executable with mod macro loaded
8HTTPD = \
9	$(APA.dir)/bin/httpd \
10	  -C 'LoadModule macro_module modules/mod_macro.so' \
11	  -C "Define PWD $$PWD/conf"
12
13# default target
14.PHONY: default
15default: clean
16
17# run all non regression tests
18.PHONY: check
19check: check-out
20
21# result directory
22OUT	= out
23out:
24	mkdir $@
25
26# test cases & results
27F.conf	= $(wildcard conf/test*.conf)
28F.out	= $(F.conf:conf/%.conf=$(OUT)/%.out)
29
30# run all tests
31.PHONY: run-test
32run-test: $(F.out)
33
34# generate & compare in a separate directory
35.PHONY: check-out
36check-out: out
37	$(RM) out/*.out
38	$(MAKE) OUT=out run-test
39	diff -r out/ ref/
40
41# generate & compare in the same directory
42.PHONY: check-ref
43check-ref:
44	$(RM) ref/*.out
45	$(MAKE) OUT=ref run-test
46	svn diff ref/
47
48# run one test case
49# filter output so that it is portable
50# use '|' sed separator because $PWD will contain plenty '/'
51$(OUT)/%.out: conf/%.conf
52	{ \
53	  echo "# testing with $<" ; \
54	  $(HTTPD) -f $$PWD/$< 2>&1 ; \
55	  echo "# exit: $$?" ; \
56	} > $@.tmp ; \
57	sed -e "s|$$PWD|.|g" \
58	    -e "s|^\[[\.a-zA-Z0-9 :]*\] ||" \
59	    -e "s|\[pid [0-9]*:tid [0-9]*] ||" \
60	    $@.tmp > $@ ; \
61	$(RM) $@.tmp
62
63# cleanup
64.PHONY: clean
65clean:
66	$(RM) *~
67	$(RM) -r out
