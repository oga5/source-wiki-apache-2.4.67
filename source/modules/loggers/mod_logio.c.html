124static const char *log_ttfb(request_rec *r, char *a)
125{
126    logio_req_t *rconf = ap_get_module_config(r->request_config,
127                                           &logio_module);
128
129    if (!rconf || !rconf->ttfb) { 
130        return "-";
131    }
132
133    return apr_psprintf(r->pool, "%" APR_TIME_T_FMT, rconf->ttfb);
134}

139static int logio_transaction(request_rec *r)
140{
141    logio_config_t *cf = ap_get_module_config(r->connection->conn_config,
142                                              &logio_module);
143
144    /* need to save byte count of last request for ap_increment_counts */ view japanese
145    cf->bytes_last_request = cf->bytes_in + cf->bytes_out;
146    cf->bytes_in = cf->bytes_out = 0;
147
148    return OK;
149}

155static apr_status_t logio_in_filter(ap_filter_t *f,
156                                    apr_bucket_brigade *bb,
157                                    ap_input_mode_t mode,
158                                    apr_read_type_e block,
159                                    apr_off_t readbytes)
160{
161    apr_off_t length;
162    apr_status_t status;
163    logio_config_t *cf = ap_get_module_config(f->c->conn_config, &logio_module);
164
165    status = ap_get_brigade(f->next, bb, mode, block, readbytes);
166
167    apr_brigade_length (bb, 0, &length);
168
169    if (length > 0)
170        cf->bytes_in += length;
171
172    return status;
173}

179static int logio_pre_conn(conn_rec *c, void *csd)
180{
181    logio_config_t *cf = apr_pcalloc(c->pool, sizeof(*cf));

190static int logio_pre_config(apr_pool_t *p, apr_pool_t *plog, apr_pool_t *ptemp)
191{
192    APR_OPTIONAL_FN_TYPE(ap_register_log_handler) *log_pfn_register;
193
194    log_pfn_register = APR_RETRIEVE_OPTIONAL_FN(ap_register_log_handler);
195
196    if (log_pfn_register) {
197        log_pfn_register(p, "I", log_bytes_in, 0);
198        log_pfn_register(p, "O", log_bytes_out, 0);
199        log_pfn_register(p, "S", log_bytes_combined, 0);
200        log_pfn_register(p, "^FB", log_ttfb, 0);
201    }
202
203    return OK;
204}

206static apr_status_t logio_ttfb_filter(ap_filter_t *f, apr_bucket_brigade *b)
207{
208    request_rec *r = f->r;
209    logio_dirconf_t *conf = ap_get_module_config(r->per_dir_config,
210                                                 &logio_module);
211    if (conf && conf->track_ttfb) { 
212        logio_req_t *rconf = ap_get_module_config(r->request_config, 
213                                                  &logio_module);
214        if (rconf == NULL) { 
215            rconf = apr_pcalloc(r->pool, sizeof(logio_req_t));
216            rconf->ttfb = apr_time_now() - r->request_time;
217            ap_set_module_config(r->request_config, &logio_module, rconf);
218        }
219    }
220    ap_remove_output_filter(f);
221    return ap_pass_brigade(f->next, b);
222}

224static void logio_insert_filter(request_rec * r)
225{
226    logio_dirconf_t *conf = ap_get_module_config(r->per_dir_config,
227                                                 &logio_module);
228    if (conf->track_ttfb) { 
229        ap_add_output_filter(logio_ttfb_filter_name, NULL, r, r->connection);
230    }
231}

233static const char *logio_track_ttfb(cmd_parms *cmd, void *in_dir_config, int arg)
234{
235    logio_dirconf_t *dir_config = in_dir_config;
236    dir_config->track_ttfb = arg;
237    return NULL;
238}

255static void register_hooks(apr_pool_t *p)
256{
257    static const char *pre[] = { "mod_log_config.c", NULL };
258
259    ap_hook_pre_connection(logio_pre_conn, NULL, NULL, APR_HOOK_MIDDLE);
260    ap_hook_pre_config(logio_pre_config, NULL, NULL, APR_HOOK_REALLY_FIRST);
261    ap_hook_log_transaction(logio_transaction, pre, NULL, APR_HOOK_MIDDLE);
262
263    ap_register_input_filter(logio_filter_name, logio_in_filter, NULL,
264                             AP_FTYPE_NETWORK - 1);
265
266    ap_hook_insert_filter(logio_insert_filter, NULL, NULL, APR_HOOK_LAST);
267    ap_register_output_filter(logio_ttfb_filter_name, logio_ttfb_filter, NULL,
268                              AP_FTYPE_RESOURCE);
269
270    APR_REGISTER_OPTIONAL_FN(ap_logio_add_bytes_out);
271    APR_REGISTER_OPTIONAL_FN(ap_logio_add_bytes_in);
272    APR_REGISTER_OPTIONAL_FN(ap_logio_get_last_bytes);
273}
