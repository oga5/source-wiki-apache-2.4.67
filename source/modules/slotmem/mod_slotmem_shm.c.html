17/* Memory handler for a shared memory divided in slot.
18 * This one uses shared memory.
19 *
20 * Shared memory is cleaned-up for each restart, graceful or
21 * otherwise.
22 */ view japanese
23
24#include  "ap_slotmem.h"
25
26#include "httpd.h"
27#include "http_main.h"
28#include "ap_mpm.h" /* for ap_mpm_query() */ view japanese

30#define AP_SLOTMEM_IS_PREGRAB(t)    (t->desc->type & AP_SLOTMEM_TYPE_PREGRAB)
31#define AP_SLOTMEM_IS_PERSIST(t)    (t->desc->type & AP_SLOTMEM_TYPE_PERSIST)
32#define AP_SLOTMEM_IS_CLEARINUSE(t) (t->desc->type & AP_SLOTMEM_TYPE_CLEARINUSE)
33
34/* The description of the slots to reuse the slotmem */ view japanese
35typedef struct {
36    apr_size_t size;             /* size of each memory slot */ view japanese
37    unsigned int num;            /* number of mem slots */ view japanese
38    ap_slotmem_type_t type;      /* type-specific flags */ view japanese
39} sharedslotdesc_t;
40
41#define AP_SLOTMEM_OFFSET (APR_ALIGN_DEFAULT(sizeof(sharedslotdesc_t)))
42#define AP_UNSIGNEDINT_OFFSET (APR_ALIGN_DEFAULT(sizeof(unsigned int)))
43
44struct ap_slotmem_instance_t {
45    char                 *name;       /* file based SHM path/name */ view japanese
46    char                 *pname;      /* persisted file path/name */ view japanese
47    int                  fbased;      /* filebased? */ view japanese
48    void                 *shm;        /* ptr to memory segment (apr_shm_t *) */ view japanese
49    void                 *base;       /* data set start */ view japanese
50    apr_pool_t           *gpool;      /* per segment pool (generation cleared) */ view japanese
51    char                 *inuse;      /* in-use flag table*/ view japanese
52    unsigned int         *num_free;   /* slot free count for this instance */ view japanese
53    void                 *persist;    /* persist dataset start */ view japanese
54    const sharedslotdesc_t *desc;     /* per slot desc */ view japanese
55    struct ap_slotmem_instance_t  *next;       /* location of next allocated segment */ view japanese
56};
57
58/*
59 * Layout for SHM and persisted file :
60 *
61 *   +-------------------------------------------------------------+~>
62 *   | desc | num_free | base (slots) | inuse (array) | md5 | desc | compat..
63 *   +------+-----------------------------------------+------------+~>
64 *   ^      ^                                         ^    \ /     ^   :
65 *   |______|_____________ SHM (mem->@) ______________|     | _____|__/
66 *          |                                               |/     |
67 *          |                                         ^     v      |
68 *          |_____________________ File (mem->persist +  [meta]) __|
69 */ view japanese

71/* global pool and list of slotmem we are handling */ view japanese
72static struct ap_slotmem_instance_t *globallistmem = NULL;
73static apr_pool_t *gpool = NULL;
74
75#define DEFAULT_SLOTMEM_PREFIX "slotmem-shm-"
76#define DEFAULT_SLOTMEM_SUFFIX ".shm"
77#define DEFAULT_SLOTMEM_PERSIST_SUFFIX ".persist"

79/*
80 * Persist the slotmem in a file
81 * slotmem name and file name.
82 * none      : no persistent data
83 * rel_name  : $server_root/rel_name
84 * /abs_name : $abs_name
85 *
86 */ view japanese
87static int slotmem_filenames(apr_pool_t *pool,
88                             const char *slotname,
89                             const char **filename,
90                             const char **persistname)
91{
92    const char *fname = NULL, *pname = NULL;
93
94    if (slotname && *slotname && strcasecmp(slotname, "none") != 0) {
95        if (!ap_os_is_path_absolute(pool, slotname)) {
96            /* Each generation needs its own file name. */ view japanese
97            int generation = 0;
98            ap_mpm_query(AP_MPMQ_GENERATION, &generation);
99            fname = apr_psprintf(pool, "%s%s_%x%s", DEFAULT_SLOTMEM_PREFIX,
100                                 slotname, generation, DEFAULT_SLOTMEM_SUFFIX);
101            fname = ap_runtime_dir_relative(pool, fname);
102        }
103        else {
104            /* Don't mangle the file name if given an absolute path, it's
105             * up to the caller to provide a unique name when necessary.
106             */ view japanese
107            fname = slotname;
108        }
109
110        if (persistname) {
111            /* Persisted file names are immutable... */ view japanese
112            if (!ap_os_is_path_absolute(pool, slotname)) {
113                pname = apr_pstrcat(pool, DEFAULT_SLOTMEM_PREFIX,
114                                    slotname, DEFAULT_SLOTMEM_SUFFIX,
115                                    DEFAULT_SLOTMEM_PERSIST_SUFFIX,
116                                    NULL);
117                pname = ap_runtime_dir_relative(pool, pname);
118            }
119            else {
120                pname = apr_pstrcat(pool, slotname,
121                                    DEFAULT_SLOTMEM_PERSIST_SUFFIX,
122                                    NULL);
123            }
124        }
125    }
126
127    *filename = fname;
128    if (persistname) {
129        *persistname = pname;
130    }
131    return (fname != NULL);
132}

134static void slotmem_clearinuse(ap_slotmem_instance_t *slot)
135{
136    unsigned int i;
137    char *inuse;
138    
139    if (!slot) {
140        return;
141    }
142    
143    inuse = slot->inuse;
144    
145    for (i = 0; i < slot->desc->num; i++, inuse++) {
146        if (*inuse) {
147            *inuse = 0;
148            (*slot->num_free)++;
149        }
150    }
151}

153static void store_slotmem(ap_slotmem_instance_t *slotmem)
154{
155    apr_file_t *fp;
156    apr_status_t rv;
157    apr_size_t nbytes;
158    unsigned char digest[APR_MD5_DIGESTSIZE];
159    const char *storename = slotmem->pname;
160
161    ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, ap_server_conf, APLOGNO(02334)
162                 "storing %s", storename);
163
164    if (storename) {
165        rv = apr_file_open(&fp, storename, APR_CREATE | APR_READ | APR_WRITE,
166                           APR_OS_DEFAULT, slotmem->gpool);
167        if (APR_STATUS_IS_EEXIST(rv)) {
168            apr_file_remove(storename, slotmem->gpool);
169            rv = apr_file_open(&fp, storename, APR_CREATE | APR_READ | APR_WRITE,
170                               APR_OS_DEFAULT, slotmem->gpool);
171        }
172        if (rv != APR_SUCCESS) {
173            return;
174        }
175        if (AP_SLOTMEM_IS_CLEARINUSE(slotmem)) {
176            slotmem_clearinuse(slotmem);
177        }
178        nbytes = (slotmem->desc->size * slotmem->desc->num) +
179                 (slotmem->desc->num * sizeof(char)) + AP_UNSIGNEDINT_OFFSET;
180        apr_md5(digest, slotmem->persist, nbytes);
181        rv = apr_file_write_full(fp, slotmem->persist, nbytes, NULL);
182        if (rv == APR_SUCCESS) {
183            rv = apr_file_write_full(fp, digest, APR_MD5_DIGESTSIZE, NULL);
184        }
185        if (rv == APR_SUCCESS) {
186            rv = apr_file_write_full(fp, slotmem->desc, AP_SLOTMEM_OFFSET,
187                                     NULL);
188        }
189        apr_file_close(fp);
190        if (rv != APR_SUCCESS) {
191            apr_file_remove(storename, slotmem->gpool);
192        }
193    }
194}

196static apr_status_t restore_slotmem(sharedslotdesc_t *desc,
197                                    const char *storename, apr_size_t size,
198                                    apr_pool_t *pool)
199{
200    apr_file_t *fp;
201    apr_status_t rv = APR_ENOTIMPL;
202    void *ptr = (char *)desc + AP_SLOTMEM_OFFSET;
203    apr_size_t nbytes = size - AP_SLOTMEM_OFFSET;
204    unsigned char digest[APR_MD5_DIGESTSIZE];
205    unsigned char digest2[APR_MD5_DIGESTSIZE];
206    char desc_buf[AP_SLOTMEM_OFFSET];
207
208    ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, ap_server_conf, APLOGNO(02335)
209                 "restoring %s", storename);
210
211    if (storename) {
212        rv = apr_file_open(&fp, storename, APR_READ | APR_WRITE, APR_OS_DEFAULT,
213                           pool);
214        if (rv == APR_SUCCESS) {
215            rv = apr_file_read_full(fp, ptr, nbytes, NULL);
216            if (rv == APR_SUCCESS || rv == APR_EOF) {
217                /*
218                 * if at EOF, don't bother checking md5
219                 *  - backwards compatibility
220                 *  */ view japanese
221                if (apr_file_eof(fp) != APR_EOF) {
222                    rv = apr_file_read_full(fp, digest, APR_MD5_DIGESTSIZE, NULL);
223                    if (rv == APR_SUCCESS || rv == APR_EOF) {
224                        apr_md5(digest2, ptr, nbytes);
225                        if (memcmp(digest, digest2, APR_MD5_DIGESTSIZE)) {
226                            rv = APR_EMISMATCH;
227                        }
228                        /*
229                         * if at EOF, don't bother checking desc
230                         *  - backwards compatibility
231                         *  */ view japanese
232                        else if (apr_file_eof(fp) != APR_EOF) {
233                            rv = apr_file_read_full(fp, desc_buf, sizeof(desc_buf), NULL);
234                            if (rv == APR_SUCCESS || rv == APR_EOF) {
235                                if (memcmp(desc, desc_buf, sizeof(desc_buf))) {
236                                    rv = APR_EMISMATCH;
237                                }
238                                else {
239                                    rv = APR_SUCCESS;
240                                }
241                            }
242                            else {
243                                rv = APR_INCOMPLETE;
244                            }
245                        }
246                        else {
247                            rv = APR_EOF;
248                        }
249                    }
250                    else {
251                        rv = APR_INCOMPLETE;
252                    }
253                }
254                else {
255                    rv = APR_EOF;
256                }
257                if (rv == APR_EMISMATCH) {
258                    ap_log_error(APLOG_MARK, APLOG_ERR, 0, ap_server_conf, APLOGNO(02551)
259                                 "persisted slotmem md5/desc mismatch");
260                }
261                else if (rv == APR_EOF) {
262                    ap_log_error(APLOG_MARK, APLOG_NOTICE, 0, ap_server_conf, APLOGNO(02552)
263                                 "persisted slotmem at EOF... bypassing md5/desc match check "
264                                 "(old persist file?)");
265                    rv = APR_SUCCESS;
266                }
267            }
268            else {
269                rv = APR_INCOMPLETE;
270            }
271            if (rv == APR_INCOMPLETE) {
272                ap_log_error(APLOG_MARK, APLOG_ERR, 0, ap_server_conf, APLOGNO(02553)
273                             "persisted slotmem read had unexpected size");
274            }
275            apr_file_close(fp);
276        }
277    }
278    return rv;
279}

281/*
282 * Whether the module is called from a MPM that re-enter main() and
283 * pre/post_config phases.
284 */ view japanese
285static APR_INLINE int is_child_process(void)
286{
287#ifdef WIN32
288    return getenv("AP_PARENT_PID") != NULL;
289#else
290    return 0;
291#endif
292}

294static apr_status_t cleanup_slotmem(void *param)
295{
296    int is_child = is_child_process();
297    ap_slotmem_instance_t *next = globallistmem;
298
299    while (next) {
300        if (!is_child && AP_SLOTMEM_IS_PERSIST(next)) {
301            store_slotmem(next);
302        }
303        apr_shm_destroy(next->shm);
304        apr_shm_remove(next->name, next->gpool);
305        next = next->next;
306    }
307
308    globallistmem = NULL;
309    return APR_SUCCESS;
310}

312static apr_status_t slotmem_doall(ap_slotmem_instance_t *mem,
313                                  ap_slotmem_callback_fn_t *func,
314                                  void *data, apr_pool_t *pool)
315{
316    unsigned int i;
317    char *ptr;
318    char *inuse;
319    apr_status_t retval = APR_SUCCESS;
320
321    if (!mem) {
322        return APR_ENOSHMAVAIL;
323    }
324
325    ptr = (char *)mem->base;
326    inuse = mem->inuse;
327    for (i = 0; i < mem->desc->num; i++, inuse++) {
328        if (!AP_SLOTMEM_IS_PREGRAB(mem) || *inuse) {
329            retval = func((void *) ptr, data, pool);
330            if (retval != APR_SUCCESS)
331                break;
332        }
333        ptr += mem->desc->size;
334    }
335    return retval;
336}

338static apr_status_t slotmem_create(ap_slotmem_instance_t **new,
339                                   const char *name, apr_size_t item_size,
340                                   unsigned int item_num,
341                                   ap_slotmem_type_t type, apr_pool_t *pool)
342{
343    int fbased = 1;
344    int restored = 0;
345    char *ptr;
346    sharedslotdesc_t *desc;
347    ap_slotmem_instance_t *res;
348    ap_slotmem_instance_t *next = globallistmem;
349    const char *fname, *pname = NULL;
350    apr_shm_t *shm;
351    apr_size_t basesize = (item_size * item_num);
352    apr_size_t size = AP_SLOTMEM_OFFSET + AP_UNSIGNEDINT_OFFSET +
353                      (item_num * sizeof(char)) + basesize;
354    int persist = (type & AP_SLOTMEM_TYPE_PERSIST) != 0;
355    apr_status_t rv;
356
357    *new = NULL;
358    if (gpool == NULL) {
359        return APR_ENOSHMAVAIL;
360    }
361    if (slotmem_filenames(pool, name, &fname, persist ? &pname : NULL)) {
362        /* first try to attach to existing slotmem */
363        if (next) {
364            for (;;) {
365                if (strcmp(next->name, fname) == 0) {
366                    /* we already have it */
367                    *new = next;
368                    ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, ap_server_conf, APLOGNO(02603)
369                                 "create found %s in global list", fname);
370                    return APR_SUCCESS;
371                }
372                if (!next->next) {
373                     break;
374                }
375                next = next->next;
376            }
377        }
378        ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, ap_server_conf, APLOGNO(02602)
379                     "create didn't find %s in global list", fname);
380    }
381    else {
382        fbased = 0;
383        fname = "none";
384    }
385
386    /* first try to attach to existing shared memory */ view japanese
387    ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, ap_server_conf, APLOGNO(02300)
388                 "create %s: %"APR_SIZE_T_FMT"/%u", fname, item_size,
389                 item_num);
390
391    {
392        /* For MPMs that run pre/post_config() phases in both the parent
393         * and children processes (e.g. winnt), SHMs created by the
394         * parent exist in the children already; attach them.
395         */ view japanese
396        if (fbased) {
397            if (is_child_process()) {
398                rv = apr_shm_attach(&shm, fname, gpool);
399            }
400            else {
401                apr_shm_remove(fname, pool);
402                rv = apr_shm_create(&shm, size, fname, gpool);
403            }
404        }
405        else {
406            rv = apr_shm_create(&shm, size, NULL, gpool);
407        }
408        ap_log_error(APLOG_MARK, rv == APR_SUCCESS ? APLOG_DEBUG : APLOG_ERR,
409                     rv, ap_server_conf, APLOGNO(02611)
410                     "create: apr_shm_%s(%s) %s",
411                     fbased && is_child_process() ? "attach" : "create",
412                     fname, rv == APR_SUCCESS ? "succeeded" : "failed");
413        if (rv != APR_SUCCESS) {
414            return rv;
415        }
416
417        desc = (sharedslotdesc_t *)apr_shm_baseaddr_get(shm);
418        memset(desc, 0, size);
419        desc->size = item_size;
420        desc->num = item_num;
421        desc->type = type;
422
423        /*
424         * TODO: Error check the below... What error makes
425         * sense if the restore fails? Any?
426         * For now, we continue with a fresh new slotmem,
427         * but NOTICE in the log.
428         */ view japanese
429        if (persist) {
430            rv = restore_slotmem(desc, pname, size, pool);
431            if (rv == APR_SUCCESS) {
432                restored = 1;
433            }
434            else {
435                /* just in case, re-zero */ view japanese
436                ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, ap_server_conf,
437                             APLOGNO(02554) "could not restore %s", fname);
438                memset((char *)desc + AP_SLOTMEM_OFFSET, 0,
439                       size - AP_SLOTMEM_OFFSET);
440            }
441        }
442    }
443
444    ptr = (char *)desc + AP_SLOTMEM_OFFSET;
445
446    /* For the chained slotmem stuff */
447    res = apr_pcalloc(gpool, sizeof(ap_slotmem_instance_t));
448    res->name = apr_pstrdup(gpool, fname);
449    res->pname = apr_pstrdup(gpool, pname);
450    res->fbased = fbased;
451    res->shm = shm;
452    res->persist = (void *)ptr;
453    res->num_free = (unsigned int *)ptr;
454    ptr += AP_UNSIGNEDINT_OFFSET;
455    if (!restored) {
456        *res->num_free = item_num;
457    }
458    res->base = (void *)ptr;
459    res->desc = desc;
460    res->gpool = gpool;
461    res->next = NULL;
462    res->inuse = ptr + basesize;
463    if (fbased) {
464        if (globallistmem == NULL) {
465            globallistmem = res;
466        }
467        else {
468            next->next = res;
469        }
470    }
471
472    *new = res;
473    return APR_SUCCESS;
474}

476static apr_status_t slotmem_attach(ap_slotmem_instance_t **new,
477                                   const char *name, apr_size_t *item_size,
478                                   unsigned int *item_num, apr_pool_t *pool)
479{
480    char *ptr;
481    ap_slotmem_instance_t *res;
482    ap_slotmem_instance_t *next = globallistmem;
483    sharedslotdesc_t *desc;
484    const char *fname;
485    apr_shm_t *shm;
486    apr_status_t rv;
487
488    if (gpool == NULL) {
489        return APR_ENOSHMAVAIL;
490    }
491    if (!slotmem_filenames(pool, name, &fname, NULL)) {
492        return APR_ENOSHMAVAIL;
493    }
494
495    ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, ap_server_conf, APLOGNO(02301)
496                 "attach looking for %s", fname);
497
498    /* first try to attach to existing slotmem */ view japanese
499    if (next) {
500        for (;;) {
501            if (strcmp(next->name, fname) == 0) {
502                /* we already have it */ view japanese
503                *new = next;
504                *item_size = next->desc->size;
505                *item_num = next->desc->num;
506                ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, ap_server_conf,
507                             APLOGNO(02302)
508                             "attach found %s: %"APR_SIZE_T_FMT"/%u", fname,
509                             *item_size, *item_num);
510                return APR_SUCCESS;
511            }
512            if (!next->next) {
513                 break;
514            }
515            next = next->next;
516        }
517    }
518
519    /* next try to attach to existing shared memory */ view japanese
520    rv = apr_shm_attach(&shm, fname, gpool);
521    if (rv != APR_SUCCESS) {
522        return rv;
523    }
524
525    /* Read the description of the slotmem */ view japanese
526    desc = (sharedslotdesc_t *)apr_shm_baseaddr_get(shm);
527    ptr = (char *)desc + AP_SLOTMEM_OFFSET;
528
529    /* For the chained slotmem stuff */ view japanese
530    res = apr_pcalloc(gpool, sizeof(ap_slotmem_instance_t));
531    res->name = apr_pstrdup(gpool, fname);
532    res->fbased = 1;
533    res->shm = shm;
534    res->persist = (void *)ptr;
535    res->num_free = (unsigned int *)ptr;
536    ptr += AP_UNSIGNEDINT_OFFSET;
537    res->base = (void *)ptr;
538    res->desc = desc;
539    res->gpool = gpool;
540    res->inuse = ptr + (desc->size * desc->num);
541    res->next = NULL;
542
543    *new = res;
544    *item_size = desc->size;
545    *item_num = desc->num;
546    ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, ap_server_conf,
547                 APLOGNO(02303)
548                 "attach found %s: %"APR_SIZE_T_FMT"/%u", fname,
549                 *item_size, *item_num);
550    return APR_SUCCESS;
551}

553static apr_status_t slotmem_dptr(ap_slotmem_instance_t *slot,
554                                 unsigned int id, void **mem)
555{
556    char *ptr;
557
558    if (!slot) {
559        return APR_ENOSHMAVAIL;
560    }
561    if (id >= slot->desc->num) {
562        return APR_EINVAL;
563    }
564
565    ptr = (char *)slot->base + slot->desc->size * id;
566    if (!ptr) {
567        return APR_ENOSHMAVAIL;
568    }
569    *mem = (void *)ptr;
570    return APR_SUCCESS;
571}

573static apr_status_t slotmem_get(ap_slotmem_instance_t *slot, unsigned int id,
574                                unsigned char *dest, apr_size_t dest_len)
575{
576    void *ptr;
577    char *inuse;
578    apr_status_t ret;
579
580    if (!slot) {
581        return APR_ENOSHMAVAIL;
582    }
583
584    inuse = slot->inuse + id;
585    if (id >= slot->desc->num) {
586        return APR_EINVAL;
587    }
588    if (AP_SLOTMEM_IS_PREGRAB(slot) && !*inuse) {
589        return APR_NOTFOUND;
590    }
591    ret = slotmem_dptr(slot, id, &ptr);
592    if (ret != APR_SUCCESS) {
593        return ret;
594    }
595    *inuse = 1;
596    memcpy(dest, ptr, dest_len); /* bounds check? */
597    return APR_SUCCESS;
598}

600static apr_status_t slotmem_put(ap_slotmem_instance_t *slot, unsigned int id,
601                                unsigned char *src, apr_size_t src_len)
602{
603    void *ptr;
604    char *inuse;
605    apr_status_t ret;
606
607    if (!slot) {
608        return APR_ENOSHMAVAIL;
609    }
610
611    inuse = slot->inuse + id;
612    if (id >= slot->desc->num) {
613        return APR_EINVAL;
614    }
615    if (AP_SLOTMEM_IS_PREGRAB(slot) && !*inuse) {
616        return APR_NOTFOUND;
617    }
618    ret = slotmem_dptr(slot, id, &ptr);
619    if (ret != APR_SUCCESS) {
620        return ret;
621    }
622    *inuse=1;
623    memcpy(ptr, src, src_len); /* bounds check? */ view japanese
624    return APR_SUCCESS;
625}

632static unsigned int slotmem_num_free_slots(ap_slotmem_instance_t *slot)
633{
634    if (AP_SLOTMEM_IS_PREGRAB(slot))
635        return *slot->num_free;
636    else {
637        unsigned int i, counter=0;
638        char *inuse = slot->inuse;
639        for (i=0; i<slot->desc->num; i++, inuse++) {

653{

705static apr_status_t slotmem_release(ap_slotmem_instance_t *slot,
706                                    unsigned int id)
707{
708    char *inuse;
709
710    if (!slot) {
711        return APR_ENOSHMAVAIL;
712    }
713
714    inuse = slot->inuse;
715
716    if (id >= slot->desc->num || !inuse[id] ) {
717        ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, ap_server_conf, APLOGNO(02294)
718                     "slotmem(%s) release failed. Num %u/inuse[%u] %d",
719                     slot->name, slotmem_num_slots(slot),
720                     id, (int)inuse[id]);
721        if (id >= slot->desc->num) {
722            return APR_EINVAL;
723        } else {
724            return APR_NOTFOUND;
725        }
726    }
727    inuse[id] = 0;
728    (*slot->num_free)++;
729    return APR_SUCCESS;
