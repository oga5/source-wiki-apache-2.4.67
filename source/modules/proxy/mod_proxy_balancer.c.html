71static int proxy_balancer_canon(request_rec *r, char *url)
72{
73    char *host;
74    apr_port_t port = 0;
75    const char *err;
76
77    /* TODO: offset of BALANCER_PREFIX ?? */ view japanese
78    if (ap_cstr_casecmpn(url, "balancer:", 9) == 0) {
79        ap_log_rerror(APLOG_MARK, APLOG_TRACE1, 0, r, "canonicalising URL %s", url);
80        url += 9;
81    }
82    else {
83        return DECLINED;
84    }
85
86    /* do syntatic check.
87     * We break the URL into host, port, path
88     */ view japanese
89    err = ap_proxy_canon_netloc(r->pool, &url, NULL, NULL, &host, &port);
90    if (err) {
91        ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(01157)
92                      "error parsing URL %s: %s",
93                      url, err);
94        return HTTP_BAD_REQUEST;
95    }
96
97    /* The canon_handler hooks are run per the BalancerMember in
98     * balancer_fixup(), keep the original/raw path for now.
99     */ view japanese
100    r->filename = apr_pstrcat(r->pool, "proxy:" BALANCER_PREFIX,
101                              host, "/", url, NULL);
102
103    return OK;
104}

106static void init_balancer_members(apr_pool_t *p, server_rec *s,
107                                 proxy_balancer *balancer)
108{
109    int i;
110    proxy_worker **workers;
111
112    workers = (proxy_worker **)balancer->workers->elts;
113
114    for (i = 0; i < balancer->workers->nelts; i++) {
115        int worker_is_initialized;
116        proxy_worker *worker = *workers;
117        ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, s, APLOGNO(01158)
118                     "Looking at %s -> %s initialized?", balancer->s->name,
119                     ap_proxy_worker_name(p, worker));
120        worker_is_initialized = PROXY_WORKER_IS_INITIALIZED(worker);
121        if (!worker_is_initialized) {
122            ap_proxy_initialize_worker(worker, s, p);
123        }
124        ++workers;
125    }
126
127    /* Set default number of attempts to the number of
128     * workers.
129     */ view japanese
130    if (!balancer->s->max_attempts_set && balancer->workers->nelts > 1) {
131        balancer->s->max_attempts = balancer->workers->nelts - 1;
132        balancer->s->max_attempts_set = 1;
133    }
134}

136/* Retrieve the parameter with the given name
137 * Something like 'JSESSIONID=12345...N'
138 */ view japanese
139static char *get_path_param(apr_pool_t *pool, char *url,
140                            const char *name, int scolon_sep)
141{
142    char *path = NULL;
143    char *pathdelims = "?&";
144
145    if (scolon_sep) {
146        pathdelims = ";?&";
147    }
148    for (path = strstr(url, name); path; path = strstr(path + 1, name)) {
149        path += strlen(name);
150        if (*path == '=') {
151            /*
152             * Session path was found, get its value
153             */ view japanese
154            ++path;
155            if (*path) {
156                char *q;
157                path = apr_strtok(apr_pstrdup(pool, path), pathdelims, &q);
158                return path;
159            }
160        }
161    }
162    return NULL;
163}

165static char *get_cookie_param(request_rec *r, const char *name)
166{
167    const char *cookies;
168    const char *start_cookie;
169
170    if ((cookies = apr_table_get(r->headers_in, "Cookie"))) {
171        for (start_cookie = ap_strstr_c(cookies, name); start_cookie;
172             start_cookie = ap_strstr_c(start_cookie + 1, name)) {
173            if (start_cookie == cookies ||
174                start_cookie[-1] == ';' ||
175                start_cookie[-1] == ',' ||
176                isspace(start_cookie[-1])) {
177
178                start_cookie += strlen(name);
179                while(*start_cookie && isspace(*start_cookie))
180                    ++start_cookie;
181                if (*start_cookie++ == '=' && *start_cookie) {
182                    /*
183                     * Session cookie was found, get its value
184                     */ view japanese
185                    char *end_cookie, *cookie;
186                    cookie = apr_pstrdup(r->pool, start_cookie);
187                    if ((end_cookie = strchr(cookie, ';')) != NULL)
188                        *end_cookie = '\0';
189                    if((end_cookie = strchr(cookie, ',')) != NULL)
190                        *end_cookie = '\0';
191                    return cookie;
192                }
193            }
194        }
195    }
196    return NULL;
197}

199/* Find the worker that has the 'route' defined
200 */ view japanese
201static proxy_worker *find_route_worker(proxy_balancer *balancer,
202                                       const char *route, request_rec *r,
203                                       int recursion)
204{
205    int i;
206    int checking_standby;
207    int checked_standby;
208
209    proxy_worker **workers;
210
211    checking_standby = checked_standby = 0;
212    while (!checked_standby) {
213        workers = (proxy_worker **)balancer->workers->elts;
214        for (i = 0; i < balancer->workers->nelts; i++, workers++) {
215            proxy_worker *worker = *workers;
216            if ( (checking_standby ? !PROXY_WORKER_IS_STANDBY(worker) : PROXY_WORKER_IS_STANDBY(worker)) )
217                continue;
218            if (*(worker->s->route) && strcmp(worker->s->route, route) == 0) {
219                if (PROXY_WORKER_IS_USABLE(worker)) {
220                    return worker;
221                } else {
222                    /*
223                     * If the worker is in error state run
224                     * retry on that worker. It will be marked as
225                     * operational if the retry timeout is elapsed.
226                     * The worker might still be unusable, but we try
227                     * anyway.
228                     */ view japanese
229                    ap_proxy_retry_worker_fn("BALANCER", worker, r->server);
230                    if (PROXY_WORKER_IS_USABLE(worker)) {
231                            return worker;
232                    } else {
233                        /*
234                         * We have a worker that is unusable.
235                         * It can be in error or disabled, but in case
236                         * it has a redirection set use that redirection worker.
237                         * This enables to safely remove the member from the
238                         * balancer. Of course you will need some kind of
239                         * session replication between those two remote.
240                         * Also check that we haven't gone thru all the
241                         * balancer members by means of redirects.
242                         * This should avoid redirect cycles.
243                         */ view japanese
244                        if ((*worker->s->redirect)
245                            && (recursion < balancer->workers->nelts)) {
246                            proxy_worker *rworker = NULL;
247                            rworker = find_route_worker(balancer, worker->s->redirect,
248                                                        r, recursion + 1);
249                            /* Check if the redirect worker is usable */ view japanese
250                            if (rworker && !PROXY_WORKER_IS_USABLE(rworker)) {
251                                /*
252                                 * If the worker is in error state run
253                                 * retry on that worker. It will be marked as
254                                 * operational if the retry timeout is elapsed.
255                                 * The worker might still be unusable, but we try
256                                 * anyway.
257                                 */ view japanese
258                                ap_proxy_retry_worker_fn("BALANCER", rworker, r->server);
259                            }
260                            if (rworker && PROXY_WORKER_IS_USABLE(rworker))
261                                return rworker;
262                        }
263                    }
264                }
265            }
266        }
267        checked_standby = checking_standby++;
268    }
269    return NULL;
270}

272static proxy_worker *find_session_route(proxy_balancer *balancer,
273                                        request_rec *r,
274                                        char **route,
275                                        const char **sticky_used,
276                                        char **url)
277{
278    proxy_worker *worker = NULL;
279    char *url_with_qs;
280
281    if (!*balancer->s->sticky)
282        return NULL;
283    /*
284     * The route might be contained in the query string and *url is not
285     * supposed to contain the query string. Hence add it temporarily if
286     * present.
287     */ view japanese
288    if (r->args) {
289        url_with_qs = apr_pstrcat(r->pool, *url, "?", r->args, NULL);
290    }
291    else {
292        url_with_qs = *url;
293    }
294    /* Try to find the sticky route inside url */ view japanese
295    *route = get_path_param(r->pool, url_with_qs, balancer->s->sticky_path, balancer->s->scolonsep);
296    if (*route) {
297        ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(01159)
298                     "Found value %s for stickysession %s",
299                     *route, balancer->s->sticky_path);
300        *sticky_used =  balancer->s->sticky_path;
301    }
302    else {
303        *route = get_cookie_param(r, balancer->s->sticky);
304        if (*route) {
305            *sticky_used =  balancer->s->sticky;
306            ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(01160)
307                         "Found value %s for stickysession %s",
308                         *route, balancer->s->sticky);
309        }
310    }
311    /*
312     * If we found a value for stickysession, find the first '.' (or whatever
313     * sticky_separator is set to) within. Everything after '.' (if present)
314     * is our route. 
315     */ view japanese
316    if ((*route) && (balancer->s->sticky_separator != 0) && ((*route = strchr(*route, balancer->s->sticky_separator)) != NULL ))
317        (*route)++;
318    if ((*route) && (**route)) {
319        ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(01161) "Found route %s", *route);
320        /* We have a route in path or in cookie
321         * Find the worker that has this route defined.
322         */ view japanese
323        worker = find_route_worker(balancer, *route, r, 1);
324        if (worker && strcmp(*route, worker->s->route)) {
325            /*
326             * Notice that the route of the worker chosen is different from
327             * the route supplied by the client.
328             */ view japanese
329            apr_table_setn(r->subprocess_env, "BALANCER_ROUTE_CHANGED", "1");
330            ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(01162)
331                          "Route changed from %s to %s",
332                          *route, worker->s->route);
333        }
334        return worker;
335    }
336    else
337        return NULL;
338}

340static proxy_worker *find_best_worker(proxy_balancer *balancer,
341                                      request_rec *r)
342{
343    proxy_worker *candidate = NULL;
344    apr_status_t rv;
345
346#if APR_HAS_THREADS
347    if ((rv = PROXY_THREAD_LOCK(balancer)) != APR_SUCCESS) {
348        ap_log_rerror(APLOG_MARK, APLOG_ERR, rv, r, APLOGNO(01163)
349                      "%s: Lock failed for find_best_worker()",
350                      balancer->s->name);
351        return NULL;
352    }
353#endif
354
355    candidate = (*balancer->lbmethod->finder)(balancer, r);
356
357    if (candidate)
358        candidate->s->elected++;
359
360#if APR_HAS_THREADS
361    if ((rv = PROXY_THREAD_UNLOCK(balancer)) != APR_SUCCESS) {
362        ap_log_rerror(APLOG_MARK, APLOG_ERR, rv, r, APLOGNO(01164)
363                      "%s: Unlock failed for find_best_worker()",
364                      balancer->s->name);
365    }
366#endif
367
368    if (candidate == NULL) {
369        /* All the workers are in error state or disabled.
370         * If the balancer has a timeout sleep for a while
371         * and try again to find the worker. The chances are
372         * that some other thread will release a connection.
373         * By default the timeout is not set, and the server
374         * returns SERVER_BUSY.
375         */ view japanese
376        if (balancer->s->timeout) {
377            /* XXX: This can perhaps be build using some
378             * smarter mechanism, like tread_cond.
379             * But since the statuses can came from
380             * different children, use the provided algo.
381             */ view japanese
382            apr_interval_time_t timeout = balancer->s->timeout;
383            apr_interval_time_t step, tval = 0;
384            /* Set the timeout to 0 so that we don't
385             * end in infinite loop
386             */ view japanese
387            balancer->s->timeout = 0;
388            step = timeout / 100;
389            while (tval < timeout) {
390                apr_sleep(step);
391                /* Try again */ view japanese
392                if ((candidate = find_best_worker(balancer, r)))
393                    break;
394                tval += step;
395            }
396            /* restore the timeout */ view japanese
397            balancer->s->timeout = timeout;
398        }
399    }
400
401    return candidate;
402
403}

405static int balancer_fixup(request_rec *r, proxy_worker *worker, char **url)
406{
407    const char *path;
408    int rc;
409
410    /* Build the proxy URL from the worker URL and the actual path */ view japanese
411    path = strstr(*url, "://");
412    if (path) {
413        path = ap_strchr_c(path + 3, '/');
414    }
415    r->filename = apr_pstrcat(r->pool, "proxy:", worker->s->name_ex, path, NULL);
416
417    /* Canonicalize r->filename per the worker scheme's canon_handler hook */ view japanese
418    rc = ap_proxy_canon_url(r);
419    if (rc == OK) {
420        AP_DEBUG_ASSERT(strncmp(r->filename, "proxy:", 6) == 0);
421        *url = apr_pstrdup(r->pool, r->filename + 6);
422    }
423    return rc;
424}

426static void force_recovery(proxy_balancer *balancer, server_rec *s)
427{
428    int i;
429    int ok = 0;
430    proxy_worker **worker;
431
432    worker = (proxy_worker **)balancer->workers->elts;
433    for (i = 0; i < balancer->workers->nelts; i++, worker++) {
434        if (!((*worker)->s->status & PROXY_WORKER_IN_ERROR)) {
435            ok = 1;
436            break;
437        }
438        else {
439            /* Try if we can recover */ view japanese
440            ap_proxy_retry_worker_fn("BALANCER", *worker, s);
441            if (!((*worker)->s->status & PROXY_WORKER_IN_ERROR)) {
442                ok = 1;
443                break;
444            }
445        }
446    }
447    if (!ok && balancer->s->forcerecovery) {
448        /* If all workers are in error state force the recovery.
449         */ view japanese
450        worker = (proxy_worker **)balancer->workers->elts;
451        for (i = 0; i < balancer->workers->nelts; i++, worker++) {
452            ++(*worker)->s->retries;
453            (*worker)->s->status &= ~PROXY_WORKER_IN_ERROR;
454            ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, s, APLOGNO(01165)
455                         "%s: Forcing recovery for worker (%s:%d)",
456                         balancer->s->name, (*worker)->s->hostname_ex,
457                         (int)(*worker)->s->port);
458        }
459    }
460}

462static apr_status_t decrement_busy_count(void *worker_)

464    proxy_worker *worker = worker_;
465    
466    if (worker->s->busy) {
467        worker->s->busy--;
468    }
469
470    return APR_SUCCESS;

473static int proxy_balancer_pre_request(proxy_worker **worker,
474                                      proxy_balancer **balancer,
475                                      request_rec *r,
476                                      proxy_server_conf *conf, char **url)
477{
478    int access_status;
479    proxy_worker *runtime;
480    char *route = NULL;
481    const char *sticky = NULL;
482    apr_status_t rv;
483
484    *worker = NULL;
485    /* Step 1: check if the url is for us
486     * The url we can handle starts with 'balancer://'
487     * If balancer is already provided skip the search
488     * for balancer, because this is failover attempt.
489     */ view japanese
490    if (!*balancer &&
491        (ap_cstr_casecmpn(*url, BALANCER_PREFIX, sizeof(BALANCER_PREFIX) - 1)
492         || !(*balancer = ap_proxy_get_balancer(r->pool, conf, *url, 1))))
493        return DECLINED;
494
495    /* Step 2: Lock the LoadBalancer
496     * XXX: perhaps we need the process lock here
497     */ view japanese
498#if APR_HAS_THREADS
499    if ((rv = PROXY_THREAD_LOCK(*balancer)) != APR_SUCCESS) {
500        ap_log_rerror(APLOG_MARK, APLOG_ERR, rv, r, APLOGNO(01166)
501                      "%s: Lock failed for pre_request", (*balancer)->s->name);
502        return DECLINED;
503    }
504#endif
505
506    /* Step 3: force recovery */ view japanese
507    force_recovery(*balancer, r->server);
508
509    /* Step 3.5: Update member list for the balancer */ view japanese
510    /* TODO: Implement as provider! */ view japanese
511    ap_proxy_sync_balancer(*balancer, r->server, conf);
512
513    /* Step 4: find the session route */ view japanese
514    runtime = find_session_route(*balancer, r, &route, &sticky, url);
515    if (runtime) {
516        if ((*balancer)->lbmethod && (*balancer)->lbmethod->updatelbstatus) {
517            /* Call the LB implementation */ view japanese
518            (*balancer)->lbmethod->updatelbstatus(*balancer, runtime, r->server);
519        }
520        else { /* Use the default one */ view japanese
521            int i, total_factor = 0;
522            proxy_worker **workers;
523            /* We have a sticky load balancer
524             * Update the workers status
525             * so that even session routes get
526             * into account.
527             */ view japanese
528            workers = (proxy_worker **)(*balancer)->workers->elts;
529            for (i = 0; i < (*balancer)->workers->nelts; i++) {
530                /* Take into calculation only the workers that are
531                 * not in error state or not disabled.
532                 */ view japanese
533                if (PROXY_WORKER_IS_USABLE(*workers)) {
534                    (*workers)->s->lbstatus += (*workers)->s->lbfactor;
535                    total_factor += (*workers)->s->lbfactor;
536                }
537                workers++;
538            }
539            runtime->s->lbstatus -= total_factor;
540        }
541        runtime->s->elected++;
542
543        *worker = runtime;
544    }
545    else if (route && (*balancer)->s->sticky_force) {
546        int i, member_of = 0;
547        proxy_worker **workers;
548        /*
549         * We have a route provided that doesn't match the
550         * balancer name. See if the provider route is the
551         * member of the same balancer in which case return 503
552         */ view japanese
553        workers = (proxy_worker **)(*balancer)->workers->elts;
554        for (i = 0; i < (*balancer)->workers->nelts; i++) {
555            if (*((*workers)->s->route) && strcmp((*workers)->s->route, route) == 0) {
556                member_of = 1;
557                break;
558            }
559            workers++;
560        }
561        if (member_of) {
562            ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(01167)
563                          "%s: All workers are in error state for route (%s)",
564                          (*balancer)->s->name, route);
565#if APR_HAS_THREADS
566            if ((rv = PROXY_THREAD_UNLOCK(*balancer)) != APR_SUCCESS) {
567                ap_log_rerror(APLOG_MARK, APLOG_ERR, rv, r, APLOGNO(01168)
568                              "%s: Unlock failed for pre_request",
569                              (*balancer)->s->name);
570            }
571#endif
572            return HTTP_SERVICE_UNAVAILABLE;
573        }
574    }
575
576#if APR_HAS_THREADS
577    if ((rv = PROXY_THREAD_UNLOCK(*balancer)) != APR_SUCCESS) {
578        ap_log_rerror(APLOG_MARK, APLOG_ERR, rv, r, APLOGNO(01169)
579                      "%s: Unlock failed for pre_request",
580                      (*balancer)->s->name);
581    }
582#endif
583    if (!*worker) {
584        runtime = find_best_worker(*balancer, r);
585        if (!runtime) {
586            if ((*balancer)->workers->nelts) {
587                ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(01170)
588                              "%s: All workers are in error state",
589                              (*balancer)->s->name);
590            } else {
591                ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(01171)
592                              "%s: No workers in balancer",
593                              (*balancer)->s->name);
594            }
595
596            return HTTP_SERVICE_UNAVAILABLE;
597        }
598        if (*(*balancer)->s->sticky && runtime) {
599            /*
600             * This balancer has sticky sessions and the client either has not
601             * supplied any routing information or all workers for this route
602             * including possible redirect and hotstandby workers are in error
603             * state, but we have found another working worker for this
604             * balancer where we can send the request. Thus notice that we have
605             * changed the route to the backend.
606             */ view japanese
607            apr_table_setn(r->subprocess_env, "BALANCER_ROUTE_CHANGED", "1");
608        }
609        *worker = runtime;
610    }
611
612    (*worker)->s->busy++;
613    apr_pool_cleanup_register(r->pool, *worker, decrement_busy_count,
614                              apr_pool_cleanup_null);
615
616    /* Add balancer/worker info to env. */ view japanese
617    apr_table_setn(r->subprocess_env,
618                   "BALANCER_NAME", (*balancer)->s->name);
619    apr_table_setn(r->subprocess_env,
620                   "BALANCER_WORKER_NAME", (*worker)->s->name_ex);
621    apr_table_setn(r->subprocess_env,
622                   "BALANCER_WORKER_ROUTE", (*worker)->s->route);
623
624    /* Rewrite the url from 'balancer://url'

748/*
749 * Compute an ID for a vhost based on what makes it selected by requests.
750 * The second and more Host(s)/IP(s):port(s), and the ServerAlias(es) are
751 * optional (see make_servers_ids() below).
752 */ view japanese
753 static const char *make_server_id(server_rec *s, apr_pool_t *p, int full)
754{
755    apr_md5_ctx_t md5_ctx;
756    unsigned char md5[APR_MD5_DIGESTSIZE];
757    char id[2 * APR_MD5_DIGESTSIZE + 1];
758    char host_ip[64]; /* for any IPv[46] string */ view japanese
759    server_addr_rec *sar;
760    int i;
761
762    apr_md5_init(&md5_ctx);
763    for (sar = s->addrs; sar; sar = sar->next) {
764        host_ip[0] = '\0';
765        apr_sockaddr_ip_getbuf(host_ip, sizeof host_ip, sar->host_addr);
766        apr_md5_update(&md5_ctx, (void *)sar->virthost, strlen(sar->virthost));
767        apr_md5_update(&md5_ctx, (void *)host_ip, strlen(host_ip));
768        apr_md5_update(&md5_ctx, (void *)&sar->host_port,
769                       sizeof(sar->host_port));
770        if (!full) {
771            break;
772        }
773    }
774    if (s->server_hostname) {
775        apr_md5_update(&md5_ctx, (void *)s->server_hostname,
776                       strlen(s->server_hostname));

780            for (i = 0; i < s->names->nelts; ++i) {
781                const char *name = APR_ARRAY_IDX(s->names, i, char *);
782                apr_md5_update(&md5_ctx, (void *)name, strlen(name));
783            }
784        }
785        if (s->wild_names) {
786            for (i = 0; i < s->wild_names->nelts; ++i) {
787                const char *name = APR_ARRAY_IDX(s->wild_names, i, char *);
788                apr_md5_update(&md5_ctx, (void *)name, strlen(name));
789            }
790        }
791    }
792    apr_md5_final(md5, &md5_ctx);
793    ap_bin2hex(md5, APR_MD5_DIGESTSIZE, id);
794
795    return apr_pstrmemdup(p, id, sizeof(id) - 1);
796}

798/*
799 * First try to compute an unique ID for each vhost with minimal criteria,
800 * that is the first Host/IP:port and ServerName. For most cases this should
801 * be enough and avoids changing the ID unnecessarily across restart (or
802 * stop/start w.r.t. persisted files) for things that this module does not
803 * care about.
804 *
805 * But if it's not enough (collisions) do a second pass for the full monty,
806 * that is additionally the other Host(s)/IP(s):port(s) and ServerAlias(es).
807 *
808 * Finally, for pathological configs where this is still not enough, let's
809 * append a counter to duplicates, because we really want that ID to be unique
810 * even if the vhost will never be selected to handle requests at run time, at
811 * load time a duplicate may steal the original slotmems (depending on its
812 * balancers' configurations), see how mod_slotmem_shm reuses slots/files based
813 * solely on this ID and resets them if the sizes don't match.
814 */ view japanese
815static apr_array_header_t *make_servers_ids(server_rec *main_s, apr_pool_t *p)
816{
817    server_rec *s = main_s;
818    apr_array_header_t *ids = apr_array_make(p, 10, sizeof(const char *));
819    apr_hash_t *dups = apr_hash_make(p);
820    int idx, *dup, full_monty = 0;
821    const char *id;
822
823    for (idx = 0, s = main_s; s; s = s->next, ++idx) {
824        id = make_server_id(s, p, 0);
825        dup = apr_hash_get(dups, id, APR_HASH_KEY_STRING);
826        apr_hash_set(dups, id, APR_HASH_KEY_STRING,
827                     apr_pmemdup(p, &idx, sizeof(int)));
828        if (dup) {
829            full_monty = 1;
830            APR_ARRAY_IDX(ids, *dup, const char *) = NULL;
831            APR_ARRAY_PUSH(ids, const char *) = NULL;
832        }
833        else {
834            APR_ARRAY_PUSH(ids, const char *) = id;
835        }
836    }
837    if (full_monty) {
838        apr_hash_clear(dups);
839        for (idx = 0, s = main_s; s; s = s->next, ++idx) {
840            id = APR_ARRAY_IDX(ids, idx, const char *);
841            if (id) {
842                /* Preserve non-duplicates */ view japanese
843                continue;
844            }
845            id = make_server_id(s, p, 1);
846            if (apr_hash_get(dups, id, APR_HASH_KEY_STRING)) {
847                id = apr_psprintf(p, "%s_%x", id, idx);
848            }
849            else {
850                apr_hash_set(dups, id, APR_HASH_KEY_STRING, (void *)-1);
851            }
852            APR_ARRAY_IDX(ids, idx, const char *) = id;
853        }
854    }
855
856    return ids;
857}

1896/* Manages the loadfactors and member status
1897 *   The balancer, worker and nonce are obtained from
1898 *   the request args (?b=...&w=...&nonce=....).
1899 *   All other params are pulled from any POST
1900 *   data that exists.
1901 * TODO:
1902 *   /.../<whatever>/balancer/worker/nonce
1903 */ view japanese
1904static int balancer_handler(request_rec *r)
1905{
1906    void *sconf;
1907    proxy_server_conf *conf;
1908    proxy_balancer *balancer, *bsel = NULL;
1909    proxy_worker *wsel = NULL;
1910    apr_table_t *params;
1911    int i;
1912    const char *name, *ref;
1913    apr_status_t rv;
1914
1915    /* is this for us? */ view japanese
1916    if (strcmp(r->handler, "balancer-manager")) {
1917        return DECLINED;
1918    }
1919
1920    r->allowed = 0
1921    | (AP_METHOD_BIT << M_GET)
1922    | (AP_METHOD_BIT << M_POST);
1923    if ((r->method_number != M_GET) && (r->method_number != M_POST)) {
1924        return DECLINED;
1925    }
1926
1927    sconf = r->server->module_config;
1928    conf = (proxy_server_conf *) ap_get_module_config(sconf, &proxy_module);
1929    params = apr_table_make(r->pool, 10);
1930
1931    balancer = (proxy_balancer *)conf->balancers->elts;
1932    for (i = 0; i < conf->balancers->nelts; i++, balancer++) {
1933#if APR_HAS_THREADS
1934        if ((rv = PROXY_THREAD_LOCK(balancer)) != APR_SUCCESS) {
1935            ap_log_rerror(APLOG_MARK, APLOG_ERR, rv, r, APLOGNO(01189)
1936                          "%s: Lock failed for balancer_handler",
1937                          balancer->s->name);
1938        }
1939#endif
1940        ap_proxy_sync_balancer(balancer, r->server, conf);
1941#if APR_HAS_THREADS
1942        if ((rv = PROXY_THREAD_UNLOCK(balancer)) != APR_SUCCESS) {
1943            ap_log_rerror(APLOG_MARK, APLOG_ERR, rv, r, APLOGNO(01190)
1944                          "%s: Unlock failed for balancer_handler",
1945                          balancer->s->name);
1946        }
1947#endif
1948    }
1949
1950    if (r->args && (r->method_number == M_GET)) {
1951        const char *allowed[] = { "w", "b", "nonce", "xml", NULL };
1952        ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(01191) "parsing r->args");
1953
1954        push2table(r->args, params, allowed, r->pool);
1955    }
1956    if (r->method_number == M_POST) {
1957        apr_bucket_brigade *ib;
1958        apr_size_t len = 1024;
1959        char *buf = apr_pcalloc(r->pool, len+1);
1960
1961        ib = apr_brigade_create(r->connection->pool, r->connection->bucket_alloc);
1962        rv = ap_get_brigade(r->input_filters, ib, AP_MODE_READBYTES,
1963                                APR_BLOCK_READ, len);
1964        if (rv != APR_SUCCESS) {
1965            return ap_map_http_request_error(rv, HTTP_BAD_REQUEST);
1966        }
1967        apr_brigade_flatten(ib, buf, &len);
1968        buf[len] = '\0';
1969        push2table(buf, params, NULL, r->pool);
1970    }
1971
1972    /* Ignore parameters if this looks like XSRF */ view japanese
1973    ref = apr_table_get(r->headers_in, "Referer");
1974    if (apr_table_elts(params)
1975        && (!ref || !safe_referer(r, ref))) {
1976        ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(10187)
1977                      "ignoring params in balancer-manager cross-site access %s: %s", ref, ap_get_server_name(r));
1978        apr_table_clear(params);
1979    }
1980
1981    /* Process the parameters */ view japanese
1982    if ((name = apr_table_get(params, "b")))
1983        bsel = ap_proxy_get_balancer(r->pool, conf,
1984            apr_pstrcat(r->pool, BALANCER_PREFIX, name, NULL), 0);
1985
1986    if ((name = apr_table_get(params, "w"))) {
1987        wsel = ap_proxy_get_worker(r->pool, bsel, conf, name);
1988    }
1989
1990
1991    /* Check that the supplied nonce matches this server's nonce;
1992     * otherwise ignore all parameters, to prevent a CSRF
1993     * attack. */ view japanese
1994    if (bsel
1995        && (*bsel->s->nonce
1996            && ((name = apr_table_get(params, "nonce")) != NULL
1997                && strcmp(bsel->s->nonce, name) == 0))) {
1998        /* Process the parameters and add the worker to the balancer */ view japanese
1999        rv = balancer_process_balancer_worker(r, conf, bsel, wsel, params);
2000        if (rv != APR_SUCCESS) {
2001            return HTTP_BAD_REQUEST;
2002        }
2003    }
2004
2005    /* display the HTML or XML page */ view japanese
2006    if (apr_table_get(params, "xml")) {
2007        balancer_display_page(r, conf, bsel, wsel, 1);
2008    } else {
2009        balancer_display_page(r, conf, bsel, wsel, 0);
2010    }
2011    return DONE;
2012}
