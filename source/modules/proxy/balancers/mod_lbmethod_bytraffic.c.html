28static int is_best_bytraffic(proxy_worker *current, proxy_worker *prev_best, void *baton)
29{
30    apr_off_t *min_traffic = (apr_off_t *)baton;
31    apr_off_t traffic = (current->s->transferred / current->s->lbfactor)
32                        + (current->s->read / current->s->lbfactor);
33
34    if (!prev_best || (traffic < *min_traffic)) {
35        *min_traffic = traffic;
36
37        return TRUE;
38    }
39
40    return FALSE;
41}
42
43/*
44 * The idea behind the find_best_bytraffic scheduler is the following:
45 *
46 * We know the amount of traffic (bytes in and out) handled by each
47 * worker. We normalize that traffic by each workers' weight. So assuming
48 * a setup as below:
49 *
50 * worker     a    b    c
51 * lbfactor   1    1    3
52 *
53 * the scheduler will allow worker c to handle 3 times the
54 * traffic of a and b. If each request/response results in the
55 * same amount of traffic, then c would be accessed 3 times as
56 * often as a or b. If, for example, a handled a request that
57 * resulted in a large i/o bytecount, then b and c would be
58 * chosen more often, to even things out.
59 */ view japanese
60static proxy_worker *find_best_bytraffic(proxy_balancer *balancer,
61                                         request_rec *r)
62{
63    apr_off_t min_traffic = 0;
64
65    return ap_proxy_balancer_get_best_worker_fn(balancer, r, is_best_bytraffic,
66                                             &min_traffic);
67}

70static apr_status_t reset(proxy_balancer *balancer, server_rec *s)
71{
72    int i;
73    proxy_worker **worker;
74    worker = (proxy_worker **)balancer->workers->elts;
75    for (i = 0; i < balancer->workers->nelts; i++, worker++) {
76        (*worker)->s->transferred = 0;
77        (*worker)->s->read = 0;
78    }
79    return APR_SUCCESS;
80}

98static int lbmethod_bytraffic_post_config(apr_pool_t *pconf, apr_pool_t *plog,
99        apr_pool_t *ptemp, server_rec *s)
100{
101
102    /* lbmethod_bytraffic_post_config() will be called twice during startup.  So, don't
103     * set up the static data the 1st time through. */ view japanese
104    if (ap_state_query(AP_SQ_MAIN_STATE) == AP_SQ_MS_CREATE_PRE_CONFIG) {
105        return OK;
106    }
107
108    ap_proxy_balancer_get_best_worker_fn =
109                 APR_RETRIEVE_OPTIONAL_FN(proxy_balancer_get_best_worker);
110    if (!ap_proxy_balancer_get_best_worker_fn) {
111        ap_log_error(APLOG_MARK, APLOG_EMERG, 0, s, APLOGNO(10150)
112                     "mod_proxy must be loaded for mod_lbmethod_bytraffic");
113        return !OK;
114    }
115
116    return OK;
117}

/*
 * find_best_bytraffic スケジューラの考え方は以下の通りです。
 *
 * 各ワーカーが処理したトラフィック量（入出力バイト数）を把握しています。
 * このトラフィック量を各ワーカーの重みで正規化します。以下のような構成を想定します。
 *
 * worker     a    b    c
 * lbfactor   1    1    3
 *
 * このスケジューラは、ワーカー c が a や b の 3 倍のトラフィックを処理できるようにします。
 * 各リクエスト/レスポンスのトラフィック量が同じであれば、c は a や b の 3 倍の頻度でアクセスされます。
 * 例えば、a が大きな I/O バイト数を伴うリクエストを処理した場合、バランスを取るために b や c がより頻繁に選択されるようになります。
 */
