28static int is_best_byrequests(proxy_worker *current, proxy_worker *prev_best, void *baton)
29{
30    int *total_factor = (int *)baton;
31
32    current->s->lbstatus += current->s->lbfactor;
33    *total_factor += current->s->lbfactor;
34
35    return (!prev_best || (current->s->lbstatus > prev_best->s->lbstatus));
36}

38/*
39 * The idea behind the find_best_byrequests scheduler is the following:
40 *
41 * lbfactor is "how much we expect this worker to work", or "the worker's
42 * normalized work quota".
43 *
44 * lbstatus is "how urgent this worker has to work to fulfill its quota
45 * of work".
46 *
47 * We distribute each worker's work quota to the worker, and then look
48 * which of them needs to work most urgently (biggest lbstatus).  This
49 * worker is then selected for work, and its lbstatus reduced by the
50 * total work quota we distributed to all workers.  Thus the sum of all
51 * lbstatus does not change.(*)
52 *
53 * If some workers are disabled, the others will
54 * still be scheduled correctly.
55 *
56 * If a balancer is configured as follows:
57 *
58 * worker     a    b    c    d
59 * lbfactor  25   25   25   25
60 *
61 * And b gets disabled, the following schedule is produced:
62 *
63 *    a c d a c d a c d ...
64 *
65 * Note that the above lbfactor setting is the *exact* same as:
66 *
67 * worker     a    b    c    d
68 * lbfactor   1    1    1    1
69 *
70 * Asymmetric configurations work as one would expect. For
71 * example:
72 *
73 * worker     a    b    c    d
74 * lbfactor   1    1    1    2
75 *
76 * would have a, b and c all handling about the same
77 * amount of load with d handling twice what a or b
78 * or c handles individually. So we could see:
79 *
80 *   b a d c d a c d b d ...
81 *
82 */ view japanese
83static proxy_worker *find_best_byrequests(proxy_balancer *balancer,
84                                request_rec *r)
85{
86    int total_factor = 0;
87    proxy_worker *worker = ap_proxy_balancer_get_best_worker_fn(balancer, r, is_best_byrequests, &total_factor);
88
89    if (worker) {
90        worker->s->lbstatus -= total_factor;
91    }
92
93    return worker;
94}

97static apr_status_t reset(proxy_balancer *balancer, server_rec *s)
98{
99    int i;
100    proxy_worker **worker;
101    worker = (proxy_worker **)balancer->workers->elts;
102    for (i = 0; i < balancer->workers->nelts; i++, worker++) {
103        (*worker)->s->lbstatus = 0;
104    }
105    return APR_SUCCESS;
106}

130static int lbmethod_byrequests_post_config(apr_pool_t *pconf, apr_pool_t *plog,
131        apr_pool_t *ptemp, server_rec *s)
132{
133
134    /* lbmethod_byrequests_post_config() will be called twice during startup.  So, don't
135     * set up the static data the 1st time through. */ view japanese
136    if (ap_state_query(AP_SQ_MAIN_STATE) == AP_SQ_MS_CREATE_PRE_CONFIG) {
137        return OK;
138    }
139
140    ap_proxy_balancer_get_best_worker_fn =
141                 APR_RETRIEVE_OPTIONAL_FN(proxy_balancer_get_best_worker);
142    if (!ap_proxy_balancer_get_best_worker_fn) {
143        ap_log_error(APLOG_MARK, APLOG_EMERG, 0, s, APLOGNO(10152)
144                     "mod_proxy must be loaded for mod_lbmethod_byrequests");
145        return !OK;
146    }
147
148    return OK;
149}

/*
 * find_best_byrequests スケジューラの背後にある考え方は以下の通りです。
 *
 * lbfactor は「このワーカーにどれだけ働いてもらうことを期待するか」、
 * すなわち「ワーカーの正規化された作業割り当て量」です。
 *
 * lbstatus は「このワーカーが割り当てられた作業量を満たすために、
 * どれだけ緊急に働く必要があるか」です。
 *
 * 各ワーカーの作業割り当て量をワーカーに分配し、その中で最も緊急に
 * 作業が必要な（lbstatus が最大の）ワーカーを探します。このワーカーが
 * 作業対象として選択され、その lbstatus は全ワーカーに分配した作業割り当て量の
 * 合計分だけ減少します。したがって、すべての lbstatus の合計は変化しません。(*)
 *
 * 一部のワーカーが無効化されていても、他のワーカーは正しくスケジュールされます。
 *
 * バランサーが以下のように設定されている場合：
 *
 * worker     a    b    c    d
 * lbfactor  25   25   25   25
 *
 * b が無効化されると、以下のスケジュールが生成されます：
 *
 *    a c d a c d a c d ...
 *
 * 上記の lbfactor 設定は、以下と*全く*同じであることに注意してください：
 *
 * worker     a    b    c    d
 * lbfactor   1    1    1    1
 *
 * 非対称な設定も期待通りに動作します。例えば：
 *
 * worker     a    b    c    d
 * lbfactor   1    1    1    2
 *
 * この場合、a、b、c はすべてほぼ同じ負荷を処理し、d は a、b、c 個別の
 * 2倍の負荷を処理することになります。そのため、以下のような結果になります：
 *
 *   b a d c d a c d b d ...
 *
 */
