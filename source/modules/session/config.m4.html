15dnl Session modules; modules that are capable of storing key value pairs in
16dnl various places, such as databases, LDAP, or cookies.
17dnl
18session_cookie_objects='mod_session_cookie.lo'
19session_crypto_objects='mod_session_crypto.lo'
20session_dbd_objects='mod_session_dbd.lo'
21
22case "$host" in
23  *os2*)
24    # OS/2 DLLs must resolve all symbols at build time
25    # and we need some from main session module
26    session_cookie_objects="$session_cookie_objects mod_session.la"
27    session_crypto_objects="$session_crypto_objects mod_session.la"
28    session_dbd_objects="$session_dbd_objects mod_session.la"
29    ;;
30esac
31
32APACHE_MODULE(session, session module, , , most)
33APACHE_MODULE(session_cookie, session cookie module, $session_cookie_objects, , $session_mods_enable,,session)
34
35if test "$enable_session_crypto" != ""; then
36  session_mods_enable_crypto=$enable_session_crypto
37else
38  session_mods_enable_crypto=$session_mods_enable
39fi
40if test "$session_mods_enable_crypto" != "no"; then
41  saved_CPPFLAGS="$CPPFLAGS"
42  CPPFLAGS="$CPPFLAGS $APR_INCLUDES $APU_INCLUDES"
43  AC_TRY_COMPILE([#include <apr_crypto.h>],[
44#if APU_HAVE_CRYPTO == 0
45#error no crypto support
46#endif
47  ], [ap_HAVE_APR_CRYPTO="yes"], [ap_HAVE_APR_CRYPTO="no"])
48  CPPFLAGS="$saved_CPPFLAGS"
49  if test $ap_HAVE_APR_CRYPTO = "no"; then
50    AC_MSG_WARN([Your APR does not include SSL/EVP support. To enable it: configure --with-crypto])
51    if test "$enable_session_crypto" != "" -a "$enable_session_crypto" != "no"; then
52        AC_MSG_ERROR([mod_session_crypto cannot be enabled])
53    fi
54    session_mods_enable_crypto="no"
55  fi
56fi
57APACHE_MODULE(session_crypto, session crypto module, $session_crypto_objects, , $session_mods_enable_crypto, [
58if test "$session_mods_enable_crypto" = "no" ; then
59  enable_session_crypto=no
60fi
61],session)
62
63APACHE_MODULE(session_dbd, session dbd module, $session_dbd_objects, , $session_mods_enable,,session)
