47/**
48 * Should the session be included within this URL.
49 *
50 * This function tests whether a session is valid for this URL. It uses the
51 * include and exclude arrays to determine whether they should be included.
52 */ view japanese
53static int session_included(request_rec * r, session_dir_conf * conf)
54{
55
56    const char **includes = (const char **) conf->includes->elts;
57    const char **excludes = (const char **) conf->excludes->elts;
58    int included = 1;                /* defaults to included */ view japanese
59    int i;
60
61    if (conf->includes->nelts) {
62        included = 0;
63        for (i = 0; !included && i < conf->includes->nelts; i++) {
64            const char *include = includes[i];
65            if (strncmp(r->uri, include, strlen(include)) == 0) {
66                included = 1;
67            }
68        }
69    }
70
71    if (conf->excludes->nelts) {
72        for (i = 0; included && i < conf->excludes->nelts; i++) {
73            const char *exclude = excludes[i];
74            if (strncmp(r->uri, exclude, strlen(exclude)) == 0) {
75                included = 0;
76            }
77        }
78    }
79
80    return included;
81}

83/**
84 * Load the session.
85 *
86 * If the session doesn't exist, a blank one will be created.
87 *
88 * @param r The request
89 * @param z A pointer to where the session will be written.
90 */ view japanese
91static apr_status_t ap_session_load(request_rec * r, session_rec ** z)
92{
93
94    session_dir_conf *dconf = ap_get_module_config(r->per_dir_config,
95                                                   &session_module);
96    apr_time_t now;
97    session_rec *zz = NULL;
98    int rv = 0;
99
100    /* is the session enabled? */ view japanese
101    if (!dconf || !dconf->enabled) {
102        return APR_SUCCESS;
103    }
104
105    /* should the session be loaded at all? */ view japanese
106    if (!session_included(r, dconf)) {
107        ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(01814)
108                      "excluded by configuration for: %s", r->uri);
109        return APR_SUCCESS;
110    }
111
112    /* load the session from the session hook */ view japanese
113    rv = ap_run_session_load(r, &zz);
114    if (DECLINED == rv) {
115        ap_log_rerror(APLOG_MARK, APLOG_WARNING, 0, r, APLOGNO(01815)
116                      "session is enabled but no session modules have been configured, "
117                      "session not loaded: %s", r->uri);
118        return APR_EGENERAL;
119    }
120    else if (OK != rv) {
121        ap_log_rerror(APLOG_MARK, APLOG_ERR, rv, r, APLOGNO(01816)
122                      "error while loading the session, "
123                      "session not loaded: %s", r->uri);
124        return rv;
125    }
126
127    /* found a session that hasn't expired? */ view japanese
128    now = apr_time_now();
129
130    if (zz) {
131        /* load the session attributes */ view japanese
132        rv = ap_run_session_decode(r, zz);
133 
134        /* having a session we cannot decode is just as good as having
135           none at all */ view japanese
136       if (OK != rv) {
137            ap_log_rerror(APLOG_MARK, APLOG_ERR, rv, r, APLOGNO(01817)
138                    "error while decoding the session, "
139                    "session not loaded: %s", r->uri);
140            zz = NULL;
141        }
142
143       /* invalidate session if session is expired */ view japanese
144        if (zz && zz->expiry && zz->expiry < now) {
145            ap_log_rerror(APLOG_MARK, APLOG_TRACE2, 0, r, "session is expired");
146            zz = NULL;
147        }
148    }
149
150    /* no luck, create a blank session */ view japanese
151    if (!zz) {
152        zz = (session_rec *) apr_pcalloc(r->pool, sizeof(session_rec));
153        zz->pool = r->pool;
154        zz->entries = apr_table_make(zz->pool, 10);
155    }
156
157    /* make sure the expiry and maxage are set, if present */ view japanese
158    if (dconf->maxage) {
159        if (!zz->expiry) {
160            zz->expiry = now + dconf->maxage * APR_USEC_PER_SEC;
161        }
162        zz->maxage = dconf->maxage;
163    }
164
165    *z = zz;
166
167    return APR_SUCCESS;
168
169}

171/**
172 * Save the session.
173 *
174 * In most implementations the session is only saved if the dirty flag is
175 * true. This prevents the session being saved unnecessarily.
176 *
177 * @param r The request
178 * @param z A pointer to where the session will be written.
179 */ view japanese
180static apr_status_t ap_session_save(request_rec * r, session_rec * z)
181{
182    if (z) {
183        apr_time_t now = apr_time_now();
184        apr_time_t initialExpiry = z->expiry;
185        int rv = 0;
186
187        session_dir_conf *dconf = ap_get_module_config(r->per_dir_config,
188                                                       &session_module);
189
190        /* sanity checks, should we try save at all? */ view japanese
191        if (z->written) {
192            ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(01818)
193                          "attempt made to save the session twice, "
194                          "session not saved: %s", r->uri);
195            return APR_EGENERAL;
196        }
197        if (z->expiry && z->expiry < now) {
198            ap_log_rerror(APLOG_MARK, APLOG_WARNING, 0, r, APLOGNO(01819)
199                          "attempt made to save a session when the session had already expired, "
200                          "session not saved: %s", r->uri);
201            return APR_EGENERAL;
202        }
203
204        /* reset the expiry back to maxage, if the expiry is present */ view japanese
205        if (dconf->maxage) {
206            z->expiry = now + dconf->maxage * APR_USEC_PER_SEC;
207            z->maxage = dconf->maxage;
208        }
209
210        /* reset the expiry before saving if present */ view japanese
211        if (z->dirty && z->maxage) {
212            z->expiry = now + z->maxage * APR_USEC_PER_SEC;
213        } 
214
215        /* don't save if the only change is the expiry by a small amount */ view japanese
216        if (!z->dirty && dconf->expiry_update_time
217                && (z->expiry - initialExpiry < dconf->expiry_update_time)) {
218            return APR_SUCCESS;
219        }
220
221        /* also don't save sessions that didn't change at all */ view japanese
222        if (!z->dirty && !z->maxage) {
223            return APR_SUCCESS;
224        }
225
226        /* encode the session */ view japanese
227        rv = ap_run_session_encode(r, z);
228        if (OK != rv) {
229            ap_log_rerror(APLOG_MARK, APLOG_ERR, rv, r, APLOGNO(01820)
230                          "error while encoding the session, "
231                          "session not saved: %s", r->uri);
232            return rv;
233        }
234
235        /* try the save */ view japanese
236        rv = ap_run_session_save(r, z);
237        if (DECLINED == rv) {
238            ap_log_rerror(APLOG_MARK, APLOG_WARNING, 0, r, APLOGNO(01821)
239                          "session is enabled but no session modules have been configured, "
240                          "session not saved: %s", r->uri);
241            return APR_EGENERAL;
242        }
243        else if (OK != rv) {
244            ap_log_rerror(APLOG_MARK, APLOG_ERR, rv, r, APLOGNO(01822)
245                          "error while saving the session, "
246                          "session not saved: %s", r->uri);
247            return rv;
248        }
249        else {
250            z->written = 1;
251        }
252    }
253
254    return APR_SUCCESS;
255
256}

258/**
259 * Get a particular value from the session.
260 * @param r The current request.
261 * @param z The current session. If this value is NULL, the session will be
262 * looked up in the request, created if necessary, and saved to the request
263 * notes.
264 * @param key The key to get.
265 * @param value The buffer to write the value to.
266 */ view japanese
267static apr_status_t ap_session_get(request_rec * r, session_rec * z,
268        const char *key, const char **value)
269{
270    if (!z) {
271        apr_status_t rv;
272        rv = ap_session_load(r, &z);
273        if (APR_SUCCESS != rv) {
274            return rv;
275        }
276    }
277    if (z && z->entries) {
278        *value = apr_table_get(z->entries, key);
279    }
280
281    return OK;
282}

284/**
285 * Set a particular value to the session.
286 *
287 * Using this method ensures that the dirty flag is set correctly, so that
288 * the session can be saved efficiently.
289 * @param r The current request.
290 * @param z The current session. If this value is NULL, the session will be
291 * looked up in the request, created if necessary, and saved to the request
292 * notes.
293 * @param key The key to set. The existing key value will be replaced.
294 * @param value The value to set.
295 */ view japanese
296static apr_status_t ap_session_set(request_rec * r, session_rec * z,
297        const char *key, const char *value)
298{
299    if (!z) {
300        apr_status_t rv;
301        rv = ap_session_load(r, &z);
302        if (APR_SUCCESS != rv) {
303            return rv;
304        }
305    }
306    if (z) {
307        if (value) {
308            apr_table_set(z->entries, key, value);
309        }
310        else {
311            apr_table_unset(z->entries, key);
312        }
313        z->dirty = 1;
314    }
315    return APR_SUCCESS;
316}

318static int identity_count(void *v, const char *key, const char *val)
319{
320    apr_size_t *count = v;
321
322    *count += strlen(key) * 3 + strlen(val) * 3 + 2;
323    return 1;
324}
325
326static int identity_concat(void *v, const char *key, const char *val)
327{
328    char *slider = v;
329    apr_size_t length = strlen(slider);
330
331    slider += length;
332    if (length) {
333        *slider = '&';
334        slider++;
335    }
336    ap_escape_urlencoded_buffer(slider, key);
337    slider += strlen(slider);
338    *slider = '=';
339    slider++;
340    ap_escape_urlencoded_buffer(slider, val);
341    return 1;
342}

344/**
345 * Default identity encoding for the session.
346 *
347 * By default, the name value pairs in the session are URLEncoded, separated
348 * by equals, and then in turn separated by ampersand, in the format of an
349 * html form.
350 *
351 * This was chosen to make it easy for external code to unpack a session,
352 * should there be a need to do so.
353 *
354 * @param r The request pointer.
355 * @param z A pointer to where the session will be written.
356 */ view japanese
357static apr_status_t session_identity_encode(request_rec * r, session_rec * z)
358{
359    char *buffer = NULL;
360    apr_size_t length = 0;
361
362    if (z->expiry) {
363        char *expiry = apr_psprintf(z->pool, "%" APR_INT64_T_FMT, z->expiry);
364        apr_table_setn(z->entries, SESSION_EXPIRY, expiry);
365    }
366    apr_table_do(identity_count, &length, z->entries, NULL);
367    buffer = apr_pcalloc(r->pool, length + 1);
368    apr_table_do(identity_concat, buffer, z->entries, NULL);
369    z->encoded = buffer;
370    return OK;
371
372}

374/**
375 * Default identity decoding for the session.
376 *
377 * By default, the name value pairs in the session are URLEncoded, separated
378 * by equals, and then in turn separated by ampersand, in the format of an
379 * html form.
380 *
381 * This was chosen to make it easy for external code to unpack a session,
382 * should there be a need to do so.
383 *
384 * This function reverses that process, and populates the session table.
385 *
386 * Name / value pairs that are not encoded properly are ignored.
387 *
388 * @param r The request pointer.
389 * @param z A pointer to where the session will be written.
390 */ view japanese
391static apr_status_t session_identity_decode(request_rec * r, session_rec * z)
392{
393
394    char *last = NULL;
395    char *encoded, *pair;
396    const char *sep = "&";
397
398    /* sanity check - anything to decode? */ view japanese
399    if (!z->encoded) {
400        return OK;
401    }
402
403    /* decode what we have */ view japanese
404    encoded = apr_pstrdup(r->pool, z->encoded);
405    pair = apr_strtok(encoded, sep, &last);
406    while (pair && pair[0]) {
407        char *plast = NULL;
408        const char *psep = "=";
409        char *key = apr_strtok(pair, psep, &plast);
410        if (key && *key) {
411            char *val = apr_strtok(NULL, sep, &plast);
412            if (!val || !*val) {
413                apr_table_unset(z->entries, key);
414            }
415            else if (!ap_unescape_urlencoded(key) && !ap_unescape_urlencoded(val)) {
416                if (!strcmp(SESSION_EXPIRY, key)) {
417                    z->expiry = (apr_time_t) apr_atoi64(val);
418                }
419                else {
420                    apr_table_set(z->entries, key, val);
421                }
422            }
423        }
424        pair = apr_strtok(NULL, sep, &last);
425    }
426    z->encoded = NULL;
427    return OK;
428
429}

431/**
432 * Ensure any changes to the session are committed.
433 *
434 * This is done in an output filter so that our options for where to
435 * store the session can include storing the session within a cookie:
436 * As an HTTP header, the cookie must be set before the output is
437 * written, but after the handler is run.
438 *
439 * NOTE: It is possible for internal redirects to cause more than one
440 * request to be present, and each request might have a session
441 * defined. We need to go through each session in turn, and save each
442 * one.
443 *
444 * The same session might appear in more than one request. The first
445 * attempt to save the session will be called
446 */ view japanese
447static apr_status_t session_output_filter(ap_filter_t * f,
448        apr_bucket_brigade * in)
449{
450
451    /* save all the sessions in all the requests */ view japanese
452    request_rec *r = f->r->main;
453    if (!r) {
454        r = f->r;
455    }
456    while (r) {
457        session_rec *z = NULL;
458        session_dir_conf *conf = ap_get_module_config(r->per_dir_config,
459                                                      &session_module);
460
461        /* load the session, or create one if necessary */ view japanese
462        /* when unset or on error, z will be NULL */ view japanese
463        ap_session_load(r, &z);
464        if (!z || z->written) {
465            r = r->next;
466            continue;
467        }
468
469        /* if a header was specified, insert the new values from the header */ view japanese
470        if (conf->header_set) {
471            const char *override = apr_table_get(r->err_headers_out, conf->header);
472            if (!override) {
473                override = apr_table_get(r->headers_out, conf->header);
474            }
475            if (override) {
476                apr_table_unset(r->err_headers_out, conf->header);
477                apr_table_unset(r->headers_out, conf->header);
478                z->encoded = override;
479                z->dirty = 1;
480                session_identity_decode(r, z);
481            }
482        }
483
484        /* save away the session, and we're done */ view japanese
485        /* when unset or on error, we've complained to the log */ view japanese
486        ap_session_save(r, z);
487
488        r = r->next;
489    }
490
491    /* remove ourselves from the filter chain */ view japanese
492    ap_remove_output_filter(f);
493
494    /* send the data up the stack */ view japanese
495    return ap_pass_brigade(f->next, in);
496
497}

507/**
508 * Fixups hook.
509 *
510 * Load the session within a fixup - this ensures that the session is
511 * properly loaded prior to the handler being called.
512 *
513 * The fixup is also responsible for injecting the session into the CGI
514 * environment, should the admin have configured it so.
515 *
516 * @param r The request
517 */ view japanese
518static int session_fixups(request_rec * r)
519{
520    session_dir_conf *conf = ap_get_module_config(r->per_dir_config,
521                                                  &session_module);
522
523    session_rec *z = NULL;
524
525    /* if an error occurs or no session has been configured, we ignore
526     * the broken session and allow it to be recreated from scratch on save
527     * if necessary.
528     */ view japanese
529    ap_session_load(r, &z);
530
531    if (conf->env) {
532        if (z) {
533            session_identity_encode(r, z);
534            if (z->encoded) {
535                apr_table_set(r->subprocess_env, HTTP_SESSION, z->encoded);
536                z->encoded = NULL;
537            }
538        }
539        apr_table_unset(r->headers_in, "Session");
540    }
541
542    return OK;
543
544}

547static void *create_session_dir_config(apr_pool_t * p, char *dummy)
548{
549    session_dir_conf *new =
550    (session_dir_conf *) apr_pcalloc(p, sizeof(session_dir_conf));
551
552    new->includes = apr_array_make(p, 10, sizeof(const char **));
553    new->excludes = apr_array_make(p, 10, sizeof(const char **));
554
555    return (void *) new;
556}
557
558static void *merge_session_dir_config(apr_pool_t * p, void *basev, void *addv)
559{
560    session_dir_conf *new = (session_dir_conf *) apr_pcalloc(p, sizeof(session_dir_conf));
561    session_dir_conf *add = (session_dir_conf *) addv;
562    session_dir_conf *base = (session_dir_conf *) basev;
563
564    new->enabled = (add->enabled_set == 0) ? base->enabled : add->enabled;
565    new->enabled_set = add->enabled_set || base->enabled_set;
566    new->maxage = (add->maxage_set == 0) ? base->maxage : add->maxage;
567    new->maxage_set = add->maxage_set || base->maxage_set;
568    new->header = (add->header_set == 0) ? base->header : add->header;
569    new->header_set = add->header_set || base->header_set;
570    new->env = (add->env_set == 0) ? base->env : add->env;
571    new->env_set = add->env_set || base->env_set;
572    new->includes = apr_array_append(p, base->includes, add->includes);
573    new->excludes = apr_array_append(p, base->excludes, add->excludes);
574    new->expiry_update_time = (add->expiry_update_set == 0)
575                                ? base->expiry_update_time
576                                : add->expiry_update_time;
577    new->expiry_update_set = add->expiry_update_set || base->expiry_update_set;
578
579    return new;
580}
