54/**
55 * Initialise the database.
56 *
57 * If the mod_dbd module is missing, this method will return APR_EGENERAL.
58 */ view japanese
59static apr_status_t dbd_init(request_rec *r, const char *query, ap_dbd_t **dbdp,
60                             apr_dbd_prepared_t **statementp)
61{
62    ap_dbd_t *dbd;
63    apr_dbd_prepared_t *statement;
64
65    if (!session_dbd_prepare_fn || !session_dbd_acquire_fn) {
66        session_dbd_prepare_fn = APR_RETRIEVE_OPTIONAL_FN(ap_dbd_prepare);
67        session_dbd_acquire_fn = APR_RETRIEVE_OPTIONAL_FN(ap_dbd_acquire);
68        if (!session_dbd_prepare_fn || !session_dbd_acquire_fn) {
69            ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(01850)
70                          "You must load mod_dbd to enable AuthDBD functions");
71            return APR_EGENERAL;
72        }
73    }
74
75    dbd = session_dbd_acquire_fn(r);
76    if (!dbd) {
77        ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(01851)
78                      "failed to acquire database connection");
79        return APR_EGENERAL;
80    }
81
82    statement = apr_hash_get(dbd->prepared, query, APR_HASH_KEY_STRING);
83    if (!statement) {
84        ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(01852)
85                      "failed to find the prepared statement called '%s'", query);
86        return APR_EGENERAL;
87    }
88
89    *dbdp = dbd;
90    *statementp = statement;
91
92    return APR_SUCCESS;
93}

95/**
96 * Load the session by the key specified.
97 *
98 * The session value is allocated using the passed apr_pool_t.
99 */ view japanese
100static apr_status_t dbd_load(apr_pool_t *p, request_rec * r,
101                             const char *key, const char **val)
102{
103
104    apr_status_t rv;
105    ap_dbd_t *dbd = NULL;
106    apr_dbd_prepared_t *statement = NULL;
107    apr_dbd_results_t *res = NULL;
108    apr_dbd_row_t *row = NULL;
109    apr_int64_t expiry = (apr_int64_t) apr_time_now();
110
111    session_dbd_dir_conf *conf = ap_get_module_config(r->per_dir_config,
112                                                      &session_dbd_module);
113
114    if (conf->selectlabel == NULL) {
115        ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(01853)
116                      "no SessionDBDselectlabel has been specified");
117        return APR_EGENERAL;
118    }
119
120    rv = dbd_init(r, conf->selectlabel, &dbd, &statement);
121    if (rv) {
122        return rv;
123    }
124    rv = apr_dbd_pvbselect(dbd->driver, r->pool, dbd->handle, &res, statement,
125                          0, key, &expiry, NULL);
126    if (rv) {
127        ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(01854)
128                      "query execution error saving session '%s' "
129                      "in database using query '%s': %s", key, conf->selectlabel,
130                      apr_dbd_error(dbd->driver, dbd->handle, rv));
131        return APR_EGENERAL;
132    }
133    for (rv = apr_dbd_get_row(dbd->driver, r->pool, res, &row, -1);
134         rv != -1;
135         rv = apr_dbd_get_row(dbd->driver, r->pool, res, &row, -1)) {
136        if (rv != 0) {
137            ap_log_rerror(APLOG_MARK, APLOG_ERR, rv, r, APLOGNO(01855)
138                          "error retrieving results while saving '%s' "
139                          "in database using query '%s': %s", key, conf->selectlabel,
140                           apr_dbd_error(dbd->driver, dbd->handle, rv));
141            return APR_EGENERAL;
142        }
143        if (*val == NULL) {
144            *val = apr_pstrdup(p, apr_dbd_get_entry(dbd->driver, row, 0));
145        }
146        /* we can't break out here or row won't get cleaned up */ view japanese
147    }
148
149    return APR_SUCCESS;
150
151}

153/**
154 * Load the session by firing off a dbd query.
155 *
156 * If the session is anonymous, the session key will be extracted from
157 * the cookie specified. Failing that, the session key will be extracted
158 * from the GET parameters.
159 *
160 * If the session is keyed by the username, the session will be extracted
161 * by that.
162 *
163 * If no session is found, an empty session will be created.
164 *
165 * On success, this returns OK.
166 */ view japanese
167static apr_status_t session_dbd_load(request_rec * r, session_rec ** z)
168{
169
170    session_dbd_dir_conf *conf = ap_get_module_config(r->per_dir_config,
171                                                      &session_dbd_module);
172
173    apr_status_t ret = APR_SUCCESS;
174    session_rec *zz = NULL;
175    const char *name = NULL;
176    const char *note = NULL;
177    const char *val = NULL;
178    const char *key = NULL;
179    request_rec *m = r->main ? r->main : r;
180
181    /* is our session in a cookie? */ view japanese
182    if (conf->name2_set) {
183        name = conf->name2;
184    }
185    else if (conf->name_set) {
186        name = conf->name;
187    }
188    else if (conf->peruser_set && r->user) {
189        name = r->user;
190    }
191    else {
192        return DECLINED;
193    }
194
195    /* first look in the notes */ view japanese
196    note = apr_pstrcat(m->pool, MOD_SESSION_DBD, name, NULL);
197    zz = (session_rec *)apr_table_get(m->notes, note);
198    if (zz) {
199        *z = zz;
200        return OK;
201    }
202
203    /* load anonymous sessions */ view japanese
204    if (conf->name_set || conf->name2_set) {
205
206        /* load an RFC2109 or RFC2965 compliant cookie */ view japanese
207        ap_cookie_read(r, name, &key, conf->remove);
208        if (key) {
209            ret = dbd_load(m->pool, r, key, &val);
210            if (ret != APR_SUCCESS) {
211                return ret;
212            }
213        }
214
215    }
216
217    /* load named session */ view japanese
218    else if (conf->peruser) {
219        if (r->user) {
220            ret = dbd_load(m->pool, r, r->user, &val);
221            if (ret != APR_SUCCESS) {
222                return ret;
223            }
224        }
225    }
226
227    /* otherwise not for us */ view japanese
228    else {
229        return DECLINED;
230    }
231
232    /* create a new session and return it */ view japanese
233    zz = (session_rec *) apr_pcalloc(m->pool, sizeof(session_rec));
234    zz->pool = m->pool;
235    zz->entries = apr_table_make(zz->pool, 10);
236    if (key && val) {
237        apr_uuid_t *uuid = apr_pcalloc(zz->pool, sizeof(apr_uuid_t));
238        if (APR_SUCCESS == apr_uuid_parse(uuid, key)) {
239            zz->uuid = uuid;
240        }
241    }
242    zz->encoded = val;
243    *z = zz;
244
245    /* put the session in the notes so we don't have to parse it again */ view japanese
246    apr_table_setn(m->notes, note, (char *)zz);
247
248    /* don't cache pages with a session */
249    apr_table_addn(r->headers_out, "Cache-Control", "no-cache, private");
250
251    return OK;
252
253}

255/**
256 * Save the session by the key specified.
257 */ view japanese
258static apr_status_t dbd_save(request_rec * r, const char *oldkey,
259        const char *newkey, const char *val, apr_int64_t expiry)
260{
261
262    apr_status_t rv;
263    ap_dbd_t *dbd = NULL;
264    apr_dbd_prepared_t *statement;
265    int rows = 0;
266
267    session_dbd_dir_conf *conf = ap_get_module_config(r->per_dir_config,
268                                                      &session_dbd_module);
269
270    if (conf->updatelabel == NULL) {
271        ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(01856)
272                      "no SessionDBDupdatelabel has been specified");
273        return APR_EGENERAL;
274    }
275
276    rv = dbd_init(r, conf->updatelabel, &dbd, &statement);
277    if (rv) {
278        return rv;
279    }
280
281    if (oldkey) {
282        rv = apr_dbd_pvbquery(dbd->driver, r->pool, dbd->handle, &rows,
283                statement, val, &expiry, newkey, oldkey, NULL);
284        if (rv) {
285            ap_log_rerror(
286                    APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(01857) "query execution error updating session '%s' "
287                    "using database query '%s': %s/%s", oldkey, newkey, conf->updatelabel, apr_dbd_error(dbd->driver, dbd->handle, rv));
288            return APR_EGENERAL;
289        }
290
291        /*
292         * if some rows were updated it means a session existed and was updated,
293         * so we are done.
294         */ view japanese
295        if (rows != 0) {
296            return APR_SUCCESS;
297        }
298    }
299
300    if (conf->insertlabel == NULL) {
301        ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(01858)
302                      "no SessionDBDinsertlabel has been specified");
303        return APR_EGENERAL;
304    }
305
306    rv = dbd_init(r, conf->insertlabel, &dbd, &statement);
307    if (rv) {
308        return rv;
309    }
310    rv = apr_dbd_pvbquery(dbd->driver, r->pool, dbd->handle, &rows, statement,
311                          val, &expiry, newkey, NULL);
312    if (rv) {
313        ap_log_rerror(APLOG_MARK, APLOG_ERR, rv, r, APLOGNO(01859)
314                      "query execution error inserting session '%s' "
315                      "in database with '%s': %s", newkey, conf->insertlabel,
316                      apr_dbd_error(dbd->driver, dbd->handle, rv));
317        return APR_EGENERAL;
318    }
319
320    /*
321     * if some rows were inserted it means a session was inserted, so we are
322     * done.
323     */ view japanese
324    if (rows != 0) {
325        return APR_SUCCESS;
326    }
327
328    ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(01860)
329                  "the session insert query did not cause any rows to be added "
330                  "to the database for session '%s', session not inserted", newkey);
331
332    return APR_EGENERAL;
333
334}

336/**
337 * Remove the session by the key specified.
338 */ view japanese
339static apr_status_t dbd_remove(request_rec * r, const char *key)
340{
341
342    apr_status_t rv;
343    ap_dbd_t *dbd;
344    apr_dbd_prepared_t *statement;
345    int rows = 0;
346
347    session_dbd_dir_conf *conf = ap_get_module_config(r->per_dir_config,
348                                                      &session_dbd_module);
349
350    if (conf->deletelabel == NULL) {
351        ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(01862)
352                      "no SessionDBDdeletelabel has been specified");
353        return APR_EGENERAL;
354    }
355
356    rv = dbd_init(r, conf->deletelabel, &dbd, &statement);
357    if (rv != APR_SUCCESS) {
358        /* No need to do additional error logging here, it has already
359           been done in dbd_init if needed */ view japanese
360        return rv;
361    }
362
363    rv = apr_dbd_pvbquery(dbd->driver, r->pool, dbd->handle, &rows, statement,
364                          key, NULL);
365    if (rv != APR_SUCCESS) {
366        ap_log_rerror(APLOG_MARK, APLOG_ERR, rv, r, APLOGNO(01864)
367                      "query execution error removing session '%s' "
368                      "from database", key);
369        return rv;
370    }
371
372    return APR_SUCCESS;
373
374}

390/**
391 * Save the session by firing off a dbd query.
392 *
393 * If the session is anonymous, save the session and write a cookie
394 * containing the uuid.
395 *
396 * If the session is keyed to the username, save the session using
397 * the username as a key.
398 *
399 * On success, this method will return APR_SUCCESS.
400 *
401 * @param r The request pointer.
402 * @param z A pointer to where the session will be written.
403 */ view japanese
404static apr_status_t session_dbd_save(request_rec * r, session_rec * z)
405{
406
407    apr_status_t ret = APR_SUCCESS;
408    session_dbd_dir_conf *conf = ap_get_module_config(r->per_dir_config,
409                                                      &session_dbd_module);
410
411    /* support anonymous sessions */ view japanese
412    if (conf->name_set || conf->name2_set) {
413        char *oldkey = NULL, *newkey = NULL;
414
415        /* if the session is new or changed, make a new session ID */ view japanese
416        if (z->uuid) {
417            oldkey = apr_pcalloc(r->pool, APR_UUID_FORMATTED_LENGTH + 1);
418            apr_uuid_format(oldkey, z->uuid);
419        }
420        if (z->dirty || !oldkey) {
421            z->uuid = apr_pcalloc(z->pool, sizeof(apr_uuid_t));
422            apr_uuid_get(z->uuid);
423            newkey = apr_pcalloc(r->pool, APR_UUID_FORMATTED_LENGTH + 1);
424            apr_uuid_format(newkey, z->uuid);
425        }
426        else {
427            newkey = oldkey;
428        }
429
430        /* save the session with the uuid as key */ view japanese
431        if (z->encoded && z->encoded[0]) {
432            ret = dbd_save(r, oldkey, newkey, z->encoded, z->expiry);
433        }
434        else {
435            ret = dbd_remove(r, oldkey);
436        }
437        if (ret != APR_SUCCESS) {
438            return ret;
439        }
440
441        /* create RFC2109 compliant cookie */ view japanese
442        if (conf->name_set) {
443            ap_cookie_write(r, conf->name, newkey, conf->name_attrs, z->maxage,
444                            r->headers_out, r->err_headers_out, NULL);
445        }
446
447        /* create RFC2965 compliant cookie */ view japanese
448        if (conf->name2_set) {
449            ap_cookie_write2(r, conf->name2, newkey, conf->name2_attrs, z->maxage,
450                             r->headers_out, r->err_headers_out, NULL);
451        }
452
453        return OK;
454
455    }
456
457    /* save named session */ view japanese
458    else if (conf->peruser) {
459
460        /* don't cache pages with a session */ view japanese
461        apr_table_addn(r->headers_out, "Cache-Control", "no-cache, private");
462
463        if (r->user) {
464            ret = dbd_save(r, r->user, r->user, z->encoded, z->expiry);
465            if (ret != APR_SUCCESS) {
466                return ret;
467            }
468            return OK;
469        }
470        else {
471            ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(01865)
472               "peruser sessions can only be saved if a user is logged in, "
473                          "session not saved: %s", r->uri);
474        }
475    }
476
477    return DECLINED;
478
479}

494{
495    session_dbd_dir_conf *new =
496    (session_dbd_dir_conf *) apr_pcalloc(p, sizeof(session_dbd_dir_conf));

500    new->selectlabel = "selectsession";

504

508static void *merge_session_dbd_dir_config(apr_pool_t * p, void *basev, void *addv)
509{
510    session_dbd_dir_conf *new = (session_dbd_dir_conf *) apr_pcalloc(p, sizeof(session_dbd_dir_conf));
511    session_dbd_dir_conf *add = (session_dbd_dir_conf *) addv;
512    session_dbd_dir_conf *base = (session_dbd_dir_conf *) basev;
513
514    new->name = (add->name_set == 0) ? base->name : add->name;
515    new->name_attrs = (add->name_set == 0) ? base->name_attrs : add->name_attrs;
516    new->name_set = add->name_set || base->name_set;
517    new->name2 = (add->name2_set == 0) ? base->name2 : add->name2;
518    new->name2_attrs = (add->name2_set == 0) ? base->name2_attrs : add->name2_attrs;
519    new->name2_set = add->name2_set || base->name2_set;
520    new->peruser = (add->peruser_set == 0) ? base->peruser : add->peruser;
521    new->peruser_set = add->peruser_set || base->peruser_set;
522    new->remove = (add->remove_set == 0) ? base->remove : add->remove;
523    new->remove_set = add->remove_set || base->remove_set;
524    new->selectlabel = (!add->selectlabel) ? base->selectlabel : add->selectlabel;
525    new->updatelabel = (!add->updatelabel) ? base->updatelabel : add->updatelabel;
526    new->insertlabel = (!add->insertlabel) ? base->insertlabel : add->insertlabel;
527    new->deletelabel = (!add->deletelabel) ? base->deletelabel : add->deletelabel;
528
529    return new;
530}

532/**
533 * Sanity check a given string that it exists, is not empty,
534 * and does not contain special characters.
535 */ view japanese
536static const char *check_string(cmd_parms * cmd, const char *string)
537{
538    if (APR_SUCCESS != ap_cookie_check_string(string)) {
539        return apr_pstrcat(cmd->pool, cmd->directive->directive,
540                           " cannot contain '=', ';' or '&'.",
541                           NULL);
542    }
543    return NULL;
544}

568static const char *set_cookie_name(cmd_parms * cmd, void *config, const char *args)
569{
570    char *last;
571    char *line = apr_pstrdup(cmd->pool, args);
572    session_dbd_dir_conf *conf = (session_dbd_dir_conf *) config;
573    char *cookie = apr_strtok(line, " \t", &last);
574    if (!cookie) {
575        return apr_pstrcat(cmd->pool, cmd->directive->directive,
576                           " requires at least one argument!",
577                           NULL);
578    }
579    conf->name = cookie;
580    conf->name_set = 1;
581    while (apr_isspace(*last)) {
582        last++;
583    }
584    conf->name_attrs = last;
585    return check_string(cmd, cookie);
586}
