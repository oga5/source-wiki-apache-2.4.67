17#ifndef __mod_h2__h2_headers__
18#define __mod_h2__h2_headers__
19
20#include "h2.h"
21
22#if !AP_HAS_RESPONSE_BUCKETS
23
24struct h2_bucket_beam;
25
26typedef struct h2_headers h2_headers;
27struct h2_headers {
28    int         status;
29    apr_table_t *headers;
30    apr_table_t *notes;
31    apr_off_t   raw_bytes;      /* RAW network bytes that generated this request - if known. */
32};
33
34
35extern const apr_bucket_type_t h2_bucket_type_headers;
36
37#define H2_BUCKET_IS_HEADERS(e)     (e->type == &h2_bucket_type_headers)
38
39apr_bucket * h2_bucket_headers_make(apr_bucket *b, h2_headers *r); 
40
41apr_bucket * h2_bucket_headers_create(apr_bucket_alloc_t *list, 
42                                       h2_headers *r);
43                                       
44h2_headers *h2_bucket_headers_get(apr_bucket *b);
45
46/**
47 * Create the headers from the given status and headers
48 * @param status the headers status
49 * @param header the headers of the headers
50 * @param notes  the notes carried by the headers
51 * @param raw_bytes the raw network bytes (if known) used to transmit these
52 * @param pool the memory pool to use
53 */
54h2_headers *h2_headers_create(int status, const apr_table_t *header,
55                              const apr_table_t *notes, apr_off_t raw_bytes,
56                              apr_pool_t *pool);
57
58/**
59 * Create the headers from the given request_rec.
60 * @param r the request record which was processed
61 * @param status the headers status
62 * @param header the headers of the headers
63 * @param pool the memory pool to use
64 */
65h2_headers *h2_headers_rcreate(request_rec *r, int status, 
66                               const apr_table_t *header, apr_pool_t *pool);
67
68/**
69 * Copy the headers into another pool. This will not copy any
70 * header strings.
71 */
72h2_headers *h2_headers_copy(apr_pool_t *pool, h2_headers *h);
73
74/**
75 * Clone the headers into another pool. This will also clone any
76 * header strings.
77 */
78h2_headers *h2_headers_clone(apr_pool_t *pool, h2_headers *h);
79
80/**
81 * Create the headers for the given error.
82 * @param type the error code
83 * @param req the original h2_request
84 * @param pool the memory pool to use
85 */
86h2_headers *h2_headers_die(apr_status_t type,
87                           const struct h2_request *req, apr_pool_t *pool);
88
89int h2_headers_are_final_response(h2_headers *headers);
90
91/**
92 * Give the number of bytes of all contained header strings.
93 */
94apr_size_t h2_headers_length(h2_headers *headers);
95
96/**
97 * For H2HEADER buckets, return the length of all contained header strings.
98 * For all other buckets, return 0.
99 */
100apr_size_t h2_bucket_headers_headers_length(apr_bucket *b);
101
102apr_bucket *h2_bucket_headers_clone(apr_bucket *b, apr_pool_t *pool,
103                                    apr_bucket_alloc_t *list);
104
105#endif /* !AP_HAS_RESPONSE_BUCKETS */
106
107#endif /* defined(__mod_h2__h2_headers__) */
