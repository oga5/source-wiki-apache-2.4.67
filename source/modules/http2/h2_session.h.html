22/**
23 * A HTTP/2 connection, a session with a specific client.
24 * 
25 * h2_session sits on top of a httpd conn_rec* instance and takes complete
26 * control of the connection data. It receives protocol frames from the
27 * client. For new HTTP/2 streams it creates secondary connections
28 * to execute the requests in h2 workers.
29 */
30

50typedef enum {
51    H2_SESSION_EV_INIT,             /* session was initialized */
52    H2_SESSION_EV_INPUT_PENDING,    /* c1 input may have data pending */
53    H2_SESSION_EV_INPUT_EXHAUSTED,  /* c1 input exhausted */
54    H2_SESSION_EV_LOCAL_GOAWAY,     /* we send a GOAWAY */
55    H2_SESSION_EV_REMOTE_GOAWAY,    /* remote send us a GOAWAY */
56    H2_SESSION_EV_CONN_ERROR,       /* connection error */
57    H2_SESSION_EV_PROTO_ERROR,      /* protocol error */
58    H2_SESSION_EV_CONN_TIMEOUT,     /* connection timeout */
59    H2_SESSION_EV_NGH2_DONE,        /* nghttp2 wants neither read nor write anything */
60    H2_SESSION_EV_MPM_STOPPING,     /* the process is stopping */
61    H2_SESSION_EV_PRE_CLOSE,        /* connection will close after this */
62    H2_SESSION_EV_NO_MORE_STREAMS,  /* no more streams to process */
63    H2_SESSION_EV_BAD_CLIENT,       /* client misbehaving badly */
64} h2_session_event_t;
65
66typedef struct h2_session {
67    int child_num;                  /* child number this session runs in */
68    apr_uint32_t id;                /* identifier of this session, unique per child */
69    conn_rec *c1;                   /* the main connection this session serves */
70    request_rec *r;                 /* the request that started this in case
71                                     * of 'h2c', NULL otherwise */
72    server_rec *s;                  /* server/vhost we're starting on */
73    apr_pool_t *pool;               /* pool to use in session */
74    struct h2_mplx *mplx;           /* multiplexer for stream data */
75    struct h2_workers *workers;     /* for executing streams */
76    struct h2_c1_io_in_ctx_t *cin;  /* connection input filter context */
77    h2_c1_io io;                    /* io on httpd conn filters */
78    unsigned int padding_max;       /* max number of padding bytes */
79    int padding_always;             /* padding has precedence over I/O optimizations */
80    struct nghttp2_session *ngh2;   /* the nghttp2 session (internal use) */
81
82    h2_session_state state;         /* state session is in */
83    
84    h2_session_props local;         /* properties of local session */
85    h2_session_props remote;        /* properites of remote session */
86    
87    unsigned int reprioritize  : 1; /* scheduled streams priority changed */
88    unsigned int flush         : 1; /* flushing output necessary */
89    apr_interval_time_t  wait_us;   /* timeout during BUSY_WAIT state, micro secs */
90    
91    struct h2_push_diary *push_diary; /* remember pushes, avoid duplicates */
92    
93    struct h2_stream_monitor *monitor;/* monitor callbacks for streams */
94    unsigned int open_streams;      /* number of streams processing */
95
96    unsigned int streams_done;      /* number of http/2 streams handled */
97    unsigned int responses_submitted; /* number of http/2 responses submitted */
98    unsigned int streams_reset;     /* number of http/2 streams reset by client */
99    unsigned int pushes_promised;   /* number of http/2 push promises submitted */
100    unsigned int pushes_submitted;  /* number of http/2 pushed responses submitted */
101    unsigned int pushes_reset;      /* number of http/2 pushed reset by client */
102    unsigned int max_stream_errors; /* max client stream errors tolerated */
103    unsigned int stream_errors;     /* number of stream errors by client */
104
105    apr_size_t frames_received;     /* number of http/2 frames received */
106    apr_size_t frames_sent;         /* number of http/2 frames sent */
107    
108    apr_size_t max_stream_count;    /* max number of open streams */
109    apr_size_t max_stream_mem;      /* max buffer memory for a single stream */
110    apr_size_t max_data_frame_len;  /* max amount of bytes for a single DATA frame */
111
112    apr_size_t idle_frames;         /* number of rcvd frames that kept session in idle state */
113    apr_interval_time_t idle_delay; /* Time we delay processing rcvd frames in idle state */
114    
115    apr_bucket_brigade *bbtmp;      /* brigade for keeping temporary data */
116
117    char status[64];                /* status message for scoreboard */
118    int last_status_code;           /* the one already reported */
119    const char *last_status_msg;    /* the one already reported */
120
121    int input_flushed;              /* stream input was flushed */
122    struct h2_iqueue *out_c1_blocked;  /* all streams with output blocked on c1 buffer full */
123    struct h2_iqueue *ready_to_process;  /* all streams ready for processing */
124
125    h2_hd_scratch hd_scratch;
126
127} h2_session;

129const char *h2_session_state_str(h2_session_state state);

131/**
132 * Create a new h2_session for the given connection.
133 * The session will apply the configured parameter.
134 * @param psession pointer receiving the created session on success or NULL
135 * @param c       the connection to work on
136 * @param r       optional request when protocol was upgraded
137 * @param cfg     the module config to apply
138 * @param workers the worker pool to use
139 * @return the created session
140 */
141apr_status_t h2_session_create(h2_session **psession,
142                               conn_rec *c, request_rec *r, server_rec *, 
143                               struct h2_workers *workers);
144
145void h2_session_event(h2_session *session, h2_session_event_t ev, 
146                      int err, const char *msg);
147
148/**
149 * Process the given HTTP/2 session until it is ended or a fatal
150 * error occurred.
151 *
152 * @param session the sessionm to process
153 * @param async if mpm is async
154 * @param pkeepalive on return, != 0 if connection to be put into keepalive
155 *                   behaviour and timouts
156 */
157apr_status_t h2_session_process(h2_session *session, int async, int *pkeepalive);
158
159/**
160 * Last chance to do anything before the connection is closed.
161 */
162apr_status_t h2_session_pre_close(h2_session *session, int async);
163
164/**
165 * Called when a serious error occurred and the session needs to terminate
166 * without further connection io.
167 * @param session the session to abort
168 * @param reason  the apache status that caused the abort
169 */
170void h2_session_abort(h2_session *session, apr_status_t reason);

172/**
173 * Returns if client settings have push enabled.
174 * @param != 0 iff push is enabled in client settings

176int h2_session_push_enabled(h2_session *session);
