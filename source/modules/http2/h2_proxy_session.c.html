70/*
71 * The H2_PING connection sub-state: a state independant of the H2_SESSION state
72 * of the connection:
73 * - H2_PING_ST_NONE: no interference with request handling, ProxyTimeout in effect.
74 *   When entered, all suspended streams are unsuspended again.
75 * - H2_PING_ST_AWAIT_ANY: new requests are suspended, a possibly configured "ping"
76 *   timeout is in effect. Any frame received transits to H2_PING_ST_NONE.
77 * - H2_PING_ST_AWAIT_PING: same as above, but only a PING frame transits 
78 *   to H2_PING_ST_NONE.
79 *
80 * An AWAIT state is entered on a new connection or when re-using a connection and
81 * the last frame received has been some time ago. The latter sends a PING frame
82 * and insists on an answer, the former is satisfied by any frame received from the
83 * backend.
84 *
85 * This works for new connections as there is always at least one SETTINGS frame
86 * that the backend sends. When re-using connection, we send a PING and insist on
87 * receiving one back, as there might be frames in our connection buffers from
88 * some time ago. Since some servers have protections against PING flooding, we
89 * only ever have one PING unanswered.
90 *
91 * Requests are suspended while in a PING state, as we do not want to send data
92 * before we can be reasonably sure that the connection is working (at least on
93 * the h2 protocol level). This also means that the session can do blocking reads
94 * when expecting PING answers.
95 */ view japanese
96static void set_ping_timeout(h2_proxy_session *session)
97{
98    if (session->ping_timeout != -1 && session->save_timeout == -1) {
99        apr_socket_t *socket = NULL;
100
101        socket = ap_get_conn_socket(session->c);
102        if (socket) {
103            apr_socket_timeout_get(socket, &session->save_timeout);
104            apr_socket_timeout_set(socket, session->ping_timeout);
105        }
106    }
107}

109static void unset_ping_timeout(h2_proxy_session *session)
110{
111    if (session->save_timeout != -1) {
112        apr_socket_t *socket = NULL;
113
114        socket = ap_get_conn_socket(session->c);
115        if (socket) {
116            apr_socket_timeout_set(socket, session->save_timeout);
117            session->save_timeout = -1;
118        }
119    }
120}

122static void enter_ping_state(h2_proxy_session *session, h2_ping_state_t state)
123{
124    if (session->ping_state == state) return;
125    switch (session->ping_state) {
126    case H2_PING_ST_NONE:
127        /* leaving NONE, enforce timeout, send frame maybe */ view japanese
128        if (H2_PING_ST_AWAIT_PING == state) {
129            unset_ping_timeout(session);
130            nghttp2_submit_ping(session->ngh2, 0, (const uint8_t *)"nevergonnagiveyouup");
131        }
132        set_ping_timeout(session);
133        session->ping_state = state;
134        break;
135    default:
136        /* no switching between the != NONE states */ view japanese
137        if (H2_PING_ST_NONE == state) {
138            session->ping_state = state;
139            unset_ping_timeout(session);
140            ping_arrived(session);
141        }
142        break;
143    }
144}

146static void ping_new_session(h2_proxy_session *session, proxy_conn_rec *p_conn)
147{
148    session->save_timeout = -1;
149    session->ping_timeout = (p_conn->worker->s->ping_timeout_set?
150                             p_conn->worker->s->ping_timeout : -1);
151    session->ping_state = H2_PING_ST_NONE;
152    enter_ping_state(session, H2_PING_ST_AWAIT_ANY);
153}

155static void ping_reuse_session(h2_proxy_session *session)
156{
157    if (H2_PING_ST_NONE == session->ping_state) {
158        apr_interval_time_t age = apr_time_now() - session->last_frame_received;
159        if (age > apr_time_from_sec(1)) {
160            enter_ping_state(session, H2_PING_ST_AWAIT_PING);
161        }
162    }
163}

165static void ping_ev_frame_received(h2_proxy_session *session, const nghttp2_frame *frame)
166{
167    session->last_frame_received = apr_time_now();
168    switch (session->ping_state) {
169    case H2_PING_ST_NONE:
170        /* nop */
171        break;
172    case H2_PING_ST_AWAIT_ANY:
173        enter_ping_state(session, H2_PING_ST_NONE);
174        break;
175    case H2_PING_ST_AWAIT_PING:
176        if (NGHTTP2_PING == frame->hd.type) {
177            enter_ping_state(session, H2_PING_ST_NONE);
178        }
179        /* we may receive many other frames while we are waiting for the
180         * PING answer. They may come all from our connection buffers and
181         * say nothing about the current state of the backend. */ view japanese
182        break;
183    }
184}

186static apr_status_t proxy_session_pre_close(void *theconn)
187{
188    proxy_conn_rec *p_conn = (proxy_conn_rec *)theconn;
189    h2_proxy_session *session = p_conn->data;
190
191    if (session && session->ngh2) {
192        ap_log_cerror(APLOG_MARK, APLOG_TRACE1, 0, session->c, 
193                      "proxy_session(%s): pool cleanup, state=%d, streams=%d",
194                      session->id, session->state, 
195                      (int)h2_proxy_ihash_count(session->streams));
196        session->aborted = 1;
197        dispatch_event(session, H2_PROXYS_EV_PRE_CLOSE, 0, NULL);
198        nghttp2_session_del(session->ngh2);
199        session->ngh2 = NULL;
200        p_conn->data = NULL;
201    }
202    return APR_SUCCESS;
203}

205static int proxy_pass_brigade(apr_bucket_alloc_t *bucket_alloc,
206                              proxy_conn_rec *p_conn,
207                              conn_rec *origin, apr_bucket_brigade *bb,
208                              int flush)
209{
210    apr_status_t status;
211    apr_off_t transferred;
212
213    if (flush) {
214        apr_bucket *e = apr_bucket_flush_create(bucket_alloc);
215        APR_BRIGADE_INSERT_TAIL(bb, e);
216    }
217    apr_brigade_length(bb, 0, &transferred);
218    if (transferred != -1)
219        p_conn->worker->s->transferred += transferred;
220    status = ap_pass_brigade(origin->output_filters, bb);
221    /* Cleanup the brigade now to avoid buckets lifetime
222     * issues in case of error returned below. */ view japanese
223    apr_brigade_cleanup(bb);
224    if (status != APR_SUCCESS) {
225        ap_log_cerror(APLOG_MARK, APLOG_DEBUG, status, origin, APLOGNO(03357)
226                      "pass output failed to %pI (%s)",
227                      p_conn->addr, p_conn->hostname);
228    }
229    return status;
230}

232static ssize_t raw_send(nghttp2_session *ngh2, const uint8_t *data,
233                        size_t length, int flags, void *user_data)
234{
235    h2_proxy_session *session = user_data;
236    apr_bucket *b;
237    apr_status_t status;
238    int flush = 1;
239
240    if (data) {
241        b = apr_bucket_transient_create((const char*)data, length, 
242                                        session->c->bucket_alloc);
243        APR_BRIGADE_INSERT_TAIL(session->output, b);
244    }
245
246    status = proxy_pass_brigade(session->c->bucket_alloc,  
247                                session->p_conn, session->c, 
248                                session->output, flush);
249    ap_log_cerror(APLOG_MARK, APLOG_TRACE1, status, session->c, 
250                  "h2_proxy_sesssion(%s): raw_send %d bytes, flush=%d", 
251                  session->id, (int)length, flush);
252    if (status != APR_SUCCESS) {
253        return NGHTTP2_ERR_CALLBACK_FAILURE;
254    }
255    return length;
256}

258static int on_frame_recv(nghttp2_session *ngh2, const nghttp2_frame *frame,
259                         void *user_data) 
260{
261    h2_proxy_session *session = user_data;
262    h2_proxy_stream *stream;
263    request_rec *r;
264    int n;
265    
266    if (APLOGcdebug(session->c)) {
267        char buffer[256];
268        
269        h2_proxy_util_frame_print(frame, buffer, sizeof(buffer)/sizeof(buffer[0]));
270        ap_log_cerror(APLOG_MARK, APLOG_DEBUG, 0, session->c, APLOGNO(03341)
271                      "h2_proxy_session(%s): recv FRAME[%s]",
272                      session->id, buffer);
273    }
274
275    ping_ev_frame_received(session, frame);
276    /* Action for frame types: */ view japanese
277    switch (frame->hd.type) {
278        case NGHTTP2_HEADERS:
279            stream = nghttp2_session_get_stream_user_data(ngh2, frame->hd.stream_id);
280            if (!stream) {
281                return NGHTTP2_ERR_CALLBACK_FAILURE;
282            }
283            r = stream->r;
284            if (r->status >= 100 && r->status < 200) {
285                /* By default, we will forward all interim responses when
286                 * we are sitting on a HTTP/2 connection to the client */ view japanese
287                int forward = session->h2_front;
288                switch(r->status) {
289                    case 100:
290                        if (stream->waiting_on_100) {
291                            stream->waiting_on_100 = 0;
292                            r->status_line = ap_get_status_line(r->status);
293                            forward = 1;
294                        } 
295                        break;
296                    case 103:
297                        /* workaround until we get this into http protocol base
298                         * parts. without this, unknown codes are converted to
299                         * 500... */ view japanese
300                        r->status_line = "103 Early Hints";
301                        break;
302                    default:
303                        r->status_line = ap_get_status_line(r->status);
304                        break;
305                }
306                ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(03487) 
307                              "h2_proxy_session(%s): got interim HEADERS, "
308                              "status=%d, will forward=%d",
309                              session->id, r->status, forward);
310                if (forward) {
311                    ap_send_interim_response(r, 1);
312                }
313            }
314            else if (r->status >= 400) {
315                proxy_dir_conf *dconf;
316                dconf = ap_get_module_config(r->per_dir_config, &proxy_module);
317                if (ap_proxy_should_override(dconf, r->status)) {
318                    apr_table_setn(r->notes, "proxy-error-override", "1");
319                    nghttp2_submit_rst_stream(ngh2, NGHTTP2_FLAG_NONE,
320                          frame->hd.stream_id, NGHTTP2_STREAM_CLOSED);
321                }
322            }
323            stream_resume(stream);
324            break;
325        case NGHTTP2_PING:
326            break;
327        case NGHTTP2_PUSH_PROMISE:
328            break;
329        case NGHTTP2_SETTINGS:
330            if (frame->settings.niv > 0) {
331                n = nghttp2_session_get_remote_settings(ngh2, NGHTTP2_SETTINGS_MAX_CONCURRENT_STREAMS);
332                if (n > 0) {
333                    session->remote_max_concurrent = n;
334                }
335            }
336            break;
337        case NGHTTP2_GOAWAY:
338            /* we expect the remote server to tell us the highest stream id
339             * that it has started processing. */ view japanese
340            session->last_stream_id = frame->goaway.last_stream_id;
341            dispatch_event(session, H2_PROXYS_EV_REMOTE_GOAWAY, 0, NULL);
342            break;
343        default:
344            break;
345    }
346    return 0;
347}

370static void process_proxy_header(apr_table_t *headers, h2_proxy_stream *stream, 
371                                 const char *n, const char *v)
372{
373    static const struct {
374        const char *name;
375        ap_proxy_header_reverse_map_fn func;
376    } transform_hdrs[] = {
377        { "Location", ap_proxy_location_reverse_map },
378        { "Content-Location", ap_proxy_location_reverse_map },
379        { "URI", ap_proxy_location_reverse_map },
380        { "Destination", ap_proxy_location_reverse_map },
381        { "Set-Cookie", ap_proxy_cookie_reverse_map },
382        { NULL, NULL }
383    };
384    request_rec *r = stream->r;
385    proxy_dir_conf *dconf;
386    int i;
387    
388    dconf = ap_get_module_config(r->per_dir_config, &proxy_module);
389    if (!dconf->preserve_host) {
390        for (i = 0; transform_hdrs[i].name; ++i) {
391            if (!ap_cstr_casecmp(transform_hdrs[i].name, n)) {
392                apr_table_add(headers, n, (*transform_hdrs[i].func)(r, dconf, v));
393                return;
394            }
395        }
396        if (!ap_cstr_casecmp("Link", n)) {
397            dconf = ap_get_module_config(r->per_dir_config, &proxy_module);
398            apr_table_add(headers, n, h2_proxy_link_reverse_map(r, dconf, 
399                            stream->real_server_uri, stream->p_server_uri, v));
400            return;
401        }
402    }
403    apr_table_add(headers, n, v);
404}

406static apr_status_t h2_proxy_stream_add_header_out(h2_proxy_stream *stream,
407                                                   const char *n, apr_size_t nlen,
408                                                   const char *v, apr_size_t vlen)
409{
410    if (n[0] == ':') {
411        if (!stream->data_received && !strncmp(":status", n, nlen)) {
412            char *s = apr_pstrndup(stream->r->pool, v, vlen);
413            
414            apr_table_setn(stream->r->notes, "proxy-status", s);
415            ap_log_cerror(APLOG_MARK, APLOG_TRACE2, 0, stream->cfront,
416                          "h2_proxy_stream(%s-%d): got status %s", 
417                          stream->session->id, stream->id, s);
418            stream->r->status = (int)apr_atoi64(s);
419            if (stream->r->status <= 0) {
420                stream->r->status = 500;
421                return APR_EGENERAL;
422            }
423        }
424        return APR_SUCCESS;
425    }
426    
427    ap_log_cerror(APLOG_MARK, APLOG_TRACE2, 0, stream->cfront,
428                  "h2_proxy_stream(%s-%d): on_header %s: %s", 
429                  stream->session->id, stream->id, n, v);
430    if (!h2_proxy_res_ignore_header(n, nlen)) {
431        char *hname, *hvalue;
432        apr_table_t *headers = (stream->headers_ended? 
433                               stream->r->trailers_out : stream->r->headers_out);
434    
435        hname = apr_pstrndup(stream->pool, n, nlen);
436        h2_proxy_util_camel_case_header(hname, nlen);
437        hvalue = apr_pstrndup(stream->pool, v, vlen);
438        
439        ap_log_cerror(APLOG_MARK, APLOG_TRACE2, 0, stream->cfront,
440                      "h2_proxy_stream(%s-%d): got header %s: %s", 
441                      stream->session->id, stream->id, hname, hvalue);
442        process_proxy_header(headers, stream, hname, hvalue);
443    }
444    return APR_SUCCESS;
445}

456static void h2_proxy_stream_end_headers_out(h2_proxy_stream *stream) 
457{
458    h2_proxy_session *session = stream->session;
459    request_rec *r = stream->r;
460    apr_pool_t *p = r->pool;
461    const char *buf;
462    
463    /* Now, add in the cookies from the response to the ones already saved */ view japanese
464    apr_table_do(add_header, stream->saves, r->headers_out, "Set-Cookie", NULL);
465    
466    /* and now load 'em all in */ view japanese
467    if (!apr_is_empty_table(stream->saves)) {
468        apr_table_unset(r->headers_out, "Set-Cookie");
469        r->headers_out = apr_table_overlay(p, r->headers_out, stream->saves);
470    }
471
472    if ((buf = apr_table_get(r->headers_out, "Content-Type"))) {
473        ap_set_content_type(r, apr_pstrdup(p, buf));
474    }
475    
476    /* handle Via header in response */ view japanese
477    if (session->conf->viaopt != via_off 
478        && session->conf->viaopt != via_block) {
479        const char *server_name = ap_get_server_name(stream->r);
480        apr_port_t port = ap_get_server_port(stream->r);
481        char portstr[32];
482        
483        /* If USE_CANONICAL_NAME_OFF was configured for the proxy virtual host,
484         * then the server name returned by ap_get_server_name() is the
485         * origin server name (which doesn't make sense with Via: headers)
486         * so we use the proxy vhost's name instead.
487         */ view japanese
488        if (server_name == stream->r->hostname) {
489            server_name = stream->r->server->server_hostname;
490        }
491        if (ap_is_default_port(port, stream->r)) {
492            portstr[0] = '\0';
493        }
494        else {
495            apr_snprintf(portstr, sizeof(portstr), ":%d", port);
496        }
497
498        /* create a "Via:" response header entry and merge it */ view japanese
499        apr_table_add(r->headers_out, "Via",
500                       (session->conf->viaopt == via_full)
501                       ? apr_psprintf(p, "%d.%d %s%s (%s)",
502                                      HTTP_VERSION_MAJOR(r->proto_num),
503                                      HTTP_VERSION_MINOR(r->proto_num),
504                                      server_name, portstr,
505                                      AP_SERVER_BASEVERSION)
506                       : apr_psprintf(p, "%d.%d %s%s",
507                                      HTTP_VERSION_MAJOR(r->proto_num),
508                                      HTTP_VERSION_MINOR(r->proto_num),
509                                      server_name, portstr)
510                       );
511    }
512    if (r->status >= 200) stream->headers_ended = 1;
513    
514    if (APLOGrtrace2(stream->r)) {
515        ap_log_rerror(APLOG_MARK, APLOG_TRACE2, 0, stream->r, 
516                      "h2_proxy_stream(%s-%d), header_out after merging", 
517                      stream->session->id, stream->id);
518        apr_table_do(log_header, stream, stream->r->headers_out, NULL);
519    }
520}

522static int stream_response_data(nghttp2_session *ngh2, uint8_t flags,
523                                int32_t stream_id, const uint8_t *data,
524                                size_t len, void *user_data) 
525{
526    h2_proxy_session *session = user_data;
527    h2_proxy_stream *stream;
528    apr_bucket *b;
529    apr_status_t status;
530    
531    stream = nghttp2_session_get_stream_user_data(ngh2, stream_id);
532    if (!stream) {
533        ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, ap_server_conf, APLOGNO(03358)
534                     "h2_proxy_session(%s): recv data chunk for "
535                     "unknown stream %d, ignored", 
536                     session->id, stream_id);
537        return 0;
538    }
539    
540    if (!stream->data_received) {
541        /* last chance to manipulate response headers.
542         * after this, only trailers */ view japanese
543        h2_proxy_stream_end_headers_out(stream);
544    }
545    stream->data_received += len;
546    b = apr_bucket_transient_create((const char*)data, len,
547                                    stream->cfront->bucket_alloc);
548    APR_BRIGADE_INSERT_TAIL(stream->output, b);
549    /* always flush after a DATA frame, as we have no other indication
550     * of buffer use */ view japanese
551    b = apr_bucket_flush_create(stream->cfront->bucket_alloc);
552    APR_BRIGADE_INSERT_TAIL(stream->output, b);
553
554    status = ap_pass_brigade(stream->r->output_filters, stream->output);
555    ap_log_rerror(APLOG_MARK, APLOG_DEBUG, status, stream->r, APLOGNO(03359)
556                  "h2_proxy_session(%s): stream=%d, response DATA %ld, %ld"
557                  " total", session->id, stream_id, (long)len,
558                  (long)stream->data_received);
559    if (status != APR_SUCCESS) {
560        ap_log_rerror(APLOG_MARK, APLOG_DEBUG, status, stream->r, APLOGNO(03344)
561                      "h2_proxy_session(%s): passing output on stream %d", 
562                      session->id, stream->id);
563        nghttp2_submit_rst_stream(ngh2, NGHTTP2_FLAG_NONE,
564                                  stream_id, NGHTTP2_STREAM_CLOSED);
565        return NGHTTP2_ERR_STREAM_CLOSING;
566    }
567    return 0;
568}

570static int on_stream_close(nghttp2_session *ngh2, int32_t stream_id,
571                           uint32_t error_code, void *user_data) 
572{
573    h2_proxy_session *session = user_data;
574    h2_proxy_stream *stream;
575    if (!session->aborted) {
576        ap_log_cerror(APLOG_MARK, APLOG_DEBUG, 0, session->c, APLOGNO(03360)
577                      "h2_proxy_session(%s): stream=%d, closed, err=%d",

613static ssize_t stream_request_data(nghttp2_session *ngh2, int32_t stream_id, 
614                                   uint8_t *buf, size_t length,
615                                   uint32_t *data_flags, 
616                                   nghttp2_data_source *source, void *user_data)
617{
618    h2_proxy_stream *stream;
619    apr_status_t status = APR_SUCCESS;
620    
621    *data_flags = 0;
622    stream = nghttp2_session_get_stream_user_data(ngh2, stream_id);
623    if (!stream) {
624        ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, ap_server_conf, APLOGNO(03361)
625                     "h2_proxy_stream(NULL): data_read, stream %d not found", 
626                     stream_id);
627        return NGHTTP2_ERR_CALLBACK_FAILURE;
628    }
629    
630    if (stream->session->ping_state != H2_PING_ST_NONE) {
631        /* suspend until we hear from the other side */ view japanese
632        stream->waiting_on_ping = 1;
633        status = APR_EAGAIN;
634    }
635    else if (stream->r->expecting_100) {
636        /* suspend until the answer comes */ view japanese
637        stream->waiting_on_100 = 1;
638        status = APR_EAGAIN;
639    }
640    else if (APR_BRIGADE_EMPTY(stream->input)) {
641        status = ap_get_brigade(stream->r->input_filters, stream->input,
642                                AP_MODE_READBYTES, APR_NONBLOCK_READ,
643                                H2MAX(APR_BUCKET_BUFF_SIZE, length));
644        ap_log_rerror(APLOG_MARK, APLOG_TRACE2, status, stream->r, 
645                      "h2_proxy_stream(%s-%d): request body read", 
646                      stream->session->id, stream->id);
647    }
648
649    if (status == APR_SUCCESS) {
650        size_t readlen = 0;
651        while (status == APR_SUCCESS 
652               && (readlen < length)
653               && !APR_BRIGADE_EMPTY(stream->input)) {
654            apr_bucket* b = APR_BRIGADE_FIRST(stream->input);
655            if (APR_BUCKET_IS_METADATA(b)) {
656                if (APR_BUCKET_IS_EOS(b)) {
657                    *data_flags |= NGHTTP2_DATA_FLAG_EOF;
658                }
659                else {
660                    /* we do nothing more regarding any meta here */ view japanese
661                }
662            }
663            else {
664                const char *bdata = NULL;
665                apr_size_t blen = 0;
666                status = apr_bucket_read(b, &bdata, &blen, APR_BLOCK_READ);
667                
668                if (status == APR_SUCCESS && blen > 0) {
669                    size_t copylen = H2MIN(length - readlen, blen);
670                    memcpy(buf, bdata, copylen);
671                    buf += copylen;
672                    readlen += copylen;
673                    if (copylen < blen) {
674                        /* We have data left in the bucket. Split it. */ view japanese
675                        status = apr_bucket_split(b, copylen);
676                    }
677                }
678            }
679            apr_bucket_delete(b);
680        }
681
682        stream->data_sent += readlen;
683        ap_log_rerror(APLOG_MARK, APLOG_DEBUG, status, stream->r, APLOGNO(03468) 
684                      "h2_proxy_stream(%d): request DATA %ld, %ld"
685                      " total, flags=%d", stream->id, (long)readlen, (long)stream->data_sent,
686                      (int)*data_flags);
687        if ((*data_flags & NGHTTP2_DATA_FLAG_EOF) && !apr_is_empty_table(stream->r->trailers_in)) {
688            ap_log_rerror(APLOG_MARK, APLOG_DEBUG, status, stream->r, APLOGNO(10179) 
689                          "h2_proxy_stream(%d): submit trailers", stream->id);
690            *data_flags |= NGHTTP2_DATA_FLAG_NO_END_STREAM;
691            submit_trailers(stream);
692        } 
693        return readlen;
694    }
695    else if (APR_STATUS_IS_EAGAIN(status)) {
696        /* suspended stream, needs to be re-awakened */ view japanese
697        ap_log_rerror(APLOG_MARK, APLOG_TRACE2, status, stream->r, 
698                      "h2_proxy_stream(%s-%d): suspending", 
699                      stream->session->id, stream_id);
700        stream->suspended = 1;
701        h2_proxy_iq_add(stream->session->suspended, stream->id, NULL, NULL);
702        return NGHTTP2_ERR_DEFERRED;
703    }
704    else {
705        nghttp2_submit_rst_stream(ngh2, NGHTTP2_FLAG_NONE, 
706                                  stream_id, NGHTTP2_STREAM_CLOSED);
707        return NGHTTP2_ERR_STREAM_CLOSING;
708    }
709}

825static apr_status_t open_stream(h2_proxy_session *session, const char *url,
826                                request_rec *r, int standalone,
827                                h2_proxy_stream **pstream)
828{
829    h2_proxy_stream *stream;
830    apr_uri_t puri;
831    const char *authority, *scheme, *path, *orig_host;
832    apr_status_t status;
833    proxy_dir_conf *dconf;
834
835    stream = apr_pcalloc(r->pool, sizeof(*stream));
836
837    stream->pool = r->pool;
838    stream->url = url;
839    stream->r = r;
840    stream->cfront = r->connection;
841    stream->standalone = standalone;
842    stream->session = session;
843    stream->state = H2_STREAM_ST_IDLE;
844    
845    stream->input = apr_brigade_create(stream->pool, stream->cfront->bucket_alloc);
846    stream->output = apr_brigade_create(stream->pool, stream->cfront->bucket_alloc);
847    
848    stream->req = h2_proxy_req_create(1, stream->pool);
849
850    status = apr_uri_parse(stream->pool, url, &puri);
851    if (status != APR_SUCCESS)
852        return status;
853    
854    scheme = (strcmp(puri.scheme, "h2")? "http" : "https");
855    orig_host = apr_table_get(r->headers_in, "Host");
856    if (orig_host == NULL) {
857        orig_host = r->hostname;
858    }
859
860    dconf = ap_get_module_config(r->per_dir_config, &proxy_module);
861    if (dconf->preserve_host) {
862        authority = orig_host;
863        if (!authority) {
864            /* Duplicate mod_proxy behaviour if ProxyPreserveHost is
865             * used but an "HTTP/0.9" request is received without a
866             * Host: header */ view japanese
867            authority = r->server->server_hostname;
868            ap_log_rerror(APLOG_MARK, APLOG_WARNING, 0, r, APLOGNO(10511)
869                          "incoming HTTP/0.9 request (with no Host header) "
870                          "and preserve host set, "
871                          "forcing hostname to be %s for uri %s",
872                          authority, r->uri);
873            apr_table_setn(r->headers_in, "Host", authority);
874        }
875    }
876    else {
877        authority = puri.hostname;
878        if (!ap_strchr_c(authority, ':') && puri.port
879            && apr_uri_port_of_scheme(scheme) != puri.port) {
880            /* port info missing and port is not default for scheme: append */ view japanese
881            authority = apr_psprintf(stream->pool, "%s:%d", authority, puri.port);
882        }
883        ap_log_cerror(APLOG_MARK, APLOG_TRACE2, 0, stream->cfront,
884                      "authority=%s from uri.hostname=%s and uri.port=%d",
885                      authority, puri.hostname, puri.port);
886    }
887    /* See #235, we use only :authority when available and remove Host:
888     * since differing values are not acceptable, see RFC 9113 ch. 8.3.1 */ view japanese
889    if (authority && strlen(authority)) {
890        apr_table_unset(r->headers_in, "Host");
891    }
892
893    /* we need this for mapping relative uris in headers ("Link") back
894     * to local uris */ view japanese
895    stream->real_server_uri = apr_psprintf(stream->pool, "%s://%s", scheme, authority);  view japanese
896    stream->p_server_uri = apr_psprintf(stream->pool, "%s://%s", puri.scheme, authority);  view japanese
897    path = apr_uri_unparse(stream->pool, &puri, APR_URI_UNP_OMITSITEPART);
898
899    h2_proxy_req_make(stream->req, stream->pool, r->method, scheme,
900                authority, path, r->headers_in);
901
902    if (dconf->add_forwarded_headers) {
903        if (PROXYREQ_REVERSE == r->proxyreq) {
904            /* Add X-Forwarded-For: so that the upstream has a chance to
905             * determine, where the original request came from.
906             */ view japanese
907            apr_table_mergen(stream->req->headers, "X-Forwarded-For",
908                             r->useragent_ip);
909
910            /* Add X-Forwarded-Host: so that upstream knows what the
911             * original request hostname was.
912             */ view japanese
913            if (orig_host) {
914                apr_table_mergen(stream->req->headers, "X-Forwarded-Host",
915                                 orig_host);
916            }
917
918            /* Add X-Forwarded-Server: so that upstream knows what the
919             * name of this proxy server is (if there are more than one)
920             * XXX: This duplicates Via: - do we strictly need it?
921             */ view japanese
922            apr_table_mergen(stream->req->headers, "X-Forwarded-Server",
923                             r->server->server_hostname);
924        }
925    }
926
927    /* Tuck away all already existing cookies */ view japanese
928    stream->saves = apr_table_make(r->pool, 2);
929    apr_table_do(add_header, stream->saves, r->headers_out, "Set-Cookie", NULL);
930
931    *pstream = stream;
932    
933    return APR_SUCCESS;
934}

936static apr_status_t submit_stream(h2_proxy_session *session, h2_proxy_stream *stream)
937{
938    h2_proxy_ngheader *hd;
939    nghttp2_data_provider *pp = NULL;
940    nghttp2_data_provider provider;
941    int rv, may_have_request_body = 1;
942    apr_status_t status;
943
944    hd = h2_proxy_util_nghd_make_req(stream->pool, stream->req);
945    
946    /* If we expect a 100-continue response, we must refrain from reading
947       any input until we get it. Reading the input will possibly trigger
948       HTTP_IN filter to generate the 100-continue itself. */ view japanese
949    if (stream->waiting_on_100 || stream->waiting_on_ping) {
950        /* make a small test if we get an EOF/EOS immediately */ view japanese
951        status = ap_get_brigade(stream->r->input_filters, stream->input,
952                                AP_MODE_READBYTES, APR_NONBLOCK_READ,
953                                APR_BUCKET_BUFF_SIZE);
954        may_have_request_body = APR_STATUS_IS_EAGAIN(status)
955                                || (status == APR_SUCCESS 
956                                    && !APR_BUCKET_IS_EOS(APR_BRIGADE_FIRST(stream->input)));
957    }
958    
959    if (may_have_request_body) {
960        provider.source.fd = 0;
961        provider.source.ptr = NULL;
962        provider.read_callback = stream_request_data;
963        pp = &provider;
964    }
965
966    rv = nghttp2_submit_request(session->ngh2, NULL, 
967                                hd->nv, hd->nvlen, pp, stream);
968                                
969    ap_log_cerror(APLOG_MARK, APLOG_DEBUG, 0, stream->cfront, APLOGNO(03363)
970                  "h2_proxy_session(%s): submit %s%s -> %d", 
971                  session->id, stream->req->authority, stream->req->path,
972                  rv);
973    if (rv > 0) {
974        stream->id = rv;
975        stream->state = H2_STREAM_ST_OPEN;
976        h2_proxy_ihash_add(session->streams, stream);
977        dispatch_event(session, H2_PROXYS_EV_STREAM_SUBMITTED, rv, NULL);
978        
979        return APR_SUCCESS;
980    }
981    return APR_EGENERAL;
982}

994static apr_status_t feed_brigade(h2_proxy_session *session, apr_bucket_brigade *bb)
995{
996    apr_status_t status = APR_SUCCESS;
997    apr_size_t readlen = 0;
998    ssize_t n;
999
1000    while (status == APR_SUCCESS && !APR_BRIGADE_EMPTY(bb)) {
1001        apr_bucket* b = APR_BRIGADE_FIRST(bb);
1002        
1003        if (APR_BUCKET_IS_METADATA(b)) {
1004            /* nop */
1005        }
1006        else {
1007            const char *bdata = NULL;
1008            apr_size_t blen = 0;
1009            
1010            status = apr_bucket_read(b, &bdata, &blen, APR_BLOCK_READ);
1011            if (status == APR_SUCCESS && blen > 0) {
1012                n = nghttp2_session_mem_recv(session->ngh2, (const uint8_t *)bdata, blen);
1013                ap_log_cerror(APLOG_MARK, APLOG_TRACE1, 0, session->c, 
1014                              "h2_proxy_session(%s): feeding %ld bytes -> %ld", 
1015                              session->id, (long)blen, (long)n);
1016                if (n < 0) {
1017                    if (nghttp2_is_fatal((int)n)) {
1018                        status = APR_EGENERAL;
1019                    }
1020                }
1021                else {
1022                    size_t rlen = (size_t)n;
1023                    readlen += rlen;
1024                    if (rlen < blen) {
1025                        apr_bucket_split(b, rlen);
1026                    }
1027                }
1028            }
1029        }
1030        apr_bucket_delete(b);
1031    }
1032    
1033    ap_log_cerror(APLOG_MARK, APLOG_TRACE1, status, session->c, 
1034                  "h2_proxy_session(%s): fed %ld bytes of input to session", 
1035                  session->id, (long)readlen);
1036    if (readlen == 0 && status == APR_SUCCESS) {
1037        return APR_EAGAIN;
1038    }
1039    return status;
1040}

1042static apr_status_t h2_proxy_session_read(h2_proxy_session *session, int block, 
1043                                          apr_interval_time_t timeout)
1044{
1045    apr_status_t status = APR_SUCCESS;
1046    
1047    if (APR_BRIGADE_EMPTY(session->input)) {
1048        apr_socket_t *socket = NULL;
1049        apr_time_t save_timeout = -1;
1050        
1051        if (block && timeout > 0) {
1052            socket = ap_get_conn_socket(session->c);
1053            if (socket) {
1054                apr_socket_timeout_get(socket, &save_timeout);
1055                apr_socket_timeout_set(socket, timeout);
1056            }
1057            else {
1058                /* cannot block on timeout */ view japanese
1059                ap_log_cerror(APLOG_MARK, APLOG_WARNING, 0, session->c, APLOGNO(03379)
1060                              "h2_proxy_session(%s): unable to get conn socket", 
1061                              session->id);
1062                return APR_ENOTIMPL;
1063            }
1064        }
1065        
1066        status = ap_get_brigade(session->c->input_filters, session->input, 
1067                                AP_MODE_READBYTES, 
1068                                block? APR_BLOCK_READ : APR_NONBLOCK_READ, 
1069                                64 * 1024);
1070        ap_log_cerror(APLOG_MARK, APLOG_TRACE3, status, session->c, 
1071                      "h2_proxy_session(%s): read from conn", session->id);
1072        if (socket && save_timeout != -1) {
1073            apr_socket_timeout_set(socket, save_timeout);
1074        }
1075    }
1076    
1077    if (status == APR_SUCCESS) {
1078        status = feed_brigade(session, session->input);
1079    }
1080    else if (APR_STATUS_IS_TIMEUP(status)) {
1081        /* nop */
1082    }
1083    else if (!APR_STATUS_IS_EAGAIN(status)) {
1084        ap_log_cerror(APLOG_MARK, APLOG_DEBUG, status, session->c, APLOGNO(03380)
1085                      "h2_proxy_session(%s): read error", session->id);
1086        dispatch_event(session, H2_PROXYS_EV_CONN_ERROR, status, NULL);
1087    }
1088
1089    return status;
1090}

1131static apr_status_t check_suspended(h2_proxy_session *session)

1140        if (stream) {
1141            if (stream->waiting_on_100 || stream->waiting_on_ping) {
1142                status = APR_EAGAIN;
1143            }
1144            else {
1145                status = ap_get_brigade(stream->r->input_filters, stream->input,
1146                                        AP_MODE_READBYTES, APR_NONBLOCK_READ,
1147                                        APR_BUCKET_BUFF_SIZE);

1150                stream_resume(stream);

1152                return APR_SUCCESS;
1153            }

1157                              session->id, stream_id);

1159                check_suspended(session);
1160                return APR_SUCCESS;
1161            }
1162        }
1163        else {
1164            /* gone? */ view japanese
1165            h2_proxy_iq_remove(session->suspended, stream_id);

1330static void ev_no_io(h2_proxy_session *session, int arg, const char *msg)
1331{

1337             * in this state, we go into a tight loop and suck up
1338             * CPU cycles. Ideally, we'd like to do a blocking read, but that
1339             * is not possible if we have scheduled tasks and wait
1340             * for them to produce something. */ view japanese
1341            if (h2_proxy_ihash_empty(session->streams)) {
1342                if (!is_accepting_streams(session)) {

1350                     * switch to blocking reads */ view japanese

1381static void ev_stream_done(h2_proxy_session *session, int stream_id, 
1382                           const char *msg)
1383{
1384    h2_proxy_stream *stream;
1385    apr_bucket *b;
1386
1387    stream = nghttp2_session_get_stream_user_data(session->ngh2, stream_id);
1388    if (stream) {
1389        /* if the stream's connection is aborted, do not send anything
1390         * more on it. */ view japanese
1391        apr_status_t status = (stream->error_code == 0)? APR_SUCCESS : APR_EINVAL;
1392        int touched = (stream->data_sent || stream->data_received ||
1393                       stream_id <= session->last_stream_id);
1394        if (!stream->cfront->aborted) {
1395            ap_log_cerror(APLOG_MARK, APLOG_DEBUG, 0, stream->cfront, APLOGNO(03364)
1396                          "h2_proxy_sesssion(%s): stream(%d) closed "
1397                          "(touched=%d, error=%d)",
1398                          session->id, stream_id, touched, stream->error_code);
1399
1400            if (status != APR_SUCCESS) {
1401              /* stream failed. If we have received (and forwarded) response
1402               * data already, we need to append an error buckt to inform
1403               * consumers.
1404               * Otherwise, we have an early fail on the connection and may
1405               * retry this request on a new one. In that case, keep the
1406               * output virgin so that a new attempt can be made. */ view japanese
1407              if (stream->data_received) {
1408                int http_status = ap_map_http_request_error(status, HTTP_BAD_REQUEST);
1409                b = ap_bucket_error_create(http_status, NULL, stream->r->pool,
1410                                           stream->cfront->bucket_alloc);
1411                APR_BRIGADE_INSERT_TAIL(stream->output, b);
1412                b = apr_bucket_eos_create(stream->cfront->bucket_alloc);
1413                APR_BRIGADE_INSERT_TAIL(stream->output, b);
1414                ap_pass_brigade(stream->r->output_filters, stream->output);
1415              }
1416            }
1417            else if (!stream->data_received) {
1418                /* if the response had no body, this is the time to flush
1419                 * an empty brigade which will also write the response headers */ view japanese
1420                h2_proxy_stream_end_headers_out(stream);
1421                stream->data_received = 1;
1422                b = apr_bucket_flush_create(stream->cfront->bucket_alloc);
1423                APR_BRIGADE_INSERT_TAIL(stream->output, b);
1424                b = apr_bucket_eos_create(stream->cfront->bucket_alloc);
1425                APR_BRIGADE_INSERT_TAIL(stream->output, b);
1426                ap_pass_brigade(stream->r->output_filters, stream->output);
1427            }
1428        }
1429
1430        stream->state = H2_STREAM_ST_CLOSED;
1431        h2_proxy_ihash_remove(session->streams, stream_id);
1432        h2_proxy_iq_remove(session->suspended, stream_id);
1433        if (session->done) {
1434            session->done(session, stream->r, status, touched, stream->error_code);
1435        }
1436    }
1437    
1438    switch (session->state) {
1439        default:
1440            /* nop */
1441            break;
1442    }
1443}

1561apr_status_t h2_proxy_session_process(h2_proxy_session *session)
1562{
1563    apr_status_t status;
1564    int have_written = 0, have_read = 0;
1565
1566    ap_log_cerror(APLOG_MARK, APLOG_TRACE2, 0, session->c, 
1567                  "h2_proxy_session(%s): process", session->id);
1568           
1569run_loop:
1570    switch (session->state) {
1571        case H2_PROXYS_ST_INIT:
1572            status = session_start(session);
1573            if (status == APR_SUCCESS) {
1574                dispatch_event(session, H2_PROXYS_EV_INIT, 0, NULL);
1575                goto run_loop;
1576            }
1577            else {
1578                dispatch_event(session, H2_PROXYS_EV_CONN_ERROR, status, NULL);
1579            }
1580            break;
1581            
1582        case H2_PROXYS_ST_BUSY:
1583        case H2_PROXYS_ST_LOCAL_SHUTDOWN:
1584        case H2_PROXYS_ST_REMOTE_SHUTDOWN:
1585            have_written = send_loop(session);
1586            
1587            if (nghttp2_session_want_read(session->ngh2)) {
1588                status = h2_proxy_session_read(session, 0, 0);
1589                if (status == APR_SUCCESS) {
1590                    have_read = 1;
1591                }
1592            }
1593            
1594            if (!have_written && !have_read 
1595                && !nghttp2_session_want_write(session->ngh2)) {
1596                dispatch_event(session, H2_PROXYS_EV_NO_IO, 0, NULL);
1597                goto run_loop;
1598            }
1599            break;
1600            
1601        case H2_PROXYS_ST_WAIT:
1602            if (is_waiting_for_backend(session)) {
1603                /* we can do a blocking read with the default timeout (as
1604                 * configured via ProxyTimeout in our socket. There is
1605                 * nothing we want to send or check until we get more data
1606                 * from the backend. */ view japanese
1607                status = h2_proxy_session_read(session, 1, 0);
1608                if (status == APR_SUCCESS) {
1609                    have_read = 1;
1610                    dispatch_event(session, H2_PROXYS_EV_DATA_READ, 0, NULL);
1611                }
1612                else {
1613                    dispatch_event(session, H2_PROXYS_EV_CONN_ERROR, status, NULL);
1614                    return status;
1615                }
1616            }
1617            else if (check_suspended(session) == APR_EAGAIN) {
1618                /* no stream has become resumed. Do a blocking read with
1619                 * ever increasing timeouts... */ view japanese
1620                if (session->wait_timeout < 25) {
1621                    session->wait_timeout = 25;
1622                }
1623                else {
1624                    session->wait_timeout = H2MIN(apr_time_from_msec(100), 
1625                                                  2*session->wait_timeout);
1626                }
1627                
1628                status = h2_proxy_session_read(session, 1, session->wait_timeout);
1629                ap_log_cerror(APLOG_MARK, APLOG_TRACE3, status, session->c, 
1630                              APLOGNO(03365)
1631                              "h2_proxy_session(%s): WAIT read, timeout=%fms", 
1632                              session->id, session->wait_timeout/1000.0);
1633                if (status == APR_SUCCESS) {
1634                    have_read = 1;
1635                    dispatch_event(session, H2_PROXYS_EV_DATA_READ, 0, NULL);
1636                }
1637                else if (APR_STATUS_IS_TIMEUP(status)
1638                    || APR_STATUS_IS_EAGAIN(status)) {
1639                    /* go back to checking all inputs again */ view japanese
1640                    transit(session, "wait cycle", H2_PROXYS_ST_BUSY);
1641                }
1642            }
1643            break;
1644            
1645        case H2_PROXYS_ST_IDLE:
1646            break;
1647
1648        case H2_PROXYS_ST_DONE: /* done, session terminated */ view japanese
1649            return APR_EOF;
1650            
1651        default:
1652            ap_log_cerror(APLOG_MARK, APLOG_ERR, APR_EGENERAL, session->c,
1653                          APLOGNO(03346)"h2_proxy_session(%s): unknown state %d", 
1654                          session->id, session->state);
1655            dispatch_event(session, H2_PROXYS_EV_PROTO_ERROR, 0, NULL);
1656            break;
1657    }
1658
1659
1660    if (have_read || have_written) {
1661        session->wait_timeout = 0;
1662    }
1663    
1664    if (!nghttp2_session_want_read(session->ngh2)
1665        && !nghttp2_session_want_write(session->ngh2)) {
1666        dispatch_event(session, H2_PROXYS_EV_NGH2_DONE, 0, NULL);
1667    }
1668    
1669    return APR_SUCCESS; /* needs to be called again */ view japanese
1670}
