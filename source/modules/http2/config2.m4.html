44dnl APACHE_CHECK_NGHTTP2
45dnl
46dnl Configure for nghttp2, giving preference to
47dnl "--with-nghttp2=<path>" if it was specified.
48dnl
49AC_DEFUN([APACHE_CHECK_NGHTTP2],[
50  AC_CACHE_CHECK([for nghttp2], [ac_cv_nghttp2], [
51    dnl initialise the variables we use
52    ac_cv_nghttp2=no
53    ap_nghttp2_found=""
54    ap_nghttp2_base=""
55    ap_nghttp2_libs=""
56
57    dnl Determine the nghttp2 base directory, if any
58    AC_MSG_CHECKING([for user-provided nghttp2 base directory])
59    AC_ARG_WITH(nghttp2, APACHE_HELP_STRING(--with-nghttp2=PATH, nghttp2 installation directory), [
60      dnl If --with-nghttp2 specifies a directory, we use that directory
61      if test "x$withval" != "xyes" -a "x$withval" != "x"; then
62        dnl This ensures $withval is actually a directory and that it is absolute
63        ap_nghttp2_base="`cd $withval ; pwd`"
64      fi
65    ])
66    if test "x$ap_nghttp2_base" = "x"; then
67      AC_MSG_RESULT(none)
68    else
69      AC_MSG_RESULT($ap_nghttp2_base)
70    fi
71
72    dnl Run header and version checks
73    saved_CPPFLAGS="$CPPFLAGS"
74    saved_LIBS="$LIBS"
75    saved_LDFLAGS="$LDFLAGS"
76
77    dnl Before doing anything else, load in pkg-config variables
78    if test -n "$PKGCONFIG"; then
79      saved_PKG_CONFIG_PATH="$PKG_CONFIG_PATH"
80      AC_MSG_CHECKING([for pkg-config along $PKG_CONFIG_PATH])
81      if test "x$ap_nghttp2_base" != "x" ; then
82        if test -f "${ap_nghttp2_base}/lib/pkgconfig/libnghttp2.pc"; then
83          dnl Ensure that the given path is used by pkg-config too, otherwise
84          dnl the system libnghttp2.pc might be picked up instead.
85          PKG_CONFIG_PATH="${ap_nghttp2_base}/lib/pkgconfig${PKG_CONFIG_PATH+:}${PKG_CONFIG_PATH}"
86          export PKG_CONFIG_PATH
87        elif test -f "${ap_nghttp2_base}/lib64/pkgconfig/libnghttp2.pc"; then
88          dnl Ensure that the given path is used by pkg-config too, otherwise
89          dnl the system libnghttp2.pc might be picked up instead.
90          PKG_CONFIG_PATH="${ap_nghttp2_base}/lib64/pkgconfig${PKG_CONFIG_PATH+:}${PKG_CONFIG_PATH}"
91          export PKG_CONFIG_PATH
92        fi
93      fi
94      AC_ARG_ENABLE(nghttp2-staticlib-deps,APACHE_HELP_STRING(--enable-nghttp2-staticlib-deps,[link mod_http2 with dependencies of libnghttp2's static libraries (as indicated by "pkg-config --static"). Must be specified in addition to --enable-http2.]), [
95        if test "$enableval" = "yes"; then
96          PKGCONFIG_LIBOPTS="--static"
97        fi
98      ])
99      ap_nghttp2_libs="`$PKGCONFIG $PKGCONFIG_LIBOPTS --libs-only-l --silence-errors libnghttp2`"
100      if test $? -eq 0; then
101        ap_nghttp2_found="yes"
102        pkglookup="`$PKGCONFIG --cflags-only-I libnghttp2`"
103        APR_ADDTO(CPPFLAGS, [$pkglookup])
104        APR_ADDTO(MOD_CFLAGS, [$pkglookup])
105        pkglookup="`$PKGCONFIG $PKGCONFIG_LIBOPTS --libs-only-L libnghttp2`"
106        APR_ADDTO(LDFLAGS, [$pkglookup])
107        APR_ADDTO(MOD_LDFLAGS, [$pkglookup])
108        pkglookup="`$PKGCONFIG $PKGCONFIG_LIBOPTS --libs-only-other libnghttp2`"
109        APR_ADDTO(LDFLAGS, [$pkglookup])
110        APR_ADDTO(MOD_LDFLAGS, [$pkglookup])
111      fi
112      PKG_CONFIG_PATH="$saved_PKG_CONFIG_PATH"
113    fi
114
115    dnl fall back to the user-supplied directory if not found via pkg-config
116    if test "x$ap_nghttp2_base" != "x" -a "x$ap_nghttp2_found" = "x"; then
117      APR_ADDTO(CPPFLAGS, [-I$ap_nghttp2_base/include])
118      APR_ADDTO(MOD_CFLAGS, [-I$ap_nghttp2_base/include])
119      APR_ADDTO(LDFLAGS, [-L$ap_nghttp2_base/lib])
120      APR_ADDTO(MOD_LDFLAGS, [-L$ap_nghttp2_base/lib])
121      if test "x$ap_platform_runtime_link_flag" != "x"; then
122        APR_ADDTO(LDFLAGS, [$ap_platform_runtime_link_flag$ap_nghttp2_base/lib])
123        APR_ADDTO(MOD_LDFLAGS, [$ap_platform_runtime_link_flag$ap_nghttp2_base/lib])
124      fi
125    fi
126
127    AC_MSG_CHECKING([for nghttp2 version >= 1.2.1])
128    AC_TRY_COMPILE([#include <nghttp2/nghttp2ver.h>],[
129#if !defined(NGHTTP2_VERSION_NUM)
130#error "Missing nghttp2 version"
131#endif
132#if NGHTTP2_VERSION_NUM < 0x010201
133#error "Unsupported nghttp2 version " NGHTTP2_VERSION_TEXT
134#endif],
135      [AC_MSG_RESULT(OK)
136       ac_cv_nghttp2=yes],
137      [AC_MSG_RESULT(FAILED)])
138
139    if test "x$ac_cv_nghttp2" = "xyes"; then
140      ap_nghttp2_libs="${ap_nghttp2_libs:--lnghttp2} `$apr_config --libs`"
141      APR_ADDTO(MOD_LDFLAGS, [$ap_nghttp2_libs])
142      APR_ADDTO(LIBS, [$ap_nghttp2_libs])
143
144      dnl Run library and function checks
145      liberrors=""
146      AC_CHECK_HEADERS([nghttp2/nghttp2.h])
147      AC_CHECK_FUNCS([nghttp2_session_server_new2], [], [liberrors="yes"])
148      if test "x$liberrors" != "x"; then
149        AC_MSG_WARN([nghttp2 library is unusable])
150      fi
151dnl # nghttp2 >= 1.3.0: access to stream weights
152      AC_CHECK_FUNCS([nghttp2_stream_get_weight], [], [liberrors="yes"])
153      if test "x$liberrors" != "x"; then
154        AC_MSG_WARN([nghttp2 version >= 1.3.0 is required])
155      fi
156dnl # nghttp2 >= 1.5.0: changing stream priorities
157      AC_CHECK_FUNCS([nghttp2_session_change_stream_priority], 
158        [APR_ADDTO(MOD_CPPFLAGS, ["-DH2_NG2_CHANGE_PRIO"])], [])
159dnl # nghttp2 >= 1.14.0: invalid header callback
160      AC_CHECK_FUNCS([nghttp2_session_callbacks_set_on_invalid_header_callback], 
161        [APR_ADDTO(MOD_CPPFLAGS, ["-DH2_NG2_INVALID_HEADER_CB"])], [])
162dnl # nghttp2 >= 1.15.0: get/set stream window sizes
163      AC_CHECK_FUNCS([nghttp2_session_get_stream_local_window_size], 
164        [APR_ADDTO(MOD_CPPFLAGS, ["-DH2_NG2_LOCAL_WIN_SIZE"])], [])
165dnl # nghttp2 >= 1.15.0: don't keep info on closed streams
166      AC_CHECK_FUNCS([nghttp2_option_set_no_closed_streams],
167        [APR_ADDTO(MOD_CPPFLAGS, ["-DH2_NG2_NO_CLOSED_STREAMS"])], [])
168dnl # nghttp2 >= 1.50.0: rfc9113 leading/trailing whitespec strictness
169      AC_CHECK_FUNCS([nghttp2_option_set_no_rfc9113_leading_and_trailing_ws_validation],
170        [APR_ADDTO(MOD_CPPFLAGS, ["-DH2_NG2_RFC9113_STRICTNESS"])], [])
171    else
172      AC_MSG_WARN([nghttp2 version is too old])
173    fi
174
175    dnl restore
176    CPPFLAGS="$saved_CPPFLAGS"
177    LIBS="$saved_LIBS"
178    LDFLAGS="$saved_LDFLAGS"
179  ])
180  if test "x$ac_cv_nghttp2" = "xyes"; then
181    AC_DEFINE(HAVE_NGHTTP2, 1, [Define if nghttp2 is available])
182  fi
183])

186dnl # hook module into the Autoconf mechanism (--enable-http2)
187APACHE_MODULE(http2, [HTTP/2 protocol handling in addition to HTTP protocol 
188handling. Implemented by mod_http2. This module requires a libnghttp2 installation. 
189See --with-nghttp2 on how to manage non-standard locations. This module
190is usually linked shared and requires loading. ], $http2_objs, , most, [
191    APACHE_CHECK_OPENSSL
192    if test "$ac_cv_openssl" = "yes" ; then
193        APR_ADDTO(MOD_CPPFLAGS, ["-DH2_OPENSSL"])
194    fi
195
196    APACHE_CHECK_NGHTTP2
197    if test "$ac_cv_nghttp2" = "yes" ; then
198        if test "x$enable_http2" = "xshared"; then
199           # The only symbol which needs to be exported is the module
200           # structure, so ask libtool to hide everything else:
201           APR_ADDTO(MOD_HTTP2_LDADD, [-export-symbols-regex http2_module])
202        fi
203    else
204        enable_http2=no
205    fi
206])
