73/*******************************************************************************
74 * iqueue - sorted list of int with user defined ordering
75 ******************************************************************************/
76typedef struct h2_iqueue {
77    int *elts;
78    int head;
79    int nelts;
80    int nalloc;
81    apr_pool_t *pool;
82} h2_iqueue;
83
84/**
85 * Comparator for two int to determine their order.
86 *
87 * @param i1 first int to compare
88 * @param i2 second int to compare
89 * @param ctx provided user data
90 * @return value is the same as for strcmp() and has the effect:
91 *    == 0: s1 and s2 are treated equal in ordering
92 *     < 0: s1 should be sorted before s2
93 *     > 0: s2 should be sorted before s1
94 */
95typedef int h2_iq_cmp(int i1, int i2, void *ctx);
96
97/**
98 * Allocate a new queue from the pool and initialize.
99 * @param pool the memory pool
100 * @param capacity the initial capacity of the queue
101 */
102h2_iqueue *h2_iq_create(apr_pool_t *pool, int capacity);
103
104/**
105 * Return != 0 iff there are no ints in the queue.
106 * @param q the queue to check
107 */
108int h2_iq_empty(h2_iqueue *q);
109
110/**
111 * Return the number of int in the queue.
112 * @param q the queue to get size on
113 */
114int h2_iq_count(h2_iqueue *q);
115
116/**
117 * Add a stream id to the queue. 
118 *
119 * @param q the queue to append the id to
120 * @param sid the stream id to add
121 * @param cmp the comparator for sorting
122 * @param ctx user data for comparator
123 * @return != 0 iff id was not already there 
124 */
125int h2_iq_add(h2_iqueue *q, int sid, h2_iq_cmp *cmp, void *ctx);
126
127/**
128 * Append the id to the queue if not already present. 
129 *
130 * @param q the queue to append the id to
131 * @param sid the id to append
132 * @return != 0 iff id was not already there 
133 */
134int h2_iq_append(h2_iqueue *q, int sid);
135
136/**
137 * Remove the int from the queue. Return != 0 iff it was found.
138 * @param q the queue
139 * @param sid the stream id to remove
140 * @return != 0 iff int was found in queue
141 */
142int h2_iq_remove(h2_iqueue *q, int sid);
143
144/**
145 * Remove all entries in the queue.
146 */
147void h2_iq_clear(h2_iqueue *q);
148
149/**
150 * Sort the stream idqueue again. Call if the int ordering
151 * has changed.
152 *
153 * @param q the queue to sort
154 * @param cmp the comparator for sorting
155 * @param ctx user data for the comparator 
156 */
157void h2_iq_sort(h2_iqueue *q, h2_iq_cmp *cmp, void *ctx);
158
159/**
160 * Get the first id from the queue or 0 if the queue is empty. 
161 * The id is being removed.
162 *
163 * @param q the queue to get the first id from
164 * @return the first id of the queue, 0 if empty
165 */
166int h2_iq_shift(h2_iqueue *q);
167
168/**
169 * Get the first max ids from the queue. All these ids will be removed.
170 *
171 * @param q the queue to get the first ids from
172 * @param pint the int array to receive the values
173 * @param max the maximum number of ids to shift
174 * @return the actual number of ids shifted
175 */
176size_t h2_iq_mshift(h2_iqueue *q, int *pint, size_t max);
177
178/**
179 * Determine if int is in the queue already
180 *
181 * @param q the queue
182 * @param sid the integer id to check for
183 * @return != 0 iff sid is already in the queue
184 */
185int h2_iq_contains(h2_iqueue *q, int sid);

191/**
192 * A thread-safe FIFO queue with some extra bells and whistles, if you
193 * do not need anything special, better use 'apr_queue'.
194 */
195typedef struct h2_fifo h2_fifo;
196
197/**
198 * Create a FIFO queue that can hold up to capacity elements. Elements can
199 * appear several times.
200 */
201apr_status_t h2_fifo_create(h2_fifo **pfifo, apr_pool_t *pool, int capacity);
202
203/**
204 * Create a FIFO set that can hold up to capacity elements. Elements only
205 * appear once. Pushing an element already present does not change the
206 * queue and is successful.
207 */
208apr_status_t h2_fifo_set_create(h2_fifo **pfifo, apr_pool_t *pool, int capacity);
209
210apr_status_t h2_fifo_term(h2_fifo *fifo);
211
212int h2_fifo_count(h2_fifo *fifo);
213
214/**
215 * Push en element into the queue. Blocks if there is no capacity left.
216 * 
217 * @param fifo the FIFO queue
218 * @param elem the element to push
219 * @return APR_SUCCESS on push, APR_EAGAIN on try_push on a full queue,
220 *         APR_EEXIST when in set mode and elem already there.
221 */
222apr_status_t h2_fifo_push(h2_fifo *fifo, void *elem);
223apr_status_t h2_fifo_try_push(h2_fifo *fifo, void *elem);
224
225apr_status_t h2_fifo_pull(h2_fifo *fifo, void **pelem);
226apr_status_t h2_fifo_try_pull(h2_fifo *fifo, void **pelem);
227
228typedef enum {
229    H2_FIFO_OP_PULL,   /* pull the element from the queue, ie discard it */
230    H2_FIFO_OP_REPUSH, /* pull and immediately re-push it */
231} h2_fifo_op_t;
232
233typedef h2_fifo_op_t h2_fifo_peek_fn(void *head, void *ctx);
234
235/**
236 * Call given function on the head of the queue, once it exists, and
237 * perform the returned operation on it. The queue will hold its lock during
238 * this time, so no other operations on the queue are possible.
239 * @param fifo the queue to peek at
240 * @param fn   the function to call on the head, once available
241 * @param ctx  context to pass in call to function
242 */
243apr_status_t h2_fifo_peek(h2_fifo *fifo, h2_fifo_peek_fn *fn, void *ctx);
244
245/**
246 * Non-blocking version of h2_fifo_peek.
247 */
248apr_status_t h2_fifo_try_peek(h2_fifo *fifo, h2_fifo_peek_fn *fn, void *ctx);
249
250/**
251 * Remove the elem from the queue, will remove multiple appearances.
252 * @param elem  the element to remove
253 * @return APR_SUCCESS iff > 0 elems were removed, APR_EAGAIN otherwise.
254 */
255apr_status_t h2_fifo_remove(h2_fifo *fifo, void *elem);

446/**
447 * Prints the brigade bucket types and lengths into the given buffer
448 * up to bmax.
449 * @return number of characters printed
450 */
451apr_size_t h2_util_bb_print(char *buffer, apr_size_t bmax, 
452                            const char *tag, const char *sep, 
453                            apr_bucket_brigade *bb);
454/**
455 * Logs the bucket brigade (which bucket types with what length)
456 * to the log at the given level.
457 * @param c the connection to log for
458 * @param sid the stream identifier this brigade belongs to
459 * @param level the log level (as in APLOG_*)
460 * @param tag a short message text about the context
461 * @param bb the brigade to log
462 */
463#define h2_util_bb_log(c, sid, level, tag, bb) \
464if (APLOG_C_IS_LEVEL(c, level)) { \
465    do { \
466        char buffer[4 * 1024]; \
467        const char *line = "(null)"; \
468        apr_size_t len, bmax = sizeof(buffer)/sizeof(buffer[0]); \
469        len = h2_util_bb_print(buffer, bmax, (tag), "", (bb)); \
470        ap_log_cerror(APLOG_MARK, level, 0, (c), "bb_dump(%ld): %s", \
471            ((c)->master? (c)->master->id : (c)->id), (len? buffer : line)); \
472    } while(0); \
473}
