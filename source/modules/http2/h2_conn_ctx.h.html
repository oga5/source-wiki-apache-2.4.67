32/**
33 * The h2 module context associated with a connection.
34 *
35 * It keeps track of the different types of connections:
36 * - those from clients that use HTTP/2 protocol
37 * - those from clients that do not use HTTP/2
38 * - those created by ourself to perform work on HTTP/2 streams
39 */ view japanese
40struct h2_conn_ctx_t {
41    const char *id;                 /* c*: our identifier of this connection */ view japanese
42    server_rec *server;             /* c*: httpd server selected. */ view japanese
43    const char *protocol;           /* c1: the protocol negotiated */ view japanese
44    struct h2_session *session;     /* c1: the h2 session established */ view japanese
45    struct h2_mplx *mplx;           /* c2: the multiplexer */ view japanese
46    struct h2_c2_transit *transit;  /* c2: transit pool and bucket_alloc */ view japanese
47
48#if !AP_HAS_RESPONSE_BUCKETS
49    int pre_conn_done;               /* has pre_connection setup run? */ view japanese
50#endif
51    int stream_id;                  /* c1: 0, c2: stream id processed */ view japanese
52    apr_pool_t *req_pool;            /* c2: a c2 child pool for a request */ view japanese
53    const struct h2_request *request; /* c2: the request to process */ view japanese
54    struct h2_bucket_beam *beam_out; /* c2: data out, created from req_pool */ view japanese
55    struct h2_bucket_beam *beam_in;  /* c2: data in or NULL, borrowed from request stream */ view japanese
56    unsigned input_chunked:1;        /* c2: if input needs HTTP/1.1 chunking applied */ view japanese
57    unsigned is_upgrade:1;           /* c2: if requst is a HTTP Upgrade */ view japanese
58
59    apr_file_t *pipe_in[2];          /* c2: input produced notification pipe */ view japanese
60    apr_pollfd_t pfd;                /* c1: poll socket input, c2: NUL */ view japanese
61
62    int has_final_response;          /* final HTTP response passed on out */ view japanese
63    apr_status_t last_err;           /* APR_SUCCES or last error encountered in filters */ view japanese
64
65    apr_off_t bytes_sent;            /* c2: bytes acutaly sent via c1 */ view japanese
66    /* atomic */ apr_uint32_t started; /* c2: processing was started */ view japanese
67    apr_time_t started_at;           /* c2: when processing started */ view japanese
68    /* atomic */ apr_uint32_t done;  /* c2: processing has finished */ view japanese view japanese
69    apr_time_t done_at;              /* c2: when processing was done */ view japanese
70};
71typedef struct h2_conn_ctx_t h2_conn_ctx_t;
