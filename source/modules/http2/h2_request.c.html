47h2_request *h2_request_create(int id, apr_pool_t *pool, const char *method,
48                              const char *scheme, const char *authority,
49                              const char *path, apr_table_t *header)
50{
51    h2_request *req = apr_pcalloc(pool, sizeof(h2_request));
52
53    req->method         = method;
54    req->scheme         = scheme;
55    req->authority      = authority;
56    req->path           = path;
57    req->headers        = header? header : apr_table_make(pool, 10);
58    req->request_time   = apr_time_now();
59
60    return req;
61}

63typedef struct {
64    apr_table_t *headers;
65    apr_pool_t *pool;
66    apr_status_t status;
67    h2_hd_scratch *scratch;
68} h1_ctx;
69
70static int set_h1_header(void *ctx, const char *key, const char *value)
71{
72    h1_ctx *x = ctx;
73    int was_added;
74    h2_req_add_header(x->headers, x->pool, key, strlen(key),
75                      value, strlen(value), x->scratch, &was_added);
76    return 1;
77}

79apr_status_t h2_request_rcreate(h2_request **preq, apr_pool_t *pool, 
80                                request_rec *r, h2_hd_scratch *scratch)
81{
82    h2_request *req;
83    const char *scheme, *authority, *path;
84    h1_ctx x;
85    
86    *preq = NULL;
87    scheme = apr_pstrdup(pool, r->parsed_uri.scheme? r->parsed_uri.scheme
88              : ap_http_scheme(r));
89    authority = apr_pstrdup(pool, r->hostname);
90    path = apr_uri_unparse(pool, &r->parsed_uri, APR_URI_UNP_OMITSITEPART);
91    
92    if (!r->method || !scheme || !r->hostname || !path) {
93        return APR_EINVAL;
94    }
95
96    /* The authority we carry in h2_request is the 'authority' part of
97     * the URL for the request. r->hostname has stripped any port info that
98     * might have been present. Do we need to add it?
99     */ view japanese
100    if (!ap_strchr_c(authority, ':')) {
101        if (r->parsed_uri.port_str) {
102            /* Yes, it was there, add it again. */ view japanese
103            authority = apr_pstrcat(pool, authority, ":", r->parsed_uri.port_str, NULL);
104        }
105        else if (!r->parsed_uri.hostname && r->server && r->server->port) {
106            /* If there was no hostname in the parsed URL, the URL was relative.
107             * In that case, we restore port from our server->port, if it
108             * is known and not the default port for the scheme. */ view japanese
109            apr_port_t defport = apr_uri_port_of_scheme(scheme);
110            if (defport != r->server->port) {
111                /* port info missing and port is not default for scheme: append */ view japanese
112                authority = apr_psprintf(pool, "%s:%d", authority,
113                                         (int)r->server->port);
114            }
115        }
116    }
117
118    req = apr_pcalloc(pool, sizeof(*req));
119    req->method      = apr_pstrdup(pool, r->method);
120    req->scheme      = scheme;
121    req->authority   = authority;
122    req->path        = path;
123    req->headers     = apr_table_make(pool, 10);
124    req->http_status = H2_HTTP_STATUS_UNSET;
125    req->request_time = apr_time_now();
126
127    x.pool = pool;
128    x.headers = req->headers;
129    x.status = APR_SUCCESS;
130    x.scratch = scratch;
131    apr_table_do(set_h1_header, &x, r->headers_in, NULL);
132
133    *preq = req;
134    return x.status;
135}

137apr_status_t h2_request_add_header(h2_request *req, apr_pool_t *pool,
138                                   const char *name, size_t nlen,
139                                   const char *value, size_t vlen,
140                                   struct h2_hd_scratch *scratch,
141                                   int *pwas_added)
142{
143    apr_status_t status = APR_SUCCESS;
144
145    *pwas_added = 0;
146    if (nlen <= 0) {
147        return status;
148    }
149
150    if (name[0] == ':') {
151        /* pseudo header, see ch. 8.1.2.3, always should come first */ view japanese
152        if (!apr_is_empty_table(req->headers)) {
153            ap_log_perror(APLOG_MARK, APLOG_ERR, 0, pool,
154                          APLOGNO(02917)
155                          "h2_request: pseudo header after request start");
156            return APR_EGENERAL;
157        }
158
159        if (H2_HEADER_METHOD_LEN == nlen
160            && !strncmp(H2_HEADER_METHOD, name, nlen)) {
161            req->method = apr_pstrndup(pool, value, vlen);
162        }
163        else if (H2_HEADER_SCHEME_LEN == nlen
164                 && !strncmp(H2_HEADER_SCHEME, name, nlen)) {
165            req->scheme = apr_pstrndup(pool, value, vlen);
166        }
167        else if (H2_HEADER_PATH_LEN == nlen
168                 && !strncmp(H2_HEADER_PATH, name, nlen)) {
169            req->path = apr_pstrndup(pool, value, vlen);
170        }
171        else if (H2_HEADER_AUTH_LEN == nlen
172                 && !strncmp(H2_HEADER_AUTH, name, nlen)) {
173            req->authority = apr_pstrndup(pool, value, vlen);
174        }
175        else if (H2_HEADER_PROTO_LEN == nlen
176                 && !strncmp(H2_HEADER_PROTO, name, nlen)) {
177            req->protocol = apr_pstrndup(pool, value, vlen);
178        }
179        else {
180            char buffer[32];
181            memset(buffer, 0, 32);
182            strncpy(buffer, name, (nlen > 31)? 31 : nlen);
183            ap_log_perror(APLOG_MARK, APLOG_WARNING, 0, pool,
184                          APLOGNO(02954)
185                          "h2_request: ignoring unknown pseudo header %s",
186                          buffer);
187        }
188    }
189    else {
190        /* non-pseudo header, add to table */ view japanese
191        status = h2_req_add_header(req->headers, pool, name, nlen, value, vlen,
192                                   scratch, pwas_added);
193    }
194
195    return status;
196}

198apr_status_t h2_request_end_headers(h2_request *req, apr_pool_t *pool,
199                                    size_t raw_bytes)
200{
201    /* rfc7540, ch. 8.1.2.3: without :authority, Host: must be there */ view japanese
202    if (req->authority && !strlen(req->authority)) {
203        req->authority = NULL;
204    }
205    if (!req->authority) {
206        const char *host = apr_table_get(req->headers, "Host");
207        if (!host) {
208            return APR_BADARG;
209        }
210        req->authority = host;
211    }
212    else {
213        apr_table_setn(req->headers, "Host", req->authority);
214    }
215    req->raw_bytes += raw_bytes;
216
217    return APR_SUCCESS;
218}

220h2_request *h2_request_clone(apr_pool_t *p, const h2_request *src)
221{
222    h2_request *dst = apr_pmemdup(p, src, sizeof(*dst));
223    dst->method       = apr_pstrdup(p, src->method);
224    dst->scheme       = apr_pstrdup(p, src->scheme);
225    dst->authority    = apr_pstrdup(p, src->authority);
226    dst->path         = apr_pstrdup(p, src->path);
227    dst->protocol     = apr_pstrdup(p, src->protocol);
228    dst->headers      = apr_table_clone(p, src->headers);
229    return dst;
230}

289#if AP_HAS_RESPONSE_BUCKETS
290apr_bucket *h2_request_create_bucket(const h2_request *req, request_rec *r)
291{
292    conn_rec *c = r->connection;
293    apr_table_t *headers = apr_table_clone(r->pool, req->headers);
294    const char *uri = req->path;
295
296    AP_DEBUG_ASSERT(req->method);
297    AP_DEBUG_ASSERT(req->authority);
298    if (!ap_cstr_casecmp("CONNECT", req->method))  {
299        uri = req->authority;
300    }
301    else if (h2_config_cgeti(c, H2_CONF_PROXY_REQUESTS)) {
302        /* Forward proxying: always absolute uris */ view japanese
303        uri = apr_psprintf(r->pool, "%s://%s%s",
304                           req->scheme, req->authority,
305                           req->path ? req->path : "");
306    }
307    else if (req->scheme && ap_cstr_casecmp(req->scheme, "http")
308             && ap_cstr_casecmp(req->scheme, "https")) {
309        /* Client sent a non-http ':scheme', use an absolute URI */ view japanese
310        uri = apr_psprintf(r->pool, "%s://%s%s", view japanese
311                           req->scheme, req->authority, req->path ? req->path : "");
312    }
313
314    return ap_bucket_request_create(req->method, uri, "HTTP/2.0", headers,
315                                    r->pool, c->bucket_alloc);
316}

319static void assign_headers(request_rec *r, const h2_request *req,
320                           int no_body, int is_connect)
321{
322    const char *cl;
323
324    r->headers_in = apr_table_clone(r->pool, req->headers);
325
326    if (req->authority && !is_connect) {
327        /* for internal handling, we have to simulate that :authority
328         * came in as Host:, RFC 9113 ch. says that mismatches between
329         * :authority and Host: SHOULD be rejected as malformed. However,
330         * we are more lenient and just replace any Host: if we have
331         * an :authority.
332         */ view japanese
333        const char *orig_host = apr_table_get(req->headers, "Host");
334        if (orig_host && strcmp(req->authority, orig_host)) {
335            ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(10401)
336                          "overwriting 'Host: %s' with :authority: %s'",
337                          orig_host, req->authority);
338            apr_table_setn(r->subprocess_env, "H2_ORIGINAL_HOST", orig_host);
339        }
340        apr_table_setn(r->headers_in, "Host", req->authority);
341        ap_log_rerror(APLOG_MARK, APLOG_TRACE1, 0, r,
342                      "set 'Host: %s' from :authority", req->authority);
343    }
344
345    /* Unless we open a byte stream via CONNECT, apply content-length guards. */ view japanese
346    if (!is_connect) {
347        cl = apr_table_get(req->headers, "Content-Length");
348        if (no_body) {
349            if (!cl && apr_table_get(req->headers, "Content-Type")) {
350                /* If we have a content-type, but already seen eos, no more
351                 * data will come. Signal a zero content length explicitly.
352                 */ view japanese
353                apr_table_setn(req->headers, "Content-Length", "0");
354            }
355        }
356#if !AP_HAS_RESPONSE_BUCKETS
357        else if (!cl) {
358            /* there may be a body and we have internal HTTP/1.1 processing.
359             * If the Content-Length is unspecified, we MUST simulate
360             * chunked Transfer-Encoding.
361             *
362             * HTTP/2 does not need a Content-Length for framing. Ideally
363             * all clients set the EOS flag on the header frame if they
364             * do not intent to send a body. However, forwarding proxies
365             * might just no know at the time and send an empty DATA
366             * frame with EOS much later.
367             */ view japanese
368            apr_table_mergen(r->headers_in, "Transfer-Encoding", "chunked");
369        }
370#endif /* else AP_HAS_RESPONSE_BUCKETS */ view japanese
371  }
372}

374request_rec *h2_create_request_rec(const h2_request *req, conn_rec *c,
375                                   int no_body)
376{
377    int access_status = HTTP_OK;
378    int is_connect = !ap_cstr_casecmp("CONNECT", req->method);
379
380#if AP_MODULE_MAGIC_AT_LEAST(20120211, 106)
381    request_rec *r = ap_create_request(c);
382#else
383    request_rec *r = my_ap_create_request(c);
384#endif
385
386#if AP_MODULE_MAGIC_AT_LEAST(20120211, 107)
387    assign_headers(r, req, no_body, is_connect);
388    ap_run_pre_read_request(r, c);
389
390    /* Time to populate r with the data we have. */
391    r->request_time = req->request_time;
392    AP_DEBUG_ASSERT(req->authority);
393    if (req->http_status != H2_HTTP_STATUS_UNSET) {
394        access_status = req->http_status;
395        goto die;
396    }
397    else if (is_connect) {
398      /* CONNECT MUST NOT have scheme or path */ view japanese
399        r->the_request = apr_psprintf(r->pool, "%s %s HTTP/2.0",
400                                      req->method, req->authority);
401        if (req->scheme) {
402            ap_log_cerror(APLOG_MARK, APLOG_DEBUG, 0, c, APLOGNO(10458)
403                          "':scheme: %s' header present in CONNECT request",
404                          req->scheme);
405            access_status = HTTP_BAD_REQUEST;
406            goto die;
407        }
408        else if (req->path) {
409            ap_log_cerror(APLOG_MARK, APLOG_DEBUG, 0, c, APLOGNO(10459)
410                          "':path: %s' header present in CONNECT request",
411                          req->path);
412            access_status = HTTP_BAD_REQUEST;
413            goto die;
414        }
415    }
416    else if (req->protocol) {
417      ap_log_cerror(APLOG_MARK, APLOG_DEBUG, 0, c, APLOGNO(10470)
418                    "':protocol: %s' header present in %s request",
419                    req->protocol, req->method);
420      access_status = HTTP_BAD_REQUEST;
421      goto die;
422    }
423    else if (h2_config_cgeti(c, H2_CONF_PROXY_REQUESTS)) {
424        if (!req->scheme) {
425            ap_log_cerror(APLOG_MARK, APLOG_DEBUG, 0, c, APLOGNO(10468)
426                          "H2ProxyRequests on, but request misses :scheme");
427            access_status = HTTP_BAD_REQUEST;
428            goto die;
429        }
430        if (!req->authority) {
431            ap_log_cerror(APLOG_MARK, APLOG_DEBUG, 0, c, APLOGNO(10469)
432                          "H2ProxyRequests on, but request misses :authority");
433            access_status = HTTP_BAD_REQUEST;
434            goto die;
435        }
436        r->the_request = apr_psprintf(r->pool, "%s %s://%s%s HTTP/2.0",
437                                      req->method, req->scheme, req->authority,
438                                      req->path ? req->path : "");
439    }
440    else if (req->scheme && ap_cstr_casecmp(req->scheme, "http")
441             && ap_cstr_casecmp(req->scheme, "https")) {
442        /* Client sent a ':scheme' pseudo header for something else
443         * than what we have on this connection. Make an absolute URI. */ view japanese
444        r->the_request = apr_psprintf(r->pool, "%s %s://%s%s HTTP/2.0", view japanese
445                                      req->method, req->scheme, req->authority,
446                                      req->path ? req->path : "");
447    }
448    else if (req->path) {
449        r->the_request = apr_psprintf(r->pool, "%s %s HTTP/2.0",
450                                      req->method, req->path);
451    }
452    else {
453        /* We should only come here on a request that is errored already.
454         * create a request line that passes parsing, we'll die anyway.
455         */ view japanese
456        AP_DEBUG_ASSERT(req->http_status != H2_HTTP_STATUS_UNSET);
457        r->the_request = apr_psprintf(r->pool, "%s / HTTP/2.0", req->method);
458    }
459
460    /* Start with r->hostname = NULL, ap_check_request_header() will get it
461     * form Host: header, otherwise we get complains about port numbers.
462     */ view japanese
463    r->hostname = NULL;
464
465    /* Validate HTTP/1 request and select vhost. */ view japanese
466    if (!ap_parse_request_line(r) || !ap_check_request_header(r)) {
467        /* we may have switched to another server still */ view japanese
468        r->per_dir_config = r->server->lookup_defaults;
469        if (req->http_status != H2_HTTP_STATUS_UNSET) {
470            access_status = req->http_status;
471            /* Be safe and close the connection */
472            c->keepalive = AP_CONN_CLOSE;
473        }
474        else {
475            access_status = r->status;
476        }
477        r->status = HTTP_OK;
478        goto die;
479    }
480#else
481    {
482        const char *s;
483
484        assign_headers(r, req, no_body, is_connect);
485        ap_run_pre_read_request(r, c);
486
487        /* Time to populate r with the data we have. */ view japanese
488        r->request_time = req->request_time;
489        r->method = apr_pstrdup(r->pool, req->method);
490        /* Provide quick information about the request method as soon as known */ view japanese
491        r->method_number = ap_method_number_of(r->method);
492        if (r->method_number == M_GET && r->method[0] == 'H') {
493            r->header_only = 1;
494        }
495        ap_parse_uri(r, req->path ? req->path : "");
496        r->protocol = (char*)"HTTP/2.0";
497        r->proto_num = HTTP_VERSION(2, 0);
498        r->the_request = apr_psprintf(r->pool, "%s %s HTTP/2.0",
499                                      r->method, req->path ? req->path : "");
500
501        /* Start with r->hostname = NULL, ap_check_request_header() will get it
502         * form Host: header, otherwise we get complains about port numbers.
503         */ view japanese
504        r->hostname = NULL;
505        ap_update_vhost_from_headers(r);
506
507         /* we may have switched to another server */
508         r->per_dir_config = r->server->lookup_defaults;
509
510         s = apr_table_get(r->headers_in, "Expect");
511         if (s && s[0]) {
512            if (ap_cstr_casecmp(s, "100-continue") == 0) {
513                r->expecting_100 = 1;
514            }
515            else {
516                r->status = HTTP_EXPECTATION_FAILED;
517                access_status = r->status;
518                goto die;
519            }
520         }
521    }
522#endif
523
524    /* we may have switched to another server */ view japanese
525    r->per_dir_config = r->server->lookup_defaults;
526
527    if (req->http_status != H2_HTTP_STATUS_UNSET) {
528        access_status = req->http_status;
529        r->status = HTTP_OK;
530        /* Be safe and close the connection */ view japanese
531        c->keepalive = AP_CONN_CLOSE;
532        goto die;
533    }
534
535    /*
536     * Add the HTTP_IN filter here to ensure that ap_discard_request_body
537     * called by ap_die and by ap_send_error_response works correctly on
538     * status codes that do not cause the connection to be dropped and
539     * in situations where the connection should be kept alive.
540     */ view japanese
541    ap_add_input_filter_handle(ap_http_input_filter_handle,
542                               NULL, r, r->connection);
543
544    if ((access_status = ap_post_read_request(r))) {
545        /* Request check post hooks failed. An example of this would be a
546         * request for a vhost where h2 is disabled --> 421.
547         */ view japanese
548        ap_log_cerror(APLOG_MARK, APLOG_DEBUG, 0, c, APLOGNO(03367)
549                      "h2_request: access_status=%d, request_create failed",
550                      access_status);
551        goto die;
552    }
553
554    AP_READ_REQUEST_SUCCESS((uintptr_t)r, (char *)r->method,
555                            (char *)r->uri, (char *)r->server->defn_name,
556                            r->status);
557    return r;
558
559die:
560    if (!r->method) {
561        /* if we fail early, `r` is not properly initialized for error
562         * processing which accesses fields in message generation.
563         * Make a best effort. */ view japanese
564        if (!r->the_request) {
565                r->the_request = apr_psprintf(r->pool, "%s %s HTTP/2.0",
566                                      req->method, req->path);
567        }
568        ap_parse_request_line(r);
569    }
570    ap_log_cerror(APLOG_MARK, APLOG_TRACE2, 0, c,
571                  "ap_die(%d) for %s", access_status, r->the_request);
572    ap_die(access_status, r);
573
574    /* ap_die() sent the response through the output filters, we must now
