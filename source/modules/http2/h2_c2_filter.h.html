43/**
44 * h2_from_h1 parses a HTTP/1.1 response into
45 * - response status
46 * - a list of header values
47 * - a series of bytes that represent the response body alone, without
48 *   any meta data, such as inserted by chunked transfer encoding.
49 *
50 * All data is allocated from the stream memory pool.
51 *
52 * Again, see comments in h2_request: ideally we would take the headers
53 * and status from the httpd structures instead of parsing them here, but
54 * we need to have all handlers and filters involved in request/response
55 * processing, so this seems to be the way for now.
56 */
57struct h2_headers;
58struct h2_response_parser;
59
60apr_status_t h2_c2_filter_catch_h1_out(ap_filter_t* f, apr_bucket_brigade* bb);
61
62apr_status_t h2_c2_filter_response_out(ap_filter_t *f, apr_bucket_brigade *bb);
63
64apr_status_t h2_c2_filter_trailers_out(ap_filter_t *f, apr_bucket_brigade *bb);
65
66#endif /* else AP_HAS_RESPONSE_BUCKETS */
67
68#endif /* defined(__mod_h2__h2_c2_filter__) */
