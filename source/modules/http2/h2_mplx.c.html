112static int stream_is_running(h2_stream *stream)
113{
114    h2_conn_ctx_t *conn_ctx = h2_conn_ctx_get(stream->c2);
115    return conn_ctx && apr_atomic_read32(&conn_ctx->started) != 0
116        && apr_atomic_read32(&conn_ctx->done) == 0;
117}

129static int add_for_purge(h2_mplx *m, h2_stream *stream)
130{
131    int i;
132    for (i = 0; i < m->spurge->nelts; ++i) {
133        h2_stream *s = APR_ARRAY_IDX(m->spurge, i, h2_stream*);
134        if (s == stream)  /* already scheduled for purging */ view japanese
135            return FALSE;
136    }
137    APR_ARRAY_PUSH(m->spurge, h2_stream *) = stream;
138    return TRUE;
139}

149static void m_stream_cleanup(h2_mplx *m, h2_stream *stream)
150{
151    h2_conn_ctx_t *c2_ctx = h2_conn_ctx_get(stream->c2);
152
153    ap_log_cerror(APLOG_MARK, APLOG_TRACE2, 0, m->c1,
154                  H2_STRM_MSG(stream, "cleanup, unsubscribing from beam events"));
155    if (c2_ctx) {
156        if (c2_ctx->beam_out) {
157            h2_beam_on_was_empty(c2_ctx->beam_out, NULL, NULL);
158        }
159        if (c2_ctx->beam_in) {
160            h2_beam_on_send(c2_ctx->beam_in, NULL, NULL);
161            h2_beam_on_received(c2_ctx->beam_in, NULL, NULL);
162            h2_beam_on_eagain(c2_ctx->beam_in, NULL, NULL);
163            h2_beam_on_consumed(c2_ctx->beam_in, NULL, NULL);
164        }
165    }
166
167    ap_log_cerror(APLOG_MARK, APLOG_TRACE2, 0, m->c1,
168                  H2_STRM_MSG(stream, "cleanup, removing from registries"));
169    ap_assert(stream->state == H2_SS_CLEANUP);
170    h2_stream_cleanup(stream);
171    h2_ihash_remove(m->streams, stream->id);
172    h2_iq_remove(m->q, stream->id);
173
174    if (c2_ctx) {
175        if (!stream_is_running(stream)) {
176            ap_log_cerror(APLOG_MARK, APLOG_TRACE2, 0, m->c1,
177                          H2_STRM_MSG(stream, "cleanup, c2 is done, move to spurge"));
178            /* processing has finished */ view japanese
179            add_for_purge(m, stream);
180        }
181        else {
182            ap_log_cerror(APLOG_MARK, APLOG_TRACE2, 0, m->c1,
183                          H2_STRM_MSG(stream, "cleanup, c2 is running, abort"));
184            /* c2 is still running */ view japanese
185            h2_c2_abort(stream->c2, m->c1);
186            ap_log_cerror(APLOG_MARK, APLOG_TRACE2, 0, m->c1,
187                          H2_STRM_MSG(stream, "cleanup, c2 is done, move to shold"));
188            h2_ihash_add(m->shold, stream);
189        }
190    }
191    else {
192        /* never started */ view japanese
193        int added = add_for_purge(m, stream);
194        if (added)
195            ap_log_cerror(APLOG_MARK, APLOG_TRACE2, 0, m->c1,
196                          H2_STRM_MSG(stream, "cleanup, never started, move to spurge"));
197    }
198}

200static h2_c2_transit *c2_transit_create(h2_mplx *m)
201{
202    apr_allocator_t *allocator;
203    apr_pool_t *ptrans;
204    h2_c2_transit *transit;
205    apr_status_t rv;
206
207    /* We create a pool with its own allocator to be used for
208     * processing a request. This is the only way to have the processing
209     * independent of its parent pool in the sense that it can work in
210     * another thread.
211     */ view japanese
212
213    rv = apr_allocator_create(&allocator);
214    if (rv == APR_SUCCESS) {
215        apr_allocator_max_free_set(allocator, ap_max_mem_free);
216        rv = apr_pool_create_ex(&ptrans, m->pool, NULL, allocator);
217    }
218    if (rv != APR_SUCCESS) {
219        /* maybe the log goes through, maybe not. */ view japanese
220        ap_log_cerror(APLOG_MARK, APLOG_ERR, rv, m->c1,
221                      APLOGNO(10004) "h2_mplx: create transit pool");
222        ap_abort_on_oom();
223        return NULL; /* should never be reached. */ view japanese
224    }
225
226    apr_allocator_owner_set(allocator, ptrans);
227    apr_pool_abort_set(abort_on_oom, ptrans);
228    apr_pool_tag(ptrans, "h2_c2_transit");
229
230    transit = apr_pcalloc(ptrans, sizeof(*transit));
231    transit->pool = ptrans;
232    transit->bucket_alloc = apr_bucket_alloc_create(ptrans);
233    return transit;
234}

261/**
262 * A h2_mplx needs to be thread-safe *and* if will be called by
263 * the h2_session thread *and* the h2_worker threads. Therefore:
264 * - calls are protected by a mutex lock, m->lock
265 * - the pool needs its own allocator, since apr_allocator_t are 
266 *   not re-entrant. The separate allocator works without a 
267 *   separate lock since we already protect h2_mplx itself.
268 *   Since HTTP/2 connections can be expected to live longer than
269 *   their HTTP/1 cousins, the separate allocator seems to work better
270 *   than protecting a shared h2_session one with an own lock.
271 */ view japanese
272h2_mplx *h2_mplx_c1_create(int child_num, apr_uint32_t id, h2_stream *stream0,
273                           server_rec *s, apr_pool_t *parent,
274                           h2_workers *workers)
275{
276    h2_conn_ctx_t *conn_ctx;
277    apr_status_t status = APR_SUCCESS;
278    apr_allocator_t *allocator;
279    apr_thread_mutex_t *mutex = NULL;
280    h2_mplx *m = NULL;
281    
282    m = apr_pcalloc(parent, sizeof(h2_mplx));
283    m->stream0 = stream0;
284    m->c1 = stream0->c2;
285    m->s = s;
286    m->child_num = child_num;
287    m->id = id;
288
289    /* We create a pool with its own allocator to be used for
290     * processing secondary connections. This is the only way to have the
291     * processing independent of its parent pool in the sense that it
292     * can work in another thread. Also, the new allocator needs its own
293     * mutex to synchronize sub-pools.
294     */ view japanese
295    status = apr_allocator_create(&allocator);
296    if (status != APR_SUCCESS) {
297        allocator = NULL;
298        goto failure;
299    }
300
301    apr_allocator_max_free_set(allocator, ap_max_mem_free);
302    apr_pool_create_ex(&m->pool, parent, NULL, allocator);
303    if (!m->pool) goto failure;
304
305    apr_pool_tag(m->pool, "h2_mplx");
306    apr_allocator_owner_set(allocator, m->pool);
307
308    status = apr_thread_mutex_create(&mutex, APR_THREAD_MUTEX_DEFAULT,
309                                     m->pool);
310    if (APR_SUCCESS != status) goto failure;
311    apr_allocator_mutex_set(allocator, mutex);
312
313    status = apr_thread_mutex_create(&m->lock, APR_THREAD_MUTEX_DEFAULT,
314                                     m->pool);
315    if (APR_SUCCESS != status) goto failure;
316
317    m->max_streams = h2_config_sgeti(s, H2_CONF_MAX_STREAMS);
318    m->stream_max_mem = h2_config_sgeti(s, H2_CONF_STREAM_MAX_MEM);
319
320    m->streams = h2_ihash_create(m->pool, offsetof(h2_stream,id));
321    m->shold = h2_ihash_create(m->pool, offsetof(h2_stream,id));
322    m->spurge = apr_array_make(m->pool, 10, sizeof(h2_stream*));
323    m->q = h2_iq_create(m->pool, m->max_streams);
324
325    m->workers = workers;
326    m->processing_max = H2MIN(h2_workers_get_max_workers(workers), m->max_streams);
327    m->processing_limit = 6; /* the original h1 max parallel connections */ view japanese
328    m->last_mood_change = apr_time_now();
329    m->mood_update_interval = apr_time_from_msec(100);
330
331    status = mplx_pollset_create(m);
332    if (APR_SUCCESS != status) {
333        ap_log_cerror(APLOG_MARK, APLOG_ERR, status, m->c1, APLOGNO(10308)
334                      "nghttp2: could not create pollset");
335        goto failure;
336    }
337    m->streams_ev_in = apr_array_make(m->pool, 10, sizeof(h2_stream*));
338    m->streams_ev_out = apr_array_make(m->pool, 10, sizeof(h2_stream*));
339
340    m->streams_input_read = h2_iq_create(m->pool, 10);
341    m->streams_output_written = h2_iq_create(m->pool, 10);
342    status = apr_thread_mutex_create(&m->poll_lock, APR_THREAD_MUTEX_DEFAULT,
343                                     m->pool);
344    if (APR_SUCCESS != status) goto failure;
345
346    conn_ctx = h2_conn_ctx_get(m->c1);
347    if (conn_ctx->pfd.reqevents) {
348        apr_pollset_add(m->pollset, &conn_ctx->pfd);
349    }
350
351    m->max_spare_transits = 3;
352    m->c2_transits = apr_array_make(m->pool, (int)m->max_spare_transits,
353                                    sizeof(h2_c2_transit*));
354
355    m->producer = h2_workers_register(workers, m->pool,
356                                      apr_psprintf(m->pool, "h2-%u",
357                                      (unsigned int)m->id),
358                                      c2_prod_next, c2_prod_done,
359                                      workers_shutdown, m);
360    return m;
361
362failure:
363    if (m->pool) {
364        apr_pool_destroy(m->pool);
365    }
366    else if (allocator) {
367        apr_allocator_destroy(allocator);
368    }
369    return NULL;
370}

510void h2_mplx_c1_destroy(h2_mplx *m)
511{
512    apr_status_t status;
513    unsigned int i, wait_secs = 60;
514    int old_aborted;
515
516    ap_log_cerror(APLOG_MARK, APLOG_TRACE2, 0, m->c1,
517                  H2_MPLX_MSG(m, "start release"));
518    /* How to shut down a h2 connection:
519     * 0. abort and tell the workers that no more work will come from us */ view japanese
520    m->shutdown = m->aborted = 1;
521
522    H2_MPLX_ENTER_ALWAYS(m);
523
524    /* While really terminating any c2 connections, treat the master
525     * connection as aborted. It's not as if we could send any more data
526     * at this point. */ view japanese
527    old_aborted = m->c1->aborted;
528    m->c1->aborted = 1;
529
530    /* How to shut down a h2 connection:
531     * 1. cancel all streams still active */ view japanese
532    ap_log_cerror(APLOG_MARK, APLOG_TRACE1, 0, m->c1,
533                  H2_MPLX_MSG(m, "release, %u/%u/%d streams (total/hold/purge), %d streams"),
534                  h2_ihash_count(m->streams),
535                  h2_ihash_count(m->shold),
536                  m->spurge->nelts, m->processing_count);
537    while (!h2_ihash_iter(m->streams, m_stream_cancel_iter, m)) {
538        /* until empty */ view japanese
539    }
540    
541    /* 2. no more streams should be scheduled or in the active set */ view japanese
542    ap_assert(h2_ihash_empty(m->streams));
543    ap_assert(h2_iq_empty(m->q));
544    
545    /* 3. while workers are busy on this connection, meaning they
546     *    are processing streams from this connection, wait on them finishing
547     *    in order to wake us and let us check again. 
548     *    Eventually, this has to succeed. */ view japanese
549    if (!m->join_wait) {
550        apr_thread_cond_create(&m->join_wait, m->pool);
551    }
552
553    for (i = 0; h2_ihash_count(m->shold) > 0; ++i) {
554        status = apr_thread_cond_timedwait(m->join_wait, m->lock, apr_time_from_sec(wait_secs));
555        
556        if (APR_STATUS_IS_TIMEUP(status)) {
557            /* This can happen if we have very long running requests
558             * that do not time out on IO. */ view japanese
559            ap_log_cerror(APLOG_MARK, APLOG_DEBUG, 0, m->c1, APLOGNO(03198)
560                          H2_MPLX_MSG(m, "waited %u sec for %u streams"),
561                          i*wait_secs, h2_ihash_count(m->shold));
562            h2_ihash_iter(m->shold, m_report_stream_iter, m);
