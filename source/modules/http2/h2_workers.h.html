17#ifndef __mod_h2__h2_workers__
18#define __mod_h2__h2_workers__
19
20/* Thread pool specific to executing secondary connections.
21 * Has a minimum and maximum number of workers it creates.
22 * Starts with minimum workers and adds some on load,
23 * reduces the number again when idle.
24 */
25struct apr_thread_mutex_t;
26struct apr_thread_cond_t;
27struct h2_mplx;
28struct h2_request;
29struct h2_fifo;
30
31typedef struct h2_workers h2_workers;
32
33
34/**
35 * Create a worker set with a maximum number of 'slots', e.g. worker
36 * threads to run. Always keep `min_active` workers running. Shutdown
37 * any additional workers after `idle_secs` seconds of doing nothing.
38 *
39 * @oaram s the base server
40 * @param pool for allocations
41 * @param min_active minimum number of workers to run
42 * @param max_slots maximum number of worker slots
43 * @param idle_limit upper duration of idle after a non-minimal slots shuts down
44 */
45h2_workers *h2_workers_create(server_rec *s, apr_pool_t *pool,
46                              int max_slots, int min_active, apr_time_t idle_limit);
47
48/**
49 *  Shut down processing.
50 */
51void h2_workers_shutdown(h2_workers *workers, int graceful);
52
53/**
54 * Get the maximum number of workers.
55 */
56apr_uint32_t h2_workers_get_max_workers(h2_workers *workers);
57
58/**
59 * ap_conn_producer_t is the source of connections (conn_rec*) to run.
60 *
61 * Active producers are queried by idle workers for connections.
62 * If they do not hand one back, they become inactive and are not
63 * queried further. `h2_workers_activate()` places them on the active
64 * list again.
65 *
66 * A producer finishing MUST call `h2_workers_join()` which removes
67 * it completely from workers processing and waits for all ongoing
68 * work for this producer to be done.
69 */
70typedef struct ap_conn_producer_t ap_conn_producer_t;
71
72/**
73 * Ask a producer for the next connection to process.
74 * @param baton value from producer registration
75 * @param pconn holds the connection to process on return
76 * @param pmore if the producer has more connections that may be retrieved
77 * @return APR_SUCCESS for a connection to process, APR_EAGAIN for no
78 *         connection being available at the time.
79 */
80typedef conn_rec *ap_conn_producer_next(void *baton, int *pmore);
81
82/**
83 * Tell the producer that processing the connection is done.
84 * @param baton value from producer registration
85 * @param conn the connection that has been processed.
86 */
87typedef void ap_conn_producer_done(void *baton, conn_rec *conn);
88
89/**
90 * Tell the producer that the workers are shutting down.
91 * @param baton value from producer registration
92 * @param graceful != 0 iff shutdown is graceful
93 */
94typedef void ap_conn_producer_shutdown(void *baton, int graceful);
95
96/**
97 * Register a new producer with the given `baton` and callback functions.
98 * Will allocate internal structures from the given pool (but make no use
99 * of the pool after registration).
100 * Producers are inactive on registration. See `h2_workers_activate()`.
101 * @param producer_pool to allocate the producer from
102 * @param name descriptive name of the producer, must not be unique
103 * @param fn_next callback for retrieving connections to process
104 * @param fn_done callback for processed connections
105 * @param baton provided value passed on in callbacks
106 * @return the producer instance created
107 */
108ap_conn_producer_t *h2_workers_register(h2_workers *workers,
109                                        apr_pool_t *producer_pool,
110                                        const char *name,
111                                        ap_conn_producer_next *fn_next,
112                                        ap_conn_producer_done *fn_done,
113                                        ap_conn_producer_shutdown *fn_shutdown,
114                                        void *baton);
115
116/**
117 * Stop retrieving more connection from the producer and wait
118 * for all ongoing for from that producer to be done.
119 */
120apr_status_t h2_workers_join(h2_workers *workers, ap_conn_producer_t *producer);
121
122/**
123 * Activate a producer. A worker will query the producer for a connection
124 * to process, once a worker is available.
125 * This may be called, irregardless of the producers active/inactive.
126 */
127apr_status_t h2_workers_activate(h2_workers *workers, ap_conn_producer_t *producer);
128
129#endif /* defined(__mod_h2__h2_workers__) */
