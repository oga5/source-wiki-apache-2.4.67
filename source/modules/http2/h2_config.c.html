41#define H2_CONFIG_GET(a, b, n) \
42    (((a)->n == DEF_VAL)? (b) : (a))->n

44#define H2_CONFIG_SET(a, n, v) \
45    ((a)->n = v)

53/* Apache httpd module configuration for h2. */ view japanese
54typedef struct h2_config {
55    const char *name;
56    int h2_max_streams;              /* max concurrent # streams (http2) */ view japanese
57    int h2_window_size;              /* stream window size (http2) */ view japanese
58    int min_workers;                 /* min # of worker threads/child */ view japanese
59    int max_workers;                 /* max # of worker threads/child */ view japanese
60    apr_interval_time_t idle_limit;  /* max duration for idle workers */ view japanese
61    int stream_max_mem_size;         /* max # bytes held in memory/stream */ view japanese
62    int h2_direct;                   /* if mod_h2 is active directly */ view japanese
63    int modern_tls_only;             /* Accept only modern TLS in HTTP/2 connections */   view japanese
64    int h2_upgrade;                  /* Allow HTTP/1 upgrade to h2/h2c */
65    apr_int64_t tls_warmup_size;     /* Amount of TLS data to send before going full write size */ view japanese
66    int tls_cooldown_secs;           /* Seconds of idle time before going back to small TLS records */ view japanese
67    int h2_push;                     /* if HTTP/2 server push is enabled */
68    struct apr_hash_t *priorities;   /* map of content-type to h2_priority records */ view japanese
69    
70    int push_diary_size;             /* # of entries in push diary */ view japanese
71    int copy_files;                  /* if files shall be copied vs setaside on output */ view japanese
72    apr_array_header_t *push_list;   /* list of h2_push_res configurations */
73    apr_table_t *early_headers;      /* HTTP headers for a 103 response */
74    int early_hints;                 /* support status code 103 */
75    int padding_bits;
76    int padding_always;
77    int output_buffered;
78    apr_interval_time_t stream_timeout;/* beam timeout */
79    int max_data_frame_len;          /* max # bytes in a single h2 DATA frame */ view japanese
80    int max_hd_block_len;            /* max # bytes in a response header block */ view japanese
81    int max_stream_errors;           /* max # of tolerated stream errors */ view japanese
82    int proxy_requests;              /* act as forward proxy */ view japanese
83    int h2_websockets;               /* if mod_h2 negotiating WebSockets */ view japanese
84} h2_config;

179static void *h2_config_merge(apr_pool_t *pool, void *basev, void *addv)
180{
181    h2_config *base = (h2_config *)basev;
182    h2_config *add = (h2_config *)addv;
183    h2_config *n = (h2_config *)apr_pcalloc(pool, sizeof(h2_config));
184    char *name = apr_pstrcat(pool, "merged[", add->name, ", ", base->name, "]", NULL);
185    n->name = name;
186
187    n->h2_max_streams       = H2_CONFIG_GET(add, base, h2_max_streams);
188    n->h2_window_size       = H2_CONFIG_GET(add, base, h2_window_size);
189    n->min_workers          = H2_CONFIG_GET(add, base, min_workers);
190    n->max_workers          = H2_CONFIG_GET(add, base, max_workers);
191    n->idle_limit           = H2_CONFIG_GET(add, base, idle_limit);
192    n->stream_max_mem_size  = H2_CONFIG_GET(add, base, stream_max_mem_size);
193    n->h2_direct            = H2_CONFIG_GET(add, base, h2_direct);
194    n->modern_tls_only      = H2_CONFIG_GET(add, base, modern_tls_only);
195    n->h2_upgrade           = H2_CONFIG_GET(add, base, h2_upgrade);
196    n->tls_warmup_size      = H2_CONFIG_GET(add, base, tls_warmup_size);
197    n->tls_cooldown_secs    = H2_CONFIG_GET(add, base, tls_cooldown_secs);
198    n->h2_push              = H2_CONFIG_GET(add, base, h2_push);
199    if (add->priorities && base->priorities) {
200        n->priorities       = apr_hash_overlay(pool, add->priorities, base->priorities);
201    }
202    else {
203        n->priorities       = add->priorities? add->priorities : base->priorities;
204    }
205    n->push_diary_size      = H2_CONFIG_GET(add, base, push_diary_size);
206    n->copy_files           = H2_CONFIG_GET(add, base, copy_files);
207    n->output_buffered      = H2_CONFIG_GET(add, base, output_buffered);
208    if (add->push_list && base->push_list) {
209        n->push_list        = apr_array_append(pool, base->push_list, add->push_list);
210    }
211    else {
212        n->push_list        = add->push_list? add->push_list : base->push_list;
213    }
214    if (add->early_headers && base->early_headers) {
215        n->early_headers    = apr_table_overlay(pool, add->early_headers, base->early_headers);
216    }
217    else {
218        n->early_headers    = add->early_headers? add->early_headers : base->early_headers;
219    }
220    n->early_hints          = H2_CONFIG_GET(add, base, early_hints);
221    n->padding_bits         = H2_CONFIG_GET(add, base, padding_bits);
222    n->padding_always       = H2_CONFIG_GET(add, base, padding_always);
223    n->stream_timeout       = H2_CONFIG_GET(add, base, stream_timeout);
224    n->max_data_frame_len   = H2_CONFIG_GET(add, base, max_data_frame_len);
225    n->max_hd_block_len     = H2_CONFIG_GET(add, base, max_hd_block_len);
226    n->max_stream_errors    = H2_CONFIG_GET(add, base, max_stream_errors);
227    n->proxy_requests       = H2_CONFIG_GET(add, base, proxy_requests);
228    n->h2_websockets        = H2_CONFIG_GET(add, base, h2_websockets);
229    return n;
230}

251void *h2_config_merge_dir(apr_pool_t *pool, void *basev, void *addv)
252{
253    h2_dir_config *base = (h2_dir_config *)basev;
254    h2_dir_config *add = (h2_dir_config *)addv;
255    h2_dir_config *n = (h2_dir_config *)apr_pcalloc(pool, sizeof(h2_dir_config));
256
257    n->name = apr_pstrcat(pool, "merged[", add->name, ", ", base->name, "]", NULL);
258    n->h2_upgrade           = H2_CONFIG_GET(add, base, h2_upgrade);
259    n->h2_push              = H2_CONFIG_GET(add, base, h2_push);
260    if (add->push_list && base->push_list) {
261        n->push_list        = apr_array_append(pool, base->push_list, add->push_list);
262    }
263    else {
264        n->push_list        = add->push_list? add->push_list : base->push_list;
265    }
266    if (add->early_headers && base->early_headers) {
267        n->early_headers    = apr_table_overlay(pool, add->early_headers, base->early_headers);
268    }
269    else {
270        n->early_headers    = add->early_headers? add->early_headers : base->early_headers;
271    }
272    n->early_hints          = H2_CONFIG_GET(add, base, early_hints);
273    n->stream_timeout         = H2_CONFIG_GET(add, base, stream_timeout);
274    return n;
275}

458static void h2_config_seti(h2_dir_config *dconf, h2_config *conf, h2_config_var_t var, int val)
459{
460    int set_srv = !dconf;
461    if (dconf) {
462        switch(var) {
463            case H2_CONF_UPGRADE:
464                H2_CONFIG_SET(dconf, h2_upgrade, val);
465                break;
466            case H2_CONF_PUSH:
467                H2_CONFIG_SET(dconf, h2_push, val);
468                break;
469            case H2_CONF_EARLY_HINTS:
470                H2_CONFIG_SET(dconf, early_hints, val);
471                break;
472            default:
473                /* not handled in dir_conf */
474                set_srv = 1;
475                break;
476        }
477    }
478
479    if (set_srv) {
480        h2_srv_config_seti(conf, var, val);
481    }
482}

504static const h2_config *h2_config_get(conn_rec *c)
505{
506    h2_conn_ctx_t *conn_ctx = h2_conn_ctx_get(c);
507    
508    if (conn_ctx && conn_ctx->server) {
509        return h2_config_sget(conn_ctx->server);
510    }
511    return h2_config_sget(c->base_server);
512}
513
514int h2_config_cgeti(conn_rec *c, h2_config_var_t var)
515{
516    return (int)h2_srv_config_geti64(h2_config_get(c), var);
517}

762static const char *h2_conf_add_push_priority(cmd_parms *cmd, void *_cfg,
763                                             const char *ctype, const char *sdependency,
764                                             const char *sweight)
765{
766    h2_config *cfg = (h2_config *)h2_config_sget(cmd->server);
767    const char *sdefweight = "16";         /* default AFTER weight */ view japanese
768    h2_dependency dependency;
769    h2_priority *priority;
770    int weight;
771 
772    (void)_cfg;
773    if (!*ctype) {
774        return "1st argument must be a mime-type, like 'text/css' or '*'";
775    }
776    
777    if (!sweight) {
778        /* 2 args only, but which one? */ view japanese
779        if (apr_isdigit(sdependency[0])) {
780            sweight = sdependency;
781            sdependency = "AFTER";        /* default dependency */ view japanese
782        }
783    }
784    
785    if (!strcasecmp("AFTER", sdependency)) {
786        dependency = H2_DEPENDANT_AFTER;
787    } 
788    else if (!strcasecmp("BEFORE", sdependency)) {
789        dependency = H2_DEPENDANT_BEFORE;
790        if (sweight) {
791            return "dependency 'Before' does not allow a weight";
792        }
793    } 
794    else if (!strcasecmp("INTERLEAVED", sdependency)) {
795        dependency = H2_DEPENDANT_INTERLEAVED;
796        sdefweight = "256";        /* default INTERLEAVED weight */ view japanese
797    }
798    else {
799        return "dependency must be one of 'After', 'Before' or 'Interleaved'";
800    }
801    
802    weight = (int)apr_atoi64(sweight? sweight : sdefweight);
803    if (weight < NGHTTP2_MIN_WEIGHT) {
804        return apr_psprintf(cmd->pool, "weight must be a number >= %d",
805                            NGHTTP2_MIN_WEIGHT);
806    }
807    
808    priority = apr_pcalloc(cmd->pool, sizeof(*priority));
809    priority->dependency = dependency;
810    priority->weight = weight;
811    
812    if (!cfg->priorities) {
813        cfg->priorities = apr_hash_make(cmd->pool);
814    }
815    apr_hash_set(cfg->priorities, ctype, (apr_ssize_t)strlen(ctype), priority);
816    return NULL;
817}

894static void add_push(apr_array_header_t **plist, apr_pool_t *pool, h2_push_res *push)
895{
896    h2_push_res *new;
897    if (!*plist) {
898        *plist = apr_array_make(pool, 10, sizeof(*push));
899    }

901    new->uri_ref = push->uri_ref;
902    new->critical = push->critical;

905static const char *h2_conf_add_push_res(cmd_parms *cmd, void *dirconf,
906                                        const char *arg1, const char *arg2,
907                                        const char *arg3)
908{
909    h2_push_res push;
910    const char *last = arg3;
911    
912    memset(&push, 0, sizeof(push));
913    if (!strcasecmp("add", arg1)) {
914        push.uri_ref = arg2;
915    }
916    else {
917        push.uri_ref = arg1;
918        last = arg2;
919        if (arg3) {
920            return "too many parameter";
921        }
922    }
923    
924    if (last) {
925        if (!strcasecmp("critical", last)) {
926            push.critical = 1;
927        }
928        else {
929            return "unknown last parameter";
930        }
931    }
932
933    if (cmd->path) {
934        add_push(&(((h2_dir_config*)dirconf)->push_list), cmd->pool, &push);
935    }
936    else {
937        add_push(&(h2_config_sget(cmd->server)->push_list), cmd->pool, &push);
938    }
939    return NULL;
940}

942static const char *h2_conf_add_early_hint(cmd_parms *cmd, void *dirconf,
943                                          const char *name, const char *value)
944{
945    apr_table_t *hds, **phds;
946
947    if(!name || !*name)
948      return "Early Hint header name must not be empty";
949    if(!value)
950      return "Early Hint header value must not be empty";
951    while (apr_isspace(*value))
952        ++value;
953    if(!*value)
954      return "Early Hint header value must not be empty/only space";
955    if (*ap_scan_http_field_content(value))
956      return "Early Hint header value contains invalid characters";
957
958    if (cmd->path) {
959        phds = &((h2_dir_config*)dirconf)->early_headers;
960    }
961    else {
962        phds = &(h2_config_sget(cmd->server))->early_headers;
963    }
964    hds = *phds;
965    if (!hds) {
966      *phds = hds = apr_table_make(cmd->pool, 10);
967    }
968    apr_table_add(hds, name, value);
969
970    return NULL;
971}

1047void h2_get_workers_config(server_rec *s, int *pminw, int *pmaxw,
1048                           apr_time_t *pidle_limit)
1049{
1050    int threads_per_child = 0;
1051
1052    *pminw = h2_config_sgeti(s, H2_CONF_MIN_WORKERS);
1053    *pmaxw = h2_config_sgeti(s, H2_CONF_MAX_WORKERS);
1054
1055    ap_mpm_query(AP_MPMQ_MAX_THREADS, &threads_per_child);
1056    if (*pminw <= 0) {
1057        *pminw = threads_per_child;
1058    }
1059    if (*pmaxw <= 0) {
1060        *pmaxw = H2MAX(4, 3 * (*pminw) / 2);
1061    }
1062    *pidle_limit = h2_config_sgeti64(s, H2_CONF_MAX_WORKER_IDLE_LIMIT);
1063}
