58apr_status_t h2_c1_child_init(apr_pool_t *pool, server_rec *s)
59{
60    int minw, maxw;
61    apr_time_t idle_limit;
62
63    if (ap_mpm_query(AP_MPMQ_IS_ASYNC, &async_mpm)) {
64        /* some MPMs do not implemnent this */
65        async_mpm = 0;
66    }
67#ifdef AP_MPMQ_CAN_WAITIO
68    if (!async_mpm || ap_mpm_query(AP_MPMQ_CAN_WAITIO, &mpm_can_waitio)) {
69        mpm_can_waitio = 0;
70    }
71#endif
72
73    h2_config_init(pool);
74
75    h2_get_workers_config(s, &minw, &maxw, &idle_limit);
76    workers = h2_workers_create(s, pool, maxw, minw, idle_limit);
77 
78    h2_c_logio_add_bytes_in = APR_RETRIEVE_OPTIONAL_FN(ap_logio_add_bytes_in);
79    h2_c_logio_add_bytes_out = APR_RETRIEVE_OPTIONAL_FN(ap_logio_add_bytes_out);
80
81    return h2_mplx_c1_child_init(pool, s);
82}
83
84void h2_c1_child_stopping(apr_pool_t *pool, int graceful)
85{
86    if (workers) {
87        h2_workers_shutdown(workers, graceful);
88    }
89}

92apr_status_t h2_c1_setup(conn_rec *c, request_rec *r, server_rec *s)
93{
94    h2_session *session;
95    h2_conn_ctx_t *ctx;
96    apr_status_t rv;
97    
98    if (!workers) {
99        ap_log_cerror(APLOG_MARK, APLOG_ERR, 0, c, APLOGNO(02911) 
100                      "workers not initialized");
101        rv = APR_EGENERAL;
102        goto cleanup;
103    }
104
105    rv = h2_session_create(&session, c, r, s, workers);
106    if (APR_SUCCESS != rv) goto cleanup;
107
108    ctx = h2_conn_ctx_get(c);
109    ap_assert(ctx);
110    h2_conn_ctx_assign_session(ctx, session);
111    /* remove the input filter of mod_reqtimeout, now that the connection
112     * is established and we have switched to h2. reqtimeout has supervised
113     * possibly configured handshake timeouts and needs to get out of the way
114     * now since the rest of its state handling assumes http/1.x to take place. */
115    ap_remove_input_filter_byhandle(c->input_filters, "reqtimeout");
116
117cleanup:
118    return rv;
119}

121int h2_c1_run(conn_rec *c)
122{
123    apr_status_t status;
124    int mpm_state = 0, keepalive = 0;
125    h2_conn_ctx_t *conn_ctx = h2_conn_ctx_get(c);
126    
127    ap_assert(conn_ctx);
128    ap_assert(conn_ctx->session);
129    c->clogging_input_filters = 0;
130    do {
131        if (c->cs) {
132            c->cs->state = CONN_STATE_HANDLER;
133        }
134    
135        status = h2_session_process(conn_ctx->session, async_mpm, &keepalive);
136        if (status != APR_SUCCESS) {
137            ap_log_cerror(APLOG_MARK, APLOG_DEBUG, status, c, 
138                          H2_SSSN_LOG(APLOGNO(03045), conn_ctx->session,
139                          "process, closing conn"));
140            c->keepalive = AP_CONN_CLOSE;
141        }
142        else {
143            c->keepalive = AP_CONN_KEEPALIVE;
144        }
145        
146        if (ap_mpm_query(AP_MPMQ_MPM_STATE, &mpm_state)) {
147            break;
148        }
149    } while (!async_mpm
150             && c->keepalive == AP_CONN_KEEPALIVE 
151             && mpm_state != AP_MPMQ_STOPPING);
152
153    if (c->cs) {
154        switch (conn_ctx->session->state) {
155            case H2_SESSION_ST_INIT:
156            case H2_SESSION_ST_IDLE:
157            case H2_SESSION_ST_BUSY:
158            case H2_SESSION_ST_WAIT:
159                if (keepalive) {
160                    /* Flush then keep-alive */
161                    c->cs->sense = CONN_SENSE_DEFAULT;
162                    c->cs->state = CONN_STATE_WRITE_COMPLETION;
163                }
164                else {
165                    /* Let the MPM know that we are not done and want to wait
166                     * for read using Timeout instead of KeepAliveTimeout.
167                     * See PR 63534. 
168                     */
169                    c->cs->sense = CONN_SENSE_WANT_READ;
170#ifdef AP_MPMQ_CAN_WAITIO
171                    if (mpm_can_waitio) {
172                        /* This tells the MPM to wait for the connection to be
173                         * readable (CONN_SENSE_WANT_READ) within the configured
174                         * Timeout and then come back to the process_connection()
175                         * hooks again when ready.
176                         */
177                        c->cs->state = CONN_STATE_ASYNC_WAITIO;
178                    }
179                    else
180#endif
181                    {
182                        /* This is a compat workaround to do the same using the
183                         * CONN_STATE_WRITE_COMPLETION state but with both
184                         * CONN_SENSE_WANT_READ to wait for readability rather
185                         * than writing and c->clogging_input_filters to force
186                         * reentering the process_connection() hooks from any
187                         * state when ready. This somehow will use Timeout too.
188                         */
189                        c->cs->state = CONN_STATE_WRITE_COMPLETION;
190                        c->clogging_input_filters = 1;
191                    }
192                }
193                break;
194
195            case H2_SESSION_ST_CLEANUP:
196            case H2_SESSION_ST_DONE:
197            default:
198                c->cs->state = CONN_STATE_LINGER;
199                break;
200        }
201    }
202
203    return OK;
204}

206apr_status_t h2_c1_pre_close(struct h2_conn_ctx_t *ctx, conn_rec *c)
207{
208    h2_conn_ctx_t *conn_ctx = h2_conn_ctx_get(c);
209
210    if (conn_ctx && conn_ctx->session) {
211        apr_status_t status = h2_session_pre_close(conn_ctx->session, async_mpm);
212        return (status == APR_SUCCESS)? DONE : status;
213    }
214    return DONE;
215}

217int h2_c1_allows_direct(conn_rec *c)
218{
219    if (!c->master) {
220        int is_tls = ap_ssl_conn_is_ssl(c);
221        const char *needed_protocol = is_tls? "h2" : "h2c";
222        int h2_direct = h2_config_cgeti(c, H2_CONF_DIRECT);
223
224        if (h2_direct < 0) {
225            h2_direct = is_tls? 0 : 1;
226        }
227        return (h2_direct && ap_is_allowed_protocol(c, NULL, NULL, needed_protocol));
228    }
229    return 0;
230}
231
232int h2_c1_can_upgrade(request_rec *r)
233{
234    if (!r->connection->master) {
235        int h2_upgrade = h2_config_rgeti(r, H2_CONF_UPGRADE);
236        return h2_upgrade > 0 || (h2_upgrade < 0 && !ap_ssl_conn_is_ssl(r->connection));
237    }
238    return 0;
239}

241static int h2_c1_hook_process_connection(conn_rec* c)
242{
243    apr_status_t status;
244    h2_conn_ctx_t *ctx;
245
246    if (c->master) goto declined;
247    ctx = h2_conn_ctx_get(c);
248
249    ap_log_cerror(APLOG_MARK, APLOG_TRACE1, 0, c, "h2_h2, process_conn");
250    if (!ctx && c->keepalives == 0) {
251        const char *proto = ap_get_protocol(c);
252
253        if (APLOGctrace1(c)) {
254            ap_log_cerror(APLOG_MARK, APLOG_TRACE1, 0, c, "h2_h2, process_conn, "
255                          "new connection using protocol '%s', direct=%d, "
256                          "tls acceptable=%d", proto, h2_c1_allows_direct(c),
257                          h2_protocol_is_acceptable_c1(c, NULL, 1));
258        }
259
260        if (!strcmp(AP_PROTOCOL_HTTP1, proto)
261            && h2_c1_allows_direct(c)
262            && h2_protocol_is_acceptable_c1(c, NULL, 1)) {
263            /* Fresh connection still is on http/1.1 and H2Direct is enabled.
264             * Otherwise connection is in a fully acceptable state.
265             * -> peek at the first 24 incoming bytes
266             */
267            apr_bucket_brigade *temp;
268            char *peek = NULL;
269            apr_size_t peeklen;
270
271            temp = apr_brigade_create(c->pool, c->bucket_alloc);
272            status = ap_get_brigade(c->input_filters, temp,
273                                    AP_MODE_SPECULATIVE, APR_BLOCK_READ, 24);
274
275            if (status != APR_SUCCESS) {
276                ap_log_cerror(APLOG_MARK, APLOG_DEBUG, status, c, APLOGNO(03054)
277                              "h2_h2, error reading 24 bytes speculative");
278                apr_brigade_destroy(temp);
279                return DECLINED;
280            }
281
282            apr_brigade_pflatten(temp, &peek, &peeklen, c->pool);
283            if ((peeklen >= 24) && !memcmp(H2_MAGIC_TOKEN, peek, 24)) {
284                ap_log_cerror(APLOG_MARK, APLOG_TRACE1, 0, c,
285                              "h2_h2, direct mode detected");
286                ctx = h2_conn_ctx_create_for_c1(c, c->base_server,
287                                                ap_ssl_conn_is_ssl(c)? "h2" : "h2c");
288            }
289            else if (APLOGctrace2(c)) {
290                ap_log_cerror(APLOG_MARK, APLOG_TRACE2, 0, c,
291                              "h2_h2, not detected in %d bytes(base64): %s",
292                              (int)peeklen, h2_util_base64url_encode(peek, peeklen, c->pool));
293            }
294            apr_brigade_destroy(temp);
295        }
296    }
297
298    if (!ctx) goto declined;
299
300    ap_log_cerror(APLOG_MARK, APLOG_TRACE1, 0, c, "process_conn");
301    if (!ctx->session) {
302        status = h2_c1_setup(c, NULL, ctx->server? ctx->server : c->base_server);
303        ap_log_cerror(APLOG_MARK, APLOG_TRACE1, status, c, "conn_setup");
304        if (status != APR_SUCCESS) {
305            h2_conn_ctx_detach(c);
306            return !OK;
307        }
308    }
309    return h2_c1_run(c);
310
311declined:
312    ap_log_cerror(APLOG_MARK, APLOG_TRACE1, 0, c, "h2_h2, declined");
313    return DECLINED;
314}

316static int h2_c1_hook_pre_close(conn_rec *c)
317{
318    h2_conn_ctx_t *ctx;
319
320    /* secondary connection? */
321    if (c->master) {
322        return DECLINED;
323    }
324
325    ctx = h2_conn_ctx_get(c);
326    if (ctx) {
327        /* If the session has been closed correctly already, we will not
328         * find a h2_conn_ctx_there. The presence indicates that the session
329         * is still ongoing. */
330        return h2_c1_pre_close(ctx, c);
331    }
332    return DECLINED;
333}

335static const char* const mod_ssl[]        = { "mod_ssl.c", NULL};
336static const char* const mod_reqtimeout[] = { "mod_ssl.c", "mod_reqtimeout.c", NULL};
337
338void h2_c1_register_hooks(void)
339{
340    /* Our main processing needs to run quite late. Definitely after mod_ssl,
341     * as we need its connection filters, but also before reqtimeout as its
342     * method of timeouts is specific to HTTP/1.1 (as of now).
343     * The core HTTP/1 processing run as REALLY_LAST, so we will have
344     * a chance to take over before it.
345     */
346    ap_hook_process_connection(h2_c1_hook_process_connection,
347                               mod_reqtimeout, NULL, APR_HOOK_LAST);
348
349    /* One last chance to properly say goodbye if we have not done so
350     * already. */
351    ap_hook_pre_close_connection(h2_c1_hook_pre_close, NULL, mod_ssl, APR_HOOK_LAST);
352}
