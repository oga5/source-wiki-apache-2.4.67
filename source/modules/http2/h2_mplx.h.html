20/**
21 * The stream multiplexer. It performs communication between the
22 * primary HTTP/2 connection (c1) to the secondary connections (c2)
23 * that process the requests, aka. HTTP/2 streams.
24 *
25 * There is one h2_mplx instance for each h2_session.
26 *
27 * Naming Convention:
28 * "h2_mplx_c1_" are methods only to be called by the primary connection
29 * "h2_mplx_c2_" are methods only to be called by a secondary connection
30 * "h2_mplx_worker_" are methods only to be called by a h2 worker thread
31 */
32

56typedef struct h2_mplx h2_mplx;
57
58struct h2_mplx {
59    int child_num;                  /* child this runs in */
60    apr_uint32_t id;                /* id unique per child */
61    conn_rec *c1;                   /* the main connection */
62    apr_pool_t *pool;
63    struct h2_stream *stream0;      /* HTTP/2's stream 0 */
64    server_rec *s;                  /* server for master conn */
65
66    int shutdown;                   /* we are shutting down */
67    int aborted;                    /* we need to get out of here asap */
68    int polling;                    /* is waiting/processing pollset events */
69    ap_conn_producer_t *producer;   /* registered producer at h2_workers */
70
71    struct h2_ihash_t *streams;     /* all streams active */
72    struct h2_ihash_t *shold;       /* all streams done with c2 processing ongoing */
73    apr_array_header_t *spurge;     /* all streams done, ready for destroy */
74    
75    struct h2_iqueue *q;            /* all stream ids that need to be started */
76
77    apr_size_t stream_max_mem;      /* max memory to buffer for a stream */
78    apr_uint32_t max_streams;       /* max # of concurrent streams */
79    apr_uint32_t max_stream_id_started; /* highest stream id that started processing */
80
81    apr_uint32_t processing_count;  /* # of c2 working for this mplx */
82    apr_uint32_t processing_limit;  /* current limit on processing c2s, dynamic */
83    apr_uint32_t processing_max;    /* max, hard limit of processing c2s */
84    
85    apr_time_t last_mood_change;    /* last time, processing limit changed */
86    apr_interval_time_t mood_update_interval; /* how frequent we update at most */
87    apr_uint32_t irritations_since; /* irritations (>0) or happy events (<0) since last mood change */
88
89    apr_thread_mutex_t *lock;
90    struct apr_thread_cond_t *join_wait;
91    
92    apr_pollset_t *pollset;         /* pollset for c1/c2 IO events */
93    apr_array_header_t *streams_ev_in;
94    apr_array_header_t *streams_ev_out;
95
96    apr_thread_mutex_t *poll_lock; /* protect modifications of queues below */
97    struct h2_iqueue *streams_input_read;  /* streams whose input has been read from */
98    struct h2_iqueue *streams_output_written; /* streams whose output has been written to */
99
100    struct h2_workers *workers;     /* h2 workers process wide instance */
101
102    apr_uint32_t max_spare_transits; /* max number of transit pools idling */
103    apr_array_header_t *c2_transits; /* base pools for running c2 connections */
104};

106apr_status_t h2_mplx_c1_child_init(apr_pool_t *pool, server_rec *s);
107
108/**
109 * Create the multiplexer for the given HTTP2 session. 
110 * Implicitly has reference count 1.
111 */
112h2_mplx *h2_mplx_c1_create(int child_id, apr_uint32_t id,
113                           struct h2_stream *stream0,
114                           server_rec *s, apr_pool_t *master,
115                           struct h2_workers *workers);
116
117/**
118 * Destroy the mplx, shutting down all ongoing processing.
119 * @param m the mplx destroyed
120 * @param wait condition var to wait on for ref counter == 0
121 */ 
122void h2_mplx_c1_destroy(h2_mplx *m);
123
124/**
125 * Shut down the multiplexer gracefully. Will no longer schedule new streams
126 * but let the ongoing ones finish normally.
127 * @return the highest stream id being/been processed
128 */
129int h2_mplx_c1_shutdown(h2_mplx *m);
130
131/**
132 * Notifies mplx that a stream has been completely handled on the main
133 * connection and is ready for cleanup.
134 * 
135 * @param m the mplx itself
136 * @param stream the stream ready for cleanup
137 * @param pstream_count return the number of streams active
138 */
139apr_status_t h2_mplx_c1_stream_cleanup(h2_mplx *m, struct h2_stream *stream,
140                                       unsigned int *pstream_count);
141
142int h2_mplx_c1_stream_is_running(h2_mplx *m, struct h2_stream *stream);
143
144/**
145 * Process a stream request.
146 * 
147 * @param m the multiplexer
148 * @param read_to_process
149 * @param input_pending
150 * @param cmp the stream priority compare function
151 * @param pstream_count on return the number of streams active in mplx
152 */
153void h2_mplx_c1_process(h2_mplx *m,
154                        struct h2_iqueue *read_to_process,
155                        h2_stream_get_fn *get_stream,
156                        h2_stream_pri_cmp_fn *cmp,
157                        struct h2_session *session,
158                        unsigned int *pstream_count);
159
160/**
161 * Stream priorities have changed, reschedule pending requests.
162 * 
163 * @param m the multiplexer
164 * @param cmp the stream priority compare function
165 * @param ctx context data for the compare function
166 */
167apr_status_t h2_mplx_c1_reprioritize(h2_mplx *m, h2_stream_pri_cmp_fn *cmp,
168                                    struct h2_session *session);

178                            stream_ev_callback *on_stream_output,
179                            void *on_ctx);

181void h2_mplx_c2_input_read(h2_mplx *m, conn_rec *c2);
182void h2_mplx_c2_output_written(h2_mplx *m, conn_rec *c2);

184typedef int h2_mplx_stream_cb(struct h2_stream *s, void *userdata);
185
186/**
187 * Iterate over all streams known to mplx from the primary connection.

189 * @param cb the callback to invoke on each stream
190 * @param ctx userdata passed to the callback

192apr_status_t h2_mplx_c1_streams_do(h2_mplx *m, h2_mplx_stream_cb *cb, void *ctx);
193
194/**
195 * Return != 0 iff all open streams want to send data
196 */
197int h2_mplx_c1_all_streams_want_send_data(h2_mplx *m);

200 * Return != 0 iff all open streams have send window exhausted

202int h2_mplx_c1_all_streams_send_win_exhausted(h2_mplx *m);
203

205 * A stream has been RST_STREAM by the client. Abort

208 */
209apr_status_t h2_mplx_c1_client_rst(h2_mplx *m, int stream_id,
210                                   struct h2_stream *stream);

213 * Get readonly access to a stream for a secondary connection.

215const struct h2_stream *h2_mplx_c2_stream_get(h2_mplx *m, int stream_id);
