25/**
26 * A h2_bucket_beam solves the task of transferring buckets, esp. their data,
27 * across threads with as little copying as possible.
28 */ view japanese
29
30typedef struct h2_bucket_beam h2_bucket_beam;
31
32typedef void h2_beam_io_callback(void *ctx, h2_bucket_beam *beam,
33                                 apr_off_t bytes);
34typedef void h2_beam_ev_callback(void *ctx, h2_bucket_beam *beam);

36/**
37 * h2_blist can hold a list of buckets just like apr_bucket_brigade, but
38 * does not to any allocations or related features.
39 */ view japanese
40typedef struct {
41    APR_RING_HEAD(h2_bucket_list, apr_bucket) list;
42} h2_blist;
43
44struct h2_bucket_beam {
45    int id;
46    const char *name;
47    conn_rec *from;
48    apr_pool_t *pool;
49    h2_blist buckets_to_send;
50    h2_blist buckets_consumed;
51    h2_blist buckets_eor;
52
53    apr_size_t max_buf_size;
54    apr_interval_time_t timeout;
55
56    int aborted;
57    int closed;
58    int tx_mem_limits; /* only memory size counts on transfers */ view japanese
59    int copy_files;
60
61    struct apr_thread_mutex_t *lock;
62    struct apr_thread_cond_t *change;
63    
64    h2_beam_ev_callback *was_empty_cb; /* event: beam changed to non-empty in h2_beam_send() */ view japanese
65    void *was_empty_ctx;
66    h2_beam_ev_callback *recv_cb;      /* event: buckets were transfered in h2_beam_receive() */ view japanese
67    void *recv_ctx;
68    h2_beam_ev_callback *send_cb;      /* event: buckets were added in h2_beam_send() */ view japanese
69    void *send_ctx;
70    h2_beam_ev_callback *eagain_cb;    /* event: a receive results in ARP_EAGAIN */ view japanese
71    void *eagain_ctx;
72
73    apr_off_t recv_bytes;             /* amount of bytes transferred in h2_beam_receive() */ view japanese
74    apr_off_t recv_bytes_reported;    /* amount of bytes reported as received via callback */ view japanese
75    h2_beam_io_callback *cons_io_cb;  /* report: recv_bytes deltas for sender */ view japanese
76    void *cons_ctx;
77};

79/**
80 * Creates a new bucket beam for transfer of buckets across threads.
81 *
82 * The pool the beam is created with will be protected by the given 
83 * mutex and will be used in multiple threads. It needs a pool allocator
84 * that is only used inside that same mutex.
85 *
86 * @param pbeam         will hold the created beam on return
87 * @param c_from        connection from which buchets are sent
88 * @param pool          pool owning the beam, beam will cleanup when pool released
89 * @param id            identifier of the beam
90 * @param tag           tag identifying beam for logging
91 * @param buffer_size   maximum memory footprint of buckets buffered in beam, or
92 *                      0 for no limitation
93 * @param timeout       timeout for blocking operations
94 */ view japanese
95apr_status_t h2_beam_create(h2_bucket_beam **pbeam,
96                            conn_rec *from,
97                            apr_pool_t *pool, 
98                            int id, const char *tag,
99                            apr_size_t buffer_size,
100                            apr_interval_time_t timeout);
101
102/**
103 * Destroys the beam immediately without cleanup.
104 */  view japanese
105apr_status_t h2_beam_destroy(h2_bucket_beam *beam, conn_rec *c);
106
107/**
108 * Switch copying of file buckets on/off.
109 */ view japanese
110void h2_beam_set_copy_files(h2_bucket_beam * beam, int enabled);
111
112/**
113 * Send buckets from the given brigade through the beam.
114 * This can block of the amount of bucket data is above the buffer limit.
115 * @param beam the beam to add buckets to
116 * @param from the connection the sender operates on, must be the same as
117 *             used to create the beam
118 * @param bb the brigade to take buckets from
119 * @param block if the sending should block when the buffer is full
120 * @param pwritten on return, contains the number of data bytes sent
121 * @return APR_SUCCESS when buckets were added to the beam. This can be
122 *                     a partial transfer and other buckets may still remain in bb
123 *         APR_EAGAIN on non-blocking send when the buffer is full
124 *         APR_TIMEUP on blocking semd that time out
125 *         APR_ECONNABORTED when beam has been aborted
126 */ view japanese
127apr_status_t h2_beam_send(h2_bucket_beam *beam, conn_rec *from,
128                          apr_bucket_brigade *bb, 
129                          apr_read_type_e block,
130                          apr_off_t *pwritten);
131
132/**
133 * Receive buckets from the beam into the given brigade. The caller is
134 * operating on connection `to`.
135 * @param beam the beam to receive buckets from
136 * @param to the connection the receiver is working with
137 * @param bb the bucket brigade to append to
138 * @param block if the read should block when buckets are unavailable
139 * @param readbytes the amount of data the receiver wants
140 * @return APR_SUCCESS when buckets were appended
141 *         APR_EAGAIN on non-blocking read when no buckets are available
142 *         APR_TIMEUP on blocking reads that time out
143 *         APR_ECONNABORTED when beam has been aborted
144 */ view japanese
145apr_status_t h2_beam_receive(h2_bucket_beam *beam, conn_rec *to,
146                             apr_bucket_brigade *bb,
147                             apr_read_type_e block,
148                             apr_off_t readbytes);
