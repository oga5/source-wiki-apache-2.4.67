77static int DH_set0_pqg(DH *dh, BIGNUM *p, BIGNUM *q, BIGNUM *g)
78{
79    /* q is optional */ view japanese
80    if (p == NULL || g == NULL)
81        return 0;
82    BN_free(dh->p);
83    BN_free(dh->q);
84    BN_free(dh->g);
85    dh->p = p;
86    dh->q = q;
87    dh->g = g;
88
89    if (q != NULL) {
90        dh->length = BN_num_bits(q);
91    }
92
93    return 1;
94}

96/*
97 * Grab well-defined DH parameters from OpenSSL, see the BN_get_rfc*
98 * functions in <openssl/bn.h> for all available primes.
99 */ view japanese
100static DH *make_dh_params(BIGNUM *(*prime)(BIGNUM *))
101{
102    DH *dh = DH_new();
103    BIGNUM *p, *g;
104
105    if (!dh) {
106        return NULL;
107    }
108    p = prime(NULL);
109    g = BN_new();
110    if (g != NULL) {
111        BN_set_word(g, 2);
112    }
113    if (!p || !g || !DH_set0_pqg(dh, p, NULL, g)) {
114        DH_free(dh);
115        BN_free(p);
116        BN_free(g);
117        return NULL;
118    }
119    return dh;
120}

122/* Storage and initialization for DH parameters. */ view japanese
123static struct dhparam {
124    BIGNUM *(*const prime)(BIGNUM *); /* function to generate... */ view japanese
125    DH *dh;                           /* ...this, used for keys.... */ view japanese
126    const unsigned int min;           /* ...of length >= this. */ view japanese
127} dhparams[] = {
128    { BN_get_rfc3526_prime_8192, NULL, 6145 },
129    { BN_get_rfc3526_prime_6144, NULL, 4097 },
130    { BN_get_rfc3526_prime_4096, NULL, 3073 },
131    { BN_get_rfc3526_prime_3072, NULL, 2049 },
132    { BN_get_rfc3526_prime_2048, NULL, 1025 },
133    { BN_get_rfc2409_prime_1024, NULL, 0 }
134};
135
136static void init_dh_params(void)
137{
138    unsigned n;
139
140    for (n = 0; n < sizeof(dhparams)/sizeof(dhparams[0]); n++)
141        dhparams[n].dh = make_dh_params(dhparams[n].prime);
142}
143
144static void free_dh_params(void)
145{
146    unsigned n;
147
148    /* DH_free() is a noop for a NULL parameter, so these are harmless
149     * in the (unexpected) case where these variables are already
150     * NULL. */ view japanese
151    for (n = 0; n < sizeof(dhparams)/sizeof(dhparams[0]); n++) {
152        DH_free(dhparams[n].dh);
153        dhparams[n].dh = NULL;
154    }
155}
156
157/* Hand out the same DH structure though once generated as we leak
158 * memory otherwise and freeing the structure up after use would be
159 * hard to track and in fact is not needed at all as it is safe to
160 * use the same parameters over and over again security wise (in
161 * contrast to the keys itself) and code safe as the returned structure
162 * is duplicated by OpenSSL anyway. Hence no modification happens
163 * to our copy. */ view japanese
164DH *modssl_get_dh_params(unsigned keylen)
165{
166    unsigned n;
167
168    for (n = 0; n < sizeof(dhparams)/sizeof(dhparams[0]); n++)
169        if (keylen >= dhparams[n].min)
170            return dhparams[n].dh;
171        
172    return NULL; /* impossible to reach. */ view japanese
173}

192/* Helper functions to create the SNI vhost policy hash. The policy
193 * hash captures the configuration elements relevant to the mode
194 * selected at runtime by SSLVHostSNIPolicy. */ view japanese

196#define md5_str_update(ctx_, pfx_, str_) do { apr_md5_update(ctx_, pfx_, strlen(pfx_)); apr_md5_update(ctx_, str_, strlen(str_)); } while (0)
197#define md5_ifstr_update(ctx_, pfx_, str_) do { apr_md5_update(ctx_, pfx_, strlen(pfx_)); if (str_) apr_md5_update(ctx_, str_, strlen(str_)); } while (0)
198#define md5_fmt_update(ctx_, fmt_, i_) do { char s_[128]; apr_snprintf(s_, sizeof s_, fmt_, i_); \
199        apr_md5_update(ctx_, s_, strlen(s_)); } while (0)

201static int md5_strarray_cmp(const void *p1, const void *p2)
202{
203    return strcmp(*(char **)p1, *(char **)p2);
204}
205
206/* Hashes an array of strings in sorted order. */ view japanese
207static void md5_strarray_hash(apr_pool_t *ptemp, apr_md5_ctx_t *hash,
208                              const char *pfx, apr_array_header_t *s)
209{
210    char **elts = apr_pmemdup(ptemp, s->elts, s->nelts * sizeof *elts);
211    int i;
212
213    qsort(elts, s->nelts, sizeof(char *), md5_strarray_cmp);
214
215    apr_md5_update(hash, pfx, strlen(pfx));
216    for (i = 0; i < s->nelts; i++) {
217        md5_str_update(hash, "elm:", elts[i]);
218    }
219}

221static void hash_sni_policy_pk(apr_pool_t *ptemp, apr_md5_ctx_t *hash, modssl_ctx_t *ctx)
222{
223    md5_fmt_update(hash, "protocol:%d", ctx->protocol);
224
225    md5_ifstr_update(hash, "ciphers:", ctx->auth.cipher_suite);
226    md5_ifstr_update(hash, "tls13_ciphers:", ctx->auth.tls13_ciphers);
227
228    md5_strarray_hash(ptemp, hash, "cert_files:", ctx->pks->cert_files);
229    md5_strarray_hash(ptemp, hash, "key_files:", ctx->pks->key_files);
230}

232static void hash_sni_policy_auth(apr_md5_ctx_t *hash, modssl_ctx_t *ctx)
233{
234    modssl_pk_server_t *pks = ctx->pks;
235    modssl_auth_ctx_t *a = &ctx->auth;
236
237    md5_fmt_update(hash, "verify_depth:%d", a->verify_depth);
238    md5_fmt_update(hash, "verify_mode:%d", a->verify_mode);
239
240    md5_ifstr_update(hash, "ca_name_path:", pks->ca_name_path);
241    md5_ifstr_update(hash, "ca_name_file:", pks->ca_name_file);
242    md5_ifstr_update(hash, "ca_cert_path:", a->ca_cert_path);
243    md5_ifstr_update(hash, "ca_cert_file:", a->ca_cert_file);
244    md5_ifstr_update(hash, "crl_path:", ctx->crl_path);
245    md5_ifstr_update(hash, "crl_file:", ctx->crl_file);
246    md5_fmt_update(hash, "crl_check_mask:%d", ctx->crl_check_mask);
247    md5_fmt_update(hash, "ocsp_mask:%d", ctx->ocsp_mask);
248    md5_fmt_update(hash, "ocsp_force_default:%d", ctx->ocsp_force_default);
249    md5_ifstr_update(hash, "ocsp_responder:", ctx->ocsp_responder);
250
251#ifdef HAVE_SRP
252    md5_ifstr_update(hash, "srp_vfile:", ctx->srp_vfile);
253#endif
254
255#ifdef HAVE_SSL_CONF_CMD
256    {
257        apr_array_header_t *parms = ctx->ssl_ctx_param;
258        int n;
259
260        for (n = 0; n < parms->nelts; n++) {
261            ssl_ctx_param_t *p = &APR_ARRAY_IDX(parms, n, ssl_ctx_param_t);
262
263            md5_str_update(hash, "param:", p->name);
264            md5_str_update(hash, "value:", p->value);
265        }
266    }
267#endif
268}

271static char *create_sni_policy_hash(apr_pool_t *p, apr_pool_t *ptemp,
272                                    modssl_snivhpolicy_t policy,
273                                    SSLSrvConfigRec *sc)
274{
275    char *rv = NULL;
276#ifdef HAVE_TLSEXT
277    if (policy != MODSSL_SNIVH_STRICT && policy != MODSSL_SNIVH_INSECURE) {
278        apr_md5_ctx_t hash;
279        unsigned char digest[APR_MD5_DIGESTSIZE];
280
281        /* Create the vhost policy hash for comparison later. */ view japanese
282        apr_md5_init(&hash);
283        hash_sni_policy_auth(&hash, sc->server);
284        if (policy == MODSSL_SNIVH_SECURE)
285            hash_sni_policy_pk(ptemp, &hash, sc->server);
286        apr_md5_final(digest, &hash);
287
288        rv = apr_palloc(p, 2 * APR_MD5_DIGESTSIZE + 1);
289        ap_bin2hex(digest, APR_MD5_DIGESTSIZE, rv); /* sets final '\0' */ view japanese
290    }
291#endif
292    return rv;
293}

302int ssl_is_challenge(conn_rec *c, const char *servername, 
303                     X509 **pcert, EVP_PKEY **pkey,
304                     const char **pcert_pem, const char **pkey_pem)
305{
306    *pcert = NULL;
307    *pkey = NULL;
308    *pcert_pem = *pkey_pem = NULL;
309    if (ap_ssl_answer_challenge(c, servername, pcert_pem, pkey_pem)) {
310        return 1;
311    }
312    else if (OK == ssl_run_answer_challenge(c, servername, pcert, pkey)) {
313        return 1;
314    }
315    return 0;
316}

336/*
337 *  Per-module initialization
338 */ view japanese
339apr_status_t ssl_init_Module(apr_pool_t *p, apr_pool_t *plog,
340                             apr_pool_t *ptemp,
341                             server_rec *base_server)
342{
343    unsigned long runtime_lib_version = modssl_runtime_lib_version();
344    SSLModConfigRec *mc = myModConfig(base_server);
345    SSLSrvConfigRec *sc;
346    server_rec *s;
347    apr_status_t rv;
348    apr_array_header_t *pphrases;
349
350    AP_DEBUG_ASSERT(mc);
351
352    if (runtime_lib_version < MODSSL_LIBRARY_VERSION) {
353        ap_log_error(APLOG_MARK, APLOG_WARNING, 0, base_server, APLOGNO(01882)
354                     "Init: this version of mod_ssl was compiled against "
355                     "a newer library (%s (%s), version currently loaded is 0x%lX)"
356                     " - may result in undefined or erroneous behavior",
357                    MODSSL_LIBRARY_TEXT, MODSSL_LIBRARY_DYNTEXT,
358                    runtime_lib_version);
359    }
360
361    /* We initialize mc->pid per-process in the child init,
362     * but it should be initialized for startup before we
363     * call ssl_rand_seed() below.
364     */ view japanese
365    mc->pid = getpid();
366
367    /*
368     * Let us cleanup on restarts and exits
369     */ view japanese
370    apr_pool_cleanup_register(p, base_server,
371                              ssl_init_ModuleKill,
372                              apr_pool_cleanup_null);
373
374    /*
375     * Any init round fixes the global config
376     */ view japanese
377    ssl_config_global_create(base_server); /* just to avoid problems */ view japanese
378    ssl_config_global_fix(mc);
379
380    /*
381     *  try to fix the configuration and open the dedicated SSL
382     *  logfile as early as possible
383     */ view japanese
384    for (s = base_server; s; s = s->next) {
385        sc = mySrvConfig(s);
386
387        if (sc->server) {
388            sc->server->sc = sc;
389        }
390
391        /*
392         * Create the server host:port string because we need it a lot
393         */ view japanese
394        if (sc->vhost_id) {
395            /* already set. This should only happen if this config rec is
396             * shared with another server. Argh! */ view japanese
397            ap_log_error(APLOG_MARK, APLOG_WARNING, 0, s, APLOGNO(10104) 
398                         "%s, SSLSrvConfigRec shared from %s", 
399                         ssl_util_vhostid(p, s), sc->vhost_id);
400        }
401        sc->vhost_id = ssl_util_vhostid(p, s);
402        sc->vhost_id_len = strlen(sc->vhost_id);
403
404        /* Default to enabled if SSLEngine is not set explicitly, and
405         * the protocol is https. */ view japanese
406        if (ap_get_server_protocol(s) 
407            && strcmp("https", ap_get_server_protocol(s)) == 0
408            && sc->enabled == SSL_ENABLED_UNSET
409            && (!apr_is_empty_array(sc->server->pks->cert_files))) {
410            sc->enabled = SSL_ENABLED_TRUE;
411        }
412
413        /* Fix up stuff that may not have been set.  If sc->enabled is
414         * UNSET, then SSL is disabled on this vhost.  */ view japanese
415        if (sc->enabled == SSL_ENABLED_UNSET) {
416            sc->enabled = SSL_ENABLED_FALSE;
417        }
418
419        if (sc->session_cache_timeout == UNSET) {
420            sc->session_cache_timeout = SSL_SESSION_CACHE_TIMEOUT;
421        }
422
423        if (sc->server && sc->server->pphrase_dialog_type == SSL_PPTYPE_UNSET) {
424            sc->server->pphrase_dialog_type = SSL_PPTYPE_BUILTIN;
425        }
426    }
427
428#if APR_HAS_THREADS && MODSSL_USE_OPENSSL_PRE_1_1_API
429    ssl_util_thread_setup(p);
430#endif
431
432    /*
433     * SSL external crypto device ("engine") support
434     */ view japanese
435    if ((rv = ssl_init_Engine(base_server, p)) != APR_SUCCESS) {
436        return rv;
437    }
438
439    ap_log_error(APLOG_MARK, APLOG_INFO, 0, base_server, APLOGNO(01883)
440                 "Init: Initialized %s library", MODSSL_LIBRARY_NAME);
441
442    /*
443     * Seed the Pseudo Random Number Generator (PRNG)
444     * only need ptemp here; nothing inside allocated from the pool
445     * needs to live once we return from ssl_rand_seed().
446     */ view japanese
447    ssl_rand_seed(base_server, ptemp, SSL_RSCTX_STARTUP, "Init: ");
448
449#ifdef HAVE_FIPS
450    if (!modssl_fips_is_enabled() && mc->fips == TRUE) {
451        if (!modssl_fips_enable(1)) {
452            ap_log_error(APLOG_MARK, APLOG_EMERG, 0, base_server, APLOGNO(01885)
453                         "Could not enable FIPS mode");
454            ssl_log_ssl_error(SSLLOG_MARK, APLOG_EMERG, base_server);
455            return ssl_die(base_server);
456        }
457
458        apr_pool_cleanup_register(p, NULL, modssl_fips_cleanup,
459                                  apr_pool_cleanup_null);
460    }
461
462    /* Log actual FIPS mode which the SSL library is operating under,
463     * which may have been set outside of the mod_ssl
464     * configuration. */ view japanese
465    if (modssl_fips_is_enabled()) {
466        ap_log_error(APLOG_MARK, APLOG_NOTICE, 0, base_server, APLOGNO(01884)
467                     MODSSL_LIBRARY_NAME " has FIPS mode enabled");
468    }
469    else {
470        ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, base_server, APLOGNO(01886)
471                     MODSSL_LIBRARY_NAME " has FIPS mode disabled");
472    }
473#endif
474
475    /*
476     * initialize the mutex handling
477     */ view japanese
478    if (!ssl_mutex_init(base_server, p)) {
479        return HTTP_INTERNAL_SERVER_ERROR;
480    }
481#ifdef HAVE_OCSP_STAPLING
482    ssl_stapling_certinfo_hash_init(p);
483#endif
484
485    /*
486     * initialize session caching
487     */ view japanese
488    if ((rv = ssl_scache_init(base_server, p)) != APR_SUCCESS) {
489        return rv;
490    }
491
492    pphrases = apr_array_make(ptemp, 2, sizeof(char *));
493
494    /*
495     *  initialize servers
496     */ view japanese
497    ap_log_error(APLOG_MARK, APLOG_INFO, 0, base_server, APLOGNO(01887)
498                 "Init: Initializing (virtual) servers for SSL");
499
500    for (s = base_server; s; s = s->next) {
501        sc = mySrvConfig(s);
502        /*
503         * Either now skip this server when SSL is disabled for
504         * it or give out some information about what we're
505         * configuring.
506         */ view japanese
507
508        /*
509         * Read the server certificate and key
510         */ view japanese
511        if ((rv = ssl_init_ConfigureServer(s, p, ptemp, sc, pphrases))
512            != APR_SUCCESS) {
513            return rv;
514        }
515    }
516
517    if (pphrases->nelts > 0) {
518        memset(pphrases->elts, 0, pphrases->elt_size * pphrases->nelts);
519        pphrases->nelts = 0;
520        ap_log_error(APLOG_MARK, APLOG_INFO, 0, s, APLOGNO(02560)
521                     "Init: Wiped out the queried pass phrases from memory");
522    }
523
524    /*
525     * Configuration consistency checks
526     */ view japanese
527    if ((rv = ssl_init_CheckServers(base_server, ptemp)) != APR_SUCCESS) {
528        return rv;
529    }
530
531    for (s = base_server; s; s = s->next) {
532        SSLDirConfigRec *sdc = ap_get_module_config(s->lookup_defaults,
533                                                    &ssl_module);
534
535        sc = mySrvConfig(s);
536        if (sc->enabled == SSL_ENABLED_TRUE) {
537            if ((rv = ssl_run_init_server(s, p, 0, sc->server->ssl_ctx)) != APR_SUCCESS) {
538                return rv;
539            }
540        }
541
542        if (sdc->proxy_enabled) {
543            rv = ssl_run_init_server(s, p, 1, sdc->proxy->ssl_ctx);
544            if (rv != APR_SUCCESS) {
545                return rv;
546            }
547        }
548
549        sc->sni_policy_hash = create_sni_policy_hash(p, ptemp, mc->snivh_policy, sc);
550    }
551
552    /*
553     *  Announce mod_ssl and SSL library in HTTP Server field
554     *  as ``mod_ssl/X.X.X OpenSSL/X.X.X''
555     */ view japanese
556    ssl_add_version_components(ptemp, p, base_server);
557
558    modssl_init_app_data2_idx(); /* for modssl_get_app_data2() at request time */ view japanese
559
560#if MODSSL_USE_OPENSSL_PRE_1_1_API
561    init_dh_params();
562#else
563    init_bio_methods();
564#endif
565
566#ifdef HAVE_OPENSSL_KEYLOG
567    {
568        const char *logfn = getenv("SSLKEYLOGFILE");
569
570        if (logfn) {
571            rv = apr_file_open(&mc->keylog_file, logfn,
572                               APR_FOPEN_CREATE|APR_FOPEN_WRITE|APR_FOPEN_APPEND|APR_FOPEN_LARGEFILE,
573                               APR_FPROT_UREAD|APR_FPROT_UWRITE,
574                               mc->pPool);
575            if (rv) {
576                ap_log_error(APLOG_MARK, APLOG_NOTICE, rv, s, APLOGNO(10226)
577                             "Could not open log file '%s' configured via SSLKEYLOGFILE",
578                             logfn);
579                return rv;
580            }
581
582            ap_log_error(APLOG_MARK, APLOG_NOTICE, 0, s, APLOGNO(10227)
583                         "Init: Logging SSL private key material to %s", logfn);
584        }
585    }
586#endif
587    
588    return OK;
589}

591/*
592 * Support for external a Crypto Device ("engine"), usually
593 * a hardware accelerator card for crypto operations.
594 */ view japanese
595apr_status_t ssl_init_Engine(server_rec *s, apr_pool_t *p)
596{
597#if MODSSL_HAVE_ENGINE_API
598    SSLModConfigRec *mc = myModConfig(s);
599    ENGINE *e;
600
601    if (mc->szCryptoDevice) {
602        if (!(e = ENGINE_by_id(mc->szCryptoDevice))) {
603            ap_log_error(APLOG_MARK, APLOG_EMERG, 0, s, APLOGNO(01888)
604                         "Init: Failed to load Crypto Device API `%s'",
605                         mc->szCryptoDevice);
606            ssl_log_ssl_error(SSLLOG_MARK, APLOG_EMERG, s);
607            return ssl_die(s);
608        }
609
610#ifdef ENGINE_CTRL_CHIL_SET_FORKCHECK
611        if (strEQ(mc->szCryptoDevice, "chil")) {
612            ENGINE_ctrl(e, ENGINE_CTRL_CHIL_SET_FORKCHECK, 1, 0, 0);
613        }
614#endif
615
616        if (!ENGINE_set_default(e, ENGINE_METHOD_ALL)) {
617            ap_log_error(APLOG_MARK, APLOG_EMERG, 0, s, APLOGNO(01889)
618                         "Init: Failed to enable Crypto Device API `%s'",
619                         mc->szCryptoDevice);
620            ssl_log_ssl_error(SSLLOG_MARK, APLOG_EMERG, s);
621            return ssl_die(s);
622        }
623        ap_log_error(APLOG_MARK, APLOG_INFO, 0, s, APLOGNO(01890)
624                     "Init: loaded Crypto Device API `%s'",
625                     mc->szCryptoDevice);
626
627        ENGINE_free(e);
628    }
629#endif
630    return APR_SUCCESS;
631}

718static apr_status_t ssl_init_ctx_protocol(server_rec *s,
719                                          apr_pool_t *p,
720                                          apr_pool_t *ptemp,
721                                          modssl_ctx_t *mctx)
722{
723    SSL_CTX *ctx = NULL;
724    MODSSL_SSL_METHOD_CONST SSL_METHOD *method = NULL;
725    char *cp;
726    int protocol = mctx->protocol;
727    SSLSrvConfigRec *sc = mySrvConfig(s);
728#if OPENSSL_VERSION_NUMBER >= 0x10100000L
729    /* default is highest supported version, will be overridden below */ view japanese
730#if SSL_HAVE_PROTOCOL_TLSV1_3
731    int prot = TLS1_3_VERSION;
732#else
733    int prot = TLS1_2_VERSION;
734#endif
735#endif
736
737    /*
738     *  Create the new per-server SSL context
739     */ view japanese
740    if (protocol == SSL_PROTOCOL_NONE) {
741        ap_log_error(APLOG_MARK, APLOG_EMERG, 0, s, APLOGNO(02231)
742                "No SSL protocols available [hint: SSLProtocol]");
743        return ssl_die(s);
744    }
745
746    cp = apr_pstrcat(p,
747#ifndef OPENSSL_NO_SSL3
748                     (protocol & SSL_PROTOCOL_SSLV3 ? "SSLv3, " : ""),
749#endif
750                     (protocol & SSL_PROTOCOL_TLSV1 ? "TLSv1, " : ""),
751#ifdef HAVE_TLSV1_X
752                     (protocol & SSL_PROTOCOL_TLSV1_1 ? "TLSv1.1, " : ""),
753                     (protocol & SSL_PROTOCOL_TLSV1_2 ? "TLSv1.2, " : ""),
754#if SSL_HAVE_PROTOCOL_TLSV1_3
755                     (protocol & SSL_PROTOCOL_TLSV1_3 ? "TLSv1.3, " : ""),
756#endif
757#endif
758                     NULL);
759    cp[strlen(cp)-2] = NUL;
760
761    ap_log_error(APLOG_MARK, APLOG_TRACE3, 0, s,
762                 "Creating new SSL context (protocols: %s)", cp);
763
764#if OPENSSL_VERSION_NUMBER < 0x10100000L
765#ifndef OPENSSL_NO_SSL3
766    if (protocol == SSL_PROTOCOL_SSLV3) {
767        method = mctx->pkp ?
768            SSLv3_client_method() : /* proxy */
769            SSLv3_server_method();  /* server */
770    }
771    else
772#endif
773    if (protocol == SSL_PROTOCOL_TLSV1) {
774        method = mctx->pkp ?
775            TLSv1_client_method() : /* proxy */
776            TLSv1_server_method();  /* server */
777    }
778#ifdef HAVE_TLSV1_X
779    else if (protocol == SSL_PROTOCOL_TLSV1_1) {
780        method = mctx->pkp ?
781            TLSv1_1_client_method() : /* proxy */
782            TLSv1_1_server_method();  /* server */
783    }
784    else if (protocol == SSL_PROTOCOL_TLSV1_2) {
785        method = mctx->pkp ?
786            TLSv1_2_client_method() : /* proxy */
787            TLSv1_2_server_method();  /* server */
788    }
789#if SSL_HAVE_PROTOCOL_TLSV1_3
790    else if (protocol == SSL_PROTOCOL_TLSV1_3) {
791        method = mctx->pkp ?
792            TLSv1_3_client_method() : /* proxy */
793            TLSv1_3_server_method();  /* server */
794    }
795#endif
796#endif
797    else { /* For multiple protocols, we need a flexible method */ view japanese
798        method = mctx->pkp ?
799            SSLv23_client_method() : /* proxy */
800            SSLv23_server_method();  /* server */
801    }
802#else
803    method = mctx->pkp ?
804        TLS_client_method() : /* proxy */ view japanese
805        TLS_server_method();  /* server */ view japanese
806#endif
807    ctx = SSL_CTX_new(method);
808    if (ctx == NULL) {
809        /* Can fail for some system/install mis-configuration. */ view japanese
810        ssl_log_ssl_error(SSLLOG_MARK, APLOG_EMERG, s);
811        return ssl_die(s);
812    }
813
814    mctx->ssl_ctx = ctx;
815
816    SSL_CTX_set_options(ctx, SSL_OP_ALL);
817
818#if OPENSSL_VERSION_NUMBER < 0x10100000L  || \
819	(defined(LIBRESSL_VERSION_NUMBER) && LIBRESSL_VERSION_NUMBER < 0x20800000L)
820    /* always disable SSLv2, as per RFC 6176 */ view japanese
821    SSL_CTX_set_options(ctx, SSL_OP_NO_SSLv2);
822
823#ifndef OPENSSL_NO_SSL3
824    if (!(protocol & SSL_PROTOCOL_SSLV3)) {
825        SSL_CTX_set_options(ctx, SSL_OP_NO_SSLv3);
826    }
827#endif
828
829    if (!(protocol & SSL_PROTOCOL_TLSV1)) {
830        SSL_CTX_set_options(ctx, SSL_OP_NO_TLSv1);
831    }
832
833#ifdef HAVE_TLSV1_X
834    if (!(protocol & SSL_PROTOCOL_TLSV1_1)) {
835        SSL_CTX_set_options(ctx, SSL_OP_NO_TLSv1_1);
836    }
837
838    if (!(protocol & SSL_PROTOCOL_TLSV1_2)) {
839        SSL_CTX_set_options(ctx, SSL_OP_NO_TLSv1_2);
840    }
841#if SSL_HAVE_PROTOCOL_TLSV1_3
842    ssl_set_ctx_protocol_option(s, ctx, SSL_OP_NO_TLSv1_3,
843                                protocol & SSL_PROTOCOL_TLSV1_3, "TLSv1.3");
844#endif
845#endif
846
847#else /* #if OPENSSL_VERSION_NUMBER < 0x10100000L */ view japanese
848    /* We first determine the maximum protocol version we should provide */ view japanese
849#if SSL_HAVE_PROTOCOL_TLSV1_3
850    if (protocol & SSL_PROTOCOL_TLSV1_3) {
851        prot = TLS1_3_VERSION;
852    } else
853#endif
854    if (protocol & SSL_PROTOCOL_TLSV1_2) {
855        prot = TLS1_2_VERSION;
856    } else if (protocol & SSL_PROTOCOL_TLSV1_1) {
857        prot = TLS1_1_VERSION;
858    } else if (protocol & SSL_PROTOCOL_TLSV1) {
859        prot = TLS1_VERSION;
860#ifndef OPENSSL_NO_SSL3
861    } else if (protocol & SSL_PROTOCOL_SSLV3) {
862        prot = SSL3_VERSION;
863#endif
864    } else {
865        SSL_CTX_free(ctx);
866        mctx->ssl_ctx = NULL;
867        ap_log_error(APLOG_MARK, APLOG_EMERG, 0, s, APLOGNO(03378)
868                "No SSL protocols available [hint: SSLProtocol]");
869        return ssl_die(s);
870    }
871    SSL_CTX_set_max_proto_version(ctx, prot);
872
873    /* Next we scan for the minimal protocol version we should provide,
874     * but we do not allow holes between max and min */ view japanese
875#if SSL_HAVE_PROTOCOL_TLSV1_3
876    if (prot == TLS1_3_VERSION && protocol & SSL_PROTOCOL_TLSV1_2) {
877        prot = TLS1_2_VERSION;
878    }
879#endif
880    if (prot == TLS1_2_VERSION && protocol & SSL_PROTOCOL_TLSV1_1) {
881        prot = TLS1_1_VERSION;
882    }
883    if (prot == TLS1_1_VERSION && protocol & SSL_PROTOCOL_TLSV1) {
884        prot = TLS1_VERSION;
885    }
886#ifndef OPENSSL_NO_SSL3
887    if (prot == TLS1_VERSION && protocol & SSL_PROTOCOL_SSLV3) {
888        prot = SSL3_VERSION;
889    }
890#endif
891    SSL_CTX_set_min_proto_version(ctx, prot);
892#endif /* if OPENSSL_VERSION_NUMBER < 0x10100000L */ view japanese
893
894#ifdef SSL_OP_CIPHER_SERVER_PREFERENCE
895    if (sc->cipher_server_pref == TRUE) {
896        SSL_CTX_set_options(ctx, SSL_OP_CIPHER_SERVER_PREFERENCE);
897    }
898#endif
899
900
901#ifndef OPENSSL_NO_COMP
902    if (sc->compression != TRUE) {
903#ifdef SSL_OP_NO_COMPRESSION
904        /* OpenSSL >= 1.0 only */ view japanese
905        SSL_CTX_set_options(ctx, SSL_OP_NO_COMPRESSION);
906#else
907        sk_SSL_COMP_zero(SSL_COMP_get_compression_methods());
908#endif
909    }
910#endif
911
912#ifdef SSL_OP_NO_TICKET
913    /*
914     * Configure using RFC 5077 TLS session tickets
915     * for session resumption.
916     */ view japanese
917    if (sc->session_tickets == FALSE) {
918        SSL_CTX_set_options(ctx, SSL_OP_NO_TICKET);
919    }
920#endif
921
922#ifdef SSL_OP_ALLOW_UNSAFE_LEGACY_RENEGOTIATION
923    if (sc->insecure_reneg == TRUE) {
924        SSL_CTX_set_options(ctx, SSL_OP_ALLOW_UNSAFE_LEGACY_RENEGOTIATION);
925    }
926#endif
927
928    SSL_CTX_set_app_data(ctx, s);
929
930    /*
931     * Configure additional context ingredients
932     */ view japanese
933    SSL_CTX_set_options(ctx, SSL_OP_SINGLE_DH_USE);
934#ifdef HAVE_ECC
935    SSL_CTX_set_options(ctx, SSL_OP_SINGLE_ECDH_USE);
936#endif
937
938#ifdef SSL_OP_NO_SESSION_RESUMPTION_ON_RENEGOTIATION
939    /*
940     * Disallow a session from being resumed during a renegotiation,
941     * so that an acceptable cipher suite can be negotiated.
942     */ view japanese
943    SSL_CTX_set_options(ctx, SSL_OP_NO_SESSION_RESUMPTION_ON_RENEGOTIATION);
944#endif
945
946#ifdef SSL_MODE_RELEASE_BUFFERS
947    /* If httpd is configured to reduce mem usage, ask openssl to do so, too */ view japanese
948    if (ap_max_mem_free != APR_ALLOCATOR_MAX_FREE_UNLIMITED)
949        SSL_CTX_set_mode(ctx, SSL_MODE_RELEASE_BUFFERS);
950#endif
951
952#if OPENSSL_VERSION_NUMBER >= 0x1010100fL
953    /* For OpenSSL >=1.1.1, disable auto-retry mode so it's possible
954     * to consume handshake records without blocking for app-data.
955     * https://github.com/openssl/openssl/issues/7178 */ view japanese
956    SSL_CTX_clear_mode(ctx, SSL_MODE_AUTO_RETRY);
957#endif
958
959#ifdef HAVE_OPENSSL_KEYLOG
960    if (mctx->sc->mc->keylog_file) {
961        SSL_CTX_set_keylog_callback(ctx, modssl_callback_keylog);
962    }
963#endif
964
965#ifdef SSL_OP_NO_RENEGOTIATION
966    /* For server-side SSL_CTX, disable renegotiation by default.. */ view japanese
967    if (!mctx->pkp) {
968        SSL_CTX_set_options(ctx, SSL_OP_NO_RENEGOTIATION);
969    }
970#endif
971
972#ifdef SSL_OP_IGNORE_UNEXPECTED_EOF
973    /* For server-side SSL_CTX, enable ignoring unexpected EOF */ view japanese
974    /* (OpenSSL 1.1.1 behavioural compatibility).. */ view japanese
975    if (!mctx->pkp) {
976        SSL_CTX_set_options(ctx, SSL_OP_IGNORE_UNEXPECTED_EOF);
977    }
978#endif
979    
980    return APR_SUCCESS;
981}

1051static apr_status_t ssl_init_ctx_verify(server_rec *s,
1052                                        apr_pool_t *p,
1053                                        apr_pool_t *ptemp,
1054                                        modssl_ctx_t *mctx)
1055{
1056    SSL_CTX *ctx = mctx->ssl_ctx;
1057
1058    int verify = SSL_VERIFY_NONE;
1059    STACK_OF(X509_NAME) *ca_list;
1060
1061    if (mctx->auth.verify_mode == SSL_CVERIFY_UNSET) {
1062        mctx->auth.verify_mode = SSL_CVERIFY_NONE;
1063    }
1064
1065    if (mctx->auth.verify_depth == UNSET) {
1066        mctx->auth.verify_depth = 1;
1067    }
1068
1069    /*
1070     *  Configure callbacks for SSL context
1071     */ view japanese
1072    if (mctx->auth.verify_mode == SSL_CVERIFY_REQUIRE) {
1073        verify |= SSL_VERIFY_PEER_STRICT;
1074    }
1075
1076    if ((mctx->auth.verify_mode == SSL_CVERIFY_OPTIONAL) ||
1077        (mctx->auth.verify_mode == SSL_CVERIFY_OPTIONAL_NO_CA))
1078    {
1079        verify |= SSL_VERIFY_PEER;
1080    }
1081
1082    SSL_CTX_set_verify(ctx, verify, ssl_callback_SSLVerify);
1083
1084    /*
1085     * Configure Client Authentication details
1086     */ view japanese
1087    if (mctx->auth.ca_cert_file || mctx->auth.ca_cert_path) {
1088        ap_log_error(APLOG_MARK, APLOG_TRACE1, 0, s,
1089                     "Configuring client authentication");
1090
1091        if (!modssl_CTX_load_verify_locations(ctx, mctx->auth.ca_cert_file,
1092                                                   mctx->auth.ca_cert_path)) {
1093            ap_log_error(APLOG_MARK, APLOG_EMERG, 0, s, APLOGNO(01895)
1094                    "Unable to configure verify locations "
1095                    "for client authentication");
1096            ssl_log_ssl_error(SSLLOG_MARK, APLOG_EMERG, s);
1097            return ssl_die(s);
1098        }
1099
1100        if (mctx->pks && (mctx->pks->ca_name_file || mctx->pks->ca_name_path)) {
1101            ca_list = ssl_init_FindCAList(s, ptemp,
1102                                          mctx->pks->ca_name_file,
1103                                          mctx->pks->ca_name_path);
1104        } else
1105            ca_list = ssl_init_FindCAList(s, ptemp,
1106                                          mctx->auth.ca_cert_file,
1107                                          mctx->auth.ca_cert_path);
1108        if (sk_X509_NAME_num(ca_list) <= 0) {
1109            ap_log_error(APLOG_MARK, APLOG_EMERG, 0, s, APLOGNO(01896)
1110                    "Unable to determine list of acceptable "
1111                    "CA certificates for client authentication");
1112            return ssl_die(s);
1113        }
1114
1115        SSL_CTX_set_client_CA_list(ctx, ca_list);
1116    }
1117
1118    /*
1119     * Give a warning when no CAs were configured but client authentication
1120     * should take place. This cannot work.
1121     */ view japanese
1122    if (mctx->auth.verify_mode == SSL_CVERIFY_REQUIRE) {
1123        ca_list = SSL_CTX_get_client_CA_list(ctx);
1124
1125        if (sk_X509_NAME_num(ca_list) == 0) {
1126            ap_log_error(APLOG_MARK, APLOG_WARNING, 0, s, APLOGNO(01897)
1127                         "Init: Oops, you want to request client "
1128                         "authentication, but no CAs are known for "
1129                         "verification!?  [Hint: SSLCACertificate*]");
1130        }
1131    }
1132
1133    return APR_SUCCESS;
1134}

1136static apr_status_t ssl_init_ctx_cipher_suite(server_rec *s,
1137                                              apr_pool_t *p,
1138                                              apr_pool_t *ptemp,
1139                                              modssl_ctx_t *mctx)
1140{
1141    SSL_CTX *ctx = mctx->ssl_ctx;
1142    const char *suite;
1143
1144    /*
1145     *  Configure SSL Cipher Suite. Always disable NULL and export ciphers,
1146     *  see also ssl_engine_config.c:ssl_cmd_SSLCipherSuite().
1147     *  OpenSSL's SSL_DEFAULT_CIPHER_LIST includes !aNULL:!eNULL from 0.9.8f,
1148     *  and !EXP from 0.9.8zf/1.0.1m/1.0.2a, so append them while we support
1149     *  earlier versions.
1150     */ view japanese
1151    suite = mctx->auth.cipher_suite ? mctx->auth.cipher_suite :
1152            apr_pstrcat(ptemp, SSL_DEFAULT_CIPHER_LIST, ":!aNULL:!eNULL:!EXP",
1153                        NULL);
1154
1155    ap_log_error(APLOG_MARK, APLOG_TRACE1, 0, s,
1156                 "Configuring permitted SSL ciphers [%s]",
1157                 suite);
1158
1159    if (!SSL_CTX_set_cipher_list(ctx, suite)) {
1160        ap_log_error(APLOG_MARK, APLOG_EMERG, 0, s, APLOGNO(01898)
1161                "Unable to configure permitted SSL ciphers");
1162        ssl_log_ssl_error(SSLLOG_MARK, APLOG_EMERG, s);
1163        return ssl_die(s);
1164    }
1165#if SSL_HAVE_PROTOCOL_TLSV1_3

1264/*
1265 * Read a file that optionally contains the server certificate in PEM
1266 * format, possibly followed by a sequence of CA certificates that
1267 * should be sent to the peer in the SSL Certificate message.
1268 */ view japanese
1269static int use_certificate_chain(
1270    SSL_CTX *ctx, char *file, int skipfirst, pem_password_cb *cb)
1271{
1272    BIO *bio;
1273    X509 *x509;
1274    unsigned long err;
1275    int n;
1276
1277    if ((bio = BIO_new(BIO_s_file())) == NULL)
1278        return -1;
1279    if (BIO_read_filename(bio, file) <= 0) {
1280        BIO_free(bio);
1281        return -1;
1282    }
1283    /* optionally skip a leading server certificate */ view japanese
1284    if (skipfirst) {
1285        if ((x509 = PEM_read_bio_X509(bio, NULL, cb, NULL)) == NULL) {
1286            BIO_free(bio);
1287            return -1;
1288        }
1289        X509_free(x509);
1290    }
1291    /* free a perhaps already configured extra chain */ view japanese
1292#ifdef OPENSSL_NO_SSL_INTERN
1293    SSL_CTX_clear_extra_chain_certs(ctx);
1294#else
1295    if (ctx->extra_certs != NULL) {
1296        sk_X509_pop_free((STACK_OF(X509) *)ctx->extra_certs, X509_free);
1297        ctx->extra_certs = NULL;
1298    }
1299#endif
1300
1301    /* create new extra chain by loading the certs */ view japanese
1302    n = 0;
1303    ERR_clear_error();
1304    while ((x509 = PEM_read_bio_X509(bio, NULL, cb, NULL)) != NULL) {
1305        if (!SSL_CTX_add_extra_chain_cert(ctx, x509)) {
1306            X509_free(x509);
1307            BIO_free(bio);
1308            return -1;
1309        }
1310        n++;
1311    }
1312    /* Make sure that only the error is just an EOF */ view japanese
1313    if ((err = ERR_peek_error()) > 0) {
1314        if (!(   ERR_GET_LIB(err) == ERR_LIB_PEM
1315              && ERR_GET_REASON(err) == PEM_R_NO_START_LINE)) {
1316            BIO_free(bio);
1317            return -1;
1318        }
1319        while (ERR_get_error() > 0) ;
1320    }
1321    BIO_free(bio);
1322    return n;
1323}

1325static apr_status_t ssl_init_ctx_cert_chain(server_rec *s,
1326                                            apr_pool_t *p,
1327                                            apr_pool_t *ptemp,
1328                                            modssl_ctx_t *mctx)
1329{
1330    BOOL skip_first = FALSE;
1331    int i, n;
1332    const char *chain = mctx->cert_chain;
1333
1334    /*
1335     * Optionally configure extra server certificate chain certificates.
1336     * This is usually done by OpenSSL automatically when one of the
1337     * server cert issuers are found under SSLCACertificatePath or in
1338     * SSLCACertificateFile. But because these are intended for client
1339     * authentication it can conflict. For instance when you use a
1340     * Global ID server certificate you've to send out the intermediate
1341     * CA certificate, too. When you would just configure this with
1342     * SSLCACertificateFile and also use client authentication mod_ssl
1343     * would accept all clients also issued by this CA. Obviously this
1344     * isn't what we want in this situation. So this feature here exists
1345     * to allow one to explicitly configure CA certificates which are
1346     * used only for the server certificate chain.
1347     */ view japanese
1348    if (!chain) {
1349        return APR_SUCCESS;
1350    }
1351
1352    for (i = 0; (i < mctx->pks->cert_files->nelts) &&
1353         APR_ARRAY_IDX(mctx->pks->cert_files, i, const char *); i++) {
1354        if (strEQ(APR_ARRAY_IDX(mctx->pks->cert_files, i, const char *), chain)) {
1355            skip_first = TRUE;
1356            break;
1357        }
1358    }
1359
1360    n = use_certificate_chain(mctx->ssl_ctx, (char *)chain, skip_first, NULL);
1361    if (n < 0) {
1362        ap_log_error(APLOG_MARK, APLOG_EMERG, 0, s, APLOGNO(01903)
1363                "Failed to configure CA certificate chain!");
1364        return ssl_die(s);
1365    }
1366
1367    ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, s, APLOGNO(01904)
1368                 "Configuring server certificate chain "
1369                 "(%d CA certificate%s)",
1370                 n, n == 1 ? "" : "s");
1371
1372    return APR_SUCCESS;
1373}

1487    EVP_PKEY *pkey;
1488    int custom_dh_done = 0;
1489#ifdef HAVE_ECC
1490    EC_GROUP *ecgroup = NULL;

1492#endif

1499                (certfile = APR_ARRAY_IDX(mctx->pks->cert_files, i,
1500                                          const char *));

1515                                              SSL_FILETYPE_PEM) < 1)) {

1518                             key_id, certfile);

1520                return APR_EGENERAL;
1521            }
1522        } else {
1523            if ((SSL_CTX_use_certificate_chain_file(mctx->ssl_ctx,
1524                                                    certfile) < 1)) {
1525                ap_log_error(APLOG_MARK, APLOG_EMERG, 0, s, APLOGNO(02562)
1526                             "Failed to configure certificate %s (with chain),"
1527                             " check %s", key_id, certfile);

1556                                             mc->szCryptoDevice : "provider",
1557                                 certfile);
1558                    ssl_log_ssl_error(SSLLOG_MARK, APLOG_EMERG, s);
1559                    return APR_EGENERAL;
1560                }

1562                /* SSL_CTX now owns the cert. */ view japanese

1846static apr_status_t ssl_init_proxy_certs(server_rec *s,
1847                                         apr_pool_t *p,
1848                                         apr_pool_t *ptemp,
1849                                         modssl_ctx_t *mctx)
1850{
1851    int n, ncerts = 0;
1852    STACK_OF(X509_INFO) *sk;
1853    modssl_pk_proxy_t *pkp = mctx->pkp;
1854    STACK_OF(X509) *chain;
1855    X509_STORE_CTX *sctx;
1856    X509_STORE *store = SSL_CTX_get_cert_store(mctx->ssl_ctx);
1857    int addl_chain = 0; /* non-zero if additional chain certs were
1858                         * added to store */ view japanese
1859
1860    ap_assert(store != NULL); /* safe to assume always non-NULL? */ view japanese
1861
1862#if OPENSSL_VERSION_NUMBER >= 0x1010100fL && !defined(LIBRESSL_VERSION_NUMBER)
1863    /* For OpenSSL >=1.1.1, turn on client cert support which is
1864     * otherwise turned off by default (by design).
1865     * https://github.com/openssl/openssl/issues/6933 */ view japanese
1866    SSL_CTX_set_post_handshake_auth(mctx->ssl_ctx, 1);
1867#endif
1868    
1869    SSL_CTX_set_client_cert_cb(mctx->ssl_ctx,
1870                               ssl_callback_proxy_cert);
1871
1872    if (!(pkp->cert_file || pkp->cert_path)) {
1873        return APR_SUCCESS;
1874    }
1875
1876    sk = sk_X509_INFO_new_null();
1877
1878    if (pkp->cert_file) {
1879        load_x509_info(ptemp, sk, pkp->cert_file);
1880    }
1881
1882    if (pkp->cert_path) {
1883        ssl_init_ca_cert_path(s, ptemp, pkp->cert_path, NULL, sk);
1884    }
1885
1886    /* Check that all client certs have got certificates and private
1887     * keys.  Note the number of certs in the stack may decrease
1888     * during the loop. */ view japanese
1889    for (n = 0; n < sk_X509_INFO_num(sk); n++) {
1890        X509_INFO *inf = sk_X509_INFO_value(sk, n);
1891        int has_privkey = inf->x_pkey && inf->x_pkey->dec_pkey;
1892
1893        /* For a lone certificate in the file, trust it as a
1894         * CA/intermediate certificate. */ view japanese
1895        if (inf->x509 && !has_privkey && !inf->enc_data) {
1896            ssl_log_xerror(SSLLOG_MARK, APLOG_DEBUG, 0, ptemp, s, inf->x509,
1897                           APLOGNO(10261) "Trusting non-leaf certificate");
1898            X509_STORE_add_cert(store, inf->x509); /* increments inf->x509 */ view japanese
1899            /* Delete from the stack and iterate again. */ view japanese
1900            X509_INFO_free(inf);
1901            sk_X509_INFO_delete(sk, n);
1902            n--;
1903            addl_chain = 1;
1904            continue;
1905        }
1906
1907        if (!has_privkey || inf->enc_data) {
1908            sk_X509_INFO_free(sk);
1909            ap_log_error(APLOG_MARK, APLOG_STARTUP, 0, s, APLOGNO(02252)
1910                         "incomplete client cert configured for SSL proxy "
1911                         "(missing or encrypted private key?)");
1912            return ssl_die(s);
1913        }
1914        
1915        if (X509_check_private_key(inf->x509, inf->x_pkey->dec_pkey) != 1) {
1916            ssl_log_xerror(SSLLOG_MARK, APLOG_STARTUP, 0, ptemp, s, inf->x509,
1917                           APLOGNO(02326) "proxy client certificate and "
1918                           "private key do not match");
1919            ssl_log_ssl_error(SSLLOG_MARK, APLOG_ERR, s);
1920            return ssl_die(s);
1921        }
1922    }
1923
1924    if ((ncerts = sk_X509_INFO_num(sk)) <= 0) {
1925        sk_X509_INFO_free(sk);
1926        ap_log_error(APLOG_MARK, APLOG_WARNING, 0, s, APLOGNO(02206)
1927                     "no client certs found for SSL proxy");
1928        return APR_SUCCESS;
1929    }
1930
1931    ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, s, APLOGNO(02207)
1932                 "loaded %d client certs for SSL proxy",
1933                 ncerts);
1934    pkp->certs = sk;
1935
1936    /* If any chain certs are configured, build the ->ca_certs chains
1937     * corresponding to the loaded keypairs. */ view japanese
1938    if (!pkp->ca_cert_file && !addl_chain) {
1939        return APR_SUCCESS;
1940    }
1941
1942    /* If SSLProxyMachineCertificateChainFile is configured, load all
1943     * the CA certs and have OpenSSL attempt to construct a full chain
1944     * from each configured end-entity cert up to a root.  This will
1945     * allow selection of the correct cert given a list of root CA
1946     * names in the certificate request from the server.  */ view japanese
1947    pkp->ca_certs = (STACK_OF(X509) **) apr_pcalloc(p, ncerts * sizeof(sk));
1948    sctx = X509_STORE_CTX_new();
1949
1950    if (!sctx) {
1951        ap_log_error(APLOG_MARK, APLOG_EMERG, 0, s, APLOGNO(02208)
1952                     "SSL proxy client cert initialization failed");
1953        ssl_log_ssl_error(SSLLOG_MARK, APLOG_EMERG, s);
1954        sk_X509_INFO_free(sk);
1955        return ssl_die(s);
1956    }
1957
1958    modssl_X509_STORE_load_locations(store, pkp->ca_cert_file, NULL);
1959
1960    for (n = 0; n < ncerts; n++) {
1961        int i;
1962
1963        X509_INFO *inf = sk_X509_INFO_value(pkp->certs, n);
1964        if (!X509_STORE_CTX_init(sctx, store, inf->x509, NULL)) {
1965            sk_X509_INFO_free(sk);
1966            X509_STORE_CTX_free(sctx);
1967            return ssl_die(s);
1968        }
1969
1970        /* Attempt to verify the client cert */ view japanese
1971        if (X509_verify_cert(sctx) != 1) {
1972            int err = X509_STORE_CTX_get_error(sctx);
1973            ssl_log_xerror(SSLLOG_MARK, APLOG_WARNING, 0, ptemp, s, inf->x509,
1974                           APLOGNO(02270) "SSL proxy client cert chain "
1975                           "verification failed: %s :",
1976                           X509_verify_cert_error_string(err));
1977        }
1978
1979        /* Clear X509_verify_cert errors */ view japanese
1980        ERR_clear_error();
1981
1982        /* Obtain a copy of the verified chain */ view japanese
1983        chain = X509_STORE_CTX_get1_chain(sctx);
1984
1985        if (chain != NULL) {
1986            /* Discard end entity cert from the chain */ view japanese
1987            X509_free(sk_X509_shift(chain));
1988
1989            if ((i = sk_X509_num(chain)) > 0) {
1990                /* Store the chain for later use */ view japanese
1991                pkp->ca_certs[n] = chain;
1992            }
1993            else {
1994                /* Discard empty chain */ view japanese
1995                sk_X509_pop_free(chain, X509_free);
1996                pkp->ca_certs[n] = NULL;
1997            }
1998
1999            ssl_log_xerror(SSLLOG_MARK, APLOG_DEBUG, 0, ptemp, s, inf->x509,
2000                           APLOGNO(02271)
2001                           "loaded %i intermediate CA%s for cert %i: ",
2002                           i, i == 1 ? "" : "s", n);
2003            if (i > 0) {
2004                int j;
2005                for (j = 0; j < i; j++) {
2006                    ssl_log_xerror(SSLLOG_MARK, APLOG_DEBUG, 0, ptemp, s,
2007                                   sk_X509_value(chain, j), APLOGNO(03039)
2008                                   "%i:", j);
2009                }
2010            }
2011        }
2012
2013        /* get ready for next X509_STORE_CTX_init */ view japanese
2014        X509_STORE_CTX_cleanup(sctx);
2015    }
2016
2017    X509_STORE_CTX_free(sctx);
2018
2019    return APR_SUCCESS;
2020}
