83/** During post_config phase, ask around if someone wants to provide
84 * OCSP stapling status information for the given cert (with the also
85 * provided issuer certificate). The first hook which does not
86 * return DECLINED promises to take responsibility (and respond
87 * in later calls via hook ssl_get_stapling_status).
88 * If no hook takes over, mod_ssl's own stapling implementation will
89 * be applied (if configured).
90 */ view japanese
91APR_DECLARE_EXTERNAL_HOOK(ssl, SSL, int, init_stapling_status,
92                          (server_rec *s, apr_pool_t *p, 
93                          X509 *cert, X509 *issuer))
94
95/** Anyone answering positive to ssl_init_stapling_status for a 
96 * certificate, needs to register here and supply the actual OCSP stapling
97 * status data (OCSP_RESP) for a new connection.
98 * A hook supplying the response data must return APR_SUCCESS.
99 * The data is returned in DER encoded bytes via pder and pderlen. The
100 * returned pointer may be NULL, which indicates that data is (currently)
101 * unavailable.
102 * If DER data is returned, it MUST come from a response with
103 * status OCSP_RESPONSE_STATUS_SUCCESSFUL and V_OCSP_CERTSTATUS_GOOD
104 * or V_OCSP_CERTSTATUS_REVOKED, not V_OCSP_CERTSTATUS_UNKNOWN. This means
105 * errors in OCSP retrieval are to be handled/logged by the hook and
106 * are not done by mod_ssl.
107 * Any DER bytes returned MUST be allocated via malloc() and ownership
108 * passes to mod_ssl. Meaning, the hook must return a malloced copy of
109 * the data it has. mod_ssl (or OpenSSL) will free it. 
110 */ view japanese
111APR_DECLARE_EXTERNAL_HOOK(ssl, SSL, int, get_stapling_status,
112                          (unsigned char **pder, int *pderlen, 
113                          conn_rec *c, server_rec *s, X509 *cert))
114                          
115#endif /* __MOD_SSL_OPENSSL_H__ */ view japanese

/** ssl_init_stapling_status に対して肯定的な応答をした証明書は、
 * ここに登録し、新しい接続のための実際の OCSP ステープリング
 * ステータスデータ (OCSP_RESP) を提供する必要があります。
 * 応答データを提供するフックは APR_SUCCESS を返す必要があります。
 * データは DER エンコードされたバイト列として pder および pderlen を介して返されます。
 * 返されるポインタが NULL の場合、データが（現在）利用不可能であることを示します。
 * DER データが返される場合、それは OCSP_RESPONSE_STATUS_SUCCESSFUL かつ
 * V_OCSP_CERTSTATUS_GOOD または V_OCSP_CERTSTATUS_REVOKED であり、
 * V_OCSP_CERTSTATUS_UNKNOWN ではない応答から取得されたものでなければなりません。
 * つまり、OCSP 取得時のエラーはフック側で処理/ログ記録する必要があり、
 * mod_ssl 側では行われません。
 * 返される DER バイト列は malloc() を使用して割り当てられる必要があり、
 * その所有権は mod_ssl に移転されます。つまり、フックは保持しているデータの
 * malloc されたコピーを返す必要があります。mod_ssl（または OpenSSL）がそれを解放します。
 */
