41/* Perform a speculative (and non-blocking) read from the connection
42 * filters for the given request, to determine whether there is any
43 * pending data to read.  Return non-zero if there is, else zero. */ view japanese
44static int has_buffered_data(request_rec *r)
45{
46    apr_bucket_brigade *bb;
47    apr_off_t len;
48    apr_status_t rv;
49    int result;
50
51    bb = apr_brigade_create(r->pool, r->connection->bucket_alloc);
52
53    rv = ap_get_brigade(r->connection->input_filters, bb, AP_MODE_SPECULATIVE,
54                        APR_NONBLOCK_READ, 1);
55    result = rv == APR_SUCCESS
56        && apr_brigade_length(bb, 1, &len) == APR_SUCCESS
57        && len > 0;
58
59    apr_brigade_destroy(bb);
60
61    return result;
62}

64/* If a renegotiation is required for the location, and the request
65 * includes a message body (and the client has not requested a "100
66 * Continue" response), then the client will be streaming the request
67 * body over the wire already.  In that case, it is not possible to
68 * stop and perform a new SSL handshake immediately; once the SSL
69 * library moves to the "accept" state, it will reject the SSL packets
70 * which the client is sending for the request body.
71 *
72 * To allow authentication to complete in the hook, the solution used
73 * here is to fill a (bounded) buffer with the request body, and then
74 * to reinject that request body later.
75 *
76 * This function is called to fill the renegotiation buffer for the
77 * location as required, or fail.  Returns zero on success or HTTP_
78 * error code on failure.
79 */ view japanese
80static int fill_reneg_buffer(request_rec *r, SSLDirConfigRec *dc)
81{
82    int rv;
83    apr_size_t rsize;
84
85    /* ### this is HTTP/1.1 specific, special case for protocol? */ view japanese
86    if (r->expecting_100 || !ap_request_has_body(r)) {
87        return 0;
88    }
89
90    rsize = dc->nRenegBufferSize == UNSET ? DEFAULT_RENEG_BUFFER_SIZE : dc->nRenegBufferSize;
91    if (rsize > 0) {
92        /* Fill the I/O buffer with the request body if possible. */ view japanese
93        rv = ssl_io_buffer_fill(r, rsize);
94    }
95    else {
96        /* If the reneg buffer size is set to zero, just fail. */ view japanese
97        rv = HTTP_REQUEST_ENTITY_TOO_LARGE;
98    }
99
100    return rv;
101}

103#ifdef HAVE_TLSEXT
104/* Check whether a transition from vhost sc1 to sc2 from SNI to Host:
105 * vhost selection is permitted according to the SSLVHostSNIPolicy
106 * setting.  Returns 1 if the policy treats the vhosts as compatible,
107 * else 0. */ view japanese
108static int ssl_check_vhost_sni_policy(SSLSrvConfigRec *sc1,
109                                      SSLSrvConfigRec *sc2)
110{
111    modssl_snivhpolicy_t policy = sc1->mc->snivh_policy;
112
113    /* Policy: insecure => allow everything. */ view japanese
114    if (policy == MODSSL_SNIVH_INSECURE)
115        return 1;
116    
117    /* Policy: strict => fail for any vhost transition. */ view japanese
118    if (policy == MODSSL_SNIVH_STRICT)
119        return sc1 == sc2;
120
121    /* For authonly/secure policy, compare the hash. */ view japanese
122    AP_DEBUG_ASSERT(sc1->sni_policy_hash);
123    AP_DEBUG_ASSERT(sc2->sni_policy_hash);
124
125    return strcmp(sc1->sni_policy_hash, sc2->sni_policy_hash) == 0;
126}

129/*
130 *  Post Read Request Handler
131 */ view japanese
132int ssl_hook_ReadReq(request_rec *r)
133{
134    SSLSrvConfigRec *sc = mySrvConfig(r->server);
135    SSLConnRec *sslconn;
136#ifdef HAVE_TLSEXT
137    const char *servername;
138#endif
139    SSL *ssl;
140
141    /* If we are on a slave connection, we do not expect to have an SSLConnRec,
142     * but our master connection might. */ view japanese
143    sslconn = myConnConfig(r->connection);
144    if (!(sslconn && sslconn->ssl) && r->connection->master) {
145        sslconn = myConnConfig(r->connection->master);
146    }
147
148    if (!sslconn) {
149        return DECLINED;
150    }
151
152    if (sslconn->service_unavailable) {
153        /* This is set when the SSL properties of this connection are
154         * incomplete or if this connection was made to challenge a 
155         * particular hostname (ACME). We never serve any request on 
156         * such a connection. */ view japanese
157         /* TODO: a retry-after indicator would be nice here */ view japanese
158        return HTTP_SERVICE_UNAVAILABLE;
159    }
160
161    if (sslconn->non_ssl_request == NON_SSL_SET_ERROR_MSG) {
162        apr_table_setn(r->notes, "error-notes",
163                       "Reason: You're speaking plain HTTP to an SSL-enabled "
164                       "server port.<br />\n Instead use the HTTPS scheme to "
165                       "access this URL, please.<br />\n");
166
167        /* Now that we have caught this error, forget it. we are done
168         * with using SSL on this request.
169         */ view japanese
170        sslconn->non_ssl_request = NON_SSL_OK;
171
172        return HTTP_BAD_REQUEST;
173    }
174
175    /*
176     * Get the SSL connection structure and perform the
177     * delayed interlinking from SSL back to request_rec
178     */ view japanese
179    ssl = sslconn->ssl;
180    if (!ssl) {
181        return DECLINED;
182    }
183#ifdef HAVE_TLSEXT
184    /*
185     * Perform SNI checks only on the initial request.  In particular,
186     * if these checks detect a problem, the checks shouldn't return an
187     * error again when processing an ErrorDocument redirect for the
188     * original problem.
189     */ view japanese
190    if (r->proxyreq != PROXYREQ_PROXY && ap_is_initial_req(r)) {
191        server_rec *handshakeserver = sslconn->server;
192        SSLSrvConfigRec *hssc = mySrvConfig(handshakeserver);
193
194        AP_DEBUG_ASSERT(hssc);
195
196        if ((servername = SSL_get_servername(ssl, TLSEXT_NAMETYPE_host_name))) {
197            /*
198             * The SNI extension supplied a hostname. So don't accept requests
199             * with either no hostname or a hostname that selected a different
200             * virtual host than the one used for the handshake, causing
201             * different SSL parameters to be applied, such as SSLProtocol,
202             * SSLCACertificateFile/Path and SSLCADNRequestFile/Path which
203             * cannot be renegotiated (SSLCA* due to current limitations in
204             * OpenSSL, see:
205             * http://mail-archives.apache.org/mod_mbox/httpd-dev/200806.mbox/%3C48592955.2090303@velox.ch%3E
206             * and
207             * http://mail-archives.apache.org/mod_mbox/httpd-dev/201312.mbox/%3CCAKQ1sVNpOrdiBm-UPw1hEdSN7YQXRRjeaT-MCWbW_7mN%3DuFiOw%40mail.gmail.com%3E
208             * )
209             */ view japanese
210            if (!r->hostname) {
211                ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02031)
212                            "Hostname %s provided via SNI, but no hostname"
213                            " provided in HTTP request", servername);
214                return HTTP_BAD_REQUEST;
215            }
216        }
217        else if (((sc->strict_sni_vhost_check == SSL_ENABLED_TRUE)
218                  || hssc->strict_sni_vhost_check == SSL_ENABLED_TRUE)
219                 && r->connection->vhost_lookup_data) {
220            /*
221             * We are using a name based configuration here, but no hostname was
222             * provided via SNI. Don't allow that if are requested to do strict
223             * checking. Check whether this strict checking was set up either in the
224             * server config we used for handshaking or in our current server.
225             * This should avoid insecure configuration by accident.
226             */ view japanese
227            ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, r, APLOGNO(02033)
228                         "No hostname was provided via SNI for a name based"
229                         " virtual host");
230            apr_table_setn(r->notes, "error-notes",
231                           "Reason: The client software did not provide a "
232                           "hostname using Server Name Indication (SNI), "
233                           "which is required to access this server.<br />\n");
234            return HTTP_FORBIDDEN;
235        }
236        /* Enforce SSL SNI vhost compatibility policy. */ view japanese
237        if (!ssl_check_vhost_sni_policy(sc, hssc)) {
238            ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02032)
239                         "Hostname %s %s and hostname %s provided"
240                         " via HTTP have no compatible SSL setup for policy '%s'",
241                         servername ? servername : handshakeserver->server_hostname,
242                         servername ? "provided via SNI" : "(default host as no SNI was provided)",
243                          r->hostname, MODSSL_SNIVH_NAME(sc->mc->snivh_policy));
244            return HTTP_MISDIRECTED_REQUEST;
245        }
246    }
247#endif
248    modssl_set_app_data2(ssl, r);
249
250    /*
251     * Log information about incoming HTTPS requests
252     */ view japanese
253    if (APLOGrinfo(r) && ap_is_initial_req(r)) {
254        ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(02034)
255                     "%s HTTPS request received for child %ld (server %s)",
256                     (r->connection->keepalives <= 0 ?
257                     "Initial (No.1)" :
258                     apr_psprintf(r->pool, "Subsequent (No.%d)",
259                                  r->connection->keepalives+1)),
260                     r->connection->id,
261                     ssl_util_vhostid(r->pool, r->server));
262    }
263
264    /* SetEnvIf ssl-*-shutdown flags can only be per-server,
265     * so they won't change across keepalive requests
266     */ view japanese
267    if (sslconn->shutdown_type == SSL_SHUTDOWN_TYPE_UNSET) {
268        ssl_configure_env(r, sslconn);
269    }
270
271    return DECLINED;
272}

274/*
275 * Move SetEnvIf information from request_rec to conn_rec/BUFF
276 * to allow the close connection handler to use them.
277 */ view japanese
278
279static void ssl_configure_env(request_rec *r, SSLConnRec *sslconn)
280{
281    int i;
282    const apr_array_header_t *arr = apr_table_elts(r->subprocess_env);
283    const apr_table_entry_t *elts = (const apr_table_entry_t *)arr->elts;
284
285    sslconn->shutdown_type = SSL_SHUTDOWN_TYPE_STANDARD;
286
287    for (i = 0; i < arr->nelts; i++) {
288        const char *key = elts[i].key;
289
290        switch (*key) {
291          case 's':
292            /* being case-sensitive here.
293             * and not checking for the -shutdown since these are the only
294             * SetEnvIf "flags" we support
295             */ view japanese
296            if (!strncmp(key+1, "sl-", 3)) {
297                key += 4;
298                if (!strncmp(key, "unclean", 7)) {
299                    sslconn->shutdown_type = SSL_SHUTDOWN_TYPE_UNCLEAN;
300                }
301                else if (!strncmp(key, "accurate", 8)) {
302                    sslconn->shutdown_type = SSL_SHUTDOWN_TYPE_ACCURATE;
303                }
304                return; /* should only ever be one ssl-*-shutdown */ view japanese
305            }
306            break;
307        }
308    }
309}

311static int ssl_check_post_client_verify(request_rec *r, SSLSrvConfigRec *sc, 
312                                        SSLDirConfigRec *dc, SSLConnRec *sslconn,
313                                        SSL *ssl)
314{
315    X509 *cert;
316    
317    /*
318     * Remember the peer certificate's DN
319     */ view japanese
320    if ((cert = SSL_get_peer_certificate(ssl))) {
321        if (sslconn->client_cert) {
322            X509_free(sslconn->client_cert);
323        }
324        sslconn->client_cert = cert;
325        sslconn->client_dn = NULL;
326    }
327    
328    /*
329     * Finally check for acceptable renegotiation results
330     */ view japanese
331    if ((dc->nVerifyClient != SSL_CVERIFY_NONE) ||
332        (sc->server->auth.verify_mode != SSL_CVERIFY_NONE)) {
333        BOOL do_verify = ((dc->nVerifyClient == SSL_CVERIFY_REQUIRE) ||
334                          (sc->server->auth.verify_mode == SSL_CVERIFY_REQUIRE));
335
336        if (do_verify && (SSL_get_verify_result(ssl) != X509_V_OK)) {
337            ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02262)
338                          "Re-negotiation handshake failed: "
339                          "Client verification failed");
340
341            return HTTP_FORBIDDEN;
342        }
343
344        if (do_verify) {
345            if (cert == NULL) {
346                ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02263)
347                              "Re-negotiation handshake failed: "
348                              "Client certificate missing");
349
350                return HTTP_FORBIDDEN;
351            }
352        }
353    }
354    return OK;
355}

357/*
358 *  Access Handler, classic flavour, for SSL/TLS up to v1.2 
359 *  where everything can be renegotiated and no one is happy.
360 */ view japanese
361static int ssl_hook_Access_classic(request_rec *r, SSLSrvConfigRec *sc, SSLDirConfigRec *dc,
362                                   SSLConnRec *sslconn, SSL *ssl)
363{
364    server_rec *handshakeserver = sslconn ? sslconn->server : NULL;
365    SSLSrvConfigRec *hssc       = handshakeserver? mySrvConfig(handshakeserver) : NULL;
366    SSL_CTX *ctx = ssl ? SSL_get_SSL_CTX(ssl) : NULL;
367    BOOL renegotiate = FALSE, renegotiate_quick = FALSE;
368    X509 *peercert;
369    X509_STORE *cert_store = NULL;
370    X509_STORE_CTX *cert_store_ctx;
371    STACK_OF(SSL_CIPHER) *cipher_list_old = NULL, *cipher_list = NULL;
372    const SSL_CIPHER *cipher = NULL;
373    int depth, verify_old, verify, n, rc;
374    const char *ncipher_suite;
375
376#ifdef HAVE_SRP
377    /*
378     * Support for per-directory reconfigured SSL connection parameters
379     *
380     * We do not force any renegotiation if the user is already authenticated
381     * via SRP.
382     *
383     */ view japanese
384    if (SSL_get_srp_username(ssl)) {
385        return DECLINED;
386    }
387#endif
388
389    /*
390     * Support for per-directory reconfigured SSL connection parameters.
391     *
392     * This is implemented by forcing an SSL renegotiation with the
393     * reconfigured parameter suite. But Apache's internal API processing
394     * makes our life very hard here, because when internal sub-requests occur
395     * we nevertheless should avoid multiple unnecessary SSL handshakes (they
396     * require extra network I/O and especially time to perform).
397     *
398     * But the optimization for filtering out the unnecessary handshakes isn't
399     * obvious and trivial.  Especially because while Apache is in its
400     * sub-request processing the client could force additional handshakes,
401     * too. And these take place perhaps without our notice. So the only
402     * possibility is to explicitly _ask_ OpenSSL whether the renegotiation
403     * has to be performed or not. It has to performed when some parameters
404     * which were previously known (by us) are not those we've now
405     * reconfigured (as known by OpenSSL) or (in optimized way) at least when
406     * the reconfigured parameter suite is stronger (more restrictions) than
407     * the currently active one.
408     */ view japanese
409
410    /*
411     * Override of SSLCipherSuite
412     *
413     * We provide two options here:
414     *
415     * o The paranoid and default approach where we force a renegotiation when
416     *   the cipher suite changed in _any_ way (which is straight-forward but
417     *   often forces renegotiations too often and is perhaps not what the
418     *   user actually wanted).
419     *
420     * o The optimized and still secure way where we force a renegotiation
421     *   only if the currently active cipher is no longer contained in the
422     *   reconfigured/new cipher suite. Any other changes are not important
423     *   because it's the servers choice to select a cipher from the ones the
424     *   client supports. So as long as the current cipher is still in the new
425     *   cipher suite we're happy. Because we can assume we would have
426     *   selected it again even when other (better) ciphers exists now in the
427     *   new cipher suite. This approach is fine because the user explicitly
428     *   has to enable this via ``SSLOptions +OptRenegotiate''. So we do no
429     *   implicit optimizations.
430     */      view japanese
431    ncipher_suite = (dc->szCipherSuite? 
432                     dc->szCipherSuite : (r->server != handshakeserver)?
433                     sc->server->auth.cipher_suite : NULL);
434    
435    if (ncipher_suite && (!sslconn->cipher_suite 
436                          || strcmp(ncipher_suite, sslconn->cipher_suite))) {
437        /* remember old state */
438
439        if (dc->nOptions & SSL_OPT_OPTRENEGOTIATE) {
440            cipher = SSL_get_current_cipher(ssl);
441        }
442        else {
443            cipher_list_old = (STACK_OF(SSL_CIPHER) *)SSL_get_ciphers(ssl);
444
445            if (cipher_list_old) {
446                cipher_list_old = sk_SSL_CIPHER_dup(cipher_list_old);
447            }
448        }
449
450        /* configure new state */
451        if (r->connection->master) {
452            /* TODO: this categorically fails changed cipher suite settings
453             * on slave connections. We could do better by
454             * - create a new SSL* from our SSL_CTX and set cipher suite there,
455             *   and retrieve ciphers, free afterwards
456             * Modifying the SSL on a slave connection is no good.
457             */ view japanese
458            apr_table_setn(r->notes, "ssl-renegotiate-forbidden", "cipher-suite");
459            return HTTP_FORBIDDEN;
460        }
461
462        if (!SSL_set_cipher_list(ssl, ncipher_suite)) {
463            ap_log_rerror(APLOG_MARK, APLOG_WARNING, 0, r, APLOGNO(02253)
464                          "Unable to reconfigure (per-directory) "
465                          "permitted SSL ciphers");
466            ssl_log_ssl_error(SSLLOG_MARK, APLOG_ERR, r->server);
467
468            if (cipher_list_old) {
469                sk_SSL_CIPHER_free(cipher_list_old);
470            }
471
472            return HTTP_FORBIDDEN;
473        }
474
475        /* determine whether a renegotiation has to be forced */ view japanese
476        cipher_list = (STACK_OF(SSL_CIPHER) *)SSL_get_ciphers(ssl);
477
478        if (dc->nOptions & SSL_OPT_OPTRENEGOTIATE) {
479            /* optimized way */ view japanese
480            if ((!cipher && cipher_list) ||
481                (cipher && !cipher_list))
482            {
483                renegotiate = TRUE;
484            }
485            else if (cipher && cipher_list &&
486                     (sk_SSL_CIPHER_find(cipher_list, cipher) < 0))
487            {
488                renegotiate = TRUE;
489            }
490        }
491        else {
492            /* paranoid way */ view japanese
493            if ((!cipher_list_old && cipher_list) ||
494                (cipher_list_old && !cipher_list))
495            {
496                renegotiate = TRUE;
497            }
498            else if (cipher_list_old && cipher_list) {
499                for (n = 0;
500                     !renegotiate && (n < sk_SSL_CIPHER_num(cipher_list));
501                     n++)
502                {
503                    const SSL_CIPHER *value = sk_SSL_CIPHER_value(cipher_list, n);
504
505                    if (sk_SSL_CIPHER_find(cipher_list_old, value) < 0) {
506                        renegotiate = TRUE;
507                    }
508                }
509
510                for (n = 0;
511                     !renegotiate && (n < sk_SSL_CIPHER_num(cipher_list_old));
512                     n++)
513                {
514                    const SSL_CIPHER *value = sk_SSL_CIPHER_value(cipher_list_old, n);
515
516                    if (sk_SSL_CIPHER_find(cipher_list, value) < 0) {
517                        renegotiate = TRUE;
518                    }
519                }
520            }
521        }
522
523        /* cleanup */ view japanese
524        if (cipher_list_old) {
525            sk_SSL_CIPHER_free(cipher_list_old);
526        }
527
528        if (renegotiate) {
529            if (r->connection->master) {
530                /* The request causes renegotiation on a slave connection.
531                 * This is not allowed since we might have concurrent requests
532                 * on this connection.
533                 */ view japanese
534                apr_table_setn(r->notes, "ssl-renegotiate-forbidden", "cipher-suite");
535                return HTTP_FORBIDDEN;
536            }
537            
538#ifdef SSL_OP_CIPHER_SERVER_PREFERENCE
539            if (sc->cipher_server_pref == TRUE) {
540                SSL_set_options(ssl, SSL_OP_CIPHER_SERVER_PREFERENCE);
541            }
542#endif
543            /* tracing */ view japanese
544            ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(02220)
545                         "Reconfigured cipher suite will force renegotiation");
546        }
547    }
548
549    /*
550     * override of SSLVerifyClient
551     *
552     * We force a renegotiation if the reconfigured/new verify type is
553     * stronger than the currently active verify type.
554     *
555     * The order is: none << optional_no_ca << optional << require
556     *
557     * Additionally the following optimization is possible here: When the
558     * currently active verify type is "none" but a client certificate is
559     * already known/present, it's enough to manually force a client
560     * verification but at least skip the I/O-intensive renegotiation
561     * handshake.
562     */ view japanese
563    if ((dc->nVerifyClient != SSL_CVERIFY_UNSET) ||
564        (sc->server->auth.verify_mode != SSL_CVERIFY_UNSET)) {
565
566        /* remember old state */ view japanese
567        verify_old = SSL_get_verify_mode(ssl);
568        /* configure new state */ view japanese
569        verify = SSL_VERIFY_NONE;
570
571        if ((dc->nVerifyClient == SSL_CVERIFY_REQUIRE) ||
572            (sc->server->auth.verify_mode == SSL_CVERIFY_REQUIRE)) {
573            verify |= SSL_VERIFY_PEER_STRICT;
574        }
575
576        if ((dc->nVerifyClient == SSL_CVERIFY_OPTIONAL) ||
577            (dc->nVerifyClient == SSL_CVERIFY_OPTIONAL_NO_CA) ||
578            (sc->server->auth.verify_mode == SSL_CVERIFY_OPTIONAL) ||
579            (sc->server->auth.verify_mode == SSL_CVERIFY_OPTIONAL_NO_CA))
580        {
581            verify |= SSL_VERIFY_PEER;
582        }
583
584        /* TODO: this seems premature since we do not know if there
585         *       are any changes required.
586         */ view japanese
587        SSL_set_verify(ssl, verify, ssl_callback_SSLVerify);
588        SSL_set_verify_result(ssl, X509_V_OK);
589
590        /* determine whether we've to force a renegotiation */ view japanese
591        if (!renegotiate && verify != verify_old) {
592            if (((verify_old == SSL_VERIFY_NONE) &&
593                 (verify     != SSL_VERIFY_NONE)) ||
594
595                (!(verify_old & SSL_VERIFY_PEER) &&
596                  (verify     & SSL_VERIFY_PEER)) ||
597
598                (!(verify_old & SSL_VERIFY_FAIL_IF_NO_PEER_CERT) &&
599                  (verify     & SSL_VERIFY_FAIL_IF_NO_PEER_CERT)))
600            {
601                renegotiate = TRUE;
602                if (r->connection->master) {
603                    /* The request causes renegotiation on a slave connection.
604                     * This is not allowed since we might have concurrent requests
605                     * on this connection.
606                     */ view japanese
607                    apr_table_setn(r->notes, "ssl-renegotiate-forbidden", "verify-client");
608                    SSL_set_verify(ssl, verify_old, ssl_callback_SSLVerify);
609                    return HTTP_FORBIDDEN;
610                }
611                /* optimization */ view japanese
612
613                if ((dc->nOptions & SSL_OPT_OPTRENEGOTIATE) &&
614                    (verify_old == SSL_VERIFY_NONE) &&
615                    ((peercert = SSL_get_peer_certificate(ssl)) != NULL))
616                {
617                    renegotiate_quick = TRUE;
618                    X509_free(peercert);
619                }
620
621                ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(02255)
622                              "Changed client verification type will force "
623                              "%srenegotiation",
624                              renegotiate_quick ? "quick " : "");
625            }
626            else if (verify != SSL_VERIFY_NONE) {
627                /*
628                 * override of SSLVerifyDepth
629                 *
630                 * The depth checks are handled by us manually inside the
631                 * verify callback function and not by OpenSSL internally
632                 * (and our function is aware of both the per-server and
633                 * per-directory contexts). So we cannot ask OpenSSL about
634                 * the currently verify depth. Instead we remember it in our
635                 * SSLConnRec attached to the SSL* of OpenSSL.  We've to force
636                 * the renegotiation if the reconfigured/new verify depth is
637                 * less than the currently active/remembered verify depth
638                 * (because this means more restriction on the certificate
639                 * chain).
640                 */
641                n = (sslconn->verify_depth != UNSET)
642                    ? sslconn->verify_depth
643                    : hssc->server->auth.verify_depth;
644                /* determine the new depth */
645                sslconn->verify_depth = (dc->nVerifyDepth != UNSET)
646                                        ? dc->nVerifyDepth
647                                        : sc->server->auth.verify_depth;
648                if (sslconn->verify_depth < n) {
649                    renegotiate = TRUE;
650                    ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(02254)
651                                  "Reduced client verification depth will "
652                                  "force renegotiation");
653                }
654            }
655        }
656        /* If we're handling a request for a vhost other than the default one,
657         * then we need to make sure that client authentication is properly
658         * enforced. For clients supplying an SNI extension, the peer
659         * certificate verification has happened in the handshake already
660         * (and r->server == handshakeserver). For non-SNI requests,
661         * an additional check is needed here. If client authentication
662         * is configured as mandatory, then we can only proceed if the
663         * CA list doesn't have to be changed (OpenSSL doesn't provide
664         * an option to change the list for an existing session).
665         */ view japanese
666        if ((r->server != handshakeserver)
667            && renegotiate
668            && ((verify & SSL_VERIFY_PEER) ||
669                (verify & SSL_VERIFY_FAIL_IF_NO_PEER_CERT))) {
670#define MODSSL_CFG_CA_NE(f, sc1, sc2) \
671            (sc1->server->auth.f && \
672             (!sc2->server->auth.f || \
673              strNE(sc1->server->auth.f, sc2->server->auth.f)))
674
675            if (MODSSL_CFG_CA_NE(ca_cert_file, sc, hssc) ||
676                MODSSL_CFG_CA_NE(ca_cert_path, sc, hssc)) {
677                if (verify & SSL_VERIFY_FAIL_IF_NO_PEER_CERT) {
678                    ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, r, APLOGNO(02256)
679                         "Non-default virtual host with SSLVerify set to "
680                         "'require' and VirtualHost-specific CA certificate "
681                         "list is only available to clients with TLS server "
682                         "name indication (SNI) support");
683                    SSL_set_verify(ssl, verify_old, NULL);
684                    return HTTP_FORBIDDEN;
685                } else
686                    /* let it pass, possibly with an "incorrect" peer cert,
687                     * so make sure the SSL_CLIENT_VERIFY environment variable
688                     * will indicate partial success only, later on.
689                     */ view japanese
690                    sslconn->verify_info = "GENEROUS";
691            }
692        }
693    }
694
695    /* Fill reneg buffer if required. */
696    if (renegotiate && !renegotiate_quick) {
697        rc = fill_reneg_buffer(r, dc);
698        if (rc) {
699            ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02257)
700                          "could not buffer message body to allow "
701                          "SSL renegotiation to proceed");
702            return rc;
703        }
704    }
705
706    /*
707     * now do the renegotiation if anything was actually reconfigured
708     */ view japanese
709    if (renegotiate) {
710        /*
711         * Now we force the SSL renegotiation by sending the Hello Request
712         * message to the client. Here we have to do a workaround: Actually
713         * OpenSSL returns immediately after sending the Hello Request (the
714         * intent AFAIK is because the SSL/TLS protocol says it's not a must
715         * that the client replies to a Hello Request). But because we insist
716         * on a reply (anything else is an error for us) we have to go to the
717         * ACCEPT state manually. Using SSL_set_accept_state() doesn't work
718         * here because it resets too much of the connection.  So we set the
719         * state explicitly and continue the handshake manually.
720         */ view japanese
721        ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, r, APLOGNO(02221)
722                      "Requesting connection re-negotiation");
723
724        if (renegotiate_quick) {
725            STACK_OF(X509) *cert_stack;
726            X509 *cert;
727
728            /* perform just a manual re-verification of the peer */ view japanese
729            ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(02258)
730                         "Performing quick renegotiation: "
731                         "just re-verifying the peer");
732
733            cert_stack = (STACK_OF(X509) *)SSL_get_peer_cert_chain(ssl);
734
735            cert = SSL_get_peer_certificate(ssl);
736
737            if (!cert_stack || (sk_X509_num(cert_stack) == 0)) {
738                if (!cert) {
739                    ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02222)
740                                  "Cannot find peer certificate chain");
741
742                    return HTTP_FORBIDDEN;
743                }
744
745                /* client cert is in the session cache, but there is
746                 * no chain, since ssl3_get_client_certificate()
747                 * sk_X509_shift-ed the peer cert out of the chain.
748                 * we put it back here for the purpose of quick_renegotiation.
749                 */ view japanese
750                cert_stack = sk_X509_new_null();
751                sk_X509_push(cert_stack, cert);
752            }
753
754            if (!(cert_store ||
755                  (cert_store = SSL_CTX_get_cert_store(ctx))))
756            {
757                ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02223)
758                              "Cannot find certificate storage");
759
760                return HTTP_FORBIDDEN;
761            }
762
763            if (!cert) {
764                cert = sk_X509_value(cert_stack, 0);
765            }
766
767            cert_store_ctx = X509_STORE_CTX_new();
768            X509_STORE_CTX_init(cert_store_ctx, cert_store, cert, cert_stack);
769            depth = SSL_get_verify_depth(ssl);
770
771            if (depth >= 0) {
772                X509_STORE_CTX_set_depth(cert_store_ctx, depth);
773            }
774
775            X509_STORE_CTX_set_ex_data(cert_store_ctx,
776                                       SSL_get_ex_data_X509_STORE_CTX_idx(),
777                                       (char *)ssl);
778
779            if (!X509_verify_cert(cert_store_ctx)) {
780                ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02224)
781                              "Re-negotiation verification step failed");
782                ssl_log_ssl_error(SSLLOG_MARK, APLOG_ERR, r->server);
783            }
784
785            SSL_set_verify_result(ssl, X509_STORE_CTX_get_error(cert_store_ctx));
786            X509_STORE_CTX_cleanup(cert_store_ctx);
787            X509_STORE_CTX_free(cert_store_ctx);
788
789            if (cert_stack != SSL_get_peer_cert_chain(ssl)) {
790                /* we created this ourselves, so free it */ view japanese
791                sk_X509_pop_free(cert_stack, X509_free);
792            }
793        }
794        else {
795            char peekbuf[1];
796            const char *reneg_support;
797            request_rec *id = r->main ? r->main : r;
798
799            /* Additional mitigation for CVE-2009-3555: At this point,
800             * before renegotiating, an (entire) request has been read
801             * from the connection.  An attacker may have sent further
802             * data to "prefix" any subsequent request by the victim's
803             * client after the renegotiation; this data may already
804             * have been read and buffered.  Forcing a connection
805             * closure after the response ensures such data will be
806             * discarded.  Legimately pipelined HTTP requests will be
807             * retried anyway with this approach. */ view japanese
808            if (has_buffered_data(r)) {
809                ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02259)
810                              "insecure SSL re-negotiation required, but "
811                              "a pipelined request is present; keepalive "
812                              "disabled");
813                r->connection->keepalive = AP_CONN_CLOSE;
814            }
815
816#if defined(SSL_get_secure_renegotiation_support)
817            reneg_support = SSL_get_secure_renegotiation_support(ssl) ?
818                            "client does" : "client does not";
819#else
820            reneg_support = "server does not";
821#endif
822            /* Perform a full renegotiation. */ view japanese
823            ap_log_rerror(APLOG_MARK, APLOG_DEBUG, 0, r, APLOGNO(02260)
824                          "Performing full renegotiation: complete handshake "
825                          "protocol (%s support secure renegotiation)",
826                          reneg_support);
827
828            SSL_set_session_id_context(ssl,
829                                       (unsigned char *)&id,
830                                       sizeof(id));
831
832            /* Toggle the renegotiation state to allow the new
833             * handshake to proceed. */ view japanese
834            modssl_set_reneg_state(sslconn, RENEG_ALLOW);
835
836            SSL_renegotiate(ssl);
837            SSL_do_handshake(ssl);
838
839            if (!SSL_is_init_finished(ssl)) {
840                ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02225)
841                              "Re-negotiation request failed");
842                ssl_log_ssl_error(SSLLOG_MARK, APLOG_ERR, r->server);
843
844                r->connection->keepalive = AP_CONN_CLOSE;
845                return HTTP_FORBIDDEN;
846            }
847
848            ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, r, APLOGNO(02226)
849                          "Awaiting re-negotiation handshake");
850
851            /* XXX: Should replace setting state with SSL_renegotiate(ssl);
852             * However, this causes failures in perl-framework currently,
853             * perhaps pre-test if we have already negotiated?
854             */ view japanese
855            /* Need to trigger renegotiation handshake by reading.
856             * Peeking 0 bytes actually works.
857             * See: http://marc.info/?t=145493359200002&r=1&w=2
858             */
859            SSL_peek(ssl, peekbuf, 0);
860
861            modssl_set_reneg_state(sslconn, RENEG_REJECT);
862
863            if (!SSL_is_init_finished(ssl)) {
864                ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02261)
865                              "Re-negotiation handshake failed");
866                ssl_log_ssl_error(SSLLOG_MARK, APLOG_ERR, r->server);
867
868                r->connection->keepalive = AP_CONN_CLOSE;
869                return HTTP_FORBIDDEN;
870            }
871
872            /* Full renegotiation successful, we now have handshaken with
873             * this server's parameters.
874             */ view japanese
875            sslconn->server = r->server;
876        }
877
878        /*
879         * Finally check for acceptable renegotiation results
880         */ view japanese
881        if (OK != (rc = ssl_check_post_client_verify(r, sc, dc, sslconn, ssl))) {
882            return rc;
883        }
884
885        /*
886         * Also check that SSLCipherSuite has been enforced as expected.
887         */ view japanese
888        if (cipher_list) {
889            cipher = SSL_get_current_cipher(ssl);
890            if (sk_SSL_CIPHER_find(cipher_list, cipher) < 0) {
891                ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02264)
892                             "SSL cipher suite not renegotiated: "
893                             "access to %s denied using cipher %s",
894                              r->filename,
895                              SSL_CIPHER_get_name(cipher));
896                return HTTP_FORBIDDEN;
897            }
898        }
899        /* remember any new cipher suite used in renegotiation */ view japanese
900        if (ncipher_suite) {
901            sslconn->cipher_suite = ncipher_suite;
902        }
903    }
904
905    return DECLINED;
906}

1011

1013                ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(10158)
1014                              "cannot perform post-handshake authentication");
1015                ssl_log_ssl_error(SSLLOG_MARK, APLOG_ERR, r->server);
1016                apr_table_setn(r->notes, "error-notes",
1017                               "Reason: Cannot perform Post-Handshake Authentication.<br />");
1018                SSL_set_verify(ssl, vmode_inplace, NULL);
1019                return HTTP_FORBIDDEN;
1020            }
1021            
1022            modssl_set_app_data2(ssl, r);
1023
1024            SSL_do_handshake(ssl);
1025            /* Need to trigger renegotiation handshake by reading.
1026             * Peeking 0 bytes actually works.
1027             * See: http://marc.info/?t=145493359200002&r=1&w=2
1028             */ view japanese
1029            SSL_peek(ssl, peekbuf, 0);
1030
1031            modssl_set_app_data2(ssl, NULL);
1032
1033            /*
1034             * Finally check for acceptable renegotiation results
1035             */ view japanese
1036            if (OK != (rc = ssl_check_post_client_verify(r, sc, dc, sslconn, ssl))) {
1037                SSL_set_verify(ssl, vmode_inplace, NULL);
1038                return rc;
1039            }
1040        }
1041    }

1047int ssl_hook_Access(request_rec *r)
1048{
1049    SSLDirConfigRec *dc         = myDirConfig(r);
1050    SSLSrvConfigRec *sc         = mySrvConfig(r->server);
1051    SSLConnRec *sslconn         = myConnConfig(r->connection);
1052    SSL *ssl                    = sslconn ? sslconn->ssl : NULL;
1053    apr_array_header_t *requires;
1054    ssl_require_t *ssl_requires;
1055    int ok, i, ret;
1056
1057    /* On a slave connection, we do not expect to have an SSLConnRec, but
1058     * our master connection might have one. */ view japanese
1059    if (!(sslconn && ssl) && r->connection->master) {
1060        sslconn         = myConnConfig(r->connection->master);
1061        ssl             = sslconn ? sslconn->ssl : NULL;
1062    }
1063
1064    /*
1065     * We should have handshaken here, otherwise we are being 
1066     * redirected (ErrorDocument) from a renegotiation failure below. 
1067     * The access is still forbidden in the latter case, let ap_die() handle
1068     * this recursive (same) error.
1069     */ view japanese
1070    if (ssl && !SSL_is_init_finished(ssl)) {
1071        return HTTP_FORBIDDEN;
1072    }
1073
1074    /*
1075     * Support for SSLRequireSSL directive
1076     */ view japanese
1077    if (dc->bSSLRequired && !ssl) {
1078        ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02219)
1079                      "access to %s failed, reason: %s",
1080                      r->filename, "SSL connection required");
1081
1082        /* remember forbidden access for strict require option */
1083        apr_table_setn(r->notes, "ssl-access-forbidden", "1");
1084
1085        return HTTP_FORBIDDEN;
1086    }
1087
1088    /*
1089     * Check to see whether SSL is in use; if it's not, then no
1090     * further access control checks are relevant.  (the test for
1091     * sc->enabled is probably strictly unnecessary)
1092     */ view japanese
1093    if (sc->enabled == SSL_ENABLED_FALSE || !ssl) {
1094        return DECLINED;
1095    }
1096
1097#if SSL_HAVE_PROTOCOL_TLSV1_3
1098    /* TLSv1.3+ is less complicated here. Branch off into a new codeline
1099     * and avoid messing with the past. */ view japanese
1100    if (SSL_version(ssl) >= TLS1_3_VERSION) {
1101        ret = ssl_hook_Access_modern(r, sc, dc, sslconn, ssl);
1102    }
1103    else
1104#endif
1105    {
1106        ret = ssl_hook_Access_classic(r, sc, dc, sslconn, ssl);
1107    }
1108
1109    if (ret != DECLINED) {
1110        return ret;
1111    }
1112
1113    /* If we're trying to have the user name set from a client
1114     * certificate then we need to set it here. This should be safe as
1115     * the user name probably isn't important from an auth checking point
1116     * of view as the certificate supplied acts in that capacity.
1117     * However, if FakeAuth is being used then this isn't the case so
1118     * we need to postpone setting the username until later.
1119     */ view japanese
1120    if ((dc->nOptions & SSL_OPT_FAKEBASICAUTH) == 0 && dc->szUserName) {
1121        char *val = ssl_var_lookup(r->pool, r->server, r->connection,
1122                                   r, (char *)dc->szUserName);
1123        if (val && val[0])
1124            r->user = val;
1125        else
1126            ap_log_rerror(APLOG_MARK, APLOG_WARNING, 0, r, APLOGNO(02227)
1127                          "Failed to set r->user to '%s'", dc->szUserName);
1128    }
1129
1130    /*
1131     * Check SSLRequire boolean expressions
1132     */ view japanese
1133    requires = dc->aRequirement;
1134    ssl_requires = (ssl_require_t *)requires->elts;
1135
1136    for (i = 0; i < requires->nelts; i++) {
1137        ssl_require_t *req = &ssl_requires[i];
1138        const char *errstring;
1139        ok = ap_expr_exec(r, req->mpExpr, &errstring);
1140
1141        if (ok < 0) {
1142            ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02265)
1143                          "access to %s failed, reason: Failed to execute "
1144                          "SSL requirement expression: %s",
1145                          r->filename, errstring);
1146
1147            /* remember forbidden access for strict require option */
1148            apr_table_setn(r->notes, "ssl-access-forbidden", "1");
1149
1150            return HTTP_FORBIDDEN;
1151        }
1152
1153        if (ok != 1) {
1154            ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, r, APLOGNO(02266)
1155                          "Access to %s denied for %s "
1156                          "(requirement expression not fulfilled)",
1157                          r->filename, r->useragent_ip);
1158
1159            ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, r, APLOGNO(02228)
1160                          "Failed expression: %s", req->cpExpr);
1161
1162            ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02229)
1163                          "access to %s failed, reason: %s",
1164                          r->filename,
1165                          "SSL requirement expression not fulfilled");
1166
1167            /* remember forbidden access for strict require option */ view japanese

1184/*
1185 *  Authentication Handler:
1186 *  Fake a Basic authentication from the X509 client certificate.
1187 *
1188 *  This must be run fairly early on to prevent a real authentication from
1189 *  occurring, in particular it must be run before anything else that
1190 *  authenticates a user.  This means that the Module statement for this
1191 *  module should be LAST in the Configuration file.
1192 */ view japanese
1193int ssl_hook_UserCheck(request_rec *r)
1194{
1195    SSLConnRec *sslconn;
1196    SSLDirConfigRec *dc = myDirConfig(r);
1197    char *clientdn;
1198    const char *auth_line, *username, *password;
1199
1200    /*
1201     * Additionally forbid access (again)
1202     * when strict require option is used.
1203     */
1204    if ((dc->nOptions & SSL_OPT_STRICTREQUIRE) &&
1205        (apr_table_get(r->notes, "ssl-access-forbidden")))
1206    {
1207        return HTTP_FORBIDDEN;
1208    }
1209
1210    /*
1211     * We decline when we are in a subrequest.  The Authorization header
1212     * would already be present if it was added in the main request.
1213     */ view japanese
1214    if (!ap_is_initial_req(r)) {
1215        return DECLINED;
1216    }
1217
1218    /*
1219     * Make sure the user is not able to fake the client certificate
1220     * based authentication by just entering an X.509 Subject DN
1221     * ("/XX=YYY/XX=YYY/..") as the username and "password" as the
1222     * password.
1223     */ view japanese
1224    if ((auth_line = apr_table_get(r->headers_in, "Authorization"))) {
1225        if (strcEQ(ap_getword(r->pool, &auth_line, ' '), "Basic")) {
1226            while ((*auth_line == ' ') || (*auth_line == '\t')) {
1227                auth_line++;
1228            }
1229
1230            auth_line = ap_pbase64decode(r->pool, auth_line);
1231            username = ap_getword_nulls(r->pool, &auth_line, ':');
1232            password = auth_line;
1233
1234            if ((username[0] == '/') && strEQ(password, "password")) {
1235                ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, APLOGNO(02035)
1236                    "Encountered FakeBasicAuth spoof: %s", username);
1237                return HTTP_FORBIDDEN;
1238            }
1239        }
1240    }
1241
1242    /*
1243     * We decline operation in various situations...
1244     * - TLS not enabled
1245     * - client did not present a certificate
1246     * - SSLOptions +FakeBasicAuth not configured
1247     * - r->user already authenticated
1248     */ view japanese
1249    if (!modssl_request_is_tls(r, &sslconn)
1250        || !sslconn->client_cert
1251        || !(dc->nOptions & SSL_OPT_FAKEBASICAUTH)
1252        || r->user) {
1253        return DECLINED;
1254    }
1255
1256    if (!sslconn->client_dn) {
1257        X509_NAME *name = X509_get_subject_name(sslconn->client_cert);
1258        char *cp = X509_NAME_oneline(name, NULL, 0);
1259        sslconn->client_dn = apr_pstrdup(r->connection->pool, cp);
1260        OPENSSL_free(cp);
1261    }
1262
1263    clientdn = (char *)sslconn->client_dn;
1264
1265    /*
1266     * Fake a password - which one would be immaterial, as, it seems, an empty
1267     * password in the users file would match ALL incoming passwords, if only
1268     * we were using the standard crypt library routine. Unfortunately, OpenSSL
1269     * "fixes" a "bug" in crypt and thus prevents blank passwords from
1270     * working.  (IMHO what they really fix is a bug in the users of the code
1271     * - failing to program correctly for shadow passwords).  We need,
1272     * therefore, to provide a password. This password can be matched by
1273     * adding the string "xxj31ZMTZzkVA" as the password in the user file.
1274     * This is just the crypted variant of the word "password" ;-)
1275     */ view japanese
1276    auth_line = apr_pstrcat(r->pool, "Basic ",
1277                            ap_pbase64encode(r->pool,
1278                                             apr_pstrcat(r->pool, clientdn,
1279                                                         ":password", NULL)),
1280                            NULL);
1281    apr_table_setn(r->headers_in, "Authorization", auth_line);
1282
1283    ap_log_rerror(APLOG_MARK, APLOG_INFO, 0, r, APLOGNO(02036)
1284                  "Faking HTTP Basic Auth header: \"Authorization: %s\"",
1285                  auth_line);
1286
1287    return DECLINED;
1288}

1562/*
1563 * This OpenSSL callback function is called when OpenSSL
1564 * does client authentication and verifies the certificate chain.
1565 */ view japanese
1566int ssl_callback_SSLVerify(int ok, X509_STORE_CTX *ctx)
1567{
1568    /* Get Apache context back through OpenSSL context */
1569    SSL *ssl = X509_STORE_CTX_get_ex_data(ctx,
1570                                          SSL_get_ex_data_X509_STORE_CTX_idx());
1571    conn_rec *conn      = (conn_rec *)SSL_get_app_data(ssl);
1572    request_rec *r      = (request_rec *)modssl_get_app_data2(ssl);
1573    server_rec *s       = r ? r->server : mySrvFromConn(conn);
1574
1575    SSLSrvConfigRec *sc = mySrvConfig(s);
1576    SSLConnRec *sslconn = myConnConfig(conn);
1577    SSLDirConfigRec *dc = r ? myDirConfig(r) : sslconn->dc;
1578    modssl_ctx_t *mctx  = myConnCtxConfig(conn, sc);
1579    int crl_check_mode  = mctx->crl_check_mask & ~SSL_CRLCHECK_FLAGS;
1580
1581    /* Get verify ingredients */ view japanese
1582    int errnum   = X509_STORE_CTX_get_error(ctx);
1583    int errdepth = X509_STORE_CTX_get_error_depth(ctx);
1584    int depth = UNSET;
1585    int verify = SSL_CVERIFY_UNSET;
1586
1587    /*
1588     * Log verification information
1589     */ view japanese
1590    ssl_log_cxerror(SSLLOG_MARK, APLOG_DEBUG, 0, conn,
1591                    X509_STORE_CTX_get_current_cert(ctx), APLOGNO(02275)
1592                    "Certificate Verification, depth %d, "
1593                    "CRL checking mode: %s (%x)", errdepth,
1594                    crl_check_mode == SSL_CRLCHECK_CHAIN ? "chain" :
1595                    crl_check_mode == SSL_CRLCHECK_LEAF  ? "leaf"  : "none",
1596                    mctx->crl_check_mask);
1597
1598    /*
1599     * Check for optionally acceptable non-verifiable issuer situation
1600     */ view japanese
1601    if (dc) {
1602        if (conn->outgoing) {
1603            verify = dc->proxy->auth.verify_mode;
1604        }
1605        else {
1606            verify = dc->nVerifyClient;
1607        }
1608    }
1609    if (!dc || (verify == SSL_CVERIFY_UNSET)) {
1610        verify = mctx->auth.verify_mode;
1611    }
1612
1613    if (verify == SSL_CVERIFY_NONE) {
1614        /*
1615         * SSLProxyVerify is either not configured or set to "none".
1616         * (this callback doesn't happen in the server context if SSLVerify
1617         *  is not configured or set to "none")
1618         */ view japanese
1619        return TRUE;
1620    }
1621
1622    if (ssl_verify_error_is_optional(errnum) &&
1623        (verify == SSL_CVERIFY_OPTIONAL_NO_CA))
1624    {
1625        ap_log_cerror(APLOG_MARK, APLOG_DEBUG, 0, conn, APLOGNO(02037)
1626                      "Certificate Verification: Verifiable Issuer is "
1627                      "configured as optional, therefore we're accepting "
1628                      "the certificate");
1629
1630        sslconn->verify_info = "GENEROUS";
1631        ok = TRUE;
1632    }
1633
1634    /*
1635     * Expired certificates vs. "expired" CRLs: by default, OpenSSL
1636     * turns X509_V_ERR_CRL_HAS_EXPIRED into a "certificate_expired(45)"
1637     * SSL alert, but that's not really the message we should convey to the
1638     * peer (at the very least, it's confusing, and in many cases, it's also
1639     * inaccurate, as the certificate itself may very well not have expired
1640     * yet). We set the X509_STORE_CTX error to something which OpenSSL's
1641     * s3_both.c:ssl_verify_alarm_type() maps to SSL_AD_CERTIFICATE_UNKNOWN,
1642     * i.e. the peer will receive a "certificate_unknown(46)" alert.
1643     * We do not touch errnum, though, so that later on we will still log
1644     * the "real" error, as returned by OpenSSL.
1645     */ view japanese
1646    if (!ok && errnum == X509_V_ERR_CRL_HAS_EXPIRED) {
1647        X509_STORE_CTX_set_error(ctx, -1);
1648    }
1649
1650    if (!ok && errnum == X509_V_ERR_UNABLE_TO_GET_CRL
1651            && (mctx->crl_check_mask & SSL_CRLCHECK_NO_CRL_FOR_CERT_OK)) {
1652        ap_log_cerror(APLOG_MARK, APLOG_TRACE3, 0, conn,
1653                      "Certificate Verification: Temporary error (%d): %s: "
1654                      "optional therefore we're accepting the certificate",
1655                      errnum, X509_verify_cert_error_string(errnum));
1656        X509_STORE_CTX_set_error(ctx, X509_V_OK);
1657        errnum = X509_V_OK;
1658        ok = TRUE;
1659    }
1660
1661#ifndef OPENSSL_NO_OCSP
1662    /*
1663     * Perform OCSP-based revocation checks
1664     */ view japanese
1665    if (ok && ((mctx->ocsp_mask & SSL_OCSPCHECK_CHAIN) ||
1666         (errdepth == 0 && (mctx->ocsp_mask & SSL_OCSPCHECK_LEAF)))) {     
1667        /* If there was an optional verification error, it's not
1668         * possible to perform OCSP validation since the issuer may be
1669         * missing/untrusted.  Fail in that case. */ view japanese
1670        if (ssl_verify_error_is_optional(errnum)) {
1671            X509_STORE_CTX_set_error(ctx, X509_V_ERR_APPLICATION_VERIFICATION);
1672            errnum = X509_V_ERR_APPLICATION_VERIFICATION;
1673            ap_log_cerror(APLOG_MARK, APLOG_ERR, 0, conn, APLOGNO(02038)
1674                          "cannot perform OCSP validation for cert "
1675                          "if issuer has not been verified "
1676                          "(optional_no_ca configured)");
1677            ok = FALSE;
1678        } else {
1679            ok = modssl_verify_ocsp(ctx, sc, s, conn, conn->pool);
1680            if (!ok) {
1681                errnum = X509_STORE_CTX_get_error(ctx);
1682            }
1683        }
1684    }
1685#endif
1686
1687    /*
1688     * If we already know it's not ok, log the real reason
1689     */ view japanese
1690    if (!ok) {
1691        if (APLOGcinfo(conn)) {
1692            ssl_log_cxerror(SSLLOG_MARK, APLOG_INFO, 0, conn,
1693                            X509_STORE_CTX_get_current_cert(ctx), APLOGNO(02276)
1694                            "Certificate Verification: Error (%d): %s",
1695                            errnum, X509_verify_cert_error_string(errnum));
1696        } else {
1697            ap_log_cerror(APLOG_MARK, APLOG_ERR, 0, conn, APLOGNO(02039)
1698                          "Certificate Verification: Error (%d): %s",
1699                          errnum, X509_verify_cert_error_string(errnum));
1700        }
1701
1702        if (sslconn->client_cert) {
1703            X509_free(sslconn->client_cert);
1704            sslconn->client_cert = NULL;
1705        }
1706        sslconn->client_dn = NULL;
1707        sslconn->verify_error = X509_verify_cert_error_string(errnum);
1708    }
1709
1710    /*
1711     * Finally check the depth of the certificate verification
1712     */ view japanese
1713    if (dc) {
1714        if (conn->outgoing) {
1715            depth = dc->proxy->auth.verify_depth;
1716        }
1717        else {
1718            depth = dc->nVerifyDepth;
1719        }
1720    }
1721    if (!dc || (depth == UNSET)) {
1722        depth = mctx->auth.verify_depth;
1723    }
1724
1725    if (errdepth > depth) {
1726        ap_log_cerror(APLOG_MARK, APLOG_ERR, 0, conn, APLOGNO(02040)
1727                      "Certificate Verification: Certificate Chain too long "
1728                      "(chain has %d certificates, but maximum allowed are "
1729                      "only %d)",
1730                      errdepth, depth);
1731
1732        errnum = X509_V_ERR_CERT_CHAIN_TOO_LONG;
1733        sslconn->verify_error = X509_verify_cert_error_string(errnum);
1734
1735        ok = FALSE;
1736    }
1737
1738    /*
1739     * And finally signal OpenSSL the (perhaps changed) state
1740     */ view japanese
1741    return ok;
1742}

1759 * being freed.
1760 */ view japanese

1763    *cert = info->x509; \
1764    CRYPTO_add(&(*cert)->references, +1, CRYPTO_LOCK_X509); \
1765    *pkey = info->x_pkey->dec_pkey; \
1766    CRYPTO_add(&(*pkey)->references, +1, CRYPTO_LOCK_EVP_PKEY)
1767#else
1768#define modssl_set_cert_info(info, cert, pkey) \
1769    *cert = info->x509; \

1775int ssl_callback_proxy_cert(SSL *ssl, X509 **x509, EVP_PKEY **pkey)
1776{
1777    conn_rec *c = (conn_rec *)SSL_get_app_data(ssl);
1778    server_rec *s = mySrvFromConn(c);
1779    SSLSrvConfigRec *sc = mySrvConfig(s);
1780    SSLDirConfigRec *dc = myDirConfigFromConn(c);
1781    X509_NAME *ca_name, *issuer, *ca_issuer;
1782    X509_INFO *info;
1783    X509 *ca_cert;
1784    STACK_OF(X509_NAME) *ca_list;
1785    STACK_OF(X509_INFO) *certs;
1786    STACK_OF(X509) *ca_certs;
1787    STACK_OF(X509) **ca_cert_chains;
1788    int i, j, k;
1789
1790    ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, s, APLOGNO(02267)
1791                 SSLPROXY_CERT_CB_LOG_FMT "entered",
1792                 sc->vhost_id);
1793
1794    certs = (dc && dc->proxy) ? dc->proxy->pkp->certs : NULL;
1795    if (!certs || (sk_X509_INFO_num(certs) <= 0)) {
1796        ap_log_error(APLOG_MARK, APLOG_WARNING, 0, s, APLOGNO(02268)
1797                     SSLPROXY_CERT_CB_LOG_FMT
1798                     "downstream server wanted client certificate "
1799                     "but none are configured", sc->vhost_id);
1800        return FALSE;
1801    }
1802
1803    ca_list = SSL_get_client_CA_list(ssl);
1804
1805    if (!ca_list || (sk_X509_NAME_num(ca_list) <= 0)) {
1806        /*
1807         * downstream server didn't send us a list of acceptable CA certs,
1808         * so we send the first client cert in the list.
1809         */ view japanese
1810        info = sk_X509_INFO_value(certs, 0);
1811
1812        modssl_proxy_info_log(c, info, APLOGNO(02278) "no acceptable CA list");
1813
1814        modssl_set_cert_info(info, x509, pkey);
1815
1816        return TRUE;
1817    }
1818
1819    ca_cert_chains = dc->proxy->pkp->ca_certs;
1820    for (i = 0; i < sk_X509_NAME_num(ca_list); i++) {
1821        ca_name = sk_X509_NAME_value(ca_list, i);
1822
1823        for (j = 0; j < sk_X509_INFO_num(certs); j++) {
1824            info = sk_X509_INFO_value(certs, j);
1825            issuer = X509_get_issuer_name(info->x509);
1826
1827            /* Search certs (by issuer name) one by one*/ view japanese
1828            if (X509_NAME_cmp(issuer, ca_name) == 0) {
1829                modssl_proxy_info_log(c, info, APLOGNO(02279)
1830                                      "found acceptable cert");
1831
1832                modssl_set_cert_info(info, x509, pkey);
1833
1834                return TRUE;
1835            }
1836
1837            if (ca_cert_chains) {
1838                /*
1839                 * Failed to find direct issuer - search intermediates
1840                 * (by issuer name), if provided.
1841                 */ view japanese
1842                ca_certs = ca_cert_chains[j];
1843                for (k = 0; k < sk_X509_num(ca_certs); k++) {
1844                    ca_cert = sk_X509_value(ca_certs, k);
1845                    ca_issuer = X509_get_issuer_name(ca_cert);
1846
1847                    if(X509_NAME_cmp(ca_issuer, ca_name) == 0 ) {
1848                        modssl_proxy_info_log(c, info, APLOGNO(02280)
1849                                              "found acceptable cert by intermediate CA");
1850
1851                        modssl_set_cert_info(info, x509, pkey);
1852
1853                        return TRUE;
1854                    }
1855                } /* end loop through chained certs */ view japanese
1856            }
1857        } /* end loop through available certs */ view japanese
1858    }
1859
1860    ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, s, APLOGNO(02269)
1861                 SSLPROXY_CERT_CB_LOG_FMT
1862                 "no client certificate found!?", sc->vhost_id);
1863
1864    return FALSE;
1865}

1896 *  This callback function is executed by OpenSSL whenever a new SSL_SESSION is
1897 *  added to the internal OpenSSL session cache. We use this hook to spread the
1898 *  SSL_SESSION also to the inter-process disk-cache to make share it with our
1899 *  other Apache pre-forked server processes.
1900 */ view japanese
1901int ssl_callback_NewSessionCacheEntry(SSL *ssl, SSL_SESSION *session)
1902{
1903    /* Get Apache context back through OpenSSL context */
1904    conn_rec *conn      = (conn_rec *)SSL_get_app_data(ssl);
1905    server_rec *s       = mySrvFromConn(conn);
1906    SSLSrvConfigRec *sc = mySrvConfig(s);
1907    long timeout        = sc->session_cache_timeout;
1908    BOOL rc;
1909    IDCONST unsigned char *id;
1910    unsigned int idlen;
1911
1912    /*
1913     * Set the timeout also for the internal OpenSSL cache, because this way
1914     * our inter-process cache is consulted only when it's really necessary.
1915     */ view japanese
1916    SSL_set_timeout(session, timeout);
1917
1918    /*
1919     * Store the SSL_SESSION in the inter-process cache with the
1920     * same expire time, so it expires automatically there, too.
1921     */ view japanese
1922#ifdef OPENSSL_NO_SSL_INTERN
1923    id = (unsigned char *)SSL_SESSION_get_id(session, &idlen);
1924#else
1925    id = session->session_id;
1926    idlen = session->session_id_length;
1927#endif

1929    rc = ssl_scache_store(s, id, idlen,
1930                          apr_time_from_sec(SSL_SESSION_get_time(session)
1931                                          + timeout),
1932                          session, conn->pool);

1935                    rc == TRUE ? "OK" : "BAD",
1936                    "caching", timeout);
1937
1938    /*
1939     * return 0 which means to OpenSSL that the session is still
1940     * valid and was not freed by us with SSL_SESSION_free().
1941     */ view japanese

2089/*
2090 * This callback function is executed while OpenSSL processes the SSL
2091 * handshake and does SSL record layer stuff.  It's used to trap
2092 * client-initiated renegotiations (where SSL_OP_NO_RENEGOTIATION is
2093 * not available), and for dumping everything to the log.
2094 */ view japanese
2095void ssl_callback_Info(const SSL *ssl, int where, int rc)
2096{
2097    conn_rec *c;
2098    server_rec *s;
2099
2100    /* Retrieve the conn_rec and the associated SSLConnRec. */ view japanese
2101    if ((c = (conn_rec *)SSL_get_app_data((SSL *)ssl)) == NULL) {
2102        return;
2103    }
2104
2105#ifndef SSL_OP_NO_RENEGOTIATION
2106    /* With OpenSSL < 1.1.1 (implying TLS v1.2 or earlier), this
2107     * callback is used to block client-initiated renegotiation.  With
2108     * TLSv1.3 it is unnecessary since renegotiation is forbidden at
2109     * protocol level.  Otherwise (TLSv1.2 with OpenSSL >=1.1.1),
2110     * SSL_OP_NO_RENEGOTIATION is used to block renegotiation. */ view japanese
2111    {

2113

2115            return;
2116        }
2117
2118        /* If the reneg state is to reject renegotiations, check the SSL
2119         * state machine and move to ABORT if a Client Hello is being
2120         * read. */ view japanese
2121        if (!c->outgoing &&
2122                (where & SSL_CB_HANDSHAKE_START) &&
2123                sslconn->reneg_state == RENEG_REJECT) {

2129         * subsequent client-initiated renegotiation. */ view japanese
2130        else if ((where & SSL_CB_HANDSHAKE_DONE)
2131                 && sslconn->reneg_state == RENEG_INIT) {
2132            sslconn->reneg_state = RENEG_REJECT;
2133        }
2134    }
2135#endif
2136
2137    s = mySrvFromConn(c);
2138    if (s && APLOGdebug(s)) {

2201 * client hello with a server name indication extension ("SNI", cf. RFC 6066).

2203static apr_status_t init_vhost(conn_rec *c, SSL *ssl, const char *servername)
2204{
2205    if (c) {
2206        SSLConnRec *sslcon = myConnConfig(c);
2207
2208        if (sslcon->vhost_found) {

2211        }
2212        sslcon->vhost_found = -1;
2213

2215            servername = SSL_get_servername(ssl, TLSEXT_NAMETYPE_host_name);
2216        }
2217        if (servername) {

2219                                            (void *)servername)) {
2220                ap_log_cerror(APLOG_MARK, APLOG_DEBUG, 0, c, APLOGNO(02043)
2221                              "SSL virtual host for servername %s found",

2224                sslcon->vhost_found = +1;
2225                return APR_SUCCESS;
2226            }
2227            else {
2228                ap_log_cerror(APLOG_MARK, APLOG_DEBUG, 0, c, APLOGNO(02044)
2229                              "No matching SSL virtual host for servername "
2230                              "%s found (using default/first virtual host)",
2231                              servername);
2232                /*
2233                 * RFC 6066 section 3 says "It is NOT RECOMMENDED to send
2234                 * a warning-level unrecognized_name(112) alert, because
2235                 * the client's behavior in response to warning-level alerts
2236                 * is unpredictable."
2237                 *
2238                 * To maintain backwards compatibility in mod_ssl, we

2241                 * "If the server understood the ClientHello extension but

2245                 * the handshake."
2246                 */ view japanese
2247            }
2248        }
2249        else {
2250            ap_log_cerror(APLOG_MARK, APLOG_DEBUG, 0, c, APLOGNO(02645)
2251                          "Server name not provided via TLS extension "
2252                          "(using default/first virtual host)");
2253        }
2254    }
2255    
2256    return APR_NOTFOUND;
2257}
