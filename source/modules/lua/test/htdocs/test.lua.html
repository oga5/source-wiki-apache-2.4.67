18function print_args(r, simple, complex)
19  local s = "    %s: %s\n"
20  r:puts("  simple:\n")
21  for k, v in pairs(simple) do
22    r:puts(s:format(k, v))
23  end
24
25  s = "    %s: "
26  r:puts("  complex:\n")
27  for k, ary in pairs(complex) do
28    r:puts(s:format(k))
29    for i=1, #ary do
30      r:puts(ary[i])
31      if i < #ary then r:puts(", ") end
32    end
33    r:puts("\n")
34  end
35end
