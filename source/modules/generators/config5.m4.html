14# Is mod_cgid needed?
15case $host in
16    *mingw*)
17        dnl No fork+thread+fd issues, and cgid doesn't work anyway.
18        cgid_needed="no"
19        ;;
20    *)
21        if ap_mpm_is_threaded; then
22            dnl if we are using a threaded MPM on Unix, we can get better
23            dnl performance with mod_cgid, and also avoid potential issues
24            dnl with forking from a threaded process.
25            cgid_needed="yes"
26        else
27            dnl if we are using a non-threaded MPM, it makes little sense to
28            dnl use mod_cgid, and it just opens up holes we don't need.
29            cgid_needed="no"
30        fi
31        ;;
32esac
33
34if test $cgid_needed = "yes"; then
35    APACHE_MODULE(cgid, CGI scripts.  Enabled by default with threaded MPMs, , , most, [
36    case $host in
37      *-solaris2*)
38        case `uname -r` in
39          5.10)
40          dnl Does the system have the appropriate patches?
41          case `uname -p` in
42            i386)
43              patch_id="120665"
44              ;;
45            sparc)
46              patch_id="120664"
47              ;;
48            *)
49              AC_MSG_WARN([Unknown platform])
50              patch_id="120664"
51              ;;
52          esac
53          AC_MSG_CHECKING([for Solaris patch $patch_id])
54          showrev -p | grep "$patch_id" >/dev/null 2>&1
55          if test $? -eq 1; then
56          dnl Solaris 11 (next release) as of snv_19 doesn't have this problem.
57          dnl It may be possible to use /kernel/drv/tl from later releases.
58          AC_MSG_ERROR([Please apply either patch # 120664 (Sparc) or # 120665 (x86).
59Without these patches, mod_cgid is non-functional on Solaris 10 due to an OS
60bug with AF_UNIX sockets.
61If you can not apply these patches, you can do one of the following:
62 - run configure with --disable-cgid
63 - switch to the prefork MPM
64For more info: <http://issues.apache.org/bugzilla/show_bug.cgi?id=34264>])
65          else
66            AC_MSG_RESULT(yes)
67          fi
68          ;;
69        esac
70        ;;
71    esac
72  ])
73    APACHE_MODULE(cgi, CGI scripts.  Enabled by default with non-threaded MPMs, , , no)
74else
75    APACHE_MODULE(cgi, CGI scripts.  Enabled by default with non-threaded MPMs, , , most)
76    APACHE_MODULE(cgid, CGI scripts.  Enabled by default with threaded MPMs, , , no)
