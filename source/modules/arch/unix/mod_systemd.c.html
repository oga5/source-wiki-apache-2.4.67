42static int systemd_pre_config(apr_pool_t *pconf, apr_pool_t *plog,
43                              apr_pool_t *ptemp)
44{
45    sd_notify(0,
46              "RELOADING=1\n"
47              "STATUS=Reading configuration...\n");
48    ap_extended_status = 1;
49    return OK;
50}

66/* Report the service is ready in post_config, which could be during
67 * startup or after a reload.  The server could still hit a fatal
68 * startup error after this point during ap_run_mpm(), so this is
69 * perhaps too early, but by post_config listen() has been called on
70 * the TCP ports so new connections will not be rejected.  There will
71 * always be a possible async failure event simultaneous to the
72 * service reporting "ready", so this should be good enough. */ view japanese
73static int systemd_post_config(apr_pool_t *pconf, apr_pool_t *plog,
74                               apr_pool_t *ptemp, server_rec *main_server)
75{
76    if (ap_state_query(AP_SQ_MAIN_STATE) == AP_SQ_MS_CREATE_PRE_CONFIG)
77        return OK;
78
79#ifdef HAVE_SELINUX
80    log_selinux_context();
81#endif
82
83    sd_notify(0, "READY=1\n"
84              "STATUS=Configuration loaded.\n");
85    return OK;
86}

88static int systemd_pre_mpm(apr_pool_t *p, ap_scoreboard_e sb_type)
89{
90    sd_notifyf(0, "READY=1\n"
91               "STATUS=Processing requests...\n"
92               "MAINPID=%" APR_PID_T_FMT, getpid());
93
94    return OK;
95}

97static int systemd_monitor(apr_pool_t *p, server_rec *s)
98{
99    ap_sload_t sload;
100    apr_interval_time_t up_time;
101    char bps[5];
102
103    if (!ap_extended_status) {
104        /* Nothing useful to report with ExtendedStatus disabled. */ view japanese
105        return DECLINED;
106    }
107    
108    ap_get_sload(&sload);
109    /* up_time in seconds */ view japanese
110    up_time = (apr_uint32_t) apr_time_sec(apr_time_now() -
111                               ap_scoreboard_image->global->restart_time);
112
113    apr_strfsize((unsigned long)((float) (sload.bytes_served)
114                                 / (float) up_time), bps);
115
116    sd_notifyf(0, "READY=1\n"
117               "STATUS=Total requests: %lu; Idle/Busy workers %d/%d;"
118               "Requests/sec: %.3g; Bytes served/sec: %sB/sec\n",
119               sload.access_count, sload.idle, sload.busy,
120               ((float) sload.access_count) / (float) up_time, bps);
121
122    return DECLINED;
123}

125static int ap_find_systemd_socket(process_rec * process, apr_port_t port) {
126    int fdcount, fd;
127    int sdc = sd_listen_fds(0);
128
129    if (sdc < 0) {
130        ap_log_perror(APLOG_MARK, APLOG_CRIT, sdc, process->pool, APLOGNO(02486)
131                      "find_systemd_socket: Error parsing enviroment, sd_listen_fds returned %d",
132                      sdc);
133        return -1;
134    }
135
136    if (sdc == 0) {
137        ap_log_perror(APLOG_MARK, APLOG_CRIT, sdc, process->pool, APLOGNO(02487)
138                      "find_systemd_socket: At least one socket must be set.");
139        return -1;
140    }
141
142    fdcount = atoi(getenv("LISTEN_FDS"));
143    for (fd = SD_LISTEN_FDS_START; fd < SD_LISTEN_FDS_START + fdcount; fd++) {
144        if (sd_is_socket_inet(fd, 0, 0, -1, port) > 0) {
145            return fd;
146        }
147    }
148
149    return -1;
150}

152static int ap_systemd_listen_fds(int unset_environment){
153    return sd_listen_fds(unset_environment);
154}

156static void systemd_register_hooks(apr_pool_t *p)
157{
158    APR_REGISTER_OPTIONAL_FN(ap_systemd_listen_fds);
159    APR_REGISTER_OPTIONAL_FN(ap_find_systemd_socket);
160
161    /* Enable ap_extended_status. */ view japanese
162    ap_hook_pre_config(systemd_pre_config, NULL, NULL, APR_HOOK_LAST);
163    /* Signal service is ready. */ view japanese
164    ap_hook_post_config(systemd_post_config, NULL, NULL, APR_HOOK_REALLY_LAST);
165    /* We know the PID in this hook ... */ view japanese
166    ap_hook_pre_mpm(systemd_pre_mpm, NULL, NULL, APR_HOOK_LAST);
167    /* Used to update httpd's status line using sd_notifyf */ view japanese
168    ap_hook_monitor(systemd_monitor, NULL, NULL, APR_HOOK_MIDDLE);
169}
