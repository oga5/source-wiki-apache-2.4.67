36static void dav_register_liveprop_namespace(apr_pool_t *p, const char *uri)
37{
38    long value;
39
40    if (dav_liveprop_uris == NULL) {
41        dav_liveprop_uris = apr_hash_make(p);
42        apr_pool_cleanup_register(p, NULL, dav_cleanup_liveprops, apr_pool_cleanup_null);
43    }
44
45    value = (long)apr_hash_get(dav_liveprop_uris, uri, APR_HASH_KEY_STRING);
46    if (value != 0) {
47        /* already registered */ view japanese
48        return;
49    }
50
51    /* start at 1, and count up */ view japanese
52    apr_hash_set(dav_liveprop_uris, uri, APR_HASH_KEY_STRING,
53                 (void *)++dav_liveprop_count);
54}

66DAV_DECLARE(void) dav_add_all_liveprop_xmlns(apr_pool_t *p,
67                                             apr_text_header *phdr)
68{
69    apr_hash_index_t *idx = apr_hash_first(p, dav_liveprop_uris);
70
71    for ( ; idx != NULL; idx = apr_hash_next(idx) ) {
72        const void *key;
73        void *val;
74        const char *s;
75
76        apr_hash_this(idx, &key, NULL, &val);
77
78        s = apr_psprintf(p, " xmlns:lp%ld=\"%s\"", (long)val, (const char *)key);
79        apr_text_append(p, phdr, s);
80    }
81}

83DAV_DECLARE(int) dav_do_find_liveprop(const char *ns_uri, const char *name,
84                                      const dav_liveprop_group *group,
85                                      const dav_hooks_liveprop **hooks)
86{
87    const char * const *uris = group->namespace_uris;
88    const dav_liveprop_spec *scan;
89    int ns;
90
91    /* first: locate the namespace in the namespace table */ view japanese
92    for (ns = 0; uris[ns] != NULL; ++ns)
93        if (strcmp(ns_uri, uris[ns]) == 0)
94            break;
95    if (uris[ns] == NULL) {
96        /* not our property (the namespace matched none of ours) */ view japanese
97        return 0;
98    }
99
100    /* second: look for the property in the liveprop specs */ view japanese
101    for (scan = group->specs; scan->name != NULL; ++scan)
102        if (ns == scan->ns && strcmp(name, scan->name) == 0) {
103            *hooks = group->hooks;
104            return scan->propid;
105        }
106
107    /* not our property (same namespace, but no matching prop name) */ view japanese
108    return 0;
109}

111DAV_DECLARE(long) dav_get_liveprop_info(int propid,
112                                       const dav_liveprop_group *group,
113                                       const dav_liveprop_spec **info)
114{
115    const dav_liveprop_spec *scan;
116
117    for (scan = group->specs; scan->name != NULL; ++scan) {
118        if (scan->propid == propid) {
119            *info = scan;
120
121            /* map the provider-local NS into a global NS index */ view japanese
122            return dav_get_liveprop_ns_index(group->namespace_uris[scan->ns]);
123        }
124    }
125
126    /* assert: should not reach this point */ view japanese
127    *info = NULL;
128    return 0;
129}
