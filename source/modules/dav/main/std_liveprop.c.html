59static dav_prop_insert dav_core_insert_prop(const dav_resource *resource,
60                                            int propid, dav_prop_insert what,
61                                            apr_text_header *phdr)
62{
63    const char *value = NULL;
64    const char *s;
65    apr_pool_t *p = resource->pool;
66    const dav_liveprop_spec *info;
67    long global_ns;
68
69    switch (propid)
70    {
71    case DAV_PROPID_resourcetype:
72        { /* additional type info provided by external modules ? */ view japanese
73            int i;
74
75            apr_array_header_t *extensions =
76                ap_list_provider_names(p, DAV_RESOURCE_TYPE_GROUP, "0");
77            ap_list_provider_names_t *entry =
78                (ap_list_provider_names_t *)extensions->elts;
79
80            for (i = 0; i < extensions->nelts; i++, entry++) {
81                const dav_resource_type_provider *res_hooks =
82                    dav_get_resource_type_providers(entry->provider_name);
83                const char *name = NULL, *uri = NULL;
84
85                if (!res_hooks || !res_hooks->get_resource_type)
86                    continue;
87
88                if (!res_hooks->get_resource_type(resource, &name, &uri) &&
89                    name) {
90
91                    if (!uri || !strcasecmp(uri, "DAV:"))
92                        value = apr_pstrcat(p, value ? value : "",
93                                            "<D:", name, "/>", NULL);
94                    else
95                        value = apr_pstrcat(p, value ? value : "",
96                                            "<x:", name,
97                                            " xmlns:x=\"", uri,
98                                            "\"/>", NULL);
99                }
100            }
101        }
102        switch (resource->type) {
103        case DAV_RESOURCE_TYPE_VERSION:
104            if (resource->baselined) {
105                value = apr_pstrcat(p, value ? value : "", "<D:baseline/>", NULL);
106                break;
107            }
108            /* fall through */ view japanese
109        case DAV_RESOURCE_TYPE_REGULAR:
110        case DAV_RESOURCE_TYPE_WORKING:
111            if (resource->collection) {
112                value = apr_pstrcat(p, value ? value : "", "<D:collection/>", NULL);
113            }
114            else {
115                /* ### should we denote lock-null resources? */ view japanese
116                if (value == NULL) {
117                    value = "";        /* becomes: <D:resourcetype/> */ view japanese
118                }
119            }
120            break;
121        case DAV_RESOURCE_TYPE_HISTORY:
122            value = apr_pstrcat(p, value ? value : "", "<D:version-history/>", NULL);
123            break;
124        case DAV_RESOURCE_TYPE_WORKSPACE:
125            value = apr_pstrcat(p, value ? value : "", "<D:collection/>", NULL);
126            break;
127        case DAV_RESOURCE_TYPE_ACTIVITY:
128            value = apr_pstrcat(p, value ? value : "", "<D:activity/>", NULL);
129            break;
130
131        default:
132            /* ### bad juju */ view japanese
133            return DAV_PROP_INSERT_NOTDEF;
134        }
135        break;
136
137    case DAV_PROPID_comment:
138    case DAV_PROPID_creator_displayname:
139    case DAV_PROPID_displayname:
140    case DAV_PROPID_source:
141    default:
142        /*
143        ** This property is known, but not defined as a liveprop. However,
144        ** it may be a dead property.
145        */ view japanese
146        return DAV_PROP_INSERT_NOTDEF;
147    }
148
149    /* assert: value != NULL */ view japanese
150
151    /* get the information and global NS index for the property */ view japanese
152    global_ns = dav_get_liveprop_info(propid, &dav_core_liveprop_group, &info);
153
154    /* assert: info != NULL && info->name != NULL */ view japanese
155
156    if (what == DAV_PROP_INSERT_SUPPORTED) {
157        s = apr_pstrcat(p,
158                         "<D:supported-live-property D:name=\"", info->name,
159                         "\" D:namespace=\"", dav_core_namespace_uris[info->ns],
160                         "\"/>" DEBUG_CR, NULL);
161    }
162    else if (what == DAV_PROP_INSERT_VALUE && *value != '\0') {
163        s = apr_psprintf(p, "<lp%ld:%s>%s</lp%ld:%s>" DEBUG_CR,
164                         global_ns, info->name, value, global_ns, info->name);
165    }
166    else {
167        s = apr_psprintf(p, "<lp%ld:%s/>" DEBUG_CR, global_ns, info->name);
168    }
169    apr_text_append(p, phdr, s);
170
171    /* we inserted what was asked for */ view japanese
172    return what;
173}
