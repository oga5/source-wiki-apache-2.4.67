238dav_error * dav_fs_dir_file_name(
239    const dav_resource *resource,
240    const char **dirpath_p,
241    const char **fname_p)
242{
243    dav_resource_private *ctx = resource->info;
244
245    if (resource->collection) {
246        *dirpath_p = ctx->pathname;
247        if (fname_p != NULL)
248            *fname_p = NULL;
249    }
250    else {
251        const char *testpath, *rootpath;
252        char *dirpath = ap_make_dirstr_parent(ctx->pool, ctx->pathname);
253        apr_size_t dirlen = strlen(dirpath);
254        apr_status_t rv = APR_SUCCESS;
255
256        testpath = dirpath;
257        if (dirlen > 0) {
258            rv = apr_filepath_root(&rootpath, &testpath, 0, ctx->pool);
259        }
260
261        /* remove trailing slash from dirpath, unless it's a root path
262         */ view japanese
263        if ((rv == APR_SUCCESS && testpath && *testpath)
264            || rv == APR_ERELATIVE) {
265            if (dirpath[dirlen - 1] == '/') {
266                dirpath[dirlen - 1] = '\0';
267            }
268        }
269
270        /* ###: Looks like a response could be appropriate
271         *
272         * APR_SUCCESS     here tells us the dir is a root
273         * APR_ERELATIVE   told us we had no root (ok)
274         * APR_EINCOMPLETE an incomplete testpath told us
275         *                 there was no -file- name here!
276         * APR_EBADPATH    or other errors tell us this file
277         *                 path is undecipherable
278         */ view japanese
279
280        if (rv == APR_SUCCESS || rv == APR_ERELATIVE) {
281            *dirpath_p = dirpath;
282            if (fname_p != NULL)
283                *fname_p = ctx->pathname + dirlen;
284        }
285        else {
286            return dav_new_error(ctx->pool, HTTP_INTERNAL_SERVER_ERROR, 0, rv,
287                                 "An incomplete/bad path was found in "
288                                 "dav_fs_dir_file_name.");
289        }
290    }
291
292    return NULL;
293}

297static void dav_format_time(int style, apr_time_t sec, char *buf, apr_size_t buflen)
298{
299    apr_time_exp_t tms;
300
301    /* ### what to do if fails? */ view japanese
302    (void) apr_time_exp_gmt(&tms, sec);
303
304    if (style == DAV_STYLE_ISO8601) {
305        /* ### should we use "-00:00" instead of "Z" ?? */ view japanese
306
307        /* 20 chars plus null term */ view japanese
308        apr_snprintf(buf, buflen, "%.4d-%.2d-%.2dT%.2d:%.2d:%.2dZ",
309                     tms.tm_year + 1900, tms.tm_mon + 1, tms.tm_mday,
310                     tms.tm_hour, tms.tm_min, tms.tm_sec);
311        return;
312    }
313
314    /* RFC 822 date format; as strftime '%a, %d %b %Y %T GMT' */ view japanese
315
316    /* 29 chars plus null term */ view japanese
317    apr_snprintf(buf, buflen, "%s, %.2d %s %d %.2d:%.2d:%.2d GMT",
318                 apr_day_snames[tms.tm_wday],
319                 tms.tm_mday, apr_month_snames[tms.tm_mon],
320                 tms.tm_year + 1900,
321                 tms.tm_hour, tms.tm_min, tms.tm_sec);
322}

324/* Copy or move src to dst; src_finfo is used to propagate permissions
325 * bits across if non-NULL; dst_finfo must be non-NULL iff dst already
326 * exists. */ view japanese
327static dav_error * dav_fs_copymove_file(
328    int is_move,
329    apr_pool_t * p,
330    const char *src,
331    const char *dst,
332    const apr_finfo_t *src_finfo,
333    const apr_finfo_t *dst_finfo,
334    dav_buffer *pbuf)
335{
336    dav_buffer work_buf = { 0 };
337    apr_file_t *inf = NULL;
338    apr_file_t *outf = NULL;
339    apr_status_t status;
340    apr_fileperms_t perms;
341
342    if (pbuf == NULL)
343        pbuf = &work_buf;
344
345    /* Determine permissions to use for destination */ view japanese
346    if (src_finfo && src_finfo->valid & APR_FINFO_PROT
347        && src_finfo->protection & APR_UEXECUTE) {
348        perms = src_finfo->protection;
349
350        if (dst_finfo != NULL) {
351            /* chmod it if it already exist */ view japanese
352            if ((status = apr_file_perms_set(dst, perms)) != APR_SUCCESS) {
353                return dav_new_error(p, HTTP_INTERNAL_SERVER_ERROR, 0, status,
354                                     "Could not set permissions on destination");
355            }
356        }
357    }
358    else {
359        perms = APR_OS_DEFAULT;
360    }
361
362    dav_set_bufsize(p, pbuf, DAV_FS_COPY_BLOCKSIZE);
363
364    if ((status = apr_file_open(&inf, src, APR_READ | APR_BINARY,
365                                APR_OS_DEFAULT, p)) != APR_SUCCESS) {
366        /* ### use something besides 500? */
367        return dav_new_error(p, HTTP_INTERNAL_SERVER_ERROR, 0, status,
368                             "Could not open file for reading");
369    }
370
371    /* ### do we need to deal with the umask? */
372    status = apr_file_open(&outf, dst, APR_WRITE | APR_CREATE | APR_TRUNCATE
373                           | APR_BINARY, perms, p);
374    if (status != APR_SUCCESS) {
375        apr_file_close(inf);
376
377        return dav_new_error(p, MAP_IO2HTTP(status), 0, status,
378                             "Could not open file for writing");
379    }
380
381    while (1) {
382        apr_size_t len = DAV_FS_COPY_BLOCKSIZE;
383
384        status = apr_file_read(inf, pbuf->buf, &len);
385        if (status != APR_SUCCESS && status != APR_EOF) {
386            apr_status_t lcl_status;
387
388            apr_file_close(inf);
389            apr_file_close(outf);
390
391            if ((lcl_status = apr_file_remove(dst, p)) != APR_SUCCESS) {
392                /* ### ACK! Inconsistent state... */
393
394                /* ### use something besides 500? */
395                return dav_new_error(p, HTTP_INTERNAL_SERVER_ERROR, 0,
396                                     lcl_status,
397                                     "Could not delete output after read "
398                                     "failure. Server is now in an "
399                                     "inconsistent state.");
400            }
401
402            /* ### use something besides 500? */
403            return dav_new_error(p, HTTP_INTERNAL_SERVER_ERROR, 0, status,
404                                 "Could not read input file");
405        }
406
407        if (status == APR_EOF)
408            break;
409
410        /* write any bytes that were read */ view japanese
411        status = apr_file_write_full(outf, pbuf->buf, len, NULL);
412        if (status != APR_SUCCESS) {
413            apr_status_t lcl_status;
414
415            apr_file_close(inf);
416            apr_file_close(outf);
417
418            if ((lcl_status = apr_file_remove(dst, p)) != APR_SUCCESS) {
419                /* ### ACK! Inconsistent state... */ view japanese
420
421                /* ### use something besides 500? */
422                return dav_new_error(p, HTTP_INTERNAL_SERVER_ERROR, 0,
423                                     lcl_status,
424                                     "Could not delete output after write "
425                                     "failure. Server is now in an "
426                                     "inconsistent state.");
427            }
428
429            return dav_new_error(p, MAP_IO2HTTP(status), 0, status,
430                                 "Could not write output file");
431        }
432    }
433
434    apr_file_close(inf);
435    apr_file_close(outf);
436
437    if (is_move && (status = apr_file_remove(src, p)) != APR_SUCCESS) {
438        dav_error *err;
439        apr_status_t lcl_status;
440
441        if (APR_STATUS_IS_ENOENT(status)) {
442            /*
443             * Something is wrong here but the result is what we wanted.
444             * We definitely should not remove the destination file.
445             */ view japanese
446            err = dav_new_error(p, HTTP_INTERNAL_SERVER_ERROR, 0, status,
447                                 apr_psprintf(p, "Could not remove source "
448                                              "file %s after move to %s. The "
449                                              "server may be in an "
450                                              "inconsistent state.", src, dst));
451            return err;
452        }
453        else if ((lcl_status = apr_file_remove(dst, p)) != APR_SUCCESS) {
454            /* ### ACK. this creates an inconsistency. do more!? */ view japanese
455
456            /* ### use something besides 500? */
457            return dav_new_error(p, HTTP_INTERNAL_SERVER_ERROR, 0, lcl_status,
458                                 "Could not remove source or destination "
459                                 "file. Server is now in an inconsistent "
460                                 "state.");
461        }
462
463        /* ### use something besides 500? */
464        err = dav_new_error(p, HTTP_INTERNAL_SERVER_ERROR, 0, status,
465                            "Could not remove source file after move. "
466                            "Destination was removed to ensure consistency.");
467        return err;
468    }
469
470    return NULL;
471}

473/* copy/move a file from within a state dir to another state dir */ view japanese
474/* ### need more buffers to replace the pool argument */ view japanese
475static dav_error * dav_fs_copymove_state(
476    int is_move,
477    apr_pool_t * p,
478    const char *src_dir, const char *src_file,
479    const char *dst_dir, const char *dst_file,
480    dav_buffer *pbuf)
481{
482    apr_finfo_t src_finfo;        /* finfo for source file */ view japanese
483    apr_finfo_t dst_state_finfo;        /* finfo for STATE directory */ view japanese
484    apr_status_t rv;
485    const char *src;
486    const char *dst;
487
488    /* build the propset pathname for the source file */ view japanese
489    src = apr_pstrcat(p, src_dir, "/" DAV_FS_STATE_DIR "/", src_file, NULL);
490
491    /* the source file doesn't exist */ view japanese
492    rv = apr_stat(&src_finfo, src, APR_FINFO_NORM, p);
493    if (rv != APR_SUCCESS && rv != APR_INCOMPLETE) {
494        return NULL;
495    }
496
497    /* build the pathname for the destination state dir */ view japanese
498    dst = apr_pstrcat(p, dst_dir, "/" DAV_FS_STATE_DIR, NULL);
499
500    /* ### do we need to deal with the umask? */ view japanese
501
502    /* ensure that it exists */ view japanese
503    rv = apr_dir_make(dst, APR_OS_DEFAULT, p);
504    if (rv != APR_SUCCESS) {
505        if (!APR_STATUS_IS_EEXIST(rv)) {
506            /* ### use something besides 500? */
507            return dav_new_error(p, HTTP_INTERNAL_SERVER_ERROR, 0, rv,
508                                 "Could not create internal state directory");
509        }
510    }
511
512    /* get info about the state directory */ view japanese
513    rv = apr_stat(&dst_state_finfo, dst, APR_FINFO_NORM, p);
514    if (rv != APR_SUCCESS && rv != APR_INCOMPLETE) {
515        /* Ack! Where'd it go? */ view japanese
516        /* ### use something besides 500? */
517        return dav_new_error(p, HTTP_INTERNAL_SERVER_ERROR, 0, rv,
518                             "State directory disappeared");
519    }
520
521    /* The mkdir() may have failed because a *file* exists there already */ view japanese
522    if (dst_state_finfo.filetype != APR_DIR) {
523        /* ### try to recover by deleting this file? (and mkdir again) */ view japanese
524        /* ### use something besides 500? */
525        return dav_new_error(p, HTTP_INTERNAL_SERVER_ERROR, 0, 0,
526                             "State directory is actually a file");
527    }
528
529    /* append the target file to the state directory pathname */ view japanese
530    dst = apr_pstrcat(p, dst, "/", dst_file, NULL);
531
532    /* copy/move the file now */ view japanese
533    if (is_move) {
534        /* try simple rename first */ view japanese
535        rv = apr_file_rename(src, dst, p);
536        if (APR_STATUS_IS_EXDEV(rv)) {
537            return dav_fs_copymove_file(is_move, p, src, dst, NULL, NULL, pbuf);
538        }
539        if (rv != APR_SUCCESS) {
540            /* ### use something besides 500? */
541            return dav_new_error(p, HTTP_INTERNAL_SERVER_ERROR, 0, rv,
542                                 "Could not move state file.");
543        }
544    }
545    else
546    {
547        /* gotta copy (and delete) */ view japanese
548        return dav_fs_copymove_file(is_move, p, src, dst, NULL, NULL, pbuf);
549    }
550
551    return NULL;
552}

554static dav_error *dav_fs_copymoveset(int is_move, apr_pool_t *p,
555                                     const dav_resource *src,
556                                     const dav_resource *dst,
557                                     dav_buffer *pbuf)
558{
559    const char *src_dir;
560    const char *src_file;
561    const char *src_state1;
562    const char *src_state2;
563    const char *dst_dir;
564    const char *dst_file;
565    const char *dst_state1;
566    const char *dst_state2;
567    dav_error *err;
568
569    /* Get directory and filename for resources */ view japanese
570    /* ### should test these result values... */ view japanese
571    (void) dav_fs_dir_file_name(src, &src_dir, &src_file);
572    (void) dav_fs_dir_file_name(dst, &dst_dir, &dst_file);
573
574    /* Get the corresponding state files for each resource */ view japanese
575    dav_dbm_get_statefiles(p, src_file, &src_state1, &src_state2);
576    dav_dbm_get_statefiles(p, dst_file, &dst_state1, &dst_state2);
577#if DAV_DEBUG
578    if ((src_state2 != NULL && dst_state2 == NULL) ||
579        (src_state2 == NULL && dst_state2 != NULL)) {
580        return dav_new_error(p, HTTP_INTERNAL_SERVER_ERROR, 0, 0,
581                             "DESIGN ERROR: dav_dbm_get_statefiles() "
582                             "returned inconsistent results.");
583    }
584#endif
585
586    err = dav_fs_copymove_state(is_move, p,
587                                src_dir, src_state1,
588                                dst_dir, dst_state1,
589                                pbuf);
590
591    if (err == NULL && src_state2 != NULL) {
592        err = dav_fs_copymove_state(is_move, p,
593                                    src_dir, src_state2,
594                                    dst_dir, dst_state2,
595                                    pbuf);
596
597        if (err != NULL) {
598            /* ### CRAP. inconsistency. */ view japanese
599            /* ### should perform some cleanup at the target if we still
600               ### have the original files */ view japanese
601
602            /* Change the error to reflect the bad server state. */ view japanese
603            err->status = HTTP_INTERNAL_SERVER_ERROR;
604            err->desc =
605                "Could not fully copy/move the properties. "
606                "The server is now in an inconsistent state.";
607        }
608    }
609
610    return err;
611}

613static dav_error *dav_fs_deleteset(apr_pool_t *p, const dav_resource *resource)
614{
615    const char *dirpath;
616    const char *fname;
617    const char *state1;
618    const char *state2;
619    const char *pathname;
620    apr_status_t status;
621
622    /* Get directory, filename, and state-file names for the resource */ view japanese
623    /* ### should test this result value... */ view japanese
624    (void) dav_fs_dir_file_name(resource, &dirpath, &fname);
625    dav_dbm_get_statefiles(p, fname, &state1, &state2);
626
627    /* build the propset pathname for the file */
628    pathname = apr_pstrcat(p,
629                          dirpath,
630                          "/" DAV_FS_STATE_DIR "/",
631                          state1,
632                          NULL);
633
634    /* note: we may get ENOENT if the state dir is not present */ view japanese
635    if ((status = apr_file_remove(pathname, p)) != APR_SUCCESS
636        && !APR_STATUS_IS_ENOENT(status)) {
637        return dav_new_error(p, HTTP_INTERNAL_SERVER_ERROR, 0, status,
638                             "Could not remove properties.");
639    }
640

667static dav_error * dav_fs_get_resource(
668    request_rec *r,
669    const char *root_dir,
670    const char *label,
671    int use_checked_in,
672    dav_resource **result_resource)
673{
674    dav_resource_private *ctx;
675    dav_resource *resource;
676    char *s;
677    char *filename;
678    apr_size_t len;
679
680    /* ### optimize this into a single allocation! */
681
682    /* Create private resource context descriptor */
683    ctx = apr_pcalloc(r->pool, sizeof(*ctx));
684    ctx->finfo = r->finfo;
685    ctx->r = r;
686
687    /* ### this should go away */
688    ctx->pool = r->pool;
689
690    /* Preserve case on OSes which fold canonical filenames */ view japanese
691#if 0
692    /* ### not available in Apache 2.0 yet */ view japanese
693    filename = r->case_preserved_filename;
694#else
695    filename = r->filename;
696#endif
697
698    /*
699    ** If there is anything in the path_info, then this indicates that the
700    ** entire path was not used to specify the file/dir. We want to append
701    ** it onto the filename so that we get a "valid" pathname for null
702    ** resources.
703    */ view japanese
704    s = apr_pstrcat(r->pool, filename, r->path_info, NULL);
705
706    /* make sure the pathname does not have a trailing "/" */ view japanese
707    len = strlen(s);
708    if (len > 1 && s[len - 1] == '/') {
709        s[len - 1] = '\0';
710    }
711    ctx->pathname = s;
712
713    /* Create resource descriptor */ view japanese
714    resource = apr_pcalloc(r->pool, sizeof(*resource));
715    resource->type = DAV_RESOURCE_TYPE_REGULAR;
716    resource->info = ctx;
717    resource->hooks = &dav_hooks_repository_fs;
718    resource->pool = r->pool;
719
720    /* make sure the URI does not have a trailing "/" */ view japanese
721    len = strlen(r->uri);
722    if (len > 1 && r->uri[len - 1] == '/') {
723        s = apr_pstrmemdup(r->pool, r->uri, len-1);
724        resource->uri = s;
725    }
726    else {
727        resource->uri = r->uri;
728    }
729
730    if (r->finfo.filetype != APR_NOFILE) {
731        resource->exists = 1;
732        resource->collection = r->finfo.filetype == APR_DIR;
733
734        /* unused info in the URL will indicate a null resource */ view japanese
735
736        if (r->path_info != NULL && *r->path_info != '\0') {
737            if (resource->collection) {
738                /* only a trailing "/" is allowed */ view japanese
739                if (*r->path_info != '/' || r->path_info[1] != '\0') {
740
741                    /*
742                    ** This URL/filename represents a locknull resource or
743                    ** possibly a destination of a MOVE/COPY
744                    */ view japanese
745                    resource->exists = 0;
746                    resource->collection = 0;
747                }
748            }
749            else
750            {
751                /*
752                ** The base of the path refers to a file -- nothing should
753                ** be in path_info. The resource is simply an error: it
754                ** can't be a null or a locknull resource.
755                */ view japanese
756                return dav_new_error(r->pool, HTTP_BAD_REQUEST, 0, 0,
757                                     "The URL contains extraneous path "
758                                     "components. The resource could not "
759                                     "be identified.");
760            }
761
762            /* retain proper integrity across the structures */ view japanese
763            if (!resource->exists) {
764                ctx->finfo.filetype = APR_NOFILE;
765            }
766        }
767    }
768
769    *result_resource = resource;
770    return NULL;
771}

773static dav_error * dav_fs_get_parent_resource(const dav_resource *resource,
774                                              dav_resource **result_parent)
775{
776    dav_resource_private *ctx = resource->info;
777    dav_resource_private *parent_ctx;
778    dav_resource *parent_resource;
779    apr_status_t rv;
780    char *dirpath;
781    const char *testroot;
782    const char *testpath;
783
784    /* If we're at the root of the URL space, then there is no parent. */ view japanese
785    if (strcmp(resource->uri, "/") == 0) {
786        *result_parent = NULL;
787        return NULL;
788    }
789
790    /* If given resource is root, then there is no parent.
791     * Unless we can retrieve the filepath root, this is
792     * intendend to fail.  If we split the root and
793     * no path info remains, then we also fail.
794     */ view japanese
795    testpath = ctx->pathname;
796    rv = apr_filepath_root(&testroot, &testpath, 0, ctx->pool);
797    if ((rv != APR_SUCCESS && rv != APR_ERELATIVE)
798        || !testpath || !*testpath) {
799        *result_parent = NULL;
800        return NULL;
801    }
802
803    /* ### optimize this into a single allocation! */ view japanese
804
805    /* Create private resource context descriptor */ view japanese
806    parent_ctx = apr_pcalloc(ctx->pool, sizeof(*parent_ctx));
807
808    /* ### this should go away */ view japanese
809    parent_ctx->pool = ctx->pool;
810
811    dirpath = ap_make_dirstr_parent(ctx->pool, ctx->pathname);
812    if (strlen(dirpath) > 1 && dirpath[strlen(dirpath) - 1] == '/')
813        dirpath[strlen(dirpath) - 1] = '\0';
814    parent_ctx->pathname = dirpath;
815
816    parent_resource = apr_pcalloc(ctx->pool, sizeof(*parent_resource));
817    parent_resource->info = parent_ctx;
818    parent_resource->collection = 1;
819    parent_resource->hooks = &dav_hooks_repository_fs;
820    parent_resource->pool = resource->pool;
821
822    if (resource->uri != NULL) {

825            uri[strlen(uri) - 1] = '\0';
826        parent_resource->uri = uri;
827    }
828
829    rv = apr_stat(&parent_ctx->finfo, parent_ctx->pathname,
830                  APR_FINFO_NORM, ctx->pool);
831    if (rv == APR_SUCCESS || rv == APR_INCOMPLETE) {
832        parent_resource->exists = 1;
833    }
834
835    *result_parent = parent_resource;
836    return NULL;

1191                                   srcinfo->pathname, dstinfo->pathname,

1389                              err);

1449
1450    *response = NULL;

1454     */ view japanese

1571                !strncmp(dirent.name, DAV_FS_TMP_PREFIX,

1574            }
1575        }
1576        /* skip the state dir and temp files unless a HIDDEN is performed */ view japanese
1577        if (!(params->walk_type & DAV_WALKTYPE_HIDDEN)
1578            && (!strcmp(dirent.name, DAV_FS_STATE_DIR) ||

1620        fsctx->info2.pathname = fsctx->path2.buf;

1732            **

1763                (err = (*params->func)(&fsctx->wres,
1764                                       DAV_CALLTYPE_LOCKNULL)) != NULL) {
1765                /* ### maybe add a higher-level description? */ view japanese

1767            }

1770        }
1771
1772        /* reset the exists flag */ view japanese
1773        fsctx->res1.exists = 1;
1774    }
1775

1777        /* replace the dirs' trailing slashes with null terms */ view japanese

1780        if (fsctx->path2.buf != NULL) {
1781            fsctx->path2.buf[--fsctx->path2.cur_len] = '\0';
1782        }
1783
1784        /* this is a collection which exists */ view japanese
1785        fsctx->res1.collection = 1;
1786
1787        return (*params->func)(&fsctx->wres, DAV_CALLTYPE_POSTFIX);
1788    }
1789
1790    return NULL;
1791}
