120define dump_bucket_ex
121    # arg0 == bucket
122    # arg1 == suppress header?
123    set $bucket = (struct apr_bucket *)$arg0
124    set $sh = $arg1
125    set $refcount = -1
126
127    print_bkt_datacol "bucket" "%-9s" $bucket->type->name $sh
128    printf "(%12lx)", (unsigned long)$bucket
129    print_bkt_datacol "length" "%-6ld" (long)($bucket->length) $sh
130    print_bkt_datacol "data" "%12lx" $bucket->data $sh
131
132    if !$sh
133        printf "\n    "
134    end
135
136    if (($bucket->type == &apr_bucket_type_eos)   || \
137        ($bucket->type == &apr_bucket_type_flush))
138
139        # metadata buckets, no content
140        print_bkt_datacol "contents" "%c" ' ' $sh
141        printf "                     "
142        print_bkt_datacol "rc" "n/%c" 'a' $sh
143
144    else
145    if ($bucket->type == &ap_bucket_type_error)
146
147        # metadata bucket, no content but it does have an error code in it
148        print_bkt_datacol "contents" "%c" ' ' $sh
149        set $status = ((ap_bucket_error *)$bucket->data)->status
150        printf " (status=%3d)        ", $status
151        print_bkt_datacol "rc" "n/%c" 'a' $sh
152
153    else
154    if ($bucket->type == &apr_bucket_type_file)
155
156        # file bucket, can show fd and refcount
157        set $fd = ((apr_bucket_file*)$bucket->data)->fd->filedes
158        print_bkt_datacol "contents" "[***file***] fd=%-6ld" (long)$fd $sh
159        set $refcount = ((apr_bucket_refcount *)$bucket->data)->refcount
160        print_bkt_datacol "rc" "%-3d" $refcount $sh
161
162    else
163    if (($bucket->type == &apr_bucket_type_heap)      || \
164        ($bucket->type == &apr_bucket_type_pool)      || \
165        ($bucket->type == &apr_bucket_type_mmap)      || \
166        ($bucket->type == &apr_bucket_type_transient) || \
167        ($bucket->type == &apr_bucket_type_immortal))
168
169        # in-memory buckets
170
171        if $bucket->type == &apr_bucket_type_heap
172            set $refcount = ((apr_bucket_refcount *)$bucket->data)->refcount
173            set $p = (apr_bucket_heap *)$bucket->data
174            set $data = $p->base+$bucket->start
175
176        else
177        if $bucket->type == &apr_bucket_type_pool
178            set $refcount = ((apr_bucket_refcount *)$bucket->data)->refcount
179            set $p = (apr_bucket_pool *)$bucket->data
180            if !$p->pool
181                set $p = (apr_bucket_heap *)$bucket->data
182            end
183            set $data = $p->base+$bucket->start
184
185        else
186        if $bucket->type == &apr_bucket_type_mmap
187            # is this safe if not APR_HAS_MMAP?
188            set $refcount = ((apr_bucket_refcount *)$bucket->data)->refcount
189            set $p = (apr_bucket_mmap *)$bucket->data
190            set $data = ((char *)$p->mmap->mm)+$bucket->start
191
192        else
193        if (($bucket->type == &apr_bucket_type_transient) || \
194            ($bucket->type == &apr_bucket_type_immortal))
195            set $data = ((char *)$bucket->data)+$bucket->start
196
197        end
198        end
199        end
200        end
201
202        if $sh
203            printf " | ["
204        else
205            printf " contents=["
206        end
207        set $datalen = $bucket->length
208        if $isValidAddress($data) == 1
209            if $datalen > 17
210                printmem $data 17
211                printf "..."
212                set $datalen = 20
213            else
214                printmemn $data $datalen
215            end
216            printf "]"
217            while $datalen < 20
218                printf " "
219                set $datalen = $datalen + 1
220            end
221        else
222            printf "Iv addr %12lx]", $data
223        end
224
225        if $refcount != -1
226            print_bkt_datacol "rc" "%-3d" $refcount $sh
227        else
228            print_bkt_datacol "rc" "n/%c" 'a' $sh
229        end
230
231    else
232        if ($bucket->type == &apr_bucket_type_pipe)
233
234            # pipe bucket, can show fd
235            set $fd = ((apr_file_t*)$bucket->data)->filedes
236            print_bkt_datacol "contents" "[***pipe***] fd=%-3ld" (long)$fd $sh
237
238        else
239        if ($bucket->type == &apr_bucket_type_socket)
240
241            # file bucket, can show fd
242            set $fd = ((apr_socket_t*)$bucket->data)->socketdes
243            print_bkt_datacol "contents" "[**socket**] fd=%-3ld" (long)$fd $sh
244
245        else
246
247            # 3rd-party bucket type
248            print_bkt_datacol "contents" "[**opaque**]%-7c" ' ' $sh
249        end
250        end
251
252        # no refcount
253        printf "   "
254        print_bkt_datacol "rc" "n/%c" 'a' $sh
255    end
256    end
257    end
258    end
259
260    printf "\n"
261
262end

496class DumpPoolAndChilds (gdb.Command):
497  """Dump the whole pool hierarchy starting from the given pool."""
498
499  def __init__ (self):
500    super (DumpPoolAndChilds, self).__init__ ("dump_pool_and_children", gdb.COMMAND_USER)
501
502  def _allocator_free_blocks(self, alloc):
503    salloc = "%s" % (alloc)
504    if self.total_free_blocks.get(salloc) != None:
505      return self.total_free_blocks[salloc]
506    i = 0
507    dalloc = alloc.dereference()
508    max =(dalloc['free'].type.sizeof)/(dalloc['free'][0].type.sizeof)
509    kb = 0
510    while i < max:
511      node = dalloc['free'][i]
512      if node != 0:
513        while node != 0:
514          noded = node.dereference()
515          kb = kb + ((int(noded['index']) + 1) << 2)
516          node = noded['next']
517      i = i + 1
518    self.total_free_blocks[salloc] = kb
519    return kb
520
521
522  def _dump_one_pool(self, arg):
523    size = 0
524    free = 0
525    nodes = 0
526    darg = arg.dereference()
527    active = darg['active']
528    node = active
529    done = 0
530    while done == 0:
531      noded = node.dereference()
532      size = size + ((int(noded['index']) + 1) << 12)
533      free = free + (noded['endp'] - noded['first_avail'])
534      nodes = nodes + 1
535      node = noded['next']
536      if node == active:
537        done = 1
538    if darg['tag'] != 0:
539      tag = darg['tag'].string()
540    else:
541      tag = "No tag"
542    print("Pool '%s' [%s]: %d/%d free (%d blocks) allocator: %s free blocks in allocator: %i kiB" % (tag, arg, free, size, nodes, darg['allocator'], self._allocator_free_blocks(darg['allocator'])))
543    self.free = self.free + free
544    self.size = self.size + size
545    self.nodes = self.nodes + nodes
546
547  def _dump(self, arg, depth):
548    pool = arg
549    while pool:
550        print("%*c" % (depth * 4 + 1, " "), end="")
551        self._dump_one_pool(pool)
552        if pool['child'] != 0:
553            self._dump(pool['child'], depth + 1)
554        pool = pool['sibling']
555
556  def invoke (self, arg, from_tty):
557    pool = gdb.parse_and_eval(arg)
558    self.free = 0
559    self.size = 0
560    self.nodes = 0
561    self.total_free_blocks = {}
562    self._dump(pool, 0)
563    print("Total %d/%d free (%d blocks)" % (self.free, self.size, self.nodes))
564    sum = 0
565    for key in self.total_free_blocks:
566      sum = sum + self.total_free_blocks[key]
567    print("Total free allocator blocks: %i kiB" % (sum))
568
569DumpPoolAndChilds ()
570end

577class isValidAddress (gdb.Function):
578    """Determines if the argument is a valid address."""
579
580    def __init__(self):
581        super(isValidAddress, self).__init__("isValidAddress")
582
583    def invoke(self, address):
584        inf = gdb.inferiors()[0]
585        result = 1
586        try:
587            inf.read_memory(address, 8)
588        except:
589            result = 0
590        return result
