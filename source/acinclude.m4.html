216dnl APACHE_MPM_MODULE(name[, shared[, objects[, config[, path[, libs]]]]])
217dnl
218dnl Provide information for building the MPM.  (Enablement is handled using
219dnl --with-mpm/--enable-mpms-shared.)
220dnl
221dnl name     -- name of MPM, same as MPM directory name
222dnl shared   -- "shared" to indicate shared module build, empty string otherwise
223dnl objects  -- one or more .lo files to link into the MPM module (default: mpmname.lo)
224dnl config   -- configuration logic to run if the MPM is enabled
225dnl path     -- relative path to MPM (default: server/mpm/mpmname)
226dnl libs     -- libs needed by this MPM
227dnl
228AC_DEFUN([APACHE_MPM_MODULE],[
229    if ap_mpm_is_enabled $1; then
230        if test -z "$3"; then
231            objects="$1.lo"
232        else
233            objects="$3"
234        fi
235
236        if test -z "$5"; then
237            mpmpath="server/mpm/$1"
238        else
239            mpmpath=$5
240        fi
241
242        dnl VPATH support
243        test -d $mpmpath || $srcdir/build/mkdir.sh $mpmpath
244
245        APACHE_FAST_OUTPUT($mpmpath/Makefile)
246
247        if test -z "$2"; then
248            APR_ADDTO(AP_LIBS, [$6])
249            libname="lib$1.la"
250            cat >$mpmpath/modules.mk<<EOF
251$libname: $objects
252	\$(MOD_LINK) $objects
253DISTCLEAN_TARGETS = modules.mk
254static = $libname
255shared =
256EOF
257        else
258            apache_need_shared=yes
259            libname="mod_mpm_$1.la"
260            shobjects=`echo $objects | sed 's/\.lo/.slo/g'`
261            cat >$mpmpath/modules.mk<<EOF
262$libname: $shobjects
263	\$(SH_LINK) -rpath \$(libexecdir) -module -avoid-version $objects $6
264DISTCLEAN_TARGETS = modules.mk
265static =
266shared = $libname
267EOF
268            MPM_MODULES="$MPM_MODULES mpm_$1"
269            # add default MPM to LoadModule list
270            if test $1 = $default_mpm; then
271                ENABLED_MPM_MODULE="mpm_$1"
272            fi
273        fi
274        $4
275    fi

279dnl APACHE_MODULE(name, helptext[, objects[, structname[, default[, config[, prereq_module]]]]])
280dnl
281dnl default is one of:
282dnl   yes    -- enabled by default. user must explicitly disable.
283dnl   no     -- disabled under default, most, all. user must explicitly enable.
284dnl   most   -- disabled by default. enabled explicitly or with most or all.
285dnl   static -- enabled as static by default, must be explicitly changed.
286dnl   ""     -- disabled under default, most. enabled explicitly or with all.
287dnl             XXX: The arg must really be empty here. Passing an empty shell
288dnl             XXX: variable doesn't work for some reason. This should be
289dnl             XXX: fixed.
290dnl
291dnl basically: yes/no is a hard setting. "most" means follow the "most"
292dnl            setting. otherwise, fall under the "all" setting.
293dnl            explicit yes/no always overrides, except if the user selects
294dnl            "reallyall".
295dnl
296dnl prereq_module is a module (without the "mod_" prefix) that must be enabled
297dnl   if the current module is enabled.  If the current module is built
298dnl   statically, prereq_module must be built statically, too.  If these
299dnl   conditions are not fulfilled, configure will abort if the current module
300dnl   has been enabled explicitly. Otherwise, configure will disable the
301dnl   current module.
302dnl   prereq_module's APACHE_MODULE() statement must have been processed
303dnl   before the current APACHE_MODULE() statement.
304dnl
305AC_DEFUN([APACHE_MODULE],[
306  AC_MSG_CHECKING(whether to enable mod_$1)
307  define([optname],[--]ifelse($5,yes,disable,enable)[-]translit($1,_,-))dnl
308  AC_ARG_ENABLE(translit($1,_,-),APACHE_HELP_STRING(optname(),$2),force_$1=$enableval,enable_$1=ifelse($5,,maybe-all,$5))
309  undefine([optname])dnl
310  _apmod_extra_msg=""
311  dnl If the module was not explicitly requested, allow it to disable itself if
312  dnl its pre-reqs fail.
313  case "$enable_$1" in
314    yes|static|shared)
315      _apmod_required="yes"
316      ;;
317    *)
318      _apmod_required="no"
319      ;;
320  esac
321  if test "$enable_$1" = "static" -o "$enable_$1" = "shared"; then
322    :
323  elif test "$enable_$1" = "yes"; then
324    enable_$1=$module_default
325  elif test "$enable_$1" = "few"; then
326    if test "$module_selection" = "few" -o "$module_selection" = "most" -o \
327            "$module_selection" = "all" -o "$module_selection" = "reallyall"
328    then
329      enable_$1=$module_default
330    else
331      enable_$1=no
332    fi
333    _apmod_extra_msg=" ($module_selection)"
334  elif test "$enable_$1" = "most"; then
335    if test "$module_selection" = "most" -o "$module_selection" = "all" -o \
336            "$module_selection" = "reallyall"
337    then
338      enable_$1=$module_default
339    else
340      enable_$1=no
341    fi
342    _apmod_extra_msg=" ($module_selection)"
343  elif test "$enable_$1" = "all" -o "$enable_$1" = "maybe-all"; then
344    if test "$module_selection" = "all" -o "$module_selection" = "reallyall"
345    then
346      enable_$1=$module_default
347      _apmod_extra_msg=" ($module_selection)"
348    else
349      enable_$1=no
350    fi
351  elif test "$enable_$1" = "reallyall" -o "$enable_$1" = "no" ; then
352    if test "$module_selection" = "reallyall" -a "$force_$1" != "no" ; then
353      enable_$1=$module_default
354      _apmod_extra_msg=" ($module_selection)"
355    else
356      enable_$1=no
357    fi
358  else
359    enable_$1=no
360  fi
361  if test "$enable_$1" != "no"; then
362    dnl If we plan to enable it, allow the module to run some autoconf magic
363    dnl that may disable it because of missing dependencies.
364    ifelse([$6$7],,:,
365           [AC_MSG_RESULT([checking dependencies])
366            ifelse([$7],,:,[m4_foreach([prereq],[$7],
367                           [if test "$enable_[]prereq" = "no" ; then
368                              enable_$1=no
369                              AC_MSG_WARN("mod_[]prereq is disabled but required for mod_$1")
370                            elif test "$enable_$1" = "static" && test "$enable_[]prereq" != "static" ; then
371                              enable_$1=$enable_[]prereq
372                              AC_MSG_WARN("building mod_$1 shared because mod_[]prereq is built shared")
373                            el])se])
374            ifelse([$6],,:,[  $6])
375            ifelse([$7],,:,[fi])
376            AC_MSG_CHECKING(whether to enable mod_$1)
377            if test "$enable_$1" = "no"; then
378              if test "$_apmod_required" = "no"; then
379                _apmod_extra_msg=" (disabled)"
380              else
381                AC_MSG_ERROR([mod_$1 has been requested but can not be built due to prerequisite failures])
382              fi
383            fi])
384  fi
385  AC_MSG_RESULT($enable_$1$_apmod_extra_msg)
386  if test "$enable_$1" != "no"; then
387    case "$enable_$1" in
388    static*)
389      MODLIST="$MODLIST ifelse($4,,$1,$4)"
390      if test "$1" = "so"; then
391          sharedobjs=yes
392      fi
393      shared="";;
394    *)
395      sharedobjs=yes
396      shared=yes
397      DSO_MODULES="$DSO_MODULES $1"
398      if test "$5" = "yes" ; then
399        ENABLED_DSO_MODULES="${ENABLED_DSO_MODULES},$1"
400      fi
401      ;;
402    esac
403    define([modprefix], [MOD_]translit($1, [a-z-], [A-Z_]))
404    APACHE_MODPATH_ADD($1, $shared, $3,, [\$(]modprefix[_LDADD)])
405    APACHE_SUBST(modprefix[_LDADD])
406    undefine([modprefix])
407  fi
408])dnl

482dnl APACHE_CHECK_OPENSSL
483dnl
484dnl Configure for OpenSSL, giving preference to
485dnl "--with-ssl=<path>" if it was specified.
486dnl
487AC_DEFUN([APACHE_CHECK_OPENSSL],[
488  AC_CACHE_CHECK([for OpenSSL], [ac_cv_openssl], [
489    dnl initialise the variables we use
490    ac_cv_openssl=no
491    ap_openssl_found=""
492    ap_openssl_base=""
493    ap_openssl_libs=""
494    ap_openssl_mod_cflags=""
495    ap_openssl_mod_ldflags=""
496
497    dnl Determine the OpenSSL base directory, if any
498    AC_MSG_CHECKING([for user-provided OpenSSL base directory])
499    AC_ARG_WITH(ssl, APACHE_HELP_STRING(--with-ssl=PATH,OpenSSL installation directory), [
500      dnl If --with-ssl specifies a directory, we use that directory
501      if test "x$withval" != "xyes" -a "x$withval" != "x"; then
502        dnl This ensures $withval is actually a directory and that it is absolute
503        ap_openssl_base="`cd $withval ; pwd`"
504      fi
505    ])
506    if test "x$ap_openssl_base" = "x"; then
507      AC_MSG_RESULT(none)
508    else
509      AC_MSG_RESULT($ap_openssl_base)
510    fi
511
512    dnl Run header and version checks
513    saved_CPPFLAGS="$CPPFLAGS"
514    saved_LIBS="$LIBS"
515    saved_LDFLAGS="$LDFLAGS"
516
517    dnl Before doing anything else, load in pkg-config variables
518    if test -n "$PKGCONFIG"; then
519      saved_PKG_CONFIG_PATH="$PKG_CONFIG_PATH"
520      if test "x$ap_openssl_base" != "x"; then
521        if test -f "${ap_openssl_base}/lib/pkgconfig/openssl.pc"; then
522          dnl Ensure that the given path is used by pkg-config too, otherwise
523          dnl the system openssl.pc might be picked up instead.
524          PKG_CONFIG_PATH="${ap_openssl_base}/lib/pkgconfig${PKG_CONFIG_PATH+:}${PKG_CONFIG_PATH}"
525          export PKG_CONFIG_PATH
526        elif test -f "${ap_openssl_base}/lib64/pkgconfig/openssl.pc"; then
527          dnl Ensure that the given path is used by pkg-config too, otherwise
528          dnl the system openssl.pc might be picked up instead.
529          PKG_CONFIG_PATH="${ap_openssl_base}/lib64/pkgconfig${PKG_CONFIG_PATH+:}${PKG_CONFIG_PATH}"
530          export PKG_CONFIG_PATH
531        fi
532      fi
533      AC_ARG_ENABLE(ssl-staticlib-deps,APACHE_HELP_STRING(--enable-ssl-staticlib-deps,[link mod_ssl with dependencies of OpenSSL's static libraries (as indicated by "pkg-config --static"). Must be specified in addition to --enable-ssl.]), [
534        if test "$enableval" = "yes"; then
535          PKGCONFIG_LIBOPTS="--static"
536        fi
537      ])
538      ap_openssl_libs="`$PKGCONFIG $PKGCONFIG_LIBOPTS --libs-only-l --silence-errors openssl`"
539      if test $? -eq 0; then
540        ap_openssl_found="yes"
541        pkglookup="`$PKGCONFIG --cflags-only-I openssl`"
542        APR_ADDTO(CPPFLAGS, [$pkglookup])
543        APR_ADDTO(MOD_CFLAGS, [$pkglookup])
544        APR_ADDTO(ab_CFLAGS, [$pkglookup])
545        pkglookup="`$PKGCONFIG $PKGCONFIG_LIBOPTS --libs-only-L openssl`"
546        APR_ADDTO(LDFLAGS, [$pkglookup])
547        APR_ADDTO(MOD_LDFLAGS, [$pkglookup])
548        pkglookup="`$PKGCONFIG $PKGCONFIG_LIBOPTS --libs-only-other openssl`"
549        APR_ADDTO(LDFLAGS, [$pkglookup])
550        APR_ADDTO(MOD_LDFLAGS, [$pkglookup])
551      fi
552      PKG_CONFIG_PATH="$saved_PKG_CONFIG_PATH"
553    fi
554
555    dnl fall back to the user-supplied directory if not found via pkg-config
556    if test "x$ap_openssl_base" != "x" -a "x$ap_openssl_found" = "x"; then
557      APR_ADDTO(CPPFLAGS, [-I$ap_openssl_base/include])
558      APR_ADDTO(MOD_CFLAGS, [-I$ap_openssl_base/include])
559      APR_ADDTO(ab_CFLAGS, [-I$ap_openssl_base/include])
560      APR_ADDTO(LDFLAGS, [-L$ap_openssl_base/lib])
561      APR_ADDTO(MOD_LDFLAGS, [-L$ap_openssl_base/lib])
562      if test "x$ap_platform_runtime_link_flag" != "x"; then
563        APR_ADDTO(LDFLAGS, [$ap_platform_runtime_link_flag$ap_openssl_base/lib])
564        APR_ADDTO(MOD_LDFLAGS, [$ap_platform_runtime_link_flag$ap_openssl_base/lib])
565      fi
566    fi
567
568    AC_MSG_CHECKING([for OpenSSL version >= 0.9.8a])
569    AC_TRY_COMPILE([#include <openssl/opensslv.h>],[
570#if !defined(OPENSSL_VERSION_NUMBER)
571#error "Missing OpenSSL version"
572#endif
573#if OPENSSL_VERSION_NUMBER < 0x0090801f
574#error "Unsupported OpenSSL version " OPENSSL_VERSION_TEXT
575#endif],
576      [AC_MSG_RESULT(OK)
577       ac_cv_openssl=yes],
578      [AC_MSG_RESULT(FAILED)])
579
580    if test "x$ac_cv_openssl" = "xyes"; then
581      ap_openssl_libs="${ap_openssl_libs:--lssl -lcrypto} `$apr_config --libs`"
582      APR_ADDTO(MOD_LDFLAGS, [$ap_openssl_libs])
583      APR_ADDTO(LIBS, [$ap_openssl_libs])
584      APR_SETVAR(ab_LIBS, [$MOD_LDFLAGS])
585      APACHE_SUBST(ab_CFLAGS)
586      APACHE_SUBST(ab_LIBS)
587
588      dnl Run library and function checks
589      liberrors=""
590      AC_CHECK_HEADERS([openssl/engine.h])
591      AC_CHECK_FUNCS([SSL_CTX_new], [], [liberrors="yes"])
592      AC_CHECK_FUNCS([OPENSSL_init_ssl])
593      AC_CHECK_FUNCS([ENGINE_init ENGINE_load_builtin_engines RAND_egd])
594      if test "x$liberrors" != "x"; then
595        AC_MSG_WARN([OpenSSL libraries are unusable])
596      fi
597    else
598      AC_MSG_WARN([OpenSSL version is too old])
599    fi
600
601    dnl restore
602    CPPFLAGS="$saved_CPPFLAGS"
603    LIBS="$saved_LIBS"
604    LDFLAGS="$saved_LDFLAGS"
605
606    dnl cache MOD_LDFLAGS, MOD_CFLAGS
607    ap_openssl_mod_cflags=$MOD_CFLAGS
608    ap_openssl_mod_ldflags=$MOD_LDFLAGS
609  ])
610  if test "x$ac_cv_openssl" = "xyes"; then
611    AC_DEFINE(HAVE_OPENSSL, 1, [Define if OpenSSL is available])
612    APR_ADDTO(MOD_LDFLAGS, [$ap_openssl_mod_ldflags])
613    APR_ADDTO(MOD_CFLAGS, [$ap_openssl_mod_cflags])
614  fi

676dnl APACHE_CHECK_APxVER({apr|apu}, major, minor, 
677dnl                     [actions-if-ok], [actions-if-not-ok])
678dnl
679dnl Checks for APR or APR-util of given major/minor version or later; 
680dnl if so, runs actions-if-ok; otherwise runs actions-if-not-ok if given.
681dnl If the version is not satisfactory and actions-if-not-ok is not
682dnl given, then an error is printed and the configure script is aborted.
683dnl
684dnl The first argument must be [apr] or [apu].
685dnl
686AC_DEFUN([APACHE_CHECK_APxVER], [
687define(ap_ckver_major, translit($1, [apru], [APRU])[_MAJOR_VERSION])
688define(ap_ckver_minor, translit($1, [apru], [APRU])[_MINOR_VERSION])
689define(ap_ckver_cvar, [ap_cv_$1ver$2$3])
690define(ap_ckver_name, ifelse([$1],[apr],[APR],[APR-util]))
691
692ap_ckver_CPPFLAGS="$CPPFLAGS"
693CPPFLAGS="$CPPFLAGS `$[$1]_config --includes`"
694
695AC_CACHE_CHECK([for ap_ckver_name version $2.$3.0 or later], ap_ckver_cvar, [
696AC_EGREP_CPP([good], [
697#include <$1_version.h>
698#if ]ap_ckver_major[ > $2 || (]ap_ckver_major[ == $2 && ]ap_ckver_minor[ >= $3)
699good
700#endif
701], [ap_ckver_cvar=yes], [ap_ckver_cvar=no])])
702
703if test "$ap_ckver_cvar" = "yes"; then
704  ifelse([$4],[],[:],[$4])
705else
706  ifelse([$5],[],[AC_MSG_ERROR([ap_ckver_name version $2.$3.0 or later is required])], [$5])
707fi
708
709CPPFLAGS="$ap_ckver_CPPFLAGS"
710
711undefine([ap_ckver_major])
712undefine([ap_ckver_minor])
713undefine([ap_ckver_cvar])
714undefine([ap_ckver_name])
715])

760dnl APACHE_ADD_GCC_CFLAG
761dnl
762dnl Check if compiler is gcc and supports flag. If yes, add to NOTEST_CFLAGS.
763dnl NOTEST_CFLAGS is merged lately, thus it won't accumulate in CFLAGS here.
764dnl Also, AC_LANG_PROGRAM() itself is known to trigger [-Wstrict-prototypes]
765dnl with some autoconf versions, so we force -Wno-strict-prototypes for the
766dnl check to avoid spurious failures when adding flags like -Werror.
767dnl
768AC_DEFUN([APACHE_ADD_GCC_CFLAG], [
769  define([ap_gcc_ckvar], [ac_cv_gcc_]translit($1, [-:.=], [____]))
770  if test "$GCC" = "yes"; then
771    AC_CACHE_CHECK([whether gcc accepts $1], ap_gcc_ckvar, [
772      save_CFLAGS="$CFLAGS"
773      CFLAGS="$CFLAGS $1 -Wno-strict-prototypes"
774      AC_COMPILE_IFELSE([AC_LANG_PROGRAM()],
775        [ap_gcc_ckvar=yes], [ap_gcc_ckvar=no])
776      CFLAGS="$save_CFLAGS"
777    ])
778    if test "$]ap_gcc_ckvar[" = "yes" ; then
779       APR_ADDTO(NOTEST_CFLAGS,[$1])
780    fi
781  fi
782  undefine([ap_gcc_ckvar])
