59AP_DECLARE(void) ap_unixd_set_rlimit(cmd_parms *cmd, struct rlimit **plimit,
60                                     const char *arg,
61                                     const char * arg2, int type)
62{
63#if (defined(RLIMIT_CPU) || defined(RLIMIT_DATA) || defined(RLIMIT_VMEM) || defined(RLIMIT_NPROC) || defined(RLIMIT_AS)) && APR_HAVE_STRUCT_RLIMIT && APR_HAVE_GETRLIMIT
64    char *str;
65    struct rlimit *limit;
66    /* If your platform doesn't define rlim_t then typedef it in ap_config.h */ view japanese
67    rlim_t cur = 0;
68    rlim_t max = 0;
69
70    *plimit = (struct rlimit *)apr_pcalloc(cmd->pool, sizeof(**plimit));
71    limit = *plimit;
72    if ((getrlimit(type, limit)) != 0)  {
73        *plimit = NULL;
74        ap_log_error(APLOG_MARK, APLOG_ERR, errno, cmd->server, APLOGNO(02172)
75                     "%s: getrlimit failed", cmd->cmd->name);
76        return;
77    }
78
79    if (*(str = ap_getword_conf(cmd->temp_pool, &arg)) != '\0') {
80        if (!strcasecmp(str, "max")) {
81            cur = limit->rlim_max;
82        }
83        else {
84            cur = atol(str);
85        }
86    }
87    else {
88        ap_log_error(APLOG_MARK, APLOG_ERR, 0, cmd->server, APLOGNO(02173)
89                     "Invalid parameters for %s", cmd->cmd->name);
90        return;
91    }
92
93    if (arg2 && (*(str = ap_getword_conf(cmd->temp_pool, &arg2)) != '\0')) {
94        max = atol(str);
95    }
96
97    /* if we aren't running as root, cannot increase max */ view japanese
98    if (geteuid()) {
99        limit->rlim_cur = cur;
100        if (max && (max > limit->rlim_max)) {
101            ap_log_error(APLOG_MARK, APLOG_ERR, 0, cmd->server, APLOGNO(02174)
102                         "Must be uid 0 to raise maximum %s", cmd->cmd->name);
103        }
104        else if (max) {
105            limit->rlim_max = max;
106        }
107    }
108    else {
109        if (cur) {
110            limit->rlim_cur = cur;
111        }
112        if (max) {
113            limit->rlim_max = max;
114        }
115    }
116#else
117
118    ap_log_error(APLOG_MARK, APLOG_ERR, 0, cmd->server, APLOGNO(02175)
119                 "Platform does not support rlimit for %s", cmd->cmd->name);
120#endif
121}

130static apr_status_t ap_unix_create_privileged_process(
131                              apr_proc_t *newproc, const char *progname,
132                              const char * const *args,
133                              const char * const *env,
134                              apr_procattr_t *attr, ap_unix_identity_t *ugid,
135                              apr_pool_t *p)
136{
137    int i = 0;
138    const char **newargs;
139    char *newprogname;
140    char *execuser, *execgroup;
141    const char *argv0;
142
143    if (!ap_unixd_config.suexec_enabled) {
144        return apr_proc_create(newproc, progname, args, env, attr, p);
145    }
146
147    argv0 = ap_strrchr_c(progname, '/');
148    /* Allow suexec's "/" check to succeed */ view japanese
149    if (argv0 != NULL) {
150        argv0++;
151    }
152    else {
153        argv0 = progname;
154    }
155
156
157    if (ugid->userdir) {
158        execuser = apr_psprintf(p, "~%ld", (long) ugid->uid);
159    }
160    else {
161        execuser = apr_psprintf(p, "%ld", (long) ugid->uid);
162    }
163    execgroup = apr_psprintf(p, "%ld", (long) ugid->gid);
164
165    if (!execuser || !execgroup) {
166        return APR_ENOMEM;
167    }
168
169    i = 0;
170    while (args[i])
171        i++;
172    /* allocate space for 4 new args, the input args, and a null terminator */ view japanese
173    newargs = apr_palloc(p, sizeof(char *) * (i + 4));
174    newprogname = SUEXEC_BIN;
175    newargs[0] = SUEXEC_BIN;
176    newargs[1] = execuser;
177    newargs[2] = execgroup;
178    newargs[3] = apr_pstrdup(p, argv0);
179
180    /*
181    ** using a shell to execute suexec makes no sense thus
182    ** we force everything to be APR_PROGRAM, and never
183    ** APR_SHELLCMD
184    */ view japanese
185    if (apr_procattr_cmdtype_set(attr, APR_PROGRAM) != APR_SUCCESS) {
186        return APR_EGENERAL;
187    }
188
189    i = 1;
190    do {
191        newargs[i + 3] = args[i];
192    } while (args[i++]);
193
194    return apr_proc_create(newproc, newprogname, newargs, env, attr, p);
195}

197AP_DECLARE(apr_status_t) ap_os_create_privileged_process(
198    const request_rec *r,
199    apr_proc_t *newproc, const char *progname,
200    const char * const *args,
201    const char * const *env,
202    apr_procattr_t *attr, apr_pool_t *p)
203{
204    ap_unix_identity_t *ugid = ap_run_get_suexec_identity(r);
205
206    if (ugid == NULL) {
207        return apr_proc_create(newproc, progname, args, env, attr, p);
208    }
209
210    return ap_unix_create_privileged_process(newproc, progname, args, env,
211                                              attr, ugid, p);
212}

215static apr_lockmech_e proc_mutex_mech(apr_proc_mutex_t *pmutex)
216{
217    const char *mechname = apr_proc_mutex_name(pmutex);
218
219    if (!strcmp(mechname, "sysvsem")) {
220        return APR_LOCK_SYSVSEM;
221    }
222    else if (!strcmp(mechname, "flock")) {
223        return APR_LOCK_FLOCK;
224    }
225    return APR_LOCK_DEFAULT;
226}

228AP_DECLARE(apr_status_t) ap_unixd_set_proc_mutex_perms(apr_proc_mutex_t *pmutex)
229{
230    if (!geteuid()) {
231        apr_lockmech_e mech = proc_mutex_mech(pmutex);
232
233        switch(mech) {
234#if APR_HAS_SYSVSEM_SERIALIZE
235        case APR_LOCK_SYSVSEM:
236        {
237            apr_os_proc_mutex_t ospmutex;
238#if !APR_HAVE_UNION_SEMUN
239            union semun {
240                long val;
241                struct semid_ds *buf;
242                unsigned short *array;
243            };
244#endif
245            union semun ick;
246            struct semid_ds buf = { { 0 } };
247
248            apr_os_proc_mutex_get(&ospmutex, pmutex);
249            buf.sem_perm.uid = ap_unixd_config.user_id;
250            buf.sem_perm.gid = ap_unixd_config.group_id;
251            buf.sem_perm.mode = 0600;
252            ick.buf = &buf;
253            if (semctl(ospmutex.crossproc, 0, IPC_SET, ick) < 0) {
254                return errno;
255            }
256        }
257        break;
258#endif
259#if APR_HAS_FLOCK_SERIALIZE
260        case APR_LOCK_FLOCK:
261        {
262            const char *lockfile = apr_proc_mutex_lockfile(pmutex);
263
264            if (lockfile) {
265                if (chown(lockfile, ap_unixd_config.user_id,
266                          -1 /* no gid change */) < 0) { view japanese
267                    return errno;
268                }
269            }
270        }
271        break;
272#endif
273        default:
274            /* do nothing */ view japanese
275            break;
276        }
277    }
278    return APR_SUCCESS;
279}

293AP_DECLARE(apr_status_t) ap_unixd_accept(void **accepted, ap_listen_rec *lr,
294                                         apr_pool_t *ptrans)
295{
296    apr_socket_t *csd;
297    apr_status_t status;
298#ifdef _OSD_POSIX
299    int sockdes;
300#endif
301
302    *accepted = NULL;
303    status = apr_socket_accept(&csd, lr->sd, ptrans);
304    if (status == APR_SUCCESS) {
305        *accepted = csd;
306#ifdef _OSD_POSIX
307        apr_os_sock_get(&sockdes, csd);
308        if (sockdes >= FD_SETSIZE) {
309            ap_log_error(APLOG_MARK, APLOG_WARNING, 0, ap_server_conf, APLOGNO(02176)
310                         "new file descriptor %d is too large; you probably need "
311                         "to rebuild Apache with a larger FD_SETSIZE "
312                         "(currently %d)",
313                         sockdes, FD_SETSIZE);
314            apr_socket_close(csd);
315            return APR_EINTR;
316        }
317#endif
318        return APR_SUCCESS;
319    }
320
321    if (APR_STATUS_IS_EINTR(status)) {
322        return status;
323    }
324    /* Our old behaviour here was to continue after accept()
325     * errors.  But this leads us into lots of troubles
326     * because most of the errors are quite fatal.  For
327     * example, EMFILE can be caused by slow descriptor
328     * leaks (say in a 3rd party module, or libc).  It's
329     * foolish for us to continue after an EMFILE.  We also
330     * seem to tickle kernel bugs on some platforms which
331     * lead to never-ending loops here.  So it seems best
332     * to just exit in most cases.
333     */ view japanese
334    switch (status) {
335#if defined(HPUX11) && defined(ENOBUFS)
336        /* On HPUX 11.x, the 'ENOBUFS, No buffer space available'
337         * error occurs because the accept() cannot complete.
338         * You will not see ENOBUFS with 10.20 because the kernel
339         * hides any occurrence from being returned to user space.
340         * ENOBUFS with 11.x's TCP/IP stack is possible, and could
341         * occur intermittently. As a work-around, we are going to
342         * ignore ENOBUFS.
343         */ view japanese
344        case ENOBUFS:
345#endif
346
347#ifdef EPROTO
348        /* EPROTO on certain older kernels really means
349         * ECONNABORTED, so we need to ignore it for them.
350         * See discussion in new-httpd archives nh.9701
351         * search for EPROTO.
352         *
353         * Also see nh.9603, search for EPROTO:
354         * There is potentially a bug in Solaris 2.x x<6,
355         * and other boxes that implement tcp sockets in
356         * userland (i.e. on top of STREAMS).  On these
357         * systems, EPROTO can actually result in a fatal
358         * loop.  See PR#981 for example.  It's hard to
359         * handle both uses of EPROTO.
360         */ view japanese
361        case EPROTO:
362#endif
363#ifdef ECONNABORTED
364        case ECONNABORTED:
365#endif
366        /* Linux generates the rest of these, other tcp
367         * stacks (i.e. bsd) tend to hide them behind
368         * getsockopt() interfaces.  They occur when
369         * the net goes sour or the client disconnects
370         * after the three-way handshake has been done
371         * in the kernel but before userland has picked
372         * up the socket.
373         */ view japanese
374#ifdef ECONNRESET
375        case ECONNRESET:
376#endif
377#ifdef ETIMEDOUT
378        case ETIMEDOUT:
379#endif
380#ifdef EHOSTUNREACH
381        case EHOSTUNREACH:
382#endif
383#ifdef ENETUNREACH
384        case ENETUNREACH:
385#endif
386        /* EAGAIN/EWOULDBLOCK can be returned on BSD-derived
387         * TCP stacks when the connection is aborted before
388         * we call connect, but only because our listener
389         * sockets are non-blocking (AP_NONBLOCK_WHEN_MULTI_LISTEN)
390         */ view japanese
391#ifdef EAGAIN
392        case EAGAIN:
393#endif
394#ifdef EWOULDBLOCK
395#if !defined(EAGAIN) || EAGAIN != EWOULDBLOCK
396        case EWOULDBLOCK:
397#endif
398#endif
399            break;
400#ifdef ENETDOWN
401        case ENETDOWN:
402            /*
403             * When the network layer has been shut down, there
404             * is not much use in simply exiting: the parent
405             * would simply re-create us (and we'd fail again).
406             * Use the CHILDFATAL code to tear the server down.
407             * @@@ Martin's idea for possible improvement:
408             * A different approach would be to define
409             * a new APEXIT_NETDOWN exit code, the reception
410             * of which would make the parent shutdown all
411             * children, then idle-loop until it detected that
412             * the network is up again, and restart the children.
413             * Ben Hyde noted that temporary ENETDOWN situations
414             * occur in mobile IP.
415             */ view japanese
416            ap_log_error(APLOG_MARK, APLOG_EMERG, status, ap_server_conf, APLOGNO(02177)
417                         "apr_socket_accept: giving up.");
418            return APR_EGENERAL;
419#endif /*ENETDOWN*/ view japanese
420
421        default:
422            /* If the socket has been closed in ap_close_listeners()
423             * by the restart/stop action, we may get EBADF.
424             * Do not print an error in this case.
425             */ view japanese
426            if (!lr->active) {
427                ap_log_error(APLOG_MARK, APLOG_DEBUG, status, ap_server_conf, APLOGNO(02178)
428                             "apr_socket_accept failed for inactive listener");
429                return status;
430            }
431            ap_log_error(APLOG_MARK, APLOG_ERR, status, ap_server_conf, APLOGNO(02179)
432                         "apr_socket_accept: (client socket)");
433            return APR_EGENERAL;
434    }
435    return status;
436}

444    (void)unused;

449AP_DECLARE(ap_unixd_mpm_retained_data *) ap_unixd_mpm_get_retained_data(void)
450{
451    if (!retained_data) {
452        retained_data = ap_retained_data_create("ap_unixd_mpm_retained_data",
453                                                sizeof(*retained_data));
454        apr_pool_pre_cleanup_register(ap_pglobal, NULL, retained_data_cleanup);
455        retained_data->mpm_state = AP_MPMQ_STARTING;
456    }
457    return retained_data;

460static void sig_term(int sig)
461{
462    if (!retained_data) {
463        /* Main process (ap_pglobal) is dying */
464        return;
465    }
466    retained_data->mpm_state = AP_MPMQ_STOPPING;
467    if (retained_data->shutdown_pending
468            && (retained_data->is_ungraceful
469                || sig == AP_SIG_GRACEFUL_STOP)) {
470        /* Already handled */
471        return;
472    }
473
474    retained_data->shutdown_pending = 1;
475    if (sig != AP_SIG_GRACEFUL_STOP) {
476        retained_data->is_ungraceful = 1;
477    }
478}
479
480static void sig_restart(int sig)
481{
482    if (!retained_data) {
483        /* Main process (ap_pglobal) is dying */
484        return;
485    }
486    retained_data->mpm_state = AP_MPMQ_STOPPING;
487    if (retained_data->restart_pending
488            && (retained_data->is_ungraceful
489                || sig == AP_SIG_GRACEFUL)) {
490        /* Already handled */ view japanese
491        return;
492    }
493
494    retained_data->restart_pending = 1;
495    if (sig != AP_SIG_GRACEFUL) {
496        retained_data->is_ungraceful = 1;
497    }
498}

500static apr_status_t unset_signals(void *unused)
501{
502    if (!retained_data) {
503        /* Main process (ap_pglobal) is dying */ view japanese
504        return APR_SUCCESS;
505    }
506    retained_data->shutdown_pending = retained_data->restart_pending = 0;
507    retained_data->was_graceful = !retained_data->is_ungraceful;
508    retained_data->is_ungraceful = 0;
509
510    return APR_SUCCESS;
511}

520AP_DECLARE(void) ap_unixd_mpm_set_signals(apr_pool_t *pconf, int one_process)
521{
522#ifndef NO_USE_SIGACTION
523    struct sigaction sa;
524#endif
525
526    if (!one_process) {
527        ap_fatal_signal_setup(ap_server_conf, pconf);
528    }
529    else if (!ap_retained_data_get("ap_unixd_mpm_one_process_cleanup")) {
530        /* In one process mode (debug), httpd will exit immediately when asked
531         * to (SIGTERM/SIGINT) and never restart. We still want the cleanups to
532         * run though (such that e.g. temporary files/IPCs don't leak on the
533         * system), so the first time around we use atexit() to cleanup after
534         * ourselves.
535         */ view japanese
536        ap_retained_data_create("ap_unixd_mpm_one_process_cleanup", 1);
537        atexit(ap_terminate);
538    }
539
540    /* Signals' handlers depend on retained data */ view japanese
541    (void)ap_unixd_mpm_get_retained_data();
542
543#ifndef NO_USE_SIGACTION
544    memset(&sa, 0, sizeof sa);
545    sigemptyset(&sa.sa_mask);
546
547#ifdef SIGPIPE
548    sa.sa_handler = SIG_IGN;
549    if (sigaction(SIGPIPE, &sa, NULL) < 0)
550        ap_log_error(APLOG_MARK, APLOG_WARNING, errno, ap_server_conf, APLOGNO(00269)
551                     "sigaction(SIGPIPE)");
552#endif
553#ifdef SIGXCPU
554    sa.sa_handler = SIG_DFL;
555    if (sigaction(SIGXCPU, &sa, NULL) < 0)
556        ap_log_error(APLOG_MARK, APLOG_WARNING, errno, ap_server_conf, APLOGNO(00267)
557                     "sigaction(SIGXCPU)");
558#endif
559#ifdef SIGXFSZ
560    /* For systems following the LFS standard, ignoring SIGXFSZ allows
561     * a write() beyond the 2GB limit to fail gracefully with E2BIG
562     * rather than terminate the process. */ view japanese
563    sa.sa_handler = SIG_IGN;
564    if (sigaction(SIGXFSZ, &sa, NULL) < 0)
565        ap_log_error(APLOG_MARK, APLOG_WARNING, errno, ap_server_conf, APLOGNO(00268)
566                     "sigaction(SIGXFSZ)");
567#endif
568
569    sa.sa_handler = sig_term;
570    if (sigaction(SIGTERM, &sa, NULL) < 0)
571        ap_log_error(APLOG_MARK, APLOG_WARNING, errno, ap_server_conf, APLOGNO(00264)
572                     "sigaction(SIGTERM)");
573#ifdef SIGINT
574    if (sigaction(SIGINT, &sa, NULL) < 0)
575        ap_log_error(APLOG_MARK, APLOG_WARNING, errno, ap_server_conf, APLOGNO(00266)
576                     "sigaction(SIGINT)");
577#endif
578#ifdef AP_SIG_GRACEFUL_STOP
579    if (sigaction(AP_SIG_GRACEFUL_STOP, &sa, NULL) < 0)
580        ap_log_error(APLOG_MARK, APLOG_WARNING, errno, ap_server_conf, APLOGNO(00265)
581                     "sigaction(" AP_SIG_GRACEFUL_STOP_STRING ")");
582#endif
583
584    /* Don't catch restart signals in ONE_PROCESS mode :) */ view japanese
585    if (!one_process) {
586        sa.sa_handler = sig_restart;
587        if (sigaction(SIGHUP, &sa, NULL) < 0)
588            ap_log_error(APLOG_MARK, APLOG_WARNING, errno, ap_server_conf, APLOGNO(00270)
589                         "sigaction(SIGHUP)");
590        if (sigaction(AP_SIG_GRACEFUL, &sa, NULL) < 0)
591            ap_log_error(APLOG_MARK, APLOG_WARNING, errno, ap_server_conf, APLOGNO(00271)
592                         "sigaction(" AP_SIG_GRACEFUL_STRING ")");
593    }
594
595#else  /* NO_USE_SIGACTION */
596
597#ifdef SIGPIPE
598    apr_signal(SIGPIPE, SIG_IGN);
599#endif /* SIGPIPE */ view japanese
600#ifdef SIGXCPU
601    apr_signal(SIGXCPU, SIG_DFL);
602#endif /* SIGXCPU */ view japanese
603#ifdef SIGXFSZ
604    apr_signal(SIGXFSZ, SIG_IGN);
605#endif /* SIGXFSZ */ view japanese
606
607    apr_signal(SIGTERM, sig_term);
608#ifdef AP_SIG_GRACEFUL_STOP
609    apr_signal(AP_SIG_GRACEFUL_STOP, sig_term);
610#endif /* AP_SIG_GRACEFUL_STOP */ view japanese
611
612    if (!one_process) {
613        /* Don't restart in ONE_PROCESS mode :) */ view japanese
614#ifdef SIGHUP
615        apr_signal(SIGHUP, sig_restart);
616#endif /* SIGHUP */ view japanese
617#ifdef AP_SIG_GRACEFUL
618        apr_signal(AP_SIG_GRACEFUL, sig_restart);
619#endif /* AP_SIG_GRACEFUL */ view japanese
620    }
621
622#endif /* NO_USE_SIGACTION */ view japanese
623
624    apr_pool_cleanup_register(pconf, NULL, unset_signals,
625                              apr_pool_cleanup_null);
626}
