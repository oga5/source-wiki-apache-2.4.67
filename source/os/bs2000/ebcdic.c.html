185/* Translate a memory block from EBCDIC (host charset) to ASCII (net charset)
186 * dest and srce may be identical, or separate memory blocks, but
187 * should not overlap.
188 */ view japanese
189void
190ebcdic2ascii(unsigned char *dest, const unsigned char *srce, size_t count)
191{
192        while (count-- != 0) {
193                *dest++ = os_toascii[*srce++];
194        }
195}
196void
197ebcdic2ascii_strictly(unsigned char *dest, const unsigned char *srce, size_t count)
198{
199        while (count-- != 0) {
200                *dest++ = os_toascii_strictly[*srce++];
201        }
202}
203void
204ascii2ebcdic(unsigned char *dest, const unsigned char *srce, size_t count)
205{
206        while (count-- != 0) {
207                *dest++ = os_toebcdic[*srce++];
208        }
209}

 * is the fact that it maps every EBCDIC *except 0x0A* to its ASCII

 * server, protocol strings are used in the form
 *  "Content-Type: text/plain\015\012". Now all the characters in the string
 * are stored as EBCDIC, only the semantics of \012 is completely

 * mapped to \015 anyway, so there's no special case for it.
