129MACRO(DEFINE_WITH_BLANKS output_definition input_symbol input_value)
130  IF(MSVC_IDE OR ${CMAKE_MAJOR_VERSION}.${CMAKE_MINOR_VERSION}.${CMAKE_PATCH_VERSION} VERSION_GREATER 2.8.11)
131    SET(${output_definition} "-D${input_symbol}=\"${input_value}\"")
132  ELSE()
133    # command-line tool + older cmake, where extra quotes must be added and
134    # escaped to survive
135    SET(${output_definition} "-D${input_symbol}=\"\\\"${input_value}\\\"\"")
136  ENDIF()
137ENDMACRO()

139MACRO(GET_MOD_ENABLE_RANK macro_modname macro_mod_enable_val macro_output_rank)
140  IF(${macro_mod_enable_val} STREQUAL "O")
141    SET(${macro_output_rank} 0)
142  ELSEIF(${macro_mod_enable_val} STREQUAL "i")
143    SET(${macro_output_rank} 1)
144  ELSEIF(${macro_mod_enable_val} STREQUAL "I")
145    SET(${macro_output_rank} 2)
146  ELSEIF(${macro_mod_enable_val} STREQUAL "a")
147    SET(${macro_output_rank} 3)
148  ELSEIF(${macro_mod_enable_val} STREQUAL "A")
149    SET(${macro_output_rank} 4)
150  ELSE()
151    MESSAGE(FATAL_ERROR "Unexpected enablement value \"${macro_mod_enable_val}\" for ${macro_modname}")
152  ENDIF()
153ENDMACRO()

157# Figure out what APR/APU features are available
158#
159# CHECK_APR_FEATURE checks for features defined to 1 or 0 in apr.h or apu.h
160# The symbol representing the feature will be set to TRUE or FALSE for 
161# compatibility with the feature tests set by FindFooPackage.
162#
163# (unclear why CHECK_SYMBOL_EXISTS is needed, but I was getting "found" for anything 
164# not defined to either 1 or 0)
165
166MACRO(CHECK_APR_FEATURE which_define)
167  SET(CMAKE_REQUIRED_INCLUDES "${APR_INCLUDE_DIR}")
168  CHECK_SYMBOL_EXISTS(${which_define} "apr.h;apu.h" tmp_${which_define})
169  IF(${tmp_${which_define}})
170    CHECK_C_SOURCE_COMPILES("#include \"${APR_INCLUDE_DIR}/apr.h\"
171      #include \"${APR_INCLUDE_DIR}/apu.h\"
172      int main() {
173      #ifndef ${which_define}
174      #error gobble
175      #endif
176      #if !${which_define}
177      #error gobble
178      #endif
179      return 1;}" ${which_define})
180  ELSE()
181    SET(${which_define})
182  ENDIF()
183  IF(${${which_define}})
184    SET(${which_define} TRUE)
185  ELSE()
186    SET(${which_define} FALSE)
187  ENDIF()
188ENDMACRO()

190CHECK_APR_FEATURE(APR_HAS_XLATE)
191CHECK_APR_FEATURE(APU_HAVE_CRYPTO)

193# APR_HAS_LDAP is defined in apr_ldap.h, which exists only in apr 1.x, so use
194# special code instead of CHECK_APR_FEATURE()
195# As with CHECK_APR_FEATURE(), convert to a TRUE/FALSE result.
196CHECK_C_SOURCE_COMPILES("#include \"${APR_INCLUDE_DIR}/apr.h\"
197#include \"${APR_INCLUDE_DIR}/apr_ldap.h\"
198int main() {
199#if !APR_HAS_LDAP
200#error gobble
201#endif
202return 1;}" APR_HAS_LDAP)
203IF(${APR_HAS_LDAP})
204  SET(APR_HAS_LDAP TRUE)
205ELSE()
206  SET(APR_HAS_LDAP FALSE)
207ENDIF()

262# Options for each available module
263#   "A" ("A"ctive) means installed and active in default .conf, fail if can't be built
264#   "I" ("I"nactive) means installed and inactive (LoadModule commented out) in default .conf, fail if can't be built
265#   "O" ("O"mit) means not installed, no LoadModule
266#   "a" - like "A", but ignore with a warning if any prereqs aren't available
267#   "i" - like "I", but ignore with a warning if any prereqs aren't available
268
269# Current heuristic for default enablement:
270#
271#   Module requires a prereq and           -> O
272#   finding/usingprereq isn't implemented
273#   yet
274#
275#   Module is included by default in       -> a if it has prereqs, A otherwise
276#   autoconf-based build 
277#
278#   Module is included in                  -> i if it has prereqs, I otherwise
279#   --enable-modules=most 
280#
281#   Otherwise                              -> O
282#

755###########   HTTPD MODULES     ############
756SET(LoadModules)
757SET(mods_built_and_loaded)
758SET(mods_built_but_not_loaded)
759SET(mods_omitted)
760FOREACH (mod ${MODULE_PATHS})
761  # Build different forms of the module name; e.g., 
762  #   mod_name->mod_cgi, mod_module_name->cgi_module, mod_shortname->cgi
763  STRING(REGEX REPLACE ".*/(mod_[^\\+]+)" "\\1"        mod_name        ${mod})
764  STRING(REGEX REPLACE "mod_(.*)"         "\\1_module" mod_module_name ${mod_name})
765  STRING(REGEX REPLACE "mod_(.*)"         "\\1"        mod_shortname   ${mod_name})
766
767  # Is it enabled?
768  STRING(TOUPPER "ENABLE_${mod_shortname}" enable_mod)
769  SET(enable_mod_val ${${enable_mod}})
770
771  # Is ENABLE_MODULES set to a higher value?
772  GET_MOD_ENABLE_RANK(${mod_name} ${enable_mod_val} this_mod_rank)
773  IF(this_mod_rank LESS enable_modules_rank)
774    # Use the value from ENABLE_MODULES
775    SET(enable_mod_val ${ENABLE_MODULES})
776  ENDIF()
777
778  IF(NOT ${enable_mod_val} STREQUAL "O") # build of module is desired
779    SET(mod_requires "${mod_name}_requires")
780    STRING(TOUPPER ${enable_mod_val} enable_mod_val_upper)
781    IF(NOT "${${mod_requires}}" STREQUAL "") # module has some prerequisite
782      FOREACH (required ${${mod_requires}})
783        IF(NOT ${required}) # prerequisite doesn't exist
784          IF(NOT ${enable_mod_val} STREQUAL ${enable_mod_val_upper}) # lower case, so optional based on prereq
785            MESSAGE(STATUS "${mod_name} was requested but couldn't be built due to a missing prerequisite (${required})")
786            SET(enable_mod_val_upper "O") # skip due to missing prerequisite
787          ELSE() # must be upper case "A" or "I" (or coding error above)
788            MESSAGE(FATAL_ERROR "${mod_name} was requested but couldn't be built due to a missing prerequisite (${required})")
789          ENDIF()
790        ENDIF()
791      ENDFOREACH()
792    ENDIF()
793    # map a->A, i->I, O->O for remaining logic since prereq checking is over
794    SET(enable_mod_val ${enable_mod_val_upper})
795  ENDIF()
796  
797  IF(${enable_mod_val} STREQUAL "O")
798    # ignore
799    SET(mods_omitted ${mods_omitted} ${mod_name})
800  ELSE()
801    # Handle whether or not the LoadModule is commented out.
802    IF(${enable_mod_val} STREQUAL "A")
803      SET(LoadModules "${LoadModules}LoadModule ${mod_module_name} modules/${mod_name}.so\n")
804      SET(mods_built_and_loaded ${mods_built_and_loaded} ${mod_name})
805    ELSEIF(${enable_mod_val} STREQUAL "I")
806      SET(LoadModules "${LoadModules}# LoadModule ${mod_module_name} modules/${mod_name}.so\n")
807      SET(mods_built_but_not_loaded ${mods_built_but_not_loaded} ${mod_name})
808    ELSE()
809      MESSAGE(FATAL_ERROR "${enable_mod} must be set to \"A\", \"I\", or \"O\" instead of \"${enable_mod_val}\"")
810    ENDIF()
811
812    # Handle building it.
813    SET(mod_main_source "${mod_name}_main_source")
814    SET(mod_extra_sources "${mod_name}_extra_sources")
815
816    IF("${${mod_main_source}}" STREQUAL "")
817      SET(tmp_mod_main_source "${mod}.c")
818    ELSE()
819      SET(tmp_mod_main_source ${${mod_main_source}})
820    ENDIF()
821    SET(all_mod_sources ${tmp_mod_main_source} ${${mod_extra_sources}})
822    ADD_LIBRARY(${mod_name} SHARED ${all_mod_sources} build/win32/httpd.rc)
823    SET(install_modules ${install_modules} ${mod_name})
824    SET(install_modules_pdb ${install_modules_pdb} "$<TARGET_PDB_FILE:${mod_name}>")
825    IF("${${mod_name}_install_lib}")
826      SET(installed_mod_libs_exps
827          ${installed_mod_libs_exps}
828          "$<TARGET_LINKER_FILE:${mod_name}>"
829          "$<TARGET_LINKER_FILE_DIR:${mod_name}>/${mod_name}.exp"
830      )
831    ENDIF()
832    SET(mod_extra_libs "${mod_name}_extra_libs")
833    SET_TARGET_PROPERTIES(${mod_name} PROPERTIES
834      SUFFIX .so
835      LINK_FLAGS /base:@${PROJECT_BINARY_DIR}/BaseAddr.ref,${mod_name}.so
836    )
837    TARGET_LINK_LIBRARIES(${mod_name} ${${mod_extra_libs}} libhttpd ${EXTRA_LIBS} ${APR_LIBRARIES} ${HTTPD_SYSTEM_LIBS})
838    DEFINE_WITH_BLANKS(define_long_name "LONG_NAME" "${mod_name} for Apache HTTP Server")
839    SET_TARGET_PROPERTIES(${mod_name} PROPERTIES COMPILE_FLAGS "${define_long_name} -DBIN_NAME=${mod_name}.so ${EXTRA_COMPILE_FLAGS}")
840
841    # Extra defines?
842    SET(mod_extra_defines "${mod_name}_extra_defines")
843    IF(NOT ${${mod_extra_defines}} STREQUAL "")
844      SET_TARGET_PROPERTIES(${mod_name} PROPERTIES COMPILE_DEFINITIONS ${${mod_extra_defines}})
845    ENDIF()
846
847    # Extra includes?
848    SET(mod_extra_includes "${mod_name}_extra_includes")
849    IF(NOT "${${mod_extra_includes}}" STREQUAL "")
850      SET(tmp_includes ${HTTPD_INCLUDE_DIRECTORIES} ${${mod_extra_includes}})
851      SET_TARGET_PROPERTIES(${mod_name} PROPERTIES INCLUDE_DIRECTORIES "${tmp_includes}")
852      GET_PROPERTY(tmp_includes TARGET ${mod_name} PROPERTY INCLUDE_DIRECTORIES)
853    ENDIF()
854
855  ENDIF()
856ENDFOREACH()
