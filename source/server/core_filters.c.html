58/**
59 * Remove all zero length buckets from the brigade.
60 */ view japanese
61#define BRIGADE_NORMALIZE(b) \
62do { \
63    apr_bucket *e = APR_BRIGADE_FIRST(b); \
64    do {  \
65        if (e->length == 0 && !APR_BUCKET_IS_METADATA(e)) { \
66            apr_bucket *d; \
67            d = APR_BUCKET_NEXT(e); \
68            apr_bucket_delete(e); \
69            e = d; \
70        } \
71        else { \
72            e = APR_BUCKET_NEXT(e); \
73        } \
74    } while (!APR_BRIGADE_EMPTY(b) && (e != APR_BRIGADE_SENTINEL(b))); \
75} while (0)

90apr_status_t ap_core_input_filter(ap_filter_t *f, apr_bucket_brigade *b,
91                                  ap_input_mode_t mode, apr_read_type_e block,
92                                  apr_off_t readbytes)
93{
94    apr_status_t rv;
95    core_net_rec *net = f->ctx;
96    core_ctx_t *ctx = net->in_ctx;
97    const char *str;
98    apr_size_t len;
99
100    if (mode == AP_MODE_INIT) {
101        /*
102         * this mode is for filters that might need to 'initialize'
103         * a connection before reading request data from a client.
104         * NNTP over SSL for example needs to handshake before the
105         * server sends the welcome message.
106         * such filters would have changed the mode before this point
107         * is reached.  however, protocol modules such as NNTP should
108         * not need to know anything about SSL.  given the example, if
109         * SSL is not in the filter chain, AP_MODE_INIT is a noop.
110         */ view japanese
111        return APR_SUCCESS;
112    }
113
114    if (!ctx)
115    {
116        net->in_ctx = ctx = apr_palloc(f->c->pool, sizeof(*ctx));
117        ctx->b = apr_brigade_create(f->c->pool, f->c->bucket_alloc);
118        ctx->tmpbb = apr_brigade_create(f->c->pool, f->c->bucket_alloc);
119        /* seed the brigade with the client socket. */ view japanese
120        rv = ap_run_insert_network_bucket(f->c, ctx->b, net->client_socket);
121        if (rv != APR_SUCCESS)
122            return rv;
123    }
124    else if (APR_BRIGADE_EMPTY(ctx->b)) {
125        return APR_EOF;
126    }
127
128    /* ### This is bad. */ view japanese
129    BRIGADE_NORMALIZE(ctx->b);
130
131    /* check for empty brigade again *AFTER* BRIGADE_NORMALIZE()
132     * If we have lost our socket bucket (see above), we are EOF.
133     *
134     * Ideally, this should be returning SUCCESS with EOS bucket, but
135     * some higher-up APIs (spec. read_request_line via ap_rgetline)
136     * want an error code. */ view japanese
137    if (APR_BRIGADE_EMPTY(ctx->b)) {
138        return APR_EOF;
139    }
140
141    if (mode == AP_MODE_GETLINE) {
142        /* we are reading a single LF line, e.g. the HTTP headers */ view japanese
143        rv = apr_brigade_split_line(b, ctx->b, block, HUGE_STRING_LEN);
144        /* We should treat EAGAIN here the same as we do for EOF (brigade is
145         * empty).  We do this by returning whatever we have read.  This may
146         * or may not be bogus, but is consistent (for now) with EOF logic.
147         */ view japanese
148        if (APR_STATUS_IS_EAGAIN(rv) && block == APR_NONBLOCK_READ) {
149            rv = APR_SUCCESS;
150        }
151        return rv;
152    }
153
154    /* ### AP_MODE_PEEK is a horrific name for this mode because we also
155     * eat any CRLFs that we see.  That's not the obvious intention of
156     * this mode.  Determine whether anyone actually uses this or not. */ view japanese
157    if (mode == AP_MODE_EATCRLF) {
158        apr_bucket *e;
159        const char *c;
160
161        /* The purpose of this loop is to ignore any CRLF (or LF) at the end
162         * of a request.  Many browsers send extra lines at the end of POST
163         * requests.  We use the PEEK method to determine if there is more
164         * data on the socket, so that we know if we should delay sending the
165         * end of one request until we have served the second request in a
166         * pipelined situation.  We don't want to actually delay sending a
167         * response if the server finds a CRLF (or LF), becuause that doesn't
168         * mean that there is another request, just a blank line.
169         */ view japanese
170        while (1) {
171            if (APR_BRIGADE_EMPTY(ctx->b))
172                return APR_EOF;
173
174            e = APR_BRIGADE_FIRST(ctx->b);
175
176            rv = apr_bucket_read(e, &str, &len, APR_NONBLOCK_READ);
177
178            if (rv != APR_SUCCESS)
179                return rv;
180
181            c = str;
182            while (c < str + len) {
183                if (*c == APR_ASCII_LF)
184                    c++;
185                else if (*c == APR_ASCII_CR && *(c + 1) == APR_ASCII_LF)
186                    c += 2;
187                else
188                    return APR_SUCCESS;
189            }
190
191            /* If we reach here, we were a bucket just full of CRLFs, so
192             * just toss the bucket. */ view japanese
193            /* FIXME: Is this the right thing to do in the core? */ view japanese
194            apr_bucket_delete(e);
195        }
196        return APR_SUCCESS;
197    }
198
199    /* If mode is EXHAUSTIVE, we want to just read everything until the end
200     * of the brigade, which in this case means the end of the socket.
201     * To do this, we attach the brigade that has currently been setaside to
202     * the brigade that was passed down, and send that brigade back.
203     *
204     * NOTE:  This is VERY dangerous to use, and should only be done with
205     * extreme caution.  FWLIW, this would be needed by an MPM like Perchild;
206     * such an MPM can easily request the socket and all data that has been
207     * read, which means that it can pass it to the correct child process.
208     */ view japanese
209    if (mode == AP_MODE_EXHAUSTIVE) {
210        apr_bucket *e;
211
212        /* Tack on any buckets that were set aside. */ view japanese
213        APR_BRIGADE_CONCAT(b, ctx->b);
214
215        /* Since we've just added all potential buckets (which will most
216         * likely simply be the socket bucket) we know this is the end,
217         * so tack on an EOS too. */ view japanese
218        /* We have read until the brigade was empty, so we know that we
219         * must be EOS. */ view japanese
220        e = apr_bucket_eos_create(f->c->bucket_alloc);
221        APR_BRIGADE_INSERT_TAIL(b, e);
222        return APR_SUCCESS;
223    }
224
225    /* read up to the amount they specified. */ view japanese
226    if (mode == AP_MODE_READBYTES || mode == AP_MODE_SPECULATIVE) {
227        apr_bucket *e;
228
229        AP_DEBUG_ASSERT(readbytes > 0);
230
231        e = APR_BRIGADE_FIRST(ctx->b);
232        rv = apr_bucket_read(e, &str, &len, block);
233
234        if (APR_STATUS_IS_EAGAIN(rv) && block == APR_NONBLOCK_READ) {
235            /* getting EAGAIN for a blocking read is an error; for a
236             * non-blocking read, return an empty brigade. */ view japanese
237            return APR_SUCCESS;
238        }
239        else if (rv != APR_SUCCESS) {
240            return rv;
241        }
242        else if (block == APR_BLOCK_READ && len == 0) {
243            /* We wanted to read some bytes in blocking mode.  We read
244             * 0 bytes.  Hence, we now assume we are EOS.
245             *
246             * When we are in normal mode, return an EOS bucket to the
247             * caller.
248             * When we are in speculative mode, leave ctx->b empty, so
249             * that the next call returns an EOS bucket.
250             */ view japanese
251            apr_bucket_delete(e);
252
253            if (mode == AP_MODE_READBYTES) {
254                e = apr_bucket_eos_create(f->c->bucket_alloc);
255                APR_BRIGADE_INSERT_TAIL(b, e);
256            }
257            return APR_SUCCESS;
258        }
259
260        /* Have we read as much data as we wanted (be greedy)? */ view japanese
261        if (len < readbytes) {
262            apr_size_t bucket_len;
263
264            rv = APR_SUCCESS;
265            /* We already registered the data in e in len */ view japanese
266            e = APR_BUCKET_NEXT(e);
267            while ((len < readbytes) && (rv == APR_SUCCESS)
268                   && (e != APR_BRIGADE_SENTINEL(ctx->b))) {
269                /* Check for the availability of buckets with known length */ view japanese
270                if (e->length != (apr_size_t)-1) {
271                    len += e->length;
272                    e = APR_BUCKET_NEXT(e);
273                }
274                else {
275                    /*
276                     * Read from bucket, but non blocking. If there isn't any
277                     * more data, well than this is fine as well, we will
278                     * not wait for more since we already got some and we are
279                     * only checking if there isn't more.
280                     */ view japanese
281                    rv = apr_bucket_read(e, &str, &bucket_len,
282                                         APR_NONBLOCK_READ);
283                    if (rv == APR_SUCCESS) {
284                        len += bucket_len;
285                        e = APR_BUCKET_NEXT(e);
286                    }
287                }
288            }
289        }
290
291        /* We can only return at most what we read. */ view japanese
292        if (len < readbytes) {
293            readbytes = len;
294        }
295
296        rv = apr_brigade_partition(ctx->b, readbytes, &e);
297        if (rv != APR_SUCCESS) {
298            return rv;
299        }
300
301        /* Must do move before CONCAT */ view japanese
302        ctx->tmpbb = apr_brigade_split_ex(ctx->b, e, ctx->tmpbb);
303
304        if (mode == AP_MODE_READBYTES) {
305            APR_BRIGADE_CONCAT(b, ctx->b);
306        }
307        else if (mode == AP_MODE_SPECULATIVE) {
308            apr_bucket *copy_bucket;
309
310            for (e = APR_BRIGADE_FIRST(ctx->b);
311                 e != APR_BRIGADE_SENTINEL(ctx->b);
312                 e = APR_BUCKET_NEXT(e))
313            {
314                rv = apr_bucket_copy(e, &copy_bucket);
315                if (rv != APR_SUCCESS) {
316                    return rv;
317                }
318                APR_BRIGADE_INSERT_TAIL(b, copy_bucket);
319            }
320        }
321
322        /* Take what was originally there and place it back on ctx->b */ view japanese
323        APR_BRIGADE_CONCAT(ctx->b, ctx->tmpbb);
324    }
325    return APR_SUCCESS;
326}

360static int should_send_brigade(apr_bucket_brigade *bb, conn_rec *c, int *flush)
361{
362    core_server_config *conf =
363        ap_get_core_module_config(c->base_server->module_config);
364    apr_size_t total_bytes = 0, non_file_bytes = 0;
365    apr_uint32_t eor_buckets = 0;
366    apr_bucket *bucket;
367    int need_flush = 0;
368
369    /* Scan through the brigade and decide whether we need to flush it,
370     * based on the following rules:
371     *
372     *  a) The brigade contains a flush bucket: Do a blocking write
373     *     of everything up that point.
374     *
375     *  b) The request is in CONN_STATE_HANDLER state, and the brigade
376     *     contains at least flush_max_threshold bytes in non-file
377     *     buckets: Do blocking writes until the amount of data in the
378     *     buffer is less than flush_max_threshold.  (The point of this
379     *     rule is to provide flow control, in case a handler is
380     *     streaming out lots of data faster than the data can be
381     *     sent to the client.)
382     *
383     *  c) The request is in CONN_STATE_HANDLER state, and the brigade
384     *     contains at least flush_max_pipelined EOR buckets:
385     *     Do blocking writes until less than flush_max_pipelined EOR
386     *     buckets are left. (The point of this rule is to prevent too many
387     *     FDs being kept open by pipelined requests, possibly allowing a
388     *     DoS).
389     *
390     *  d) The brigade contains a morphing bucket: otherwise ap_save_brigade()
391     *     could read the whole bucket into memory.
392     */ view japanese
393    for (bucket = APR_BRIGADE_FIRST(bb);
394         bucket != APR_BRIGADE_SENTINEL(bb);
395         bucket = APR_BUCKET_NEXT(bucket)) {
396
397        if (!APR_BUCKET_IS_METADATA(bucket)) {
398            if (bucket->length == (apr_size_t)-1) {
399                if (flush) {
400                    ap_log_cerror(APLOG_MARK, APLOG_TRACE6, 0, c,
401                                  "core_output_filter: flushing because "
402                                  "of morphing bucket");
403                }
404                need_flush = 1;
405                break;
406            }
407
408            total_bytes += bucket->length;
409            if (!APR_BUCKET_IS_FILE(bucket)) {
410                non_file_bytes += bucket->length;
411                if (non_file_bytes > conf->flush_max_threshold) {
412                    if (flush) {
413                        ap_log_cerror(APLOG_MARK, APLOG_TRACE6, 0, c,
414                                      "core_output_filter: flushing because "
415                                      "of max threshold");
416                    }
417                    need_flush = 1;
418                    break;
419                }
420            }
421        }
422        else if (APR_BUCKET_IS_FLUSH(bucket)) {
423            if (flush) {
424                ap_log_cerror(APLOG_MARK, APLOG_TRACE6, 0, c,
425                              "core_output_filter: flushing because "
426                              "of FLUSH bucket");
427            }
428            need_flush = 1;
429            break;
430        }
431        else if (AP_BUCKET_IS_EOR(bucket)
432                 && conf->flush_max_pipelined >= 0
433                 && ++eor_buckets > conf->flush_max_pipelined) {
434            if (flush) {
435                ap_log_cerror(APLOG_MARK, APLOG_TRACE6, 0, c,
436                              "core_output_filter: flushing because "
437                              "of max pipelined");
438            }
439            need_flush = 1;
440            break;
441        }
442    }
443    if (flush) {
444        *flush = need_flush;
445    }
446
447    /* Also send if above flush_min_threshold, or if there are FILE buckets */ view japanese
448    return (need_flush
449            || total_bytes >= THRESHOLD_MIN_WRITE
450            || total_bytes > non_file_bytes);
451}

453apr_status_t ap_core_output_filter(ap_filter_t *f, apr_bucket_brigade *new_bb)
454{
455    conn_rec *c = f->c;
456    core_net_rec *net = f->ctx;
457    core_output_filter_ctx_t *ctx = net->out_ctx;
458    apr_bucket_brigade *bb = NULL;
459    apr_status_t rv = APR_SUCCESS;
460
461    /* Fail quickly if the connection has already been aborted. */ view japanese
462    if (c->aborted) {
463        if (new_bb != NULL) {
464            apr_brigade_cleanup(new_bb);
465        }
466        return APR_ECONNABORTED;
467    }
468
469    if (ctx == NULL) {
470        ctx = apr_pcalloc(c->pool, sizeof(*ctx));
471        net->out_ctx = (core_output_filter_ctx_t *)ctx;
472        /*
473         * Need to create buffered_bb brigade with correct lifetime. Passing
474         * NULL to ap_save_brigade() would result in a brigade
475         * allocated from bb->pool which might be wrong.
476         */ view japanese
477        ctx->buffered_bb = apr_brigade_create(c->pool, c->bucket_alloc);
478    }
479
480    if (new_bb != NULL)
481        bb = new_bb;
482
483    if ((ctx->buffered_bb != NULL) &&
484        !APR_BRIGADE_EMPTY(ctx->buffered_bb)) {
485        if (new_bb != NULL) {
486            APR_BRIGADE_PREPEND(bb, ctx->buffered_bb);
487        }
488        else {
489            bb = ctx->buffered_bb;
490        }
491    }
492    else if (new_bb == NULL) {
493        c->data_in_output_filters = 0;
494        return APR_SUCCESS;
495    }
496
497    if (!new_bb || should_send_brigade(bb, c, NULL)) {
498        apr_socket_t *sock = net->client_socket;
499        apr_interval_time_t sock_timeout = 0;
500
501        /* Non-blocking writes on the socket in any case. */ view japanese
502        apr_socket_timeout_get(sock, &sock_timeout);
503        apr_socket_timeout_set(sock, 0);
504
505        do {
506            rv = send_brigade_nonblocking(sock, bb, ctx, c);
507            if (new_bb && APR_STATUS_IS_EAGAIN(rv)) {
508                /* Scan through the brigade and decide whether we must absolutely
509                 * flush the remaining data, based on should_send_brigade() &flush
510                 * rules. If so, wait for writability and retry, otherwise we did
511                 * our best already and can wait for the next call.
512                 */ view japanese
513                int flush;
514                (void)should_send_brigade(bb, c, &flush);
515                if (flush) {
516                    apr_int32_t nfd;
517                    apr_pollfd_t pfd;
518                    memset(&pfd, 0, sizeof(pfd));
519                    pfd.reqevents = APR_POLLOUT;
520                    pfd.desc_type = APR_POLL_SOCKET;
521                    pfd.desc.s = sock;
522                    pfd.p = c->pool;
523                    do {
524                        rv = apr_poll(&pfd, 1, &nfd, sock_timeout);
525                    } while (APR_STATUS_IS_EINTR(rv));
526                }
527            }
528        } while (rv == APR_SUCCESS && !APR_BRIGADE_EMPTY(bb));
529
530        /* Restore original socket timeout before leaving. */ view japanese
531        apr_socket_timeout_set(sock, sock_timeout);
532    }
533
534    if (rv != APR_SUCCESS && !APR_STATUS_IS_EAGAIN(rv)) {
535        /* The client has aborted the connection */ view japanese
536        ap_log_cerror(
537                APLOG_MARK, APLOG_TRACE1, rv, c,
538                "core_output_filter: writing data to the network");
539        /*
540         * Set c->aborted before apr_brigade_cleanup to have the correct status
541         * when logging the request as apr_brigade_cleanup triggers the logging
542         * of the request if it contains an EOR bucket.
543         */ view japanese
544        c->aborted = 1;
545        apr_brigade_cleanup(bb);
546        return rv;
547    }
548
549    setaside_remaining_output(f, ctx, bb, c);
550    return APR_SUCCESS;
551}

553/*
554 * This function assumes that either ctx->buffered_bb == NULL, or
555 * ctx->buffered_bb is empty, or ctx->buffered_bb == bb
556 */ view japanese
557static void setaside_remaining_output(ap_filter_t *f,
558                                      core_output_filter_ctx_t *ctx,
559                                      apr_bucket_brigade *bb,
560                                      conn_rec *c)
561{
562    apr_bucket *bucket;
563
564    /* Don't set aside leading empty buckets, all previous data have been
565     * consumed so it's safe to delete them now.
566     */ view japanese
567    while (((bucket = APR_BRIGADE_FIRST(bb)) != APR_BRIGADE_SENTINEL(bb)) &&
568           (APR_BUCKET_IS_METADATA(bucket) || (bucket->length == 0))) {
569        apr_bucket_delete(bucket);
570    }
571
572    c->data_in_output_filters = 0;
573    if (!APR_BRIGADE_EMPTY(bb)) {
574        c->data_in_output_filters = 1;
575        if (bb != ctx->buffered_bb) {
576            if (!ctx->deferred_write_pool) {
577                apr_pool_create(&ctx->deferred_write_pool, c->pool);
578                apr_pool_tag(ctx->deferred_write_pool, "deferred_write");
579            }
580            ap_save_brigade(f, &(ctx->buffered_bb), &bb,
581                            ctx->deferred_write_pool);
582        }
583    }
584    else if (ctx->deferred_write_pool) {
585        /*
586         * There are no more requests in the pipeline. We can just clear the
587         * pool.
588         */ view japanese
589        apr_pool_clear(ctx->deferred_write_pool);
590    }
591}

605static APR_INLINE int is_in_memory_bucket(apr_bucket *b)
606{
607    /* These buckets' data are already in memory. */ view japanese
608    return APR_BUCKET_IS_HEAP(b)
609           || APR_BUCKET_IS_POOL(b)
610           || APR_BUCKET_IS_TRANSIENT(b)
611           || APR_BUCKET_IS_IMMORTAL(b);
612}

614#if APR_HAS_SENDFILE
615static APR_INLINE int can_sendfile_bucket(apr_bucket *b)
616{

649static apr_status_t send_brigade_nonblocking(apr_socket_t *s,
650                                             apr_bucket_brigade *bb,
651                                             core_output_filter_ctx_t *ctx,
652                                             conn_rec *c)
653{
654    apr_status_t rv = APR_SUCCESS;
655    core_server_config *conf =
656        ap_get_core_module_config(c->base_server->module_config);
657    apr_size_t nvec = 0, nbytes = 0;
658    apr_bucket *bucket, *next;
659    const char *data;
660    apr_size_t length;
661
662    for (bucket = APR_BRIGADE_FIRST(bb);
663         bucket != APR_BRIGADE_SENTINEL(bb);
664         bucket = next) {
665        next = APR_BUCKET_NEXT(bucket);
666
667#if APR_HAS_SENDFILE
668        if (can_sendfile_bucket(bucket)) {
669            if (nvec > 0) {
670                sock_nopush(s, 1);
671                rv = writev_nonblocking(s, bb, ctx, nbytes, nvec, c);
672                if (rv != APR_SUCCESS) {
673                    goto cleanup;
674                }
675                nbytes = 0;
676                nvec = 0;
677            }
678            rv = sendfile_nonblocking(s, bucket, ctx, c);
679            if (rv != APR_SUCCESS) {
680                goto cleanup;
681            }
682            continue;
683        }
684#endif /* APR_HAS_SENDFILE */ view japanese
685
686        if (bucket->length) {
687            /* Non-blocking read first, in case this is a morphing
688             * bucket type. */ view japanese
689            rv = apr_bucket_read(bucket, &data, &length, APR_NONBLOCK_READ);
690            if (APR_STATUS_IS_EAGAIN(rv)) {
691                /* Read would block; flush any pending data and retry. */ view japanese
692                if (nvec) {
693                    rv = writev_nonblocking(s, bb, ctx, nbytes, nvec, c);
694                    if (rv != APR_SUCCESS) {
695                        goto cleanup;
696                    }
697                    nbytes = 0;
698                    nvec = 0;
699                }
700                sock_nopush(s, 0);
701
702                rv = apr_bucket_read(bucket, &data, &length, APR_BLOCK_READ);
703            }
704            if (rv != APR_SUCCESS) {
705                goto cleanup;
706            }
707
708            /* reading may have split the bucket, so recompute next: */ view japanese
709            next = APR_BUCKET_NEXT(bucket);
710        }
711
712        if (!bucket->length) {
713            /* Don't delete empty buckets until all the previous ones have been
714             * sent (nvec == 0); this must happen in sequence since metabuckets
715             * like EOR could free the data still pointed to by the iovec. So
716             * unless the latter is empty, let writev_nonblocking() cleanup the
717             * brigade in order.
718             */ view japanese
719            if (!nvec) {
720                apr_bucket_delete(bucket);
721            }
722        }
723        else {
724            /* Make sure that these new data fit in our iovec. */ view japanese
725            if (nvec == ctx->nvec) {
726                if (nvec == NVEC_MAX) {
727                    sock_nopush(s, 1);
728                    rv = writev_nonblocking(s, bb, ctx, nbytes, nvec, c);
729                    if (rv != APR_SUCCESS) {
730                        goto cleanup;
731                    }
732                    nbytes = 0;
733                    nvec = 0;
734                }
735                else {
736                    struct iovec *newvec;
737                    apr_size_t newn = nvec * 2;
738                    if (newn < NVEC_MIN) {
739                        newn = NVEC_MIN;
740                    }
741                    else if (newn > NVEC_MAX) {
742                        newn = NVEC_MAX;
743                    }
744                    newvec = apr_palloc(c->pool, newn * sizeof(struct iovec));
745                    if (nvec) {
746                        memcpy(newvec, ctx->vec, nvec * sizeof(struct iovec));
747                    }
748                    ctx->vec = newvec;
749                    ctx->nvec = newn;
750                }
751            }
752            nbytes += length;
753            ctx->vec[nvec].iov_base = (void *)data;
754            ctx->vec[nvec].iov_len = length;
755            nvec++;
756        }
757
758        /* Flush above max threshold, unless the brigade still contains in
759         * memory buckets which we want to try writing in the same pass (if
760         * we are at the end of the brigade, the write will happen outside
761         * the loop anyway).
762         */ view japanese
763        if (nbytes > conf->flush_max_threshold
764                && next != APR_BRIGADE_SENTINEL(bb)
765                && next->length && !is_in_memory_bucket(next)) {
766            sock_nopush(s, 1);
767            rv = writev_nonblocking(s, bb, ctx, nbytes, nvec, c);
768            if (rv != APR_SUCCESS) {
769                goto cleanup;
770            }
771            nbytes = 0;
772            nvec = 0;
773        }
774    }
775    if (nvec > 0) {
776        rv = writev_nonblocking(s, bb, ctx, nbytes, nvec, c);
777    }
778
779cleanup:
780    sock_nopush(s, 0);
781    return rv;
782}

784static apr_status_t writev_nonblocking(apr_socket_t *s,
785                                       apr_bucket_brigade *bb,
786                                       core_output_filter_ctx_t *ctx,
787                                       apr_size_t bytes_to_write,
788                                       apr_size_t nvec,
789                                       conn_rec *c)
790{
791    apr_status_t rv;
792    struct iovec *vec = ctx->vec;
793    apr_size_t bytes_written = 0;
794    apr_size_t i, offset = 0;
795
796    do {
797        apr_size_t n = 0;
798        rv = apr_socket_sendv(s, vec + offset, nvec - offset, &n);
799        bytes_written += n;
800
801        for (i = offset; i < nvec; ) {
802            apr_bucket *bucket = APR_BRIGADE_FIRST(bb);
803            if (!bucket->length) {
804                apr_bucket_delete(bucket);
805            }
806            else if (n >= vec[i].iov_len) {
807                apr_bucket_delete(bucket);
808                n -= vec[i++].iov_len;
809                offset++;
810            }
811            else {
812                if (n) {
813                    apr_bucket_split(bucket, n);
814                    apr_bucket_delete(bucket);
815                    vec[i].iov_len -= n;
816                    vec[i].iov_base = (char *) vec[i].iov_base + n;
817                }
818                break;
819            }
820        }
821    } while (rv == APR_SUCCESS && bytes_written < bytes_to_write);
822
823    if ((ap__logio_add_bytes_out != NULL) && (bytes_written > 0)) {
824        ap__logio_add_bytes_out(c, bytes_written);
825    }
826    ctx->bytes_written += bytes_written;
827
828    ap_log_cerror(APLOG_MARK, APLOG_TRACE6, rv, c,
829                  "writev_nonblocking: %"APR_SIZE_T_FMT"/%"APR_SIZE_T_FMT,
830                  bytes_written, bytes_to_write);
831    return rv;
832}
