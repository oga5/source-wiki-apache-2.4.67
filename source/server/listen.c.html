69static apr_status_t make_sock(apr_pool_t *p, ap_listen_rec *server, int do_bind_listen)
70{
71    apr_socket_t *s = server->sd;
72    int one = 1;
73#if APR_HAVE_IPV6
74#ifdef AP_ENABLE_V4_MAPPED
75    int v6only_setting = 0;
76#else
77    int v6only_setting = 1;
78#endif
79#endif
80    apr_status_t stat;
81
82#ifndef WIN32
83    stat = apr_socket_opt_set(s, APR_SO_REUSEADDR, one);
84    if (stat != APR_SUCCESS && stat != APR_ENOTIMPL) {
85        ap_log_perror(APLOG_MARK, APLOG_CRIT, stat, p, APLOGNO(00067)
86                      "make_sock: for address %pI, apr_socket_opt_set: (SO_REUSEADDR)",
87                      server->bind_addr);
88        apr_socket_close(s);
89        return stat;
90    }
91#endif
92
93    stat = apr_socket_opt_set(s, APR_SO_KEEPALIVE, one);
94    if (stat != APR_SUCCESS && stat != APR_ENOTIMPL) {
95        ap_log_perror(APLOG_MARK, APLOG_CRIT, stat, p, APLOGNO(00068)
96                      "make_sock: for address %pI, apr_socket_opt_set: (SO_KEEPALIVE)",
97                      server->bind_addr);
98        apr_socket_close(s);
99        return stat;
100    }
101
102    /*
103     * To send data over high bandwidth-delay connections at full
104     * speed we must force the TCP window to open wide enough to keep the
105     * pipe full.  The default window size on many systems
106     * is only 4kB.  Cross-country WAN connections of 100ms
107     * at 1Mb/s are not impossible for well connected sites.
108     * If we assume 100ms cross-country latency,
109     * a 4kB buffer limits throughput to 40kB/s.
110     *
111     * To avoid this problem I've added the SendBufferSize directive
112     * to allow the web master to configure send buffer size.
113     *
114     * The trade-off of larger buffers is that more kernel memory
115     * is consumed.  YMMV, know your customers and your network!
116     *
117     * -John Heidemann <johnh@isi.edu> 25-Oct-96
118     *
119     * If no size is specified, use the kernel default.
120     */ view japanese
121    if (send_buffer_size) {
122        stat = apr_socket_opt_set(s, APR_SO_SNDBUF,  send_buffer_size);
123        if (stat != APR_SUCCESS && stat != APR_ENOTIMPL) {
124            ap_log_perror(APLOG_MARK, APLOG_WARNING, stat, p, APLOGNO(00070)
125                          "make_sock: failed to set SendBufferSize for "
126                          "address %pI, using default",
127                          server->bind_addr);
128            /* not a fatal error */
129        }
130    }
131    if (receive_buffer_size) {
132        stat = apr_socket_opt_set(s, APR_SO_RCVBUF, receive_buffer_size);
133        if (stat != APR_SUCCESS && stat != APR_ENOTIMPL) {
134            ap_log_perror(APLOG_MARK, APLOG_WARNING, stat, p, APLOGNO(00071)
135                          "make_sock: failed to set ReceiveBufferSize for "
136                          "address %pI, using default",
137                          server->bind_addr);
138            /* not a fatal error */ view japanese
139        }
140    }
141
142#if APR_TCP_NODELAY_INHERITED
143    ap_sock_disable_nagle(s);
144#endif
145
146#if defined(SO_REUSEPORT)
147    if (ap_have_so_reuseport && ap_listencbratio > 0) {
148        int thesock;
149        apr_os_sock_get(&thesock, s);
150        if (setsockopt(thesock, SOL_SOCKET, SO_REUSEPORT,
151                       (void *)&one, sizeof(int)) < 0) {
152            stat = apr_get_netos_error();
153            ap_log_perror(APLOG_MARK, APLOG_CRIT, stat, p, APLOGNO(02638)
154                          "make_sock: for address %pI, apr_socket_opt_set: "
155                          "(SO_REUSEPORT)",
156                          server->bind_addr);
157            apr_socket_close(s);
158            return stat;
159        }
160    }
161#endif
162
163    if (do_bind_listen) {
164#if APR_HAVE_IPV6
165        if (server->bind_addr->family == APR_INET6) {
166            stat = apr_socket_opt_set(s, APR_IPV6_V6ONLY, v6only_setting);
167            if (stat != APR_SUCCESS && stat != APR_ENOTIMPL) {
168                ap_log_perror(APLOG_MARK, APLOG_CRIT, stat, p, APLOGNO(00069)
169                              "make_sock: for address %pI, apr_socket_opt_set: "
170                              "(IPV6_V6ONLY)",
171                              server->bind_addr);
172                apr_socket_close(s);
173                return stat;
174            }
175        }
176#endif
177
178        if ((stat = apr_socket_bind(s, server->bind_addr)) != APR_SUCCESS) {
179            ap_log_perror(APLOG_MARK, APLOG_STARTUP|APLOG_CRIT, stat, p, APLOGNO(00072)
180                          "make_sock: could not bind to address %pI",
181                          server->bind_addr);
182            apr_socket_close(s);
183            return stat;
184        }
185
186        if ((stat = apr_socket_listen(s, ap_listenbacklog)) != APR_SUCCESS) {
187            ap_log_perror(APLOG_MARK, APLOG_STARTUP|APLOG_ERR, stat, p, APLOGNO(00073)
188                          "make_sock: unable to listen for connections "
189                          "on address %pI",
190                          server->bind_addr);
191            apr_socket_close(s);
192            return stat;
193        }
194    }
195
196#ifdef WIN32
197    /* I seriously doubt that this would work on Unix; I have doubts that
198     * it entirely solves the problem on Win32.  However, since setting
199     * reuseaddr on the listener -prior- to binding the socket has allowed
200     * us to attach to the same port as an already running instance of
201     * Apache, or even another web server, we cannot identify that this
202     * port was exclusively granted to this instance of Apache.
203     *
204     * So set reuseaddr, but do not attempt to do so until we have the
205     * parent listeners successfully bound.
206     */ view japanese
207    stat = apr_socket_opt_set(s, APR_SO_REUSEADDR, one);
208    if (stat != APR_SUCCESS && stat != APR_ENOTIMPL) {
209        ap_log_perror(APLOG_MARK, APLOG_CRIT, stat, p, APLOGNO(00074)
210                    "make_sock: for address %pI, apr_socket_opt_set: (SO_REUSEADDR)",
211                     server->bind_addr);
212        apr_socket_close(s);
213        return stat;
214    }
215#endif
216
217    server->sd = s;
218    server->active = 1;
219
220    server->accept_func = NULL;
221
222    return APR_SUCCESS;
223}

225static const char* find_accf_name(server_rec *s, const char *proto)
226{
227    const char* accf;
228    core_server_config *conf = ap_get_core_module_config(s->module_config);
229    if (!proto) {
230        return NULL;
231    }
232
233    accf = apr_table_get(conf->accf_map, proto);
234
235    if (accf && !strcmp("none", accf)) {
236        return NULL;
237    }
238
239    return accf;
240}

242static void ap_apply_accept_filter(apr_pool_t *p, ap_listen_rec *lis,
243                                           server_rec *server)
244{
245    apr_socket_t *s = lis->sd;
246    const char *accf;
247    apr_status_t rv;
248    const char *proto;
249
250    proto = lis->protocol;
251
252    if (!proto) {
253        proto = ap_get_server_protocol(server);
254    }
255
256
257    accf = find_accf_name(server, proto);
258
259    if (accf) {
260#if APR_HAS_SO_ACCEPTFILTER
261        /* In APR 1.x, the 2nd and 3rd parameters are char * instead of 
262         * const char *, so make a copy of those args here.
263         */ view japanese
264        rv = apr_socket_accept_filter(s, apr_pstrdup(p, accf),
265                                      apr_pstrdup(p, ""));
266        if (rv != APR_SUCCESS && !APR_STATUS_IS_ENOTIMPL(rv)) {
267            ap_log_perror(APLOG_MARK, APLOG_WARNING, rv, p, APLOGNO(00075)
268                          "Failed to enable the '%s' Accept Filter",
269                          accf);
270        }
271#else
272        rv = apr_socket_opt_set(s, APR_TCP_DEFER_ACCEPT, ap_listentcpdeferaccept);
273        if (rv != APR_SUCCESS && !APR_STATUS_IS_ENOTIMPL(rv)) {
274            ap_log_perror(APLOG_MARK, APLOG_WARNING, rv, p, APLOGNO(00076)
275                              "Failed to enable APR_TCP_DEFER_ACCEPT");
276        }
277#endif
278    }
279}

289static apr_status_t alloc_systemd_listener(process_rec * process,
290                                           int fd, const char *proto,
291                                           ap_listen_rec **out_rec)
292{
293    apr_status_t rv;
294    struct sockaddr sa;
295    socklen_t len = sizeof(struct sockaddr);
296    apr_os_sock_info_t si;
297    ap_listen_rec *rec;
298    *out_rec = NULL;
299
300    memset(&si, 0, sizeof(si));
301
302    rv = getsockname(fd, &sa, &len);
303
304    if (rv != 0) {
305        rv = apr_get_netos_error();
306        ap_log_perror(APLOG_MARK, APLOG_CRIT, rv, process->pool, APLOGNO(02489)
307                      "getsockname on %d failed.", fd);
308        return rv;
309    }
310
311    si.os_sock = &fd;
312    si.family = sa.sa_family;
313    si.local = &sa;
314    si.type = SOCK_STREAM;
315    si.protocol = APR_PROTO_TCP;
316
317    rec = apr_pcalloc(process->pool, sizeof(ap_listen_rec));
318
319
320    rv = apr_os_sock_make(&rec->sd, &si, process->pool);
321    if (rv != APR_SUCCESS) {
322        ap_log_perror(APLOG_MARK, APLOG_CRIT, rv, process->pool, APLOGNO(02490)
323                      "apr_os_sock_make on %d failed.", fd);
324        return rv;
325    }
326
327    rv = apr_socket_addr_get(&rec->bind_addr, APR_LOCAL, rec->sd);
328    if (rv != APR_SUCCESS) {
329        ap_log_perror(APLOG_MARK, APLOG_CRIT, rv, process->pool, APLOGNO(02491)
330                      "apr_socket_addr_get on %d failed.", fd);
331        return rv;
332    }
333
334    rec->protocol = apr_pstrdup(process->pool, proto);
335
336    *out_rec = rec;
337
338    return make_sock(process->pool, rec, 0);
339}

341static const char *set_systemd_listener(process_rec *process, apr_port_t port,
342                                        const char *proto)
343{
344    ap_listen_rec *last, *new;
345    apr_status_t rv;
346    APR_OPTIONAL_FN_TYPE(ap_find_systemd_socket) *find_systemd_socket;
347    int fd;
348
349    find_systemd_socket = APR_RETRIEVE_OPTIONAL_FN(ap_find_systemd_socket);
350
351    if (!find_systemd_socket)
352       return "Systemd socket activation is used, but mod_systemd is probably "
353               "not loaded";
354
355    fd = find_systemd_socket(process, port);
356    if (fd < 0) {
357        return "Systemd socket activation is used, but this port is not "
358                "configured in systemd";
359    }
360
361    last = ap_listeners;
362    while (last && last->next) {
363        last = last->next;
364    }
365
366    rv = alloc_systemd_listener(process, fd, proto, &new);
367    if (rv != APR_SUCCESS) {
368        return "Failed to setup socket passed by systemd using socket activation";
369    }
370
371    if (last == NULL) {
372        ap_listeners = new;
373    }
374    else {
375        last->next = new;
376    }
377
378    return NULL;
379}

382/* Returns non-zero if socket address SA matches hostname, port and
383 * scope_id.  p is used for temporary allocations. */ view japanese
384static int match_address(const apr_sockaddr_t *sa,
385                         const char *hostname, apr_port_t port,
386                         const char *scope_id, apr_pool_t *p)
387{
388    const char *old_scope = NULL;
389
390#if APR_VERSION_AT_LEAST(1,7,0)
391    /* To be clever here we could correctly match numeric and
392     * non-numeric zone ids.  Ignore failure, old_scope will be left
393     * as NULL. */ view japanese
394    (void) apr_sockaddr_zone_get(sa, &old_scope, NULL, p);
395#endif
396    
397    return port == sa->port
398        && ((!hostname && !sa->hostname)
399            || (hostname && sa->hostname && !strcmp(sa->hostname, hostname)))
400        && ((!scope_id && !old_scope)
401            || (scope_id && old_scope && !strcmp(scope_id, old_scope)));            
402}

404/* ### This logic doesn't cope with DNS changes across a restart. */ view japanese
405static int find_listeners(ap_listen_rec **from, ap_listen_rec **to,
406                          const char *addr, apr_port_t port,
407                          const char *scope_id, apr_pool_t *temp_pool)
408{
409    int found = 0;
410
411    while (*from) {
412        apr_sockaddr_t *sa = (*from)->bind_addr;
413
414        /* Some listeners are not real so they will not have a bind_addr. */ view japanese
415        if (sa) {
416            ap_listen_rec *new;
417
418            /* Re-use the existing record if it matches completely
419             * against an existing listener. */ view japanese
420            if (match_address(sa, addr, port, scope_id, temp_pool)) {
421                found = 1;
422                if (!to) {
423                    break;
424                }
425                new = *from;
426                *from = new->next;
427                new->next = *to;
428                *to = new;
429                continue;
430            }
431        }
432
433        from = &(*from)->next;
434    }
435
436    return found;
437}

439static const char *alloc_listener(process_rec *process, const char *addr,
440                                  apr_port_t port, const char* proto,
441                                  const char *scope_id, void *slave,
442                                  apr_pool_t *temp_pool)
443{
444    ap_listen_rec *last;
445    apr_status_t status;
446    apr_sockaddr_t *sa;
447
448    /* see if we've got a listener for this address:port, which is an error */ view japanese
449    if (find_listeners(&ap_listeners, NULL, addr, port, scope_id, temp_pool)) {
450        return "Cannot define multiple Listeners on the same IP:port";
451    }
452
453    /* see if we've got an old listener for this address:port */ view japanese
454    if (find_listeners(&old_listeners, &ap_listeners, addr, port,
455                       scope_id, temp_pool)) {
456        if (ap_listeners->slave != slave) {
457            return "Cannot define a slave on the same IP:port as a Listener";
458        }
459        return NULL;
460    }
461
462    if ((status = apr_sockaddr_info_get(&sa, addr, APR_UNSPEC, port, 0,
463                                        process->pool))
464        != APR_SUCCESS) {
465        ap_log_perror(APLOG_MARK, APLOG_CRIT, status, process->pool, APLOGNO(00077)
466                      "alloc_listener: failed to set up sockaddr for %s",
467                      addr);
468        return "Listen setup failed";
469    }
470
471    /* Initialize to our last configured ap_listener. */ view japanese
472    last = ap_listeners;
473    while (last && last->next) {
474        last = last->next;
475    }
476
477    while (sa) {
478        ap_listen_rec *new;
479
480        /* this has to survive restarts */ view japanese
481        new = apr_palloc(process->pool, sizeof(ap_listen_rec));
482        new->active = 0;
483        new->next = 0;
484        new->bind_addr = sa;
485        new->protocol = apr_pstrdup(process->pool, proto);
486
487        /* Go to the next sockaddr. */ view japanese
488        sa = sa->next;
489
490        status = apr_socket_create(&new->sd, new->bind_addr->family,
491                                    SOCK_STREAM, 0, process->pool);
492
493#if APR_HAVE_IPV6
494        /* What could happen is that we got an IPv6 address, but this system
495         * doesn't actually support IPv6.  Try the next address.
496         */ view japanese
497        if (status != APR_SUCCESS && !addr &&
498            new->bind_addr->family == APR_INET6) {
499            continue;
500        }
501#endif
502        if (status != APR_SUCCESS) {
503            ap_log_perror(APLOG_MARK, APLOG_CRIT, status, process->pool, APLOGNO(00078)
504                          "alloc_listener: failed to get a socket for %s",
505                          addr);
506            return "Listen setup failed";
507        }
508
509#if APR_VERSION_AT_LEAST(1,7,0)
510        if (scope_id) {
511            status = apr_sockaddr_zone_set(new->bind_addr, scope_id);
512            if (status) {
513                ap_log_perror(APLOG_MARK, APLOG_CRIT, status, process->pool, APLOGNO(10102)
514                              "alloc_listener: failed to set scope for %pI to %s",
515                              new->bind_addr, scope_id);
516                return "Listen step failed";
517            }
518        }
519#endif
520
521        /* We need to preserve the order returned by getaddrinfo() */ view japanese
522        if (last == NULL) {
523            ap_listeners = last = new;
524        } else {
525            last->next = new;
526            last = new;
527        }
528        new->slave = slave;
529    }
530
531    return NULL;
532}
533/* Evaluates to true if the (apr_sockaddr_t *) addr argument is the
534 * IPv4 match-any-address, 0.0.0.0. */ view japanese
535#define IS_INADDR_ANY(addr) ((addr)->family == APR_INET \
536                             && (addr)->sa.sin.sin_addr.s_addr == INADDR_ANY)
537
538/* Evaluates to true if the (apr_sockaddr_t *) addr argument is the
539 * IPv6 match-any-address, [::]. */ view japanese
540#define IS_IN6ADDR_ANY(addr) ((addr)->family == APR_INET6 \
541                              && IN6_IS_ADDR_UNSPECIFIED(&(addr)->sa.sin6.sin6_addr))
542
543/**
544 * Create, open, listen, and bind all sockets.
545 * @param process The process record for the currently running server
546 * @return The number of open sockets
547 */ view japanese
548static int open_listeners(apr_pool_t *pool)
549{
550    ap_listen_rec *lr;
551    ap_listen_rec *next;
552    ap_listen_rec *previous;
553    int num_open;
554    const char *userdata_key = "ap_open_listeners";
555    void *data;
556#if AP_NONBLOCK_WHEN_MULTI_LISTEN
557    int use_nonblock;
558#endif
559
560    /* Don't allocate a default listener.  If we need to listen to a
561     * port, then the user needs to have a Listen directive in their
562     * config file.
563     */ view japanese
564    num_open = 0;
565    previous = NULL;
566    for (lr = ap_listeners; lr; previous = lr, lr = lr->next) {
567        if (lr->active) {
568            ++num_open;
569        }
570        else {
571#if APR_HAVE_IPV6
572            ap_listen_rec *cur;
573            int v6only_setting;
574            int skip = 0;
575
576            /* If we have the unspecified IPv4 address (0.0.0.0) and
577             * the unspecified IPv6 address (::) is next, we need to
578             * swap the order of these in the list. We always try to
579             * bind to IPv6 first, then IPv4, since an IPv6 socket
580             * might be able to receive IPv4 packets if V6ONLY is not
581             * enabled, but never the other way around.
582             * Note: In some configurations, the unspecified IPv6 address
583             * could be even later in the list.  This logic only corrects
584             * the situation where it is next in the list, such as when
585             * apr_sockaddr_info_get() returns an IPv4 and an IPv6 address,
586             * in that order.
587             */ view japanese
588            if (lr->next != NULL
589                && IS_INADDR_ANY(lr->bind_addr)
590                && lr->bind_addr->port == lr->next->bind_addr->port
591                && IS_IN6ADDR_ANY(lr->next->bind_addr)) {
592                /* Exchange lr and lr->next */ view japanese
593                next = lr->next;
594                lr->next = next->next;
595                next->next = lr;
596                if (previous) {
597                    previous->next = next;
598                }
599                else {
600                    ap_listeners = next;
601                }
602                lr = next;
603            }
604
605            /* If we are trying to bind to 0.0.0.0 and a previous listener
606             * was :: on the same port and in turn that socket does not have
607             * the IPV6_V6ONLY flag set; we must skip the current attempt to
608             * listen (which would generate an error). IPv4 will be handled
609             * on the established IPv6 socket.
610             */ view japanese
611            if (IS_INADDR_ANY(lr->bind_addr) && previous) {
612                for (cur = ap_listeners; cur != lr; cur = cur->next) {
613                    if (lr->bind_addr->port == cur->bind_addr->port
614                        && IS_IN6ADDR_ANY(cur->bind_addr)
615                        && apr_socket_opt_get(cur->sd, APR_IPV6_V6ONLY,
616                                              &v6only_setting) == APR_SUCCESS
617                        && v6only_setting == 0) {
618
619                        /* Remove the current listener from the list */
620                        previous->next = lr->next;
621                        lr = previous; /* maintain current value of previous after
622                                        * post-loop expression is evaluated
623                                        */ view japanese
624                        skip = 1;
625                        break;
626                    }
627                }
628                if (skip) {
629                    continue;
630                }
631            }
632#endif
633            if (make_sock(pool, lr, 1) == APR_SUCCESS) {
634                ++num_open;
635            }
636            else {
637#if APR_HAVE_IPV6
638                /* If we tried to bind to ::, and the next listener is

873                }

876#if AP_NONBLOCK_WHEN_MULTI_LISTEN

878            stat = apr_socket_opt_set(duplr->sd, APR_SO_NONBLOCK, use_nonblock);

881                              "unable to control socket non-blocking status");
882                return stat;
883            }
884#endif
885            ap_apply_accept_filter(p, duplr, s);
886
887            if (last == NULL) {
888                (*buckets)[i] = last = duplr;
889            }
890            else {
891                last->next = duplr;

893            }

895        }

server/listen.c:956-969

/* 現時点では、実装がすべてのリスニングスレッド/プロセス間で接続を均等に分散するため、
         * SO_REUSEPORT が定義されている（つまり 3.9 以降の）Linux に限定しています。
         *
         * *BSD にも SO_REUSEPORT はありますが、セマンティクスが異なります。
         * 最初にバインドされたワイルドカードアドレスのソケット、または最後にバインドされた
         * 非ワイルドカードアドレスのソケットのみが接続を受け取ります（均等性は保証されません）。
         * 同じポートにバインドされた残りのソケットは接続を受け取りません。
         * これは httpd では（常に）機能するとは限りません。
         *
         * TODO: 最新の DragonFlyBSD の SO_REUSEPORT は Linux と同じセマンティクスを
         * 持っている（ようです）。そのため、いつか configure.in から HAVE_SO_REUSEPORT を
         * 利用できるようにする必要があるかもしれません。
         */

Original comment

/* This is limited to Linux with defined SO_REUSEPORT (ie. 3.9+) for
         * now since the implementation evenly distributes connections across
         * all the listening threads/processes.
         *
         * *BSDs have SO_REUSEPORT too but with a different semantic: the first
         * wildcard address bound socket or the last non-wildcard address bound
         * socket will receive connections (no evenness guarantee); the rest of
         * the sockets bound to the same port will not.
         * This can't (always) work for httpd.
         *
         * TODO: latests DragonFlyBSD's SO_REUSEPORT (seems to?) have the same
         * semantic as Linux, so we may need HAVE_SO_REUSEPORT available from
         * configure.in some day.
         */
