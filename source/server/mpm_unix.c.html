66typedef enum {
67    DO_NOTHING,
68    SEND_SIGTERM,
69    SEND_SIGTERM_NOLOG,
70    SEND_SIGKILL,
71    GIVEUP
72} action_t;

74typedef struct extra_process_t {
75    struct extra_process_t *next;
76    pid_t pid;
77    ap_generation_t gen;
78} extra_process_t;
79
80static extra_process_t *extras;
81
82AP_DECLARE(void) ap_register_extra_mpm_process(pid_t pid, ap_generation_t gen)
83{
84    extra_process_t *p = (extra_process_t *)ap_malloc(sizeof(extra_process_t));
85
86    p->next = extras;
87    p->pid = pid;
88    p->gen = gen;
89    extras = p;
90}

92AP_DECLARE(int) ap_unregister_extra_mpm_process(pid_t pid, ap_generation_t *old_gen)
93{
94    extra_process_t *cur = extras;
95    extra_process_t *prev = NULL;
96
97    while (cur && cur->pid != pid) {
98        prev = cur;
99        cur = cur->next;
100    }
101
102    if (cur) {
103        if (prev) {
104            prev->next = cur->next;
105        }
106        else {
107            extras = cur->next;
108        }
109        *old_gen = cur->gen;
110        free(cur);
111        return 1; /* found */ view japanese
112    }
113    else {
114        /* we don't know about any such process */ view japanese
115        return 0;
116    }
117}

119static int reclaim_one_pid(pid_t pid, action_t action)
120{
121    apr_proc_t proc;
122    apr_status_t waitret;
123    apr_exit_why_e why;
124    int status;
125
126    /* Ensure pid sanity. */
127    if (pid < 1) {
128        return 1;
129    }
130
131    proc.pid = pid;
132    waitret = apr_proc_wait(&proc, &status, &why, APR_NOWAIT);
133    if (waitret != APR_CHILD_NOTDONE) {
134        if (waitret == APR_CHILD_DONE)
135            ap_process_child_status(&proc, why, status);
136        return 1;
137    }
138
139    switch(action) {
140    case DO_NOTHING:
141        break;
142
143    case SEND_SIGTERM:
144        /* ok, now it's being annoying */ view japanese
145        ap_log_error(APLOG_MARK, APLOG_WARNING,
146                     0, ap_server_conf, APLOGNO(00045)
147                     "child process %" APR_PID_T_FMT
148                     " still did not exit, "
149                     "sending a SIGTERM",
150                     pid);
151        /* FALLTHROUGH */ view japanese
152    case SEND_SIGTERM_NOLOG:
153        kill(pid, SIGTERM);
154        break;
155
156    case SEND_SIGKILL:
157        ap_log_error(APLOG_MARK, APLOG_ERR,
158                     0, ap_server_conf, APLOGNO(00046)
159                     "child process %" APR_PID_T_FMT
160                     " still did not exit, "
161                     "sending a SIGKILL",
162                     pid);
163        kill(pid, SIGKILL);
164        break;
165
166    case GIVEUP:
167        /* gave it our best shot, but alas...  If this really
168         * is a child we are trying to kill and it really hasn't
169         * exited, we will likely fail to bind to the port
170         * after the restart.
171         */ view japanese
172        ap_log_error(APLOG_MARK, APLOG_ERR,
173                     0, ap_server_conf, APLOGNO(00047)
174                     "could not make child process %" APR_PID_T_FMT
175                     " exit, "
176                     "attempting to continue anyway",
177                     pid);
178        break;
179    }
180
181    return 0;
182}

184AP_DECLARE(void) ap_reclaim_child_processes(int terminate,
185                                            ap_reclaim_callback_fn_t *mpm_callback)
186{
187    apr_time_t waittime = 1024 * 16;
188    int i;
189    extra_process_t *cur_extra;
190    int not_dead_yet;
191    int max_daemons;
192    apr_time_t starttime = apr_time_now();
193    /* this table of actions and elapsed times tells what action is taken
194     * at which elapsed time from starting the reclaim
195     */ view japanese
196    struct {
197        action_t action;
198        apr_time_t action_time;
199    } action_table[] = {
200        {DO_NOTHING, 0}, /* dummy entry for iterations where we reap
201                          * children but take no action against
202                          * stragglers
203                          */ view japanese
204        {SEND_SIGTERM_NOLOG, 0}, /* skipped if terminate == 0 */ view japanese
205        {SEND_SIGTERM, apr_time_from_sec(3)},
206        {SEND_SIGTERM, apr_time_from_sec(5)},
207        {SEND_SIGTERM, apr_time_from_sec(7)},
208        {SEND_SIGKILL, apr_time_from_sec(9)},
209        {GIVEUP,       apr_time_from_sec(10)}
210    };
211    int cur_action;      /* index of action we decided to take this
212                          * iteration
213                          */ view japanese
214    int next_action = terminate ? 1 : 2; /* index of first real action */ view japanese
215
216    ap_mpm_query(AP_MPMQ_MAX_DAEMON_USED, &max_daemons);
217
218    do {
219        if (action_table[next_action].action_time > 0) {
220            apr_sleep(waittime);
221            /* don't let waittime get longer than 1 second; otherwise, we don't
222             * react quickly to the last child exiting, and taking action can
223             * be delayed
224             */ view japanese
225            waittime = waittime * 4;
226            if (waittime > apr_time_from_sec(1)) {
227                waittime = apr_time_from_sec(1);
228            }
229        }
230
231        /* see what action to take, if any */ view japanese
232        if (action_table[next_action].action_time <= apr_time_now() - starttime) {
233            cur_action = next_action;
234            ++next_action;
235        }
236        else {
237            cur_action = 0; /* nothing to do */ view japanese
238        }
239
240        /* now see who is done */
241        not_dead_yet = 0;
242        for (i = 0; i < max_daemons; ++i) {
243            process_score *ps = ap_get_scoreboard_process(i);
244            pid_t pid = ps->pid;
245
246            if (pid == 0) {
247                continue; /* not every scoreboard entry is in use */
248            }
249
250            if (reclaim_one_pid(pid, action_table[cur_action].action)) {
251                mpm_callback(i, 0, 0);
252            }
253            else {
254                ++not_dead_yet;
255            }
256        }
257
258        cur_extra = extras;
259        while (cur_extra) {
260            ap_generation_t old_gen;
261            extra_process_t *next = cur_extra->next;
262            pid_t pid = cur_extra->pid;
263
264            if (reclaim_one_pid(pid, action_table[cur_action].action)) {
265                if (ap_unregister_extra_mpm_process(pid, &old_gen) == 1) {
266                    /* cur_extra dangling pointer from here. */
267                    mpm_callback(-1, pid, old_gen);
268                }
269                else {
270                    AP_DEBUG_ASSERT(1 == 0);
271                }
272            }
273            else {
274                ++not_dead_yet;
275            }
276            cur_extra = next;
277        }
278#if APR_HAS_OTHER_CHILD
279        apr_proc_other_child_refresh_all(APR_OC_REASON_RESTART);
280#endif
281
282    } while (not_dead_yet > 0 &&
283             action_table[cur_action].action != GIVEUP);
284}

286AP_DECLARE(void) ap_relieve_child_processes(ap_reclaim_callback_fn_t *mpm_callback)
287{
288    int i;
289    extra_process_t *cur_extra;
290    int max_daemons;
291
292    ap_mpm_query(AP_MPMQ_MAX_DAEMON_USED, &max_daemons);
293
294    /* now see who is done */ view japanese
295    for (i = 0; i < max_daemons; ++i) {
296        process_score *ps = ap_get_scoreboard_process(i);
297        pid_t pid = ps->pid;
298
299        if (pid == 0) {
300            continue; /* not every scoreboard entry is in use */ view japanese
301        }
302
303        if (reclaim_one_pid(pid, DO_NOTHING)) {
304            mpm_callback(i, 0, 0);
305        }
306    }
307
308    cur_extra = extras;
309    while (cur_extra) {
310        ap_generation_t old_gen;
311        extra_process_t *next = cur_extra->next;
312        pid_t pid = cur_extra->pid;
313
314        if (reclaim_one_pid(pid, DO_NOTHING)) {
315            if (ap_unregister_extra_mpm_process(pid, &old_gen) == 1) {
316                /* cur_extra dangling pointer from here. */ view japanese
317                mpm_callback(-1, pid, old_gen);
318            }
319            else {
320                AP_DEBUG_ASSERT(1 == 0);
321            }
322        }
323        cur_extra = next;
324    }
325}

327/* Before sending the signal to the pid this function verifies that
328 * the pid is a member of the current process group; either using
329 * apr_proc_wait(), where waitpid() guarantees to fail for non-child
330 * processes; or by using getpgid() directly, if available. */ view japanese
331AP_DECLARE(apr_status_t) ap_mpm_safe_kill(pid_t pid, int sig)
332{
333#ifndef HAVE_GETPGID
334    apr_proc_t proc;
335    apr_status_t rv;
336    apr_exit_why_e why;
337    int status;
338
339    /* Ensure pid sanity */ view japanese
340    if (pid < 1) {
341        return APR_EINVAL;
342    }
343
344    proc.pid = pid;
345    rv = apr_proc_wait(&proc, &status, &why, APR_NOWAIT);
346    if (rv == APR_CHILD_DONE) {
347        /* The child already died - log the termination status if
348         * necessary: */ view japanese
349        ap_process_child_status(&proc, why, status);
350        return APR_EINVAL;
351    }
352    else if (rv != APR_CHILD_NOTDONE) {
353        /* The child is already dead and reaped, or was a bogus pid -
354         * log this either way. */ view japanese
355        ap_log_error(APLOG_MARK, APLOG_NOTICE, rv, ap_server_conf, APLOGNO(00048)
356                     "cannot send signal %d to pid %ld (non-child or "
357                     "already dead)", sig, (long)pid);
358        return APR_EINVAL;
359    }
360#else
361    pid_t pg;
362
363    /* Ensure pid sanity. */ view japanese
364    if (pid < 1) {
365        return APR_EINVAL;
366    }
367
368    pg = getpgid(pid);
369    if (pg == -1) {
370        /* Process already dead... */ view japanese
371        return errno;
372    }
373
374    if (pg != getpgrp()) {
375        ap_log_error(APLOG_MARK, APLOG_ALERT, 0, ap_server_conf, APLOGNO(00049)
376                     "refusing to send signal %d to pid %ld outside "
377                     "process group", sig, (long)pid);
378        return APR_EINVAL;
379    }
380#endif
381
382    return kill(pid, sig) ? errno : APR_SUCCESS;
383}

386AP_DECLARE(int) ap_process_child_status(apr_proc_t *pid, apr_exit_why_e why,
387                                        int status)
388{
389    int signum = status;
390    const char *sigdesc;
391
392    /* Child died... if it died due to a fatal error,
393     * we should simply bail out.  The caller needs to
394     * check for bad rc from us and exit, running any
395     * appropriate cleanups.
396     *
397     * If the child died due to a resource shortage,
398     * the parent should limit the rate of forking
399     */ view japanese
400    if (APR_PROC_CHECK_EXIT(why)) {
401        if (status == APEXIT_CHILDSICK) {
402            return status;
403        }
404
405        if (status == APEXIT_CHILDFATAL) {
406            ap_log_error(APLOG_MARK, APLOG_ALERT,
407                         0, ap_server_conf, APLOGNO(00050)
408                         "Child %" APR_PID_T_FMT
409                         " returned a Fatal error... Apache is exiting!",
410                         pid->pid);
411            return APEXIT_CHILDFATAL;
412        }
413
414        return 0;
415    }
416
417    if (APR_PROC_CHECK_SIGNALED(why)) {
418        sigdesc = apr_signal_description_get(signum);
419
420        switch (signum) {
421        case SIGTERM:
422        case SIGHUP:
423        case AP_SIG_GRACEFUL:
424        case SIGKILL:
425            break;
426
427        default:
428            if (APR_PROC_CHECK_CORE_DUMP(why)) {
429                ap_log_error(APLOG_MARK, APLOG_NOTICE,
430                             0, ap_server_conf, APLOGNO(00051)
431                             "child pid %ld exit signal %s (%d), "
432                             "possible coredump in %s",
433                             (long)pid->pid, sigdesc, signum,
434                             ap_coredump_dir);
435            }
436            else {
437                ap_log_error(APLOG_MARK, APLOG_NOTICE,
438                             0, ap_server_conf, APLOGNO(00052)
439                             "child pid %ld exit signal %s (%d)",
440                             (long)pid->pid, sigdesc, signum);
441            }
442        }
443    }
444    return 0;
445}

447AP_DECLARE(apr_status_t) ap_mpm_pod_open(apr_pool_t *p, ap_pod_t **pod)
448{
449    apr_status_t rv;
450
451    *pod = apr_palloc(p, sizeof(**pod));
452    rv = apr_file_pipe_create_ex(&((*pod)->pod_in), &((*pod)->pod_out),
453                                 APR_WRITE_BLOCK, p);
454    if (rv != APR_SUCCESS) {
455        return rv;
456    }
457
458    apr_file_pipe_timeout_set((*pod)->pod_in, 0);
459    (*pod)->p = p;
460
461    /* close these before exec. */
462    apr_file_inherit_unset((*pod)->pod_in);
463    apr_file_inherit_unset((*pod)->pod_out);
464
465    return APR_SUCCESS;
466}

468AP_DECLARE(apr_status_t) ap_mpm_pod_check(ap_pod_t *pod)
469{
470    char c;
471    apr_size_t len = 1;
472    apr_status_t rv;
473
474    rv = apr_file_read(pod->pod_in, &c, &len);
475
476    if ((rv == APR_SUCCESS) && (len == 1)) {
477        return APR_SUCCESS;
478    }
479
480    if (rv != APR_SUCCESS) {
481        return rv;
482    }
483
484    return AP_NORESTART;
485}

620/* This function connects to the server and sends enough data to
621 * ensure the child wakes up and processes a new connection.  This
622 * permits the MPM to skip the poll when there is only one listening
623 * socket, because it provides a alternate way to unblock an accept()
624 * when the pod is used.  */ view japanese
625static apr_status_t dummy_connection(ap_pod_t *pod)
626{
627    const char *data;
628    apr_status_t rv;
629    apr_socket_t *sock;
630    apr_pool_t *p;
631    apr_size_t len;
632    ap_listen_rec *lp;
633
634    /* create a temporary pool for the socket.  pconf stays around too long */ view japanese
635    rv = apr_pool_create(&p, pod->p);
636    if (rv != APR_SUCCESS) {
637        return rv;
638    }
639    apr_pool_tag(p, "dummy_connection");
640
641    /* If possible, find a listener which is configured for
642     * plain-HTTP, not SSL; using an SSL port would either be
643     * expensive to do correctly (performing a complete SSL handshake)
644     * or cause log spam by doing incorrectly (simply sending EOF). */ view japanese
645    lp = ap_listeners;
646    while (lp && lp->protocol && ap_cstr_casecmp(lp->protocol, "http") != 0) {
647        lp = lp->next;
648    }
649    if (!lp) {
650        lp = ap_listeners;
651    }
652
653    rv = apr_socket_create(&sock, lp->bind_addr->family, SOCK_STREAM, 0, p);
654    if (rv != APR_SUCCESS) {
655        ap_log_error(APLOG_MARK, APLOG_WARNING, rv, ap_server_conf, APLOGNO(00054)
656                     "get socket to connect to listener");
657        apr_pool_destroy(p);
658        return rv;
659    }
660
661    /* on some platforms (e.g., FreeBSD), the kernel won't accept many
662     * queued connections before it starts blocking local connects...
663     * we need to keep from blocking too long and instead return an error,
664     * because the MPM won't want to hold up a graceful restart for a
665     * long time
666     */ view japanese
667    rv = apr_socket_timeout_set(sock, apr_time_from_sec(3));
668    if (rv != APR_SUCCESS) {
669        ap_log_error(APLOG_MARK, APLOG_WARNING, rv, ap_server_conf, APLOGNO(00055)
670                     "set timeout on socket to connect to listener");
671        apr_socket_close(sock);
672        apr_pool_destroy(p);
673        return rv;
674    }
675
676    rv = apr_socket_connect(sock, lp->bind_addr);
677    if (rv != APR_SUCCESS) {
678        int log_level = APLOG_WARNING;
679
680        if (APR_STATUS_IS_TIMEUP(rv)) {
681            /* probably some server processes bailed out already and there
682             * is nobody around to call accept and clear out the kernel
683             * connection queue; usually this is not worth logging
684             */ view japanese
685            log_level = APLOG_DEBUG;

815        else {
816            status = apr_psprintf(pconf,
817                                  "httpd (pid %" APR_PID_T_FMT "?) not running",
818                                  otherpid);
819        }
820    }
821
822    if (!strcmp(dash_k_arg, "start") || dash_k_arg == dash_k_arg_noarg) {
823        if (running) {
824            printf("%s\n", status);
825            return 1;
826        }
827    }
828
829    if (!strcmp(dash_k_arg, "stop")) {
830        if (!running) {
831            printf("%s\n", status);
832        }
833        else {
834            send_signal(otherpid, SIGTERM);
835        }
836        return 1;
837    }
838
839    if (!strcmp(dash_k_arg, "restart")) {
840        if (!running) {

842        }
843        else {
844            *exit_status = send_signal(otherpid, SIGHUP);
845            return 1;
846        }
847    }
848
849    if (!strcmp(dash_k_arg, "graceful")) {
850        if (!running) {
851            printf("httpd not running, trying to start\n");
852        }
853        else {
854            *exit_status = send_signal(otherpid, AP_SIG_GRACEFUL);
855            return 1;
856        }
857    }
858
859    if (!strcmp(dash_k_arg, "graceful-stop")) {
860        if (!running) {
861            printf("%s\n", status);
862        }
863        else {
