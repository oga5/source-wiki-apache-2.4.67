237/* The LISTENER_SIGNAL signal will be sent from the main thread to the
238 * listener thread to wake it up for graceful termination (what a child
239 * process from an old generation does when the admin does "apachectl
240 * graceful").  This signal will be blocked in all threads of a child
241 * process except for the listener thread.
242 */ view japanese
243#define LISTENER_SIGNAL     SIGHUP
244
245/* The WORKER_SIGNAL signal will be sent from the main thread to the
246 * worker threads during an ungraceful restart or shutdown.
247 * This ensures that on systems (i.e., Linux) where closing the worker
248 * socket doesn't awake the worker thread when it is polling on the socket
249 * (especially in apr_wait_for_io_or_timeout() when handling
250 * Keep-Alive connections), close_worker_sockets() and join_workers()
251 * still function in timely manner and allow ungraceful shutdowns to
252 * proceed to completion.  Otherwise join_workers() doesn't return
253 * before the main process decides the child process is non-responsive
254 * and sends a SIGKILL.
255 */ view japanese
256#define WORKER_SIGNAL       AP_SIG_GRACEFUL

262static void close_worker_sockets(void)
263{
264    int i;
265    for (i = 0; i < threads_per_child; i++) {
266        if (worker_sockets[i]) {
267            apr_socket_close(worker_sockets[i]);
268            worker_sockets[i] = NULL;
269        }
270    }
271}

273static void wakeup_listener(void)
274{
275    listener_may_exit = 1;
276
277    /* Unblock the listener if it's poll()ing */ view japanese
278    if (worker_pollset && listener_is_wakeable) {
279        apr_pollset_wakeup(worker_pollset);
280    }
281
282    /* unblock the listener if it's waiting for a worker */ view japanese
283    ap_queue_info_term(worker_queue_info);
284
285    if (!listener_os_thread) {
286        /* XXX there is an obscure path that this doesn't handle perfectly:
287         *     right after listener thread is created but before
288         *     listener_os_thread is set, the first worker thread hits an
289         *     error and starts graceful termination
290         */ view japanese
291        return;
292    }
293    /*
294     * we should just be able to "kill(ap_my_pid, LISTENER_SIGNAL)" on all
295     * platforms and wake up the listener thread since it is the only thread
296     * with SIGHUP unblocked, but that doesn't work on Linux
297     */ view japanese
298#ifdef HAVE_PTHREAD_KILL
299    pthread_kill(*listener_os_thread, LISTENER_SIGNAL);
300#else
301    kill(ap_my_pid, LISTENER_SIGNAL);
302#endif
303}

311static void signal_threads(int mode)
312{
313    if (terminate_mode == mode) {
314        return;
315    }
316    terminate_mode = mode;
317    retained->mpm->mpm_state = AP_MPMQ_STOPPING;
318
319    /* in case we weren't called from the listener thread, wake up the
320     * listener thread
321     */ view japanese
322    wakeup_listener();
323
324    /* for ungraceful termination, let the workers exit now;
325     * for graceful termination, the listener thread will notify the
326     * workers to exit once it has stopped accepting new connections
327     */ view japanese
328    if (mode == ST_UNGRACEFUL) {
329        workers_may_exit = 1;
330        ap_queue_interrupt_all(worker_queue);
331        close_worker_sockets(); /* forcefully kill all current connections */ view japanese
332    }
333
334    ap_run_child_stopping(pchild, mode == ST_GRACEFUL);
335}

337static int worker_query(int query_code, int *result, apr_status_t *rv)
338{
339    *rv = APR_SUCCESS;
340    switch (query_code) {
341        case AP_MPMQ_MAX_DAEMON_USED:
342            *result = retained->max_daemons_limit;
343            break;
344        case AP_MPMQ_IS_THREADED:
345            *result = AP_MPMQ_STATIC;
346            break;
347        case AP_MPMQ_IS_FORKED:
348            *result = AP_MPMQ_DYNAMIC;
349            break;
350        case AP_MPMQ_HARD_LIMIT_DAEMONS:
351            *result = server_limit;
352            break;
353        case AP_MPMQ_HARD_LIMIT_THREADS:
354            *result = thread_limit;
355            break;
356        case AP_MPMQ_MAX_THREADS:
357            *result = threads_per_child;
358            break;
359        case AP_MPMQ_MIN_SPARE_DAEMONS:
360            *result = 0;
361            break;
362        case AP_MPMQ_MIN_SPARE_THREADS:
363            *result = min_spare_threads;
364            break;
365        case AP_MPMQ_MAX_SPARE_DAEMONS:
366            *result = 0;
367            break;
368        case AP_MPMQ_MAX_SPARE_THREADS:
369            *result = max_spare_threads;
370            break;
371        case AP_MPMQ_MAX_REQUESTS_DAEMON:
372            *result = ap_max_requests_per_child;
373            break;
374        case AP_MPMQ_MAX_DAEMONS:
375            *result = ap_daemons_limit;
376            break;
377        case AP_MPMQ_MPM_STATE:
378            *result = retained->mpm->mpm_state;
379            break;
380        case AP_MPMQ_GENERATION:
381            *result = retained->mpm->my_generation;
382            break;
383        default:
384            *rv = APR_ENOTIMPL;
385            break;
386    }
387    return OK;
388}

529static void accept_mutex_error(const char *func, apr_status_t rv, int process_slot)
530{
531    int level = APLOG_EMERG;
532
533    if (ap_scoreboard_image->parent[process_slot].generation !=
534        ap_scoreboard_image->global->running_generation) {
535        level = APLOG_DEBUG; /* common to get these at restart time */
536    }
537    else if (requests_this_child == INT_MAX
538        || ((requests_this_child == ap_max_requests_per_child)
539            && ap_max_requests_per_child)) {
540        ap_log_error(APLOG_MARK, level, rv, ap_server_conf, APLOGNO(00272)
541                     "apr_proc_mutex_%s failed "
542                     "before this child process served any requests.",
543                     func);
544        clean_child_exit(APEXIT_CHILDSICK);
545    }
546    ap_log_error(APLOG_MARK, level, rv, ap_server_conf, APLOGNO(00273)
547                 "apr_proc_mutex_%s failed. Attempting to "
548                 "shutdown process gracefully.", func);
549    signal_threads(ST_GRACEFUL);
550}

552static void * APR_THREAD_FUNC listener_thread(apr_thread_t *thd, void * dummy)
553{
554    proc_info * ti = dummy;
555    int process_slot = ti->pid;
556    void *csd = NULL;
557    apr_pool_t *ptrans = NULL;            /* Pool for per-transaction stuff */
558    apr_status_t rv;
559    ap_listen_rec *lr = NULL;
560    int have_idle_worker = 0;
561    int last_poll_idx = 0;
562
563    free(ti);
564
565    /* Unblock the signal used to wake this thread up, and set a handler for
566     * it.
567     */ view japanese
568    apr_signal(LISTENER_SIGNAL, dummy_signal_handler);
569    unblock_signal(LISTENER_SIGNAL);
570
571    /* TODO: Switch to a system where threads reuse the results from earlier
572       poll calls - manoj */ view japanese
573    while (1) {
574        /* TODO: requests_this_child should be synchronized - aaron */ view japanese
575        if (requests_this_child <= 0) {
576            check_infinite_requests();
577        }
578        if (listener_may_exit) break;
579
580        if (!have_idle_worker) {
581            rv = ap_queue_info_wait_for_idler(worker_queue_info, NULL);
582            if (APR_STATUS_IS_EOF(rv)) {
583                break; /* we've been signaled to die now */ view japanese
584            }
585            else if (rv != APR_SUCCESS) {
586                ap_log_error(APLOG_MARK, APLOG_EMERG, rv, ap_server_conf,
587                             "apr_queue_info_wait failed. Attempting to "
588                             " shutdown process gracefully.");
589                signal_threads(ST_GRACEFUL);
590                break;
591            }
592            have_idle_worker = 1;
593        }
594
595        /* We've already decremented the idle worker count inside
596         * ap_queue_info_wait_for_idler. */ view japanese
597
598        if ((rv = SAFE_ACCEPT(apr_proc_mutex_lock(my_bucket->mutex)))
599            != APR_SUCCESS) {
600
601            if (!listener_may_exit) {
602                accept_mutex_error("lock", rv, process_slot);
603            }
604            break;                    /* skip the lock release */ view japanese
605        }
606
607        if (!my_bucket->listeners->next) {
608            /* Only one listener, so skip the poll */ view japanese
609            lr = my_bucket->listeners;
610        }
611        else {
612            while (!listener_may_exit) {
613                apr_int32_t numdesc;
614                const apr_pollfd_t *pdesc;
615
616                rv = apr_pollset_poll(worker_pollset, -1, &numdesc, &pdesc);
617                if (rv != APR_SUCCESS) {
618                    if (APR_STATUS_IS_EINTR(rv)) {
619                        continue;
620                    }
621
622                    /* apr_pollset_poll() will only return errors in catastrophic
623                     * circumstances. Let's try exiting gracefully, for now. */ view japanese
624                    ap_log_error(APLOG_MARK, APLOG_ERR, rv, ap_server_conf, APLOGNO(03137)
625                                 "apr_pollset_poll: (listen)");
626                    signal_threads(ST_GRACEFUL);
627                }
628
629                if (listener_may_exit) break;
630
631                /* We can always use pdesc[0], but sockets at position N
632                 * could end up completely starved of attention in a very
633                 * busy server. Therefore, we round-robin across the
634                 * returned set of descriptors. While it is possible that
635                 * the returned set of descriptors might flip around and
636                 * continue to starve some sockets, we happen to know the
637                 * internal pollset implementation retains ordering
638                 * stability of the sockets. Thus, the round-robin should
639                 * ensure that a socket will eventually be serviced.
640                 */ view japanese
641                if (last_poll_idx >= numdesc)
642                    last_poll_idx = 0;
643
644                /* Grab a listener record from the client_data of the poll
645                 * descriptor, and advance our saved index to round-robin
646                 * the next fetch.
647                 *
648                 * ### hmm... this descriptor might have POLLERR rather
649                 * ### than POLLIN
650                 */ view japanese
651                lr = pdesc[last_poll_idx++].client_data;
652                break;
653
654            } /* while */ view japanese
655
656        } /* if/else */ view japanese
657
658        if (!listener_may_exit) {
659            /* the following pops a recycled ptrans pool off a stack */ view japanese
660            ap_queue_info_pop_pool(worker_queue_info, &ptrans);
661            if (ptrans == NULL) {
662                /* we can't use a recycled transaction pool this time.
663                 * create a new transaction pool */ view japanese
664                apr_allocator_t *allocator;
665
666                apr_allocator_create(&allocator);
667                apr_allocator_max_free_set(allocator, ap_max_mem_free);
668                apr_pool_create_ex(&ptrans, pconf, NULL, allocator);
669                apr_allocator_owner_set(allocator, ptrans);
670                apr_pool_tag(ptrans, "transaction");
671            }
672            rv = lr->accept_func(&csd, lr, ptrans);
673            /* later we trash rv and rely on csd to indicate success/failure */ view japanese
674            AP_DEBUG_ASSERT(rv == APR_SUCCESS || !csd);
675
676            if (rv == APR_EGENERAL) {
677                /* E[NM]FILE, ENOMEM, etc */ view japanese
678                resource_shortage = 1;
679                signal_threads(ST_GRACEFUL);
680            }
681            if ((rv = SAFE_ACCEPT(apr_proc_mutex_unlock(my_bucket->mutex)))
682                != APR_SUCCESS) {
683
684                if (listener_may_exit) {
685                    break;
686                }
687                accept_mutex_error("unlock", rv, process_slot);
688            }
689            if (csd != NULL) {
690                rv = ap_queue_push_socket(worker_queue, csd, NULL, ptrans);
691                if (rv) {
692                    /* trash the connection; we couldn't queue the connected
693                     * socket to a worker
694                     */ view japanese
695                    apr_socket_close(csd);
696                    ap_log_error(APLOG_MARK, APLOG_CRIT, rv, ap_server_conf, APLOGNO(03138)
697                                 "ap_queue_push_socket failed");
698                }
699                else {
700                    have_idle_worker = 0;
701                }
702            }
703        }
704        else {
705            if ((rv = SAFE_ACCEPT(apr_proc_mutex_unlock(my_bucket->mutex)))
706                != APR_SUCCESS) {
707                int level = APLOG_EMERG;
708
709                if (ap_scoreboard_image->parent[process_slot].generation !=
710                    ap_scoreboard_image->global->running_generation) {
711                    level = APLOG_DEBUG; /* common to get these at restart time */ view japanese
712                }
713                ap_log_error(APLOG_MARK, level, rv, ap_server_conf, APLOGNO(00274)
714                             "apr_proc_mutex_unlock failed. Attempting to "
715                             "shutdown process gracefully.");
716                signal_threads(ST_GRACEFUL);
717            }
718            break;
719        }
720    }
721
722    ap_close_listeners_ex(my_bucket->listeners);
723    ap_queue_info_free_idle_pools(worker_queue_info);
724    ap_queue_term(worker_queue);
725    dying = 1;
726    ap_scoreboard_image->parent[process_slot].quiescing = 1;
727
728    /* wake up the main thread */ view japanese
729    kill(ap_my_pid, SIGTERM);
730
731    apr_thread_exit(thd, APR_SUCCESS);
732    return NULL;
733}

735/* XXX For ungraceful termination/restart, we definitely don't want to
736 *     wait for active connections to finish but we may want to wait
737 *     for idle workers to get out of the queue code and release mutexes,
738 *     since those mutexes are cleaned up pretty soon and some systems
739 *     may not react favorably (i.e., segfault) if operations are attempted
740 *     on cleaned-up mutexes.
741 */ view japanese
742static void * APR_THREAD_FUNC worker_thread(apr_thread_t *thd, void * dummy)
743{
744    proc_info * ti = dummy;
745    int process_slot = ti->pid;
746    int thread_slot = ti->tid;
747    apr_socket_t *csd = NULL;
748    apr_bucket_alloc_t *bucket_alloc;
749    apr_pool_t *last_ptrans = NULL;
750    apr_pool_t *ptrans;                /* Pool for per-transaction stuff */ view japanese
751    apr_status_t rv;
752    int is_idle = 0;
753
754    free(ti);
755
756    ap_scoreboard_image->servers[process_slot][thread_slot].pid = ap_my_pid;
757    ap_scoreboard_image->servers[process_slot][thread_slot].tid = apr_os_thread_current();
758    ap_scoreboard_image->servers[process_slot][thread_slot].generation = retained->mpm->my_generation;
759    ap_update_child_status_from_indexes(process_slot, thread_slot,
760                                        SERVER_STARTING, NULL);
761
762#ifdef HAVE_PTHREAD_KILL
763    apr_signal(WORKER_SIGNAL, dummy_signal_handler);
764    unblock_signal(WORKER_SIGNAL);
765#endif
766
767    while (!workers_may_exit) {
768        if (!is_idle) {
769            rv = ap_queue_info_set_idle(worker_queue_info, last_ptrans);
770            last_ptrans = NULL;
771            if (rv != APR_SUCCESS) {
772                ap_log_error(APLOG_MARK, APLOG_EMERG, rv, ap_server_conf,
773                             "ap_queue_info_set_idle failed. Attempting to "
774                             "shutdown process gracefully.");
775                signal_threads(ST_GRACEFUL);
776                break;
777            }
778            is_idle = 1;
779        }
780
781        ap_update_child_status_from_indexes(process_slot, thread_slot,
782                                            SERVER_READY, NULL);
783worker_pop:
784        if (workers_may_exit) {
785            break;
786        }
787        rv = ap_queue_pop_socket(worker_queue, &csd, &ptrans);
788
789        if (rv != APR_SUCCESS) {
790            /* We get APR_EOF during a graceful shutdown once all the connections
791             * accepted by this server process have been handled.
792             */ view japanese
793            if (APR_STATUS_IS_EOF(rv)) {
794                break;
795            }
796            /* We get APR_EINTR whenever ap_queue_pop_*() has been interrupted
797             * from an explicit call to ap_queue_interrupt_all(). This allows
798             * us to unblock threads stuck in ap_queue_pop_*() when a shutdown
799             * is pending.
800             *
801             * If workers_may_exit is set and this is ungraceful termination/
802             * restart, we are bound to get an error on some systems (e.g.,
803             * AIX, which sanity-checks mutex operations) since the queue
804             * may have already been cleaned up.  Don't log the "error" if
805             * workers_may_exit is set.
806             */ view japanese
807            else if (APR_STATUS_IS_EINTR(rv)) {
808                goto worker_pop;
809            }
810            /* We got some other error. */ view japanese
811            else if (!workers_may_exit) {
812                ap_log_error(APLOG_MARK, APLOG_CRIT, rv, ap_server_conf, APLOGNO(03139)
813                             "ap_queue_pop_socket failed");
814            }
815            continue;
816        }
817        is_idle = 0;
818        worker_sockets[thread_slot] = csd;
819        bucket_alloc = apr_bucket_alloc_create(ptrans);
820        process_socket(thd, ptrans, csd, process_slot, thread_slot, bucket_alloc);
821        worker_sockets[thread_slot] = NULL;
822        requests_this_child--;
823        apr_pool_clear(ptrans);
824        last_ptrans = ptrans;
825    }
826
827    ap_update_child_status_from_indexes(process_slot, thread_slot,
828                                        dying ? SERVER_DEAD
829                                              : SERVER_GRACEFUL, NULL);
830
831    apr_thread_exit(thd, APR_SUCCESS);
832    return NULL;
833}

867static void setup_threads_runtime(void)
868{
869    ap_listen_rec *lr;
870    int pollset_flags;
871    apr_status_t rv;
872
873    /* All threads (listener, workers) and synchronization objects (queues,
874     * pollset, mutexes...) created here should have at least the lifetime of
875     * the connections they handle (i.e. ptrans). We can't use this thread's
876     * self pool because all these objects survive it, nor use pchild or pconf
877     * directly because this starter thread races with other modules' runtime,
878     * nor finally pchild (or subpool thereof) because it is killed explicitly
879     * before pconf (thus connections/ptrans can live longer, which matters in
880     * ONE_PROCESS mode). So this leaves us with a subpool of pconf, created
881     * before any ptrans hence destroyed after.
882     */ view japanese
883    apr_pool_create(&pruntime, pconf);
884    apr_pool_tag(pruntime, "mpm_runtime");
885
886    /* We must create the fd queues before we start up the listener
887     * and worker threads. */ view japanese
888    rv = ap_queue_create(&worker_queue, threads_per_child, pruntime);
889    if (rv != APR_SUCCESS) {
890        ap_log_error(APLOG_MARK, APLOG_ALERT, rv, ap_server_conf, APLOGNO(03140)
891                     "ap_queue_create() failed");
892        clean_child_exit(APEXIT_CHILDFATAL);
893    }
894
895    rv = ap_queue_info_create(&worker_queue_info, pruntime,
896                              threads_per_child, -1);
897    if (rv != APR_SUCCESS) {
898        ap_log_error(APLOG_MARK, APLOG_ALERT, rv, ap_server_conf, APLOGNO(03141)
899                     "ap_queue_info_create() failed");
900        clean_child_exit(APEXIT_CHILDFATAL);
901    }
902
903    /* Create the main pollset. When APR_POLLSET_WAKEABLE is asked we account
904     * for the wakeup pipe explicitely with num_listensocks+1 because some
905     * pollset implementations don't do it implicitely in APR.
906     */ view japanese
907    pollset_flags = APR_POLLSET_NOCOPY | APR_POLLSET_WAKEABLE;
908    rv = apr_pollset_create(&worker_pollset, num_listensocks + 1, pruntime,
909                            pollset_flags);
910    if (rv == APR_SUCCESS) {
911        listener_is_wakeable = 1;
912    }
913    else {
914        pollset_flags &= ~APR_POLLSET_WAKEABLE;
915        rv = apr_pollset_create(&worker_pollset, num_listensocks, pruntime,
916                                pollset_flags);
917    }
918    if (rv != APR_SUCCESS) {
919        ap_log_error(APLOG_MARK, APLOG_EMERG, rv, ap_server_conf, APLOGNO(03285)
920                     "Couldn't create pollset in thread;"
921                     " check system or user limits");
922        /* let the parent decide how bad this really is */
923        clean_child_exit(APEXIT_CHILDSICK);
924    }
925
926    for (lr = my_bucket->listeners; lr != NULL; lr = lr->next) {
927        apr_pollfd_t *pfd = apr_pcalloc(pruntime, sizeof *pfd);
928
929        pfd->desc_type = APR_POLL_SOCKET;
930        pfd->desc.s = lr->sd;
931        pfd->reqevents = APR_POLLIN;
932        pfd->client_data = lr;
933
934        rv = apr_pollset_add(worker_pollset, pfd);
935        if (rv != APR_SUCCESS) {
936            ap_log_error(APLOG_MARK, APLOG_EMERG, rv, ap_server_conf, APLOGNO(03286)
937                         "Couldn't create add listener to pollset;"
938                         " check system or user limits");
939            /* let the parent decide how bad this really is */
940            clean_child_exit(APEXIT_CHILDSICK);
941        }
942
943        lr->accept_func = ap_unixd_accept;
944    }
945
946    worker_sockets = apr_pcalloc(pruntime, threads_per_child *
947                                           sizeof(apr_socket_t *));
948}

950/* XXX under some circumstances not understood, children can get stuck
951 *     in start_threads forever trying to take over slots which will
952 *     never be cleaned up; for now there is an APLOG_DEBUG message issued
953 *     every so often when this condition occurs
954 */ view japanese
955static void * APR_THREAD_FUNC start_threads(apr_thread_t *thd, void *dummy)
956{
957    thread_starter *ts = dummy;
958    apr_thread_t **threads = ts->threads;
959    apr_threadattr_t *thread_attr = ts->threadattr;
960    int my_child_num = ts->child_num_arg;
961    proc_info *my_info;
962    apr_status_t rv;
963    int threads_created = 0;
964    int listener_started = 0;
965    int prev_threads_created;
966    int loops, i;
967
968    loops = prev_threads_created = 0;
969    while (1) {
970        /* threads_per_child does not include the listener thread */ view japanese
971        for (i = 0; i < threads_per_child; i++) {
972            int status = ap_scoreboard_image->servers[my_child_num][i].status;
973
974            if (status != SERVER_GRACEFUL && status != SERVER_DEAD) {
975                continue;
976            }
977
978            my_info = (proc_info *)ap_malloc(sizeof(proc_info));
979            my_info->pid = my_child_num;
980            my_info->tid = i;
981            my_info->sd = 0;
982
983            /* We are creating threads right now */ view japanese
984            ap_update_child_status_from_indexes(my_child_num, i,
985                                                SERVER_STARTING, NULL);
986            /* We let each thread update its own scoreboard entry.  This is
987             * done because it lets us deal with tid better.
988             */ view japanese
989            rv = ap_thread_create(&threads[i], thread_attr,
990                                  worker_thread, my_info, pruntime);
991            if (rv != APR_SUCCESS) {
992                ap_log_error(APLOG_MARK, APLOG_ALERT, rv, ap_server_conf, APLOGNO(03142)
993                             "ap_thread_create: unable to create worker thread");
994                /* let the parent decide how bad this really is */
995                clean_child_exit(APEXIT_CHILDSICK);
996            }
997            threads_created++;
998        }
999        /* Start the listener only when there are workers available */ view japanese
1000        if (!listener_started && threads_created) {
1001            create_listener_thread(ts);
1002            listener_started = 1;
1003        }
1004        if (start_thread_may_exit || threads_created == threads_per_child) {
1005            break;
1006        }
1007        /* wait for previous generation to clean up an entry */ view japanese
1008        apr_sleep(apr_time_from_sec(1));
1009        ++loops;
1010        if (loops % 120 == 0) { /* every couple of minutes */ view japanese
1011            if (prev_threads_created == threads_created) {
1012                ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, ap_server_conf,
1013                             "child %" APR_PID_T_FMT " isn't taking over "
1014                             "slots very quickly (%d of %d)",
1015                             ap_my_pid, threads_created, threads_per_child);
1016            }
1017            prev_threads_created = threads_created;
1018        }
1019    }
1020
1021    /* What state should this child_main process be listed as in the
1022     * scoreboard...?
1023     *  ap_update_child_status_from_indexes(my_child_num, i, SERVER_STARTING,
1024     *                                      (request_rec *) NULL);
1025     *
1026     *  This state should be listed separately in the scoreboard, in some kind
1027     *  of process_status, not mixed in with the worker threads' status.
1028     *  "life_status" is almost right, but it's in the worker's structure, and
1029     *  the name could be clearer.   gla
1030     */ view japanese
1031    apr_thread_exit(thd, APR_SUCCESS);
1032    return NULL;
1033}

1035static void join_workers(apr_thread_t *listener, apr_thread_t **threads,
1036                         int mode)
1037{
1038    int i;
1039    apr_status_t rv, thread_rv;
1040
1041    if (listener) {
1042        int iter;
1043
1044        /* deal with a rare timing window which affects waking up the
1045         * listener thread...  if the signal sent to the listener thread
1046         * is delivered between the time it verifies that the
1047         * listener_may_exit flag is clear and the time it enters a
1048         * blocking syscall, the signal didn't do any good...  work around
1049         * that by sleeping briefly and sending it again
1050         */ view japanese
1051
1052        iter = 0;
1053        while (!dying) {
1054            apr_sleep(apr_time_from_msec(500));
1055            if (dying || ++iter > 10) {
1056                break;
1057            }
1058            /* listener has not stopped accepting yet */ view japanese
1059            ap_log_error(APLOG_MARK, APLOG_TRACE1, 0, ap_server_conf,
1060                         "listener has not stopped accepting yet (%d iter)", iter);
1061            wakeup_listener();
1062        }
1063        if (iter > 10) {
1064            ap_log_error(APLOG_MARK, APLOG_DEBUG, 0, ap_server_conf, APLOGNO(00276)
1065                         "the listener thread didn't exit");
1066        }
1067        else {
1068            rv = apr_thread_join(&thread_rv, listener);
1069            if (rv != APR_SUCCESS) {
1070                ap_log_error(APLOG_MARK, APLOG_CRIT, rv, ap_server_conf, APLOGNO(00277)
1071                             "apr_thread_join: unable to join listener thread");
1072            }
1073        }
1074    }
1075
1076    for (i = 0; i < threads_per_child; i++) {
1077        if (threads[i]) { /* if we ever created this thread */ view japanese
1078            if (mode != ST_GRACEFUL) {
1079#ifdef HAVE_PTHREAD_KILL
1080                apr_os_thread_t *worker_os_thread;
1081
1082                apr_os_thread_get(&worker_os_thread, threads[i]);
1083                pthread_kill(*worker_os_thread, WORKER_SIGNAL);
1084#endif
1085            }
1086
1087            rv = apr_thread_join(&thread_rv, threads[i]);
1088            if (rv != APR_SUCCESS) {
1089                ap_log_error(APLOG_MARK, APLOG_CRIT, rv, ap_server_conf, APLOGNO(00278)
1090                             "apr_thread_join: unable to join worker "
1091                             "thread %d",
1092                             i);
1093            }
1094        }
1095    }
1096}

1216                         ap_thread_stacksize);
1217        }
1218    }
1219
1220    ts->threads = threads;
1221    ts->listener = NULL;
1222    ts->child_num_arg = child_num_arg;
1223    ts->threadattr = thread_attr;
1224
1225    rv = ap_thread_create(&start_thread_id, thread_attr, start_threads,
1226                          ts, pchild);
1227    if (rv != APR_SUCCESS) {
1228        ap_log_error(APLOG_MARK, APLOG_ALERT, rv, ap_server_conf, APLOGNO(00282)
1229                     "ap_thread_create: unable to create worker thread");
1230        /* let the parent decide how bad this really is */ view japanese
1231        clean_child_exit(APEXIT_CHILDSICK);
1232    }
1233
1234    retained->mpm->mpm_state = AP_MPMQ_RUNNING;
1235
1236    /* If we are only running in one_process mode, we will want to
1237     * still handle signals. */ view japanese
1238    if (one_process) {
1239        /* Block until we get a terminating signal. */ view japanese
1240        apr_signal_thread(check_signal);
1241        /* make sure the start thread has finished; signal_threads()
1242         * and join_workers() depend on that
1243         */ view japanese
1244        /* XXX join_start_thread() won't be awakened if one of our
1245         *     threads encounters a critical error and attempts to
1246         *     shutdown this child
1247         */ view japanese
1248        join_start_thread(start_thread_id);
1249        signal_threads(ST_UNGRACEFUL); /* helps us terminate a little more
1250                           * quickly than the dispatch of the signal thread
1251                           * beats the Pipe of Death and the browsers
1252                           */ view japanese
1253        /* A terminating signal was received. Now join each of the
1254         * workers to clean them up.
1255         *   If the worker already exited, then the join frees
1256         *   their resources and returns.
1257         *   If the worker hasn't exited, then this blocks until
1258         *   they have (then cleans up).
1259         */
1260        join_workers(ts->listener, threads, ST_UNGRACEFUL);
1261    }
1262    else { /* !one_process */ view japanese
1263        /* remove SIGTERM from the set of blocked signals...  if one of
1264         * the other threads in the process needs to take us down
1265         * (e.g., for MaxConnectionsPerChild) it will send us SIGTERM
1266         */ view japanese
1267        apr_signal(SIGTERM, dummy_signal_handler);
1268        unblock_signal(SIGTERM);
1269        /* Watch for any messages from the parent over the POD */ view japanese
1270        while (1) {
1271            rv = ap_mpm_podx_check(my_bucket->pod);
1272            if (rv == AP_MPM_PODX_NORESTART) {
1273                /* see if termination was triggered while we slept */ view japanese
1274                switch(terminate_mode) {
1275                case ST_GRACEFUL:
1276                    rv = AP_MPM_PODX_GRACEFUL;
1277                    break;
1278                case ST_UNGRACEFUL:
1279                    rv = AP_MPM_PODX_RESTART;
1280                    break;
1281                }
1282            }
1283            if (rv == AP_MPM_PODX_GRACEFUL || rv == AP_MPM_PODX_RESTART) {
1284                /* make sure the start thread has finished;
1285                 * signal_threads() and join_workers depend on that
1286                 */ view japanese
1287                join_start_thread(start_thread_id);
1288                signal_threads(rv == AP_MPM_PODX_GRACEFUL ? ST_GRACEFUL : ST_UNGRACEFUL);
1289                break;
1290            }
1291        }
1292
1293        /* A terminating signal was received. Now join each of the
1294         * workers to clean them up.
1295         *   If the worker already exited, then the join frees
1296         *   their resources and returns.
1297         *   If the worker hasn't exited, then this blocks until
1298         *   they have (then cleans up).
1299         */ view japanese
1300        join_workers(ts->listener, threads,
1301                     rv == AP_MPM_PODX_GRACEFUL ? ST_GRACEFUL : ST_UNGRACEFUL);
1302    }
1303
1304    free(threads);
1305
1306    clean_child_exit(resource_shortage ? APEXIT_CHILDSICK : 0);
1307}

1309static int make_child(server_rec *s, int slot, int bucket)
1310{
1311    int pid;
1312
1313    if (slot + 1 > retained->max_daemons_limit) {
1314        retained->max_daemons_limit = slot + 1;
1315    }
1316
1317    if (one_process) {
1318        my_bucket = &all_buckets[0];
1319
1320        worker_note_child_started(slot, getpid());
1321        child_main(slot, 0);
1322        /* NOTREACHED */
1323        ap_assert(0);
1324        return -1;
1325    }
1326
1327    if ((pid = fork()) == -1) {
1328        ap_log_error(APLOG_MARK, APLOG_ERR, errno, s, APLOGNO(00283)
1329                     "fork: Unable to fork new process");
1330        /* fork didn't succeed.  There's no need to touch the scoreboard;
1331         * if we were trying to replace a failed child process, then
1332         * server_main_loop() marked its workers SERVER_DEAD, and if
1333         * we were trying to replace a child process that exited normally,
1334         * its worker_thread()s left SERVER_DEAD or SERVER_GRACEFUL behind.
1335         */ view japanese
1336
1337        /* In case system resources are maxxed out, we don't want
1338           Apache running away with the CPU trying to fork over and
1339           over and over again. */ view japanese
1340        apr_sleep(apr_time_from_sec(10));
1341
1342        return -1;
1343    }
1344
1345    if (!pid) {
1346#if AP_HAS_THREAD_LOCAL
1347        ap_thread_current_after_fork();
1348#endif
1349
1350        my_bucket = &all_buckets[bucket];
1351
1352#ifdef HAVE_BINDPROCESSOR
1353        /* By default, AIX binds to a single processor.  This bit unbinds
1354         * children which will then bind to another CPU.
1355         */ view japanese
1356        int status = bindprocessor(BINDPROCESS, (int)getpid(),
1357                               PROCESSOR_CLASS_ANY);
1358        if (status != OK)
1359            ap_log_error(APLOG_MARK, APLOG_DEBUG, errno,
1360                         ap_server_conf, APLOGNO(00284)
1361                         "processor unbind failed");
1362#endif
1363        RAISE_SIGSTOP(MAKE_CHILD);
1364
1365        apr_signal(SIGTERM, just_die);
1366        child_main(slot, bucket);
1367        /* NOTREACHED */ view japanese
1368        ap_assert(0);
1369        return -1;
1370    }
1371
1372    if (ap_scoreboard_image->parent[slot].pid != 0) {
1373        /* This new child process is squatting on the scoreboard
1374         * entry owned by an exiting child process, which cannot
1375         * exit until all active requests complete.
1376         */ view japanese
1377        worker_note_child_lost_slot(slot, pid);
1378    }
1379    ap_scoreboard_image->parent[slot].quiescing = 0;
1380    worker_note_child_started(slot, pid);
1381    return 0;
1382}

1385static void startup_children(int number_to_start)
1386{
1387    int i;
1388
1389    for (i = 0; number_to_start && i < ap_daemons_limit; ++i) {

1397    }

1400static void perform_idle_server_maintenance(int child_bucket)
1401{
1402    int num_buckets = retained->mpm->num_buckets;
1403    int idle_thread_count;
1404    process_score *ps;
1405    int free_length;
1406    int totally_free_length = 0;
1407    int free_slots[MAX_SPAWN_RATE];
1408    int last_non_dead;
1409    int total_non_dead;
1410    int active_thread_count = 0;
1411    int i, j;
1412
1413    /* initialize the free_list */ view japanese
1414    free_length = 0;
1415
1416    idle_thread_count = 0;
1417    last_non_dead = -1;
1418    total_non_dead = 0;
1419
1420    for (i = 0; i < ap_daemons_limit; ++i) {
1421        /* Initialization to satisfy the compiler. It doesn't know
1422         * that threads_per_child is always > 0 */ view japanese
1423        int any_dying_threads = 0;
1424        int any_dead_threads = 0;
1425        int all_dead_threads = 1;
1426        int child_threads_active = 0;
1427
1428        if (num_buckets > 1 && (i % num_buckets) != child_bucket) {
1429            /* We only care about child_bucket in this call */ view japanese
1430            continue;
1431        }
1432        if (i >= retained->max_daemons_limit &&
1433            totally_free_length == retained->idle_spawn_rate[child_bucket]) {
1434            /* short cut if all active processes have been examined and
1435             * enough empty scoreboard slots have been found
1436             */ view japanese
1437            break;
1438        }
1439        ps = &ap_scoreboard_image->parent[i];
1440        for (j = 0; j < threads_per_child; j++) {
1441            int status = ap_scoreboard_image->servers[i][j].status;
1442
1443            /* XXX any_dying_threads is probably no longer needed    GLA */ view japanese
1444            any_dying_threads = any_dying_threads ||
1445                                (status == SERVER_GRACEFUL);
1446            any_dead_threads = any_dead_threads || (status == SERVER_DEAD);
1447            all_dead_threads = all_dead_threads &&
1448                                   (status == SERVER_DEAD ||
1449                                    status == SERVER_GRACEFUL);
1450
1451            /* We consider a starting server as idle because we started it
1452             * at least a cycle ago, and if it still hasn't finished starting
1453             * then we're just going to swamp things worse by forking more.
1454             * So we hopefully won't need to fork more if we count it.
1455             * This depends on the ordering of SERVER_READY and SERVER_STARTING.
1456             */ view japanese
1457            if (ps->pid != 0) { /* XXX just set all_dead_threads in outer for
1458                                   loop if no pid?  not much else matters */ view japanese
1459                if (status <= SERVER_READY &&
1460                        !ps->quiescing &&
1461                        ps->generation == retained->mpm->my_generation) {
1462                    ++idle_thread_count;
1463                }
1464                if (status >= SERVER_READY && status < SERVER_GRACEFUL) {
1465                    ++child_threads_active;
1466                }
1467            }
1468        }
1469        active_thread_count += child_threads_active;
1470        if (any_dead_threads
1471                && totally_free_length < retained->idle_spawn_rate[child_bucket]
1472                && free_length < MAX_SPAWN_RATE / num_buckets
1473                && (!ps->pid               /* no process in the slot */ view japanese
1474                    || ps->quiescing)) {   /* or at least one is going away */ view japanese
1475            if (all_dead_threads) {
1476                /* great! we prefer these, because the new process can
1477                 * start more threads sooner.  So prioritize this slot
1478                 * by putting it ahead of any slots with active threads.
1479                 *
1480                 * first, make room by moving a slot that's potentially still
1481                 * in use to the end of the array
1482                 */ view japanese
1483                free_slots[free_length] = free_slots[totally_free_length];
1484                free_slots[totally_free_length++] = i;
1485            }
1486            else {
1487                /* slot is still in use - back of the bus
1488                 */ view japanese
1489                free_slots[free_length] = i;
1490            }
1491            ++free_length;
1492        }
1493        else if (child_threads_active == threads_per_child) {
1494            had_healthy_child = 1;
1495        }
1496        /* XXX if (!ps->quiescing)     is probably more reliable  GLA */ view japanese
1497        if (!any_dying_threads) {
1498            ++total_non_dead;
1499        }
1500        if (ps->pid != 0) {
1501            last_non_dead = i;
1502        }
1503    }
1504
1505    retained->max_daemons_limit = last_non_dead + 1;
1506
1507    if (retained->sick_child_detected) {
1508        if (had_healthy_child) {
1509            /* Assume this is a transient error, even though it may not be.  Leave
1510             * the server up in case it is able to serve some requests or the
1511             * problem will be resolved.
1512             */ view japanese
1513            retained->sick_child_detected = 0;
1514        }
1515        else if (child_bucket < num_buckets - 1) {
1516            /* check for had_healthy_child up to the last child bucket */ view japanese
1517            return;
1518        }
1519        else {
1520            /* looks like a basket case, as no child ever fully initialized; give up.
1521             */ view japanese
1522            retained->mpm->shutdown_pending = 1;
1523            child_fatal = 1;
1524            ap_log_error(APLOG_MARK, APLOG_ALERT, 0,
1525                         ap_server_conf, APLOGNO(02325)
1526                         "A resource shortage or other unrecoverable failure "
1527                         "was encountered before any child process initialized "
1528                         "successfully... httpd is exiting!");
1529            /* the child already logged the failure details */ view japanese
1530            return;
1531        }
1532    }
1533
1534    if (idle_thread_count > max_spare_threads / num_buckets) {
1535        /* Kill off one child */ view japanese
1536        ap_mpm_podx_signal(all_buckets[child_bucket].pod,
1537                           AP_MPM_PODX_GRACEFUL);
1538        retained->idle_spawn_rate[child_bucket] = 1;
1539    }
1540    else if (idle_thread_count < min_spare_threads / num_buckets) {
1541        /* terminate the free list */ view japanese
1542        if (free_length == 0) { /* scoreboard is full, can't fork */ view japanese
1543
1544            if (active_thread_count >= max_workers / num_buckets) {
1545                /* no threads are "inactive" - starting, stopping, etc. */ view japanese
1546                /* have we reached MaxRequestWorkers, or just getting close? */ view japanese
1547                if (0 == idle_thread_count) {
1548                    if (!retained->maxclients_reported) {
1549                        /* only report this condition once */ view japanese
1550                        ap_log_error(APLOG_MARK, APLOG_ERR, 0, ap_server_conf, APLOGNO(00286)
1551                                     "server reached MaxRequestWorkers "
1552                                     "setting, consider raising the "
1553                                     "MaxRequestWorkers setting");
1554                        retained->maxclients_reported = 1;
1555                    }
1556                } else {
1557                    if (!retained->near_maxclients_reported) {
1558                        ap_log_error(APLOG_MARK, APLOG_ERR, 0, ap_server_conf, APLOGNO(00287)
1559                                     "server is within MinSpareThreads of "
1560                                     "MaxRequestWorkers, consider raising the "
1561                                     "MaxRequestWorkers setting");
1562                        retained->near_maxclients_reported = 1;
1563                    }
1564                }
1565            }
1566            else {
1567                ap_log_error(APLOG_MARK, APLOG_ERR, 0,
1568                             ap_server_conf, APLOGNO(00288)
1569                             "scoreboard is full, not at MaxRequestWorkers");
1570            }
1571            retained->idle_spawn_rate[child_bucket] = 1;
1572        }
1573        else {
1574            if (free_length > retained->idle_spawn_rate[child_bucket]) {
1575                free_length = retained->idle_spawn_rate[child_bucket];
1576            }
1577            if (retained->idle_spawn_rate[child_bucket] >= 8) {
1578                ap_log_error(APLOG_MARK, APLOG_INFO, 0,
1579                             ap_server_conf, APLOGNO(00289)
1580                             "server seems busy, (you may need "
1581                             "to increase StartServers, ThreadsPerChild "
1582                             "or Min/MaxSpareThreads), "
1583                             "spawning %d children, there are around %d idle "
1584                             "threads, and %d total children", free_length,
1585                             idle_thread_count, total_non_dead);
1586            }
1587            for (i = 0; i < free_length; ++i) {
1588                make_child(ap_server_conf, free_slots[i], child_bucket);
1589            }
1590            /* the next time around we want to spawn twice as many if this
1591             * wasn't good enough, but not if we've just done a graceful
1592             */ view japanese
1593            if (retained->hold_off_on_exponential_spawning) {
1594                --retained->hold_off_on_exponential_spawning;
1595            }
1596            else if (retained->idle_spawn_rate[child_bucket]
1597                     < MAX_SPAWN_RATE / num_buckets) {
1598                retained->idle_spawn_rate[child_bucket] *= 2;
1599            }
1600        }
1601    }
1602    else {
1603        retained->idle_spawn_rate[child_bucket] = 1;
1604    }
1605}

1607static void server_main_loop(int remaining_children_to_start)
1608{
1609    int num_buckets = retained->mpm->num_buckets;
1610    int successive_kills = 0;
1611    ap_generation_t old_gen;
1612    int child_slot;
1613    apr_exit_why_e exitwhy;
1614    int status, processed_status;
1615    apr_proc_t pid;
1616    int i;
1617
1618    while (!retained->mpm->restart_pending && !retained->mpm->shutdown_pending) {
1619        ap_wait_or_timeout(&exitwhy, &status, &pid, pconf, ap_server_conf);
1620
1621        if (pid.pid != -1) {
1622            processed_status = ap_process_child_status(&pid, exitwhy, status);
1623            child_slot = ap_find_child_by_pid(&pid);
1624            if (processed_status == APEXIT_CHILDFATAL) {
1625                /* fix race condition found in PR 39311
1626                 * A child created at the same time as a graceful happens 
1627                 * can find the lock missing and create a fatal error.
1628                 * It is not fatal for the last generation to be in this state.
1629                 */ view japanese
1630                if (child_slot < 0
1631                    || ap_get_scoreboard_process(child_slot)->generation
1632                       == retained->mpm->my_generation) {
1633                    retained->mpm->shutdown_pending = 1;
1634                    child_fatal = 1;
1635                    return;
1636                }
1637                else {
1638                    ap_log_error(APLOG_MARK, APLOG_WARNING, 0, ap_server_conf, APLOGNO(00290)
1639                                 "Ignoring fatal error in child of previous "
1640                                 "generation (pid %ld).",
1641                                 (long)pid.pid);
1642                    retained->sick_child_detected = 1;
1643                }
1644            }
1645            else if (processed_status == APEXIT_CHILDSICK) {
1646                /* tell perform_idle_server_maintenance to check into this
1647                 * on the next timer pop
1648                 */ view japanese
1649                retained->sick_child_detected = 1;
1650            }
1651            /* non-fatal death... note that it's gone in the scoreboard. */ view japanese
1652            if (child_slot >= 0) {
1653                process_score *ps;
1654
1655                for (i = 0; i < threads_per_child; i++)
1656                    ap_update_child_status_from_indexes(child_slot, i,
1657                                                        SERVER_DEAD, NULL);
1658
1659                worker_note_child_killed(child_slot, 0, 0);
1660                ps = &ap_scoreboard_image->parent[child_slot];
1661                ps->quiescing = 0;
1662                if (processed_status == APEXIT_CHILDSICK) {
1663                    /* resource shortage, minimize the fork rate */
1664                    retained->idle_spawn_rate[child_slot % num_buckets] = 1;
1665                }
1666                else if (remaining_children_to_start
1667                    && child_slot < ap_daemons_limit) {
1668                    /* we're still doing a 1-for-1 replacement of dead
1669                     * children with new children
1670                     */ view japanese
1671                    make_child(ap_server_conf, child_slot,
1672                               child_slot % num_buckets);
1673                    --remaining_children_to_start;
1674                }
1675            }
1676            else if (ap_unregister_extra_mpm_process(pid.pid, &old_gen) == 1) {
1677                worker_note_child_killed(-1, /* already out of the scoreboard */ view japanese
1678                                         pid.pid, old_gen);
1679                if (processed_status == APEXIT_CHILDSICK
1680                    && old_gen == retained->mpm->my_generation) {
1681                    /* resource shortage, minimize the fork rate */ view japanese
1682                    for (i = 0; i < num_buckets; i++) {
1683                        retained->idle_spawn_rate[i] = 1;
1684                    }
1685                }
1686#if APR_HAS_OTHER_CHILD
1687            }
1688            else if (apr_proc_other_child_alert(&pid, APR_OC_REASON_DEATH,
1689                                                status) == 0) {
1690                /* handled */ view japanese
1691#endif
1692            }
1693            else if (retained->mpm->was_graceful) {
1694                /* Great, we've probably just lost a slot in the
1695                 * scoreboard.  Somehow we don't know about this child.
1696                 */ view japanese
1697                ap_log_error(APLOG_MARK, APLOG_WARNING, 0,
1698                             ap_server_conf, APLOGNO(00291)
1699                             "long lost child came home! (pid %ld)",
1700                             (long)pid.pid);
1701            }
1702            /* Don't perform idle maintenance when a child dies,
1703             * only do it when there's a timeout.  Remember only a
1704             * finite number of children can die, and it's pretty
1705             * pathological for a lot to die suddenly.  If a child is
1706             * killed by a signal (faulting) we want to restart it ASAP
1707             * though, up to 3 successive faults or we stop this until
1708             * a timeout happens again (to avoid the flood of fork()ed
1709             * processes that keep being killed early).
1710             */ view japanese
1711            if (child_slot < 0 || !APR_PROC_CHECK_SIGNALED(exitwhy)) {
1712                continue;
1713            }
1714            if (++successive_kills >= 3) {
1715                if (successive_kills % 10 == 3) {
1716                    ap_log_error(APLOG_MARK, APLOG_WARNING, 0,
1717                                 ap_server_conf, APLOGNO(10393)
1718                                 "children are killed successively!");
1719                }
1720                continue;
1721            }
1722            ++remaining_children_to_start;
1723        }
1724        else {
1725            successive_kills = 0;
1726        }
1727
1728        if (remaining_children_to_start) {
1729            /* we hit a 1 second timeout in which none of the previous
1730             * generation of children needed to be reaped... so assume
1731             * they're all done, and pick up the slack if any is left.
1732             */ view japanese
1733            startup_children(remaining_children_to_start);
1734            remaining_children_to_start = 0;
1735            /* In any event we really shouldn't do the code below because
1736             * few of the servers we just started are in the IDLE state
1737             * yet, so we'd mistakenly create an extra server.
1738             */ view japanese
1739            continue;
1740        }
1741
1742        for (i = 0; i < num_buckets; i++) {
1743            perform_idle_server_maintenance(i);
1744        }
1745    }
1746}

1748static int worker_run(apr_pool_t *_pconf, apr_pool_t *plog, server_rec *s)
1749{
1750    int num_buckets = retained->mpm->num_buckets;
1751    int remaining_children_to_start;
1752    int i;
1753
1754    ap_log_pid(pconf, ap_pid_fname);
1755
1756    if (!retained->mpm->was_graceful) {
1757        if (ap_run_pre_mpm(s->process->pool, SB_SHARED) != OK) {
1758            retained->mpm->mpm_state = AP_MPMQ_STOPPING;
1759            return !OK;
1760        }
1761        /* fix the generation number in the global score; we just got a new,
1762         * cleared scoreboard
1763         */ view japanese
1764        ap_scoreboard_image->global->running_generation = retained->mpm->my_generation;
1765    }
1766
1767    ap_unixd_mpm_set_signals(pconf, one_process);
1768
1769    /* Don't thrash since num_buckets depends on the
1770     * system and the number of online CPU cores...
1771     */ view japanese
1772    if (ap_daemons_limit < num_buckets)
1773        ap_daemons_limit = num_buckets;
1774    if (ap_daemons_to_start < num_buckets)
1775        ap_daemons_to_start = num_buckets;
1776    /* We want to create as much children at a time as the number of buckets,
