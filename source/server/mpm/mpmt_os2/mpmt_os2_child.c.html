43/* We don't need many processes,
44 * they're only for redundancy in the event of a crash
45 */
46#define HARD_SERVER_LIMIT 10
47
48/* Limit on the total number of threads per process
49 */
50#ifndef HARD_THREAD_LIMIT
51#define HARD_THREAD_LIMIT 256
52#endif
53
54#define ID_FROM_CHILD_THREAD(c, t)    ((c * HARD_THREAD_LIMIT) + t)
55
56typedef struct {
57    apr_pool_t *pconn;
58    apr_socket_t *conn_sd;
59} worker_args_t;

61#define WORKTYPE_CONN 0
62#define WORKTYPE_EXIT 1

74extern HMTX ap_mpm_accept_mutex;

82static void clean_child_exit(int code)
83{
84    if (pchild) {
85        apr_pool_destroy(pchild);
86    }
87
88    exit(code);
89}

93void ap_mpm_child_main(apr_pool_t *pconf)
94{
95    ap_listen_rec *lr = NULL;
96    int requests_this_child = 0;
97    int rv = 0;
98    unsigned long ulTimes;
99    int my_pid = getpid();
100    ULONG rc, c;
101    HQUEUE workq;
102    apr_pollset_t *pollset;
103    int num_listeners;
104    TID server_maint_tid;
105    void *sb_mem;
106
107    /* Stop Ctrl-C/Ctrl-Break signals going to child processes */
108    DosSetSignalExceptionFocus(0, &ulTimes);
109    set_signals();
110
111    /* Create pool for child */
112    apr_pool_create(&pchild, pconf);
113    apr_pool_tag(pchild, "pchild");
114
115    ap_run_child_init(pchild, ap_server_conf);
116
117    /* Create an event semaphore used to trigger other threads to shutdown */
118    rc = DosCreateEventSem(NULL, &shutdown_event, 0, FALSE);
119
120    if (rc) {
121        ap_log_error(APLOG_MARK, APLOG_ERR, APR_FROM_OS_ERROR(rc), ap_server_conf, APLOGNO(00189)
122                     "unable to create shutdown semaphore, exiting");
123        clean_child_exit(APEXIT_CHILDFATAL);
124    }
125
126    /* Gain access to the scoreboard. */
127    rc = DosGetNamedSharedMem(&sb_mem, ap_scoreboard_fname,
128                              PAG_READ|PAG_WRITE);
129
130    if (rc) {
131        ap_log_error(APLOG_MARK, APLOG_ERR, APR_FROM_OS_ERROR(rc), ap_server_conf, APLOGNO(00190)
132                     "scoreboard not readable in child, exiting");
133        clean_child_exit(APEXIT_CHILDFATAL);
134    }
135
136    ap_calc_scoreboard_size();
137    ap_init_scoreboard(sb_mem);
138
139    /* Gain access to the accpet mutex */
140    rc = DosOpenMutexSem(NULL, &ap_mpm_accept_mutex);
141
142    if (rc) {
143        ap_log_error(APLOG_MARK, APLOG_ERR, APR_FROM_OS_ERROR(rc), ap_server_conf, APLOGNO(00191)
144                     "accept mutex couldn't be accessed in child, exiting");
145        clean_child_exit(APEXIT_CHILDFATAL);
146    }
147
148    /* Find our pid in the scoreboard so we know what slot our parent allocated us */
149    for (child_slot = 0; ap_scoreboard_image->parent[child_slot].pid != my_pid && child_slot < HARD_SERVER_LIMIT; child_slot++);
150
151    if (child_slot == HARD_SERVER_LIMIT) {
152        ap_log_error(APLOG_MARK, APLOG_ERR, 0, ap_server_conf, APLOGNO(00192)
153                     "child pid not found in scoreboard, exiting");
154        clean_child_exit(APEXIT_CHILDFATAL);
155    }
156
157    ap_my_generation = ap_scoreboard_image->parent[child_slot].generation;
158    memset(ap_scoreboard_image->servers[child_slot], 0, sizeof(worker_score) * HARD_THREAD_LIMIT);
159
160    /* Set up an OS/2 queue for passing connections & termination requests
161     * to worker threads
162     */
163    rc = DosCreateQueue(&workq, QUE_FIFO, apr_psprintf(pchild, "/queues/httpd/work.%d", my_pid));
164
165    if (rc) {
166        ap_log_error(APLOG_MARK, APLOG_ERR, APR_FROM_OS_ERROR(rc), ap_server_conf, APLOGNO(00193)
167                     "unable to create work queue, exiting");
168        clean_child_exit(APEXIT_CHILDFATAL);
169    }
170
171    /* Create initial pool of worker threads */
172    for (c = 0; c < ap_min_spare_threads; c++) {
173//        ap_scoreboard_image->servers[child_slot][c].tid = _beginthread(worker_main, NULL, 128*1024, (void *)c);
174    }
175
176    /* Start maintenance thread */
177    server_maint_tid = _beginthread(server_maintenance, NULL, 32768, NULL);
178
179    /* Set up poll */
180    for (num_listeners = 0, lr = ap_listeners; lr; lr = lr->next) {
181        num_listeners++;
182    }
183
184    apr_pollset_create(&pollset, num_listeners, pchild, 0);
185
186    for (lr = ap_listeners; lr != NULL; lr = lr->next) {
187        apr_pollfd_t pfd = { 0 };
188
189        pfd.desc_type = APR_POLL_SOCKET;
190        pfd.desc.s = lr->sd;
191        pfd.reqevents = APR_POLLIN;
192        pfd.client_data = lr;
193        apr_pollset_add(pollset, &pfd);
194    }
195
196    /* Main connection accept loop */
197    do {
198        apr_pool_t *pconn;
199        worker_args_t *worker_args;
200        int last_poll_idx = 0;
201
202        apr_pool_create(&pconn, pchild);
203        apr_pool_tag(pconn, "transaction");
204        worker_args = apr_palloc(pconn, sizeof(worker_args_t));
205        worker_args->pconn = pconn;
206
207        if (num_listeners == 1) {
208            rv = apr_socket_accept(&worker_args->conn_sd, ap_listeners->sd, pconn);
209        } else {
210            const apr_pollfd_t *poll_results;
211            apr_int32_t num_poll_results;
212
213            rc = DosRequestMutexSem(ap_mpm_accept_mutex, SEM_INDEFINITE_WAIT);
214
215            if (shutdown_pending) {
216                DosReleaseMutexSem(ap_mpm_accept_mutex);
217                break;
218            }
219
220            rv = APR_FROM_OS_ERROR(rc);
221
222            if (rv == APR_SUCCESS) {
223                rv = apr_pollset_poll(pollset, -1, &num_poll_results, &poll_results);
224                DosReleaseMutexSem(ap_mpm_accept_mutex);
225            }
226
227            if (rv == APR_SUCCESS) {
228                if (last_poll_idx >= num_listeners) {
229                    last_poll_idx = 0;
230                }
231
232                lr = poll_results[last_poll_idx++].client_data;
233                rv = apr_socket_accept(&worker_args->conn_sd, lr->sd, pconn);
234                last_poll_idx++;
235            }
236        }
237
238        if (rv != APR_SUCCESS) {
239            if (!APR_STATUS_IS_EINTR(rv)) {
240                ap_log_error(APLOG_MARK, APLOG_ERR, rv, ap_server_conf, APLOGNO(00194)
241                             "apr_socket_accept");
242                clean_child_exit(APEXIT_CHILDFATAL);
243            }
244        } else {
245            DosWriteQueue(workq, WORKTYPE_CONN, sizeof(worker_args_t), worker_args, 0);
246            requests_this_child++;
247        }
248
249        if (ap_max_requests_per_child != 0 && requests_this_child >= ap_max_requests_per_child)
250            break;
251    } while (!shutdown_pending && ap_my_generation == ap_scoreboard_image->global->running_generation);
252
253    ap_scoreboard_image->parent[child_slot].quiescing = 1;
254    DosPostEventSem(shutdown_event);
255    DosWaitThread(&server_maint_tid, DCWW_WAIT);
256
257    if (is_graceful) {
258        char someleft;
259
260        /* tell our worker threads to exit */
261        for (c=0; c<HARD_THREAD_LIMIT; c++) {
262            if (ap_scoreboard_image->servers[child_slot][c].status != SERVER_DEAD) {
263                DosWriteQueue(workq, WORKTYPE_EXIT, 0, NULL, 0);
264            }
265        }
266
267        do {
268            someleft = 0;
269
270            for (c=0; c<HARD_THREAD_LIMIT; c++) {
271                if (ap_scoreboard_image->servers[child_slot][c].status != SERVER_DEAD) {
272                    someleft = 1;
273                    DosSleep(1000);
274                    break;
275                }
276            }
277        } while (someleft);
278    } else {
279        DosPurgeQueue(workq);
280
281        for (c=0; c<HARD_THREAD_LIMIT; c++) {
282            if (ap_scoreboard_image->servers[child_slot][c].status != SERVER_DEAD) {
283                DosKillThread(ap_scoreboard_image->servers[child_slot][c].tid);
284            }
285        }
286    }
287
288    apr_pool_destroy(pchild);
289}

293void add_worker()
294{
295    int thread_slot;
296    int stacksize = ap_thread_stacksize == 0 ? 128*1024 : ap_thread_stacksize;
297
298    /* Find a free thread slot */
299    for (thread_slot=0; thread_slot < HARD_THREAD_LIMIT; thread_slot++) {
300        if (ap_scoreboard_image->servers[child_slot][thread_slot].status == SERVER_DEAD) {
301            ap_scoreboard_image->servers[child_slot][thread_slot].status = SERVER_STARTING;
302            ap_scoreboard_image->servers[child_slot][thread_slot].tid =
303                _beginthread(worker_main, NULL, stacksize, (void *)thread_slot);
304            break;
305        }
306    }
307}

311ULONG APIENTRY thread_exception_handler(EXCEPTIONREPORTRECORD *pReportRec,
312                                        EXCEPTIONREGISTRATIONRECORD *pRegRec,
313                                        CONTEXTRECORD *pContext,
314                                        PVOID p)
315{
316    int c;
317
318    if (pReportRec->fHandlerFlags & EH_NESTED_CALL) {
319        return XCPT_CONTINUE_SEARCH;
320    }
321
322    if (pReportRec->ExceptionNum == XCPT_ACCESS_VIOLATION ||
323        pReportRec->ExceptionNum == XCPT_INTEGER_DIVIDE_BY_ZERO) {
324        ap_log_error(APLOG_MARK, APLOG_ERR, 0, ap_server_conf, APLOGNO(00195)
325                     "caught exception in worker thread, initiating child shutdown pid=%d", getpid());
326        for (c=0; c<HARD_THREAD_LIMIT; c++) {
327            if (ap_scoreboard_image->servers[child_slot][c].tid == _gettid()) {
328                ap_scoreboard_image->servers[child_slot][c].status = SERVER_DEAD;
329                break;
330            }
331        }
332
333        /* Shut down process ASAP, it could be quite unhealthy & leaking resources */
334        shutdown_pending = 1;
335        ap_scoreboard_image->parent[child_slot].quiescing = 1;
336        kill(getpid(), SIGHUP);
337        DosUnwindException(UNWIND_ALL, 0, 0);
338    }
339
340    return XCPT_CONTINUE_SEARCH;
341}

345static void worker_main(void *vpArg)
346{
347    apr_thread_t *thd = NULL;
348    apr_os_thread_t osthd;
349    long conn_id;
350    conn_rec *current_conn;
351    apr_pool_t *pconn;
352    apr_allocator_t *allocator;
353    apr_bucket_alloc_t *bucket_alloc;
354    worker_args_t *worker_args;
355    HQUEUE workq;
356    PID owner;
357    int rc;
358    REQUESTDATA rd;
359    ULONG len;
360    BYTE priority;
361    int thread_slot = (int)vpArg;
362    EXCEPTIONREGISTRATIONRECORD reg_rec = { NULL, thread_exception_handler };
363    ap_sb_handle_t *sbh;
364
365    /* Trap exceptions in this thread so we don't take down the whole process */
366    DosSetExceptionHandler( &reg_rec );
367
368    osthd = apr_os_thread_current();
369    apr_os_thread_put(&thd, &osthd, pchild);
370
371    rc = DosOpenQueue(&owner, &workq,
372                      apr_psprintf(pchild, "/queues/httpd/work.%d", getpid()));
373
374    if (rc) {
375        ap_log_error(APLOG_MARK, APLOG_ERR, APR_FROM_OS_ERROR(rc), ap_server_conf, APLOGNO(00196)
376                     "unable to open work queue, exiting");
377        ap_scoreboard_image->servers[child_slot][thread_slot].tid = 0;
378    }
379
380    conn_id = ID_FROM_CHILD_THREAD(child_slot, thread_slot);
381    ap_update_child_status_from_indexes(child_slot, thread_slot, SERVER_READY,
382                                        NULL);
383
384    apr_allocator_create(&allocator);
385    apr_allocator_max_free_set(allocator, ap_max_mem_free);
386    bucket_alloc = apr_bucket_alloc_create_ex(allocator);
387
388    while (rc = DosReadQueue(workq, &rd, &len, (PPVOID)&worker_args, 0, DCWW_WAIT, &priority, NULLHANDLE),
389           rc == 0 && rd.ulData != WORKTYPE_EXIT) {
390        pconn = worker_args->pconn;
391        ap_create_sb_handle(&sbh, pconn, child_slot, thread_slot);
392        current_conn = ap_run_create_connection(pconn, ap_server_conf,
393                                                worker_args->conn_sd, conn_id,
394                                                sbh, bucket_alloc);
395
396        if (current_conn) {
397            current_conn->current_thread = thd;
398            ap_process_connection(current_conn, worker_args->conn_sd);
399            ap_lingering_close(current_conn);
400        }
401
402        apr_pool_destroy(pconn);
403        ap_update_child_status_from_indexes(child_slot, thread_slot,
404                                            SERVER_READY, NULL);
405    }
406
407    ap_update_child_status_from_indexes(child_slot, thread_slot, SERVER_DEAD,
408                                        NULL);
409
410    apr_bucket_alloc_destroy(bucket_alloc);
411    apr_allocator_destroy(allocator);
412}

416static void server_maintenance(void *vpArg)
417{
418    int num_idle, num_needed;
419    ULONG num_pending = 0;
420    int threadnum;
421    HQUEUE workq;
422    ULONG rc;
423    PID owner;
424
425    rc = DosOpenQueue(&owner, &workq,
426                      apr_psprintf(pchild, "/queues/httpd/work.%d", getpid()));
427
428    if (rc) {
429        ap_log_error(APLOG_MARK, APLOG_ERR, APR_FROM_OS_ERROR(rc), ap_server_conf, APLOGNO(00197)
430                     "unable to open work queue in maintenance thread");
431        return;
432    }
433
434    do {
435        for (num_idle=0, threadnum=0; threadnum < HARD_THREAD_LIMIT; threadnum++) {
436            num_idle += ap_scoreboard_image->servers[child_slot][threadnum].status == SERVER_READY;
437        }
438
439        DosQueryQueue(workq, &num_pending);
440        num_needed = ap_min_spare_threads - num_idle + num_pending;
441
442        if (num_needed > 0) {
443            for (threadnum=0; threadnum < num_needed; threadnum++) {
444                add_worker();
445            }
446        }
447
448        if (num_idle - num_pending > ap_max_spare_threads) {
449            DosWriteQueue(workq, WORKTYPE_EXIT, 0, NULL, 0);
450        }
451    } while (DosWaitEventSem(shutdown_event, 500) == ERROR_TIMEOUT);
452}

456/* Signal handling routines */
457
458static void sig_term(int sig)
459{
460    shutdown_pending = 1;
461    is_graceful = 0;
462    signal(SIGTERM, SIG_DFL);
463}
464
465
466
467static void sig_hup(int sig)
468{
469    shutdown_pending = 1;
470    is_graceful = 1;
471}

474
475static void set_signals()
476{
477    struct sigaction sa;
478
479    sigemptyset(&sa.sa_mask);
480    sa.sa_flags = 0;
481    sa.sa_handler = sig_term;
482
483    if (sigaction(SIGTERM, &sa, NULL) < 0)
484        ap_log_error(APLOG_MARK, APLOG_WARNING, errno, ap_server_conf, APLOGNO(00198) "sigaction(SIGTERM)");
485
486    sa.sa_handler = sig_hup;
487
488    if (sigaction(SIGHUP, &sa, NULL) < 0)
489        ap_log_error(APLOG_MARK, APLOG_WARNING, errno, ap_server_conf, APLOGNO(00199) "sigaction(SIGHUP)");
490}
