54/* The maximum length of the queue of pending connections, as defined
55 * by listen(2).  Under some systems, it should be increased if you
56 * are experiencing a heavy TCP SYN flood attack.
57 *
58 * It defaults to 511 instead of 512 because some systems store it
59 * as an 8-bit datatype; 512 truncated to 8-bits is 0, while 511 is
60 * 255 when truncated.
61 */
62#ifndef DEFAULT_LISTENBACKLOG
63#define DEFAULT_LISTENBACKLOG 511
64#endif

101/**
102 * Make sure all child processes that have been spawned by the parent process
103 * have died.  This includes process registered as "other_children".
104 *
105 * @param terminate Either 1 or 0.  If 1, send the child processes SIGTERM

107 *        on its own before signalling it.
108 * @param mpm_callback Callback invoked for each dead child process

110 * @note The MPM child processes which are reclaimed are those listed
111 * in the scoreboard as well as those currently registered via
112 * ap_register_extra_mpm_process().
113 */
114AP_DECLARE(void) ap_reclaim_child_processes(int terminate,

422/**
423 * Notification that connection handling is suspending (disassociating from the
424 * current thread)
425 * @param c The current connection
426 * @param r The current request, or NULL if there is no active request
427 * @ingroup hooks
428 * @see ap_hook_resume_connection
429 * @note This hook is not implemented by MPMs like Prefork and Worker which 
430 * handle all processing of a particular connection on the same thread.
431 * @note This hook will be called on the thread that was previously
432 * processing the connection.
433 * @note This hook is not called at the end of connection processing.  This
434 * hook only notifies a module when processing of an active connection is
435 * suspended.
436 * @note Resumption and subsequent suspension of a connection solely to perform
437 * I/O by the MPM, with no execution of non-MPM code, may not necessarily result
438 * in a call to this hook.
439 */

441                (conn_rec *c, request_rec *r))

443/**
444 * Notification that connection handling is resuming (associating with a thread)
445 * @param c The current connection
446 * @param r The current request, or NULL if there is no active request
447 * @ingroup hooks
448 * @see ap_hook_suspend_connection
449 * @note This hook is not implemented by MPMs like Prefork and Worker which 
450 * handle all processing of a particular connection on the same thread.
451 * @note This hook will be called on the thread that will resume processing
452 * the connection.
453 * @note This hook is not called at the beginning of connection processing.
454 * This hook only notifies a module when processing resumes for a
455 * previously-suspended connection.
456 * @note Resumption and subsequent suspension of a connection solely to perform
