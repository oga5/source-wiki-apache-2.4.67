29/*
30 * MODULE_MAGIC_NUMBER_MAJOR
31 * Major API changes that could cause compatibility problems for older modules
32 * such as structure size changes.  No binary compatibility is possible across
33 * a change in the major version.
34 *
35 * MODULE_MAGIC_NUMBER_MINOR
36 * Minor API changes that do not cause binary compatibility problems.
37 * Should be reset to 0 when upgrading MODULE_MAGIC_NUMBER_MAJOR.
38 *
39 * See the AP_MODULE_MAGIC_AT_LEAST macro below for an example.
40 */

624/**
625 * Determine if the server's current MODULE_MAGIC_NUMBER is at least a
626 * specified value.
627 *
628 * Useful for testing for features.
629 * For example, suppose you wish to use the apr_table_overlap
630 *    function.  You can do this:
631 *
632 * \code
633 * #if AP_MODULE_MAGIC_AT_LEAST(19980812,2)
634 *     ... use apr_table_overlap()
635 * #else
636 *     ... alternative code which doesn't use apr_table_overlap()
637 * #endif
638 * \endcode
639 *
640 * @param major The major module magic number
641 * @param minor The minor module magic number
642 * @def AP_MODULE_MAGIC_AT_LEAST(int major, int minor)
643 */
644#define AP_MODULE_MAGIC_AT_LEAST(major,minor)           \
645    ((major) < MODULE_MAGIC_NUMBER_MAJOR                \
646     || ((major) == MODULE_MAGIC_NUMBER_MAJOR           \
647         && (minor) <= MODULE_MAGIC_NUMBER_MINOR))
