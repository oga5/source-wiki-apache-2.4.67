43/**
44 * How the directives arguments should be parsed.
45 * @remark Note that for all of these except RAW_ARGS, the config routine is
46 *      passed a freshly allocated string which can be modified or stored
47 *      or whatever...
48 */ view japanese
49enum cmd_how {
50    RAW_ARGS,           /**< cmd_func parses command line itself */ view japanese
51    TAKE1,              /**< one argument only */ view japanese
52    TAKE2,              /**< two arguments only */ view japanese
53    ITERATE,            /**< one argument, occurring multiple times
54                         * (e.g., IndexIgnore)
55                         */ view japanese
56    ITERATE2,           /**< two arguments, 2nd occurs multiple times
57                         * (e.g., AddIcon)
58                         */ view japanese
59    FLAG,               /**< One of 'On' or 'Off' */ view japanese
60    NO_ARGS,            /**< No args at all, e.g. &lt;/Directory&gt; */ view japanese
61    TAKE12,             /**< one or two arguments */ view japanese
62    TAKE3,              /**< three arguments only */ view japanese
63    TAKE23,             /**< two or three arguments */ view japanese
64    TAKE123,            /**< one, two or three arguments */ view japanese
65    TAKE13,             /**< one or three arguments */ view japanese
66    TAKE_ARGV           /**< an argc and argv are passed */ view japanese
67};

203typedef struct command_struct command_rec;
204struct command_struct {
205    /** Name of this command */ view japanese
206    const char *name;
207    /** The function to be called when this directive is parsed */ view japanese

210    void *cmd_data;
