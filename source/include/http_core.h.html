51 * The most basic server code is encapsulated in a single module
52 * known as the core, which is just *barely* functional enough to
53 * serve documents, though not terribly well.

55 * Largely for NCSA back-compatibility reasons, the core needs to

91/**
92 * @defgroup get_remote_host Remote Host Resolution
93 * @ingroup APACHE_CORE_HTTPD

96/** REMOTE_HOST returns the hostname, or NULL if the hostname
97 * lookup fails.  It will force a DNS lookup according to the
98 * HostnameLookups setting.
99 */
100#define REMOTE_HOST (0)
101
102/** REMOTE_NAME returns the hostname, or the dotted quad if the
103 * hostname lookup fails.  It will force a DNS lookup according
104 * to the HostnameLookups setting.
105 */
106#define REMOTE_NAME (1)
107
108/** REMOTE_NOLOOKUP is like REMOTE_NAME except that a DNS lookup is
109 * never forced.
110 */
111#define REMOTE_NOLOOKUP (2)
112
113/** REMOTE_DOUBLE_REV will always force a DNS lookup, and also force
114 * a double reverse lookup, regardless of the HostnameLookups
115 * setting.  The result is the (double reverse checked) hostname,
116 * or NULL if any of the lookups fail.
117 */
118#define REMOTE_DOUBLE_REV (3)

391/**
392 * @brief  Per-request configuration
393*/
394typedef struct {
395    /** bucket brigade used by getline for look-ahead and
396     * ap_get_client_block for holding left-over request body */
397    struct apr_bucket_brigade *bb;
398
399    /** an array of per-request working data elements, accessed
400     * by ID using ap_get_request_note()
401     * (Use ap_register_request_note() during initialization
402     * to add elements)
403     */
404    void **notes;

406    /** Custom response strings registered via ap_custom_response(),
407     * or NULL; check per-dir config if nothing found here
408     */
409    char **response_code_strings; /* from ap_custom_response(), not from
410                                   * ErrorDocument
411                                   */
412
413    /** per-request document root of the server. This allows mass vhosting
414     * modules better compatibility with some scripts. Normally the
415     * context_* info should be used instead */
416    const char *document_root;
417
418    /*
419     * more fine-grained context information which is set by modules like
420     * mod_alias and mod_userdir
421     */
422    /** the context root directory on disk for the current resource,
423     *  without trailing slash
424     */
425    const char *context_document_root;
426    /** the URI prefix that corresponds to the context_document_root directory,
427     *  without trailing slash
428     */
429    const char *context_prefix;
430
431    /** There is a script processor installed on the output filter chain,
432     * so it needs the default_handler to deliver a (script) file into
433     * the chain so it can process it. Normally, default_handler only
434     * serves files on a GET request (assuming the file is actual content),
435     * since other methods are not content-retrieval. This flag overrides
436     * that behavior, stating that the "content" is actually a script and
437     * won't actually be delivered as the response for the non-GET method.
438     */
439    int deliver_script;
440
441    /** Should addition of charset= be suppressed for this request?
442     */
443    int suppress_charset;
