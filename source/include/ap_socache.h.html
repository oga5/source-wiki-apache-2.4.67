38/** If this flag is set, the store/retrieve/remove/status interfaces

48/** A cache instance. */ view japanese

60};

82                                             const unsigned char *data,

86/** A socache provider structure.  socache providers are registered
87 * with the ap_provider.h interface using the AP_SOCACHE_PROVIDER_*
88 * constants. */ view japanese
89typedef struct ap_socache_provider_t {
90    /** Canonical provider name. */ view japanese
91    const char *name;
92
93    /** Bitmask of AP_SOCACHE_FLAG_* flags. */ view japanese
94    unsigned int flags;
95
96    /**
97     * Create a small object cache based on the given configuration
98     * string.  The instance pointer returned in the instance
99     * parameter will be passed as the first argument to subsequent
100     * invocations.
101     *
102     * @param instance Output parameter to which instance object is written.
103     * @param arg User-specified configuration string.  May be NULL to
104     *        force use of defaults.
105     * @param tmp Pool to be used for any temporary allocations
106     * @param p Pool to be use for any allocations lasting as long as
107     * the created instance
108     * @return NULL on success, or an error string on failure.
109     */ view japanese
110    const char *(*create)(ap_socache_instance_t **instance, const char *arg,
111                          apr_pool_t *tmp, apr_pool_t *p);
112
113    /**
114     * Initialize the cache.  The cname must be of maximum length 16
115     * characters, and uniquely identifies the consumer of the cache
116     * within the server; using the module name is recommended, e.g.
117     * "mod_ssl-sess".  This string may be used within a filesystem
118     * path so use of only alphanumeric [a-z0-9_-] characters is
119     * recommended.  If hints is non-NULL, it gives a set of hints for
120     * the provider.  Returns APR error code.
121     *
122     * @param instance The cache instance
123     * @param cname A unique string identifying the consumer of this API
124     * @param hints Optional hints argument describing expected cache use
125     * @param s Server structure to which the cache is associated
126     * @param pool Pool for long-lived allocations
127     * @return APR status value indicating success.
128     */ view japanese
129    apr_status_t (*init)(ap_socache_instance_t *instance, const char *cname,
130                         const struct ap_socache_hints *hints,
131                         server_rec *s, apr_pool_t *pool);
132
133    /**
134     * Destroy a given cache instance object.
135     * @param instance The cache instance to destroy.
136     * @param s Associated server structure (for logging purposes)
137     */ view japanese
138    void (*destroy)(ap_socache_instance_t *instance, server_rec *s);
139
140    /**
141     * Store an object in a cache instance.
142     * @param instance The cache instance
143     * @param s Associated server structure (for logging purposes)
144     * @param id Unique ID for the object; binary blob
145     * @param idlen Length of id blob
146     * @param expiry Absolute time at which the object expires
147     * @param data Data to store; binary blob
148     * @param datalen Length of data blob
149     * @param pool Pool for temporary allocations.
150     * @return APR status value.
151     */ view japanese
152    apr_status_t (*store)(ap_socache_instance_t *instance, server_rec *s,
153                          const unsigned char *id, unsigned int idlen,
154                          apr_time_t expiry,
155                          unsigned char *data, unsigned int datalen,
156                          apr_pool_t *pool);
157
158    /**
159     * Retrieve a cached object.
160     * 
161     * @param instance The cache instance
162     * @param s Associated server structure (for logging purposes)
163     * @param id Unique ID for the object; binary blob
164     * @param idlen Length of id blob
165     * @param data Output buffer to place retrievd data (binary blob)
166     * @param datalen On entry, length of data buffer; on exit, the
167     * number of bytes written to the data buffer.
168     * @param pool Pool for temporary allocations.
169     * @return APR status value; APR_NOTFOUND if the object was not
170     * found
171     */ view japanese
172    apr_status_t (*retrieve)(ap_socache_instance_t *instance, server_rec *s,
173                             const unsigned char *id, unsigned int idlen,
174                             unsigned char *data, unsigned int *datalen,
175                             apr_pool_t *pool);
176
177    /**
178     * Remove an object from the cache
179     *
180     * @param instance The cache instance
181     * @param s Associated server structure (for logging purposes)
182     * @param id Unique ID for the object; binary blob
183     * @param idlen Length of id blob
184     * @param pool Pool for temporary allocations.
185     */ view japanese
186    apr_status_t (*remove)(ap_socache_instance_t *instance, server_rec *s,
187                           const unsigned char *id, unsigned int idlen,
188                           apr_pool_t *pool);
189
190    /** 
191     * Dump the status of a cache instance for mod_status.  Will use
192     * the ap_r* interfaces to produce appropriate status output.
193     * XXX: ap_r* are deprecated, bad dogfood
194     *
195     * @param instance The cache instance
196     * @param r The request structure
197     * @param flags The AP_STATUS_* constants used (see mod_status.h)
198     */ view japanese
199    void (*status)(ap_socache_instance_t *instance, request_rec *r, int flags);
200
201    /**
202     * Dump all cached objects through an iterator callback.
203     * @param instance The cache instance
204     * @param s Associated server context (for processing and logging)
205     * @param userctx User defined pointer passed through to the iterator
206     * @param iterator The user provided callback function which will receive
207     * individual calls for each unexpired id/data pair
208     * @param pool Pool for temporary allocations.
209     * @return APR status value; APR_NOTFOUND if the object was not
210     * found
211     */ view japanese
212    apr_status_t (*iterate)(ap_socache_instance_t *instance, server_rec *s,
213                            void *userctx, ap_socache_iterator_t *iterator,
214                            apr_pool_t *pool);
215
216} ap_socache_provider_t;
217
218/** The provider group used to register socache providers. */ view japanese
219#define AP_SOCACHE_PROVIDER_GROUP "socache"
220/** The provider version used to register socache providers. */ view japanese
221#define AP_SOCACHE_PROVIDER_VERSION "0"
222
223/** Default provider name. */ view japanese
224#define AP_SOCACHE_DEFAULT_PROVIDER "default"
225
226#ifdef __cplusplus
227}
228#endif

230#endif /* AP_SOCACHE_H */ view japanese
