15# configuration which somehow breaks the test suite runs.  Appears

24function install_apx() {
25    local name=$1
26    local version=$2
27    local prefix=${HOME}/root/${name}-${version}
28    local build=${HOME}/build/${name}-${version}
29    local giturl=https://github.com/apache/${name}.git
30    local config=$3
31    local buildconf=$4
32    local commit=tarball
33
34    mkdir -p ${HOME}/build
35
36    if ! test -v TEST_APR_TARBALL; then
37         case $version in
38            trunk|*.x) ref=refs/heads/${version} ;;
39            *) ref=refs/tags/${version} ;;
40         esac
41
42         # Fetch the object ID (hash) of latest commit
43         commit=`git ls-remote ${giturl} ${ref} | cut -f1`
44         if test -z "$commit"; then
45            : Could not determine latest commit hash for ${ref} in ${giturl} - check branch is valid?
46            exit 1
47         fi
48    fi
49
50    # Blow away the cached install root if the cached install is stale
51    # or doesn't match the expected configuration.
52    grep -q "${version} ${commit} ${config} CC=$CC" ${HOME}/root/.key-${name} || rm -rf ${prefix}
53
54    if test -d ${prefix}; then
55        return 0
56    fi
57
58    if test -v TEST_APR_TARBALL; then
59         curl https://archive.apache.org/dist/apr/${name}-${version}.tar.gz > apx.tar.gz
60         tar -C ${HOME}/build -xzf apx.tar.gz
61         rm apx.tar.gz
62         pushd ${build}
63    else
64         git init -q ${build}
65         pushd $build
66         # Clone and checkout the commit identified above.
67         git remote add origin ${giturl}
68         git fetch -q --depth=1 origin ${commit}
69         git checkout ${commit}
70         ./buildconf ${buildconf}
71    fi
72
73    ./configure --prefix=${prefix} ${config}
74    make -j2
75    make install
76    popd
77
78    echo ${version} ${commit} "${config}" "CC=${CC}" > ${HOME}/root/.key-${name}
79}

81# Allow to load $HOME/build/apache/httpd/.gdbinit
82echo "add-auto-load-safe-path $HOME/work/httpd/httpd/.gdbinit" >> $HOME/.gdbinit
83
84# Unless either SKIP_TESTING or NO_TEST_FRAMEWORK are set, install
85# CPAN modules required to run the Perl test framework.
86if ! test -v SKIP_TESTING -o -v NO_TEST_FRAMEWORK; then
87    # Clear CPAN cache if necessary
88    if [ -v CLEAR_CACHE ]; then rm -rf ~/perl5; fi
89
90    if ! perl -V > perlver; then
91        : Perl binary broken
92        perl -V
93        exit 1
94    fi
95
96    # Compare the current "perl -V" output with the output at the time
97    # the cache was built; flush the cache if it's changed to avoid
98    # failure later when /usr/bin/perl refuses to load a mismatched XS
99    # module.
100    if ! cmp -s perlver ~/perl5/.perlver; then
101        : Purging cache since "perl -V" output has changed
102        rm -rf ~/perl5
103    fi
104    
105    cpanm --local-lib=~/perl5 local::lib && eval $(perl -I ~/perl5/lib/perl5/ -Mlocal::lib)
106
107    pkgs="Net::SSL LWP::Protocol::https                                 \
108           LWP::Protocol::AnyEvent::http ExtUtils::Embed Test::More     \
109           AnyEvent DateTime HTTP::DAV FCGI                             \
110           AnyEvent::WebSocket::Client Apache::Test"
111
112    # CPAN modules are to be used with the system Perl and always with
113    # CC=gcc, e.g. for the CC="gcc -m32" case the builds are not correct
114    # otherwise.
115    CC=gcc cpanm --notest $pkgs
116    unset pkgs
117
118    # Cache the perl -V output for future verification.
119    mv perlver ~/perl5/.perlver
120
121    # Make a shallow clone of httpd-tests git repo.
122    git clone -q --depth=1 https://github.com/apache/httpd-tests.git test/perl-framework
123fi
124
125# For LDAP testing, run slapd listening on port 8389 and populate the
126# directory as described in t/modules/ldap.t in the test framework:
127if test -v TEST_LDAP -a -x test/perl-framework/scripts/ldap-init.sh; then
128    docker build -t httpd_ldap -f test/travis_Dockerfile_slapd.centos test/
129    pushd test/perl-framework
130       ./scripts/ldap-init.sh
131    popd
132fi
133
134if test -v TEST_SSL; then
135    pushd test/perl-framework
136       ./scripts/memcached-init.sh
137       ./scripts/redis-init.sh
138    popd
139fi
140
141if test -v TEST_OPENSSL3; then
142    # Build the requested version of OpenSSL if it's not already
143    # installed in the cached ~/root
144    if ! test -f $HOME/root/openssl-is-${TEST_OPENSSL3}; then
145        # Remove any previous install.
146        rm -rf $HOME/root/openssl3
147
148        mkdir -p build/openssl
149        pushd build/openssl
150           curl "https://www.openssl.org/source/openssl-${TEST_OPENSSL3}.tar.gz" |
151              tar -xzf -
152           cd openssl-${TEST_OPENSSL3}
153           ./Configure --prefix=$HOME/root/openssl3 shared no-tests
154           make $MFLAGS
155           make install_sw
156           touch $HOME/root/openssl-is-${TEST_OPENSSL3}
157       popd
158    fi
159
160    # Point APR/APR-util at the installed version of OpenSSL.
161    if test -v APU_VERSION; then
162        APU_CONFIG="${APU_CONFIG} --with-openssl=$HOME/root/openssl3"
163    elif test -v APR_VERSION; then
164        APR_CONFIG="${APR_CONFIG} --with-openssl=$HOME/root/openssl3"
165    else
166        : Non-system APR/APR-util must be used to build with OpenSSL 3 to avoid mismatch with system libraries
167        exit 1
168    fi
169fi
170
171if test -v APR_VERSION; then
172    install_apx apr ${APR_VERSION} "${APR_CONFIG}"
173    ldd $HOME/root/apr-${APR_VERSION}/lib/libapr-?.so || true
174    APU_CONFIG="$APU_CONFIG --with-apr=$HOME/root/apr-${APR_VERSION}"
175fi
176
177if test -v APU_VERSION; then
178    install_apx apr-util ${APU_VERSION} "${APU_CONFIG}" --with-apr=$HOME/build/apr-${APR_VERSION}
179    ldd $HOME/root/apr-util-${APU_VERSION}/lib/libaprutil-?.so || true
180fi
181
182# Since librustls is not a package (yet) on any platform, we
183# build the version we want from source
184if test -v TEST_MOD_TLS -a -v RUSTLS_VERSION; then
185    if ! test -d $HOME/root/rustls; then
186        RUSTLS_HOME="$HOME/build/rustls-ffi"
187        git clone -q --depth=1 -b "$RUSTLS_VERSION" https://github.com/rustls/rustls-ffi.git "$RUSTLS_HOME"
188        pushd "$RUSTLS_HOME"
189            make install DESTDIR="$HOME/root/rustls"
190        popd
191    fi
192fi
