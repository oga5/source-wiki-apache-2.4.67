test/test_travis_conditions.sh
1#!/bin/sh -e
2# Script to test whether travis conditions match correctly.
3# "gem install travis-conditions". Tests .travis.yml in the cwd.
4
5cond_24x="`sed -n '/&condition_24x_only/{s/.*condition_24x_only//;p;q;}' .travis.yml`"
6cond_not_24x="`sed -n '/&condition_not_24x/{s/.*condition_not_24x//;p;q;}' .travis.yml`"
7
8echo Condition 2.4.x: ${cond_24x}
9echo Condition not 2.4.x: ${cond_not_24x}
10
11function testit()
12{
13    local data=$1
14    local expect=$2
15
16    is_24x=`travis-conditions eval "$cond_24x" --data "${data}"`
17    not_24x=`travis-conditions eval "$cond_not_24x" --data "${data}"`
18
19    if [ $is_24x = $not_24x ]; then
20        echo FAIL: Tests as both 2.4.x and not 2.4.x for "$data"
21        return 1
22    elif [ $expect = 2.4.x -a $is_24x = true ]; then
23        echo PASS
24    elif [ $expect = trunk ]; then
25        echo PASS
26    else
27        echo FAIL for "$data"
28        return 1
29    fi
30    return 0
31}
32
33testit '{"tag": "2.4.49"}' 2.4.x
34testit '{"tag": "2.5.59"}' trunk
35testit '{"branch": "2.4.x"}' 2.4.x
36testit '{"branch": "candidate-2.4.49"}' 2.4.x
37testit '{"branch": "2.4.55-candidate"}' 2.4.x
38testit '{"branch": "2.4-backport-branch"}' 2.4.x
39testit '{"branch": "2.4.x-something"}' 2.4.x
40testit '{"branch": "2.5.0"}' trunk
41testit '{"branch": "2.5.x"}' trunk
42testit '{"branch": "trunk"}' trunk
