22I usually just use this perl script, which I call "tcpdumpscii".
23Then run "tcpdumpscii -s 1500 -x [other tcpdump args]".
24
25  Bill
26
27#!/import/misc/bin/perl
28#
29#
30open(TCPDUMP,"tcpdump -l @ARGV|");
31while (<TCPDUMP>) {
32	if (/^\s+(\S\S)+/) {
33		$sav = $_;
34		$asc = "";
35		while (s/\s*(\S\S)\s*//) {
36			$i = hex($1);
37			if ($i < 32 || $i > 126) {
38				$asc .= ".";
39			} else {
40				$asc .= pack(C,hex($1));
41			}
42		}
43		$foo = "." x length($asc);
44		$_ = $sav;
45		s/\t/        /g;
46		s/^$foo/$asc/;
47	}
48	print;
49}
