test/travis_run_linux.sh
1#!/bin/bash -ex
2
3# Test for APLOGNO() macro errors (duplicates, empty args) etc.  For
4# trunk, run the updater script to see if it fails.  If it succeeds
5# and changes any files (because there was a missing argument), the
6# git diff will be non-empty, so fail for that case too.  For
7# non-trunk use a grep and only catch the empty argument case.
8if test -v TEST_LOGNO; then
9    if test -f docs/log-message-tags/update-log-msg-tags; then
10        find server modules os -name \*.c | \
11            xargs perl docs/log-message-tags/update-log-msg-tags
12        git diff --exit-code .
13        : PASSED
14        exit 0
15    else
16        set -o pipefail
17        if find server modules os -name \*.c | \
18                xargs grep -C1 --color=always 'APLOGNO()'; then
19            : FAILED
20            exit 1
21        else
22            : PASSED
23            exit 0
24        fi
25    fi
26fi
27
28### Installed apr/apr-util don't include the *.m4 files but the
29### Debian packages helpfully install them, so use the system APR to buildconf
30./buildconf --with-apr=/usr/bin/apr-1-config ${BUILDCONFIG}
31
32PREFIX=${PREFIX:-$HOME/build/httpd-root}
33
34# If perl-framework testing is required it is checked out here by
35# _before_linux.sh:
36if test -d test/perl-framework; then
37    CONFIG="$CONFIG --enable-load-all-modules"
38    if grep -q ^check: Makefile.in; then
39        CONFIG="--with-test-suite=test/perl-framework $CONFIG"
40        WITH_TEST_SUITE=1
41    fi
42
43    # Use the CPAN environment.
44    eval $(perl -I ~/perl5/lib/perl5/ -Mlocal::lib)
45fi
46if test -v APR_VERSION; then
47    CONFIG="$CONFIG --with-apr=$HOME/root/apr-${APR_VERSION}"
48else
49    CONFIG="$CONFIG --with-apr=/usr"
50fi
51if test -v APU_VERSION; then
52    CONFIG="$CONFIG --with-apr-util=$HOME/root/apr-util-${APU_VERSION}"
53else
54    CONFIG="$CONFIG --with-apr-util=/usr"
55fi
56
57# Pick up the rustls install built previously.
58if test -v TEST_MOD_TLS -a RUSTLS_VERSION; then
59  CONFIG="$CONFIG --with-tls --with-rustls=$HOME/root/rustls"
60fi
61
62if test -v TEST_OPENSSL3; then
63    CONFIG="$CONFIG --with-ssl=$HOME/root/openssl3"
64    export LD_LIBRARY_PATH=$HOME/root/openssl3/lib:$HOME/root/openssl3/lib64
65fi
66
67srcdir=$PWD
68
69if test -v TEST_VPATH; then
70    mkdir ../vpath
71    cd ../vpath
72fi
73
74$srcdir/configure --prefix=$PREFIX $CONFIG
75make $MFLAGS
76
77if test -v TEST_INSTALL; then
78   make install
79   pushd $PREFIX
80     test `./bin/apxs -q PREFIX` = $PREFIX
81     test `$PWD/bin/apxs -q PREFIX` = $PREFIX
82     ./bin/apxs -g -n foobar
83     cd foobar; make
84   popd
85fi
86
87if ! test -v SKIP_TESTING; then
88    set +e
89    RV=0
90
91    if test -v TEST_MALLOC; then
92        # Enable enhanced glibc malloc debugging, see mallopt(3)
93        export MALLOC_PERTURB_=65 MALLOC_CHECK_=3
94        export LIBC_FATAL_STDERR_=1
95    fi
96
97    if test -v TEST_UBSAN; then
98        export UBSAN_OPTIONS="log_path=$PWD/ubsan.log"
99    fi
100
101    if test -v TEST_ASAN; then
102        export ASAN_OPTIONS="log_path=$PWD/asan.log:detect_leaks=0"
103    fi
104
105    # Try to keep all potential coredumps from all processes
106    sudo sysctl -w kernel.core_uses_pid=1 2>/dev/null || true
107    # Systemd based systems might process core dumps via systemd-coredump.
108    # But we want to have local unprocessed files.
109    sudo sysctl -w kernel.core_pattern=core || true
110    ulimit -c unlimited 2>/dev/null || true
111
112    if test -v WITH_TEST_SUITE; then
113        make check TESTS="${TESTS}" TEST_CONFIG="${TEST_ARGS}"
114        RV=$?
115    else
116        test -v TEST_INSTALL || make install
117        pushd test/perl-framework
118            perl Makefile.PL -apxs $PREFIX/bin/apxs
119            make test APACHE_TEST_EXTRA_ARGS="${TEST_ARGS} ${TESTS}" | tee test.log
120            RV=${PIPESTATUS[0]}
121            # re-run failing tests with -v, avoiding set -e
122            if [ $RV -ne 0 ]; then
123                #mv t/logs/error_log t/logs/error_log_save
124                FAILERS=""
125                while read FAILER; do
126                    FAILERS="$FAILERS $FAILER"
127                done < <(awk '/Failed:/{print $1}' test.log)
128                if [ -n "$FAILERS" ]; then
129                    t/TEST -v $FAILERS || true
130                fi
131                # set -e would have killed us after the original t/TEST
132                rm -f test.log
133                #mv t/logs/error_log_save t/logs/error_log
134                false
135            fi
136        popd
137    fi
138
139    # Skip further testing if a core dump was created during the test
140    # suite run above.
141    if test $RV -eq 0 && test -n "`ls test/perl-framework/t/core{,.*} 2>/dev/null`"; then
142        RV=4
143    fi
144
145    if test -v TEST_SSL -a $RV -eq 0; then
146        pushd test/perl-framework
147            # Test loading encrypted private keys
148            ./t/TEST -defines "TEST_SSL_DES3_KEY TEST_SSL_PASSPHRASE_EXEC" t/ssl
149            RV=$?
150
151            # Log the OpenSSL version.
152            grep 'mod_ssl.*compiled against' t/logs/error_log | tail -n 1
153            
154            # Test various session cache backends
155            for cache in shmcb redis:localhost:6379 memcache:localhost:11211; do
156                test $RV -eq 0 || break
157
158                SSL_SESSCACHE=$cache ./t/TEST -sslproto TLSv1.2 -defines TEST_SSL_SESSCACHE -start
159                ./t/TEST t/ssl
160                RV=$?
161                ./t/TEST -stop
162                SRV=$?
163                if test $RV -eq 0 -a $SRV -ne 0; then
164                    RV=$SRV
165                fi
166            done
167        popd
168    fi
169
170    if test -v LITMUS -a $RV -eq 0; then
171        pushd test/perl-framework
172           mkdir -p t/htdocs/modules/dav
173           ./t/TEST -start
174           # litmus uses $TESTS, so unset it.
175           unset TESTS
176           litmus http://localhost:8529/modules/dav/
177           RV=$?
178           ./t/TEST -stop
179        popd
180    fi
181
182    if test $RV -ne 0 && test -f test/perl-framework/t/logs/error_log; then
183        grep -v ':\(debug\|trace[12345678]\)\]' test/perl-framework/t/logs/error_log
184    fi
185
186    if test -v TEST_CORE -a $RV -eq 0; then
187        # Run HTTP/2 tests.
188        MPM=event py.test-3 test/modules/core
189        RV=$?
190    fi
191
192    if test -v TEST_H2 -a $RV -eq 0; then
193        # Build the test clients
194        (cd test/clients && make)
195        # Run HTTP/2 tests.
196        MPM=event py.test-3 test/modules/http2
197        RV=$?
198        if test $RV -eq 0; then
199          MPM=worker py.test-3 test/modules/http2
200          RV=$?
201        fi
202    fi
203
204    if test -v TEST_MD -a $RV -eq 0; then
205        # Run ACME tests.
206        # need the go based pebble as ACME test server
207        # which is a package on debian sid, but not on focal
208        export GOPATH=${PREFIX}/gocode
209        mkdir -p "${GOPATH}"
210        export PATH="${GOROOT}/bin:${GOPATH}/bin:${PATH}"
211        go get -u github.com/letsencrypt/pebble/...
212        (cd $GOPATH/src/github.com/letsencrypt/pebble && go install ./...)
213
214        py.test-3 test/modules/md
215        RV=$?
216    fi
217
218    # Catch cases where abort()s get logged to stderr by libraries but
219    # only cause child processes to terminate e.g. during shutdown,
220    # which may not otherwise trigger test failures.
221
222    # "glibc detected": printed with LIBC_FATAL_STDERR_/MALLOC_CHECK_
223    # glibc will abort when malloc errors are detected.  This will get
224    # caught by the segfault grep as well.
225
226    # "pool concurrency check": printed by APR built with
227    # --enable-thread-debug when an APR pool concurrency check aborts
228
229    for phrase in 'Segmentation fault' 'glibc detected' 'pool concurrency check:' 'Assertion.*failed'; do
230        # Ignore IO/debug logs
231        if grep -v ':\(debug\|trace[12345678]\)\]' test/perl-framework/t/logs/error_log | grep -q "$phrase"; then
232            grep --color=always -C5 "$phrase" test/perl-framework/t/logs/error_log
233            RV=2
234        fi
235    done
236
237    if test -v TEST_UBSAN && test -n "`ls ubsan.log.* 2>/dev/null`"; then
238        cat ubsan.log.*
239        RV=3
240    fi
241
242    if test -v TEST_ASAN && test -n "`ls asan.log.* 2>/dev/null`"; then
243        cat asan.log.*
244
245        # ASan can report memory leaks, fail on errors only
246        if grep -q "ERROR: AddressSanitizer:" `ls asan.log.*`; then
247            RV=4
248        fi
249    fi
250
251    for core in `ls test/perl-framework/t/core{,.*} test/gen/apache/core{,.*} 2>/dev/null`; do
252        gdb -ex 'thread apply all backtrace full' -batch ./httpd "$core"
253        RV=5
254    done
255
256    exit $RV
257fi
