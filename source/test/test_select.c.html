18 * produced by a select loop with an exponential backoff.

26#include <stdio.h>
27#include <stdlib.h>
28#include <sys/time.h>
29
30int main (void)
31{
32    int srv;
33    long waittime = 4096;
34    struct timeval tv;
35
36    printf("Start\n");
37    while ((waittime > 0) && (waittime < 3000000)) {
38        printf("%d\n", waittime);
39        tv.tv_sec  = waittime/1000000;
40        tv.tv_usec = waittime%1000000;
41        waittime <<= 1;
42        srv = select(0, NULL, NULL, NULL, &tv);
43    }
44    printf("End\n");
45    exit(0);
46}
