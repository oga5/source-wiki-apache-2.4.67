1# a config safe to use for stopping the server
2# this allows us to stop the server even when+
3# the config in the file is borked (as test cases may try to do that)
4#
5ServerName localhost
6ServerRoot "${server_dir}"
7
8Include "conf/modules.conf"
9
10DocumentRoot "${server_dir}/htdocs"
11
12<IfModule log_config_module>
13    LogFormat "%h %l %u %t \"%r\" %>s %O \"%{Referer}i\" \"%{User-Agent}i\" %k" combined
14    LogFormat "%h %l %u %t \"%r\" %>s %b" common
15    CustomLog "logs/access_log" combined
16
17</IfModule>
18
19TypesConfig "${gen_dir}/apache/conf/mime.types"
20
21Listen ${http_port}
22Listen ${https_port}
23
24<IfModule mod_ssl.c>
25    # provide some default
26    SSLSessionCache "shmcb:ssl_gcache_data(32000)"
27</IfModule>
28
29<VirtualHost *:${http_port}>
30    ServerName ${http_tld}
31    ServerAlias www.${http_tld}
32    <IfModule ssl_module>
33      SSLEngine off
34    </IfModule>
35    DocumentRoot "${server_dir}/htdocs"
36</VirtualHost>
37
38<Directory "${server_dir}/htdocs/cgi">
39    Options Indexes FollowSymLinks
40    AllowOverride None
41    Require all granted
42
43    AddHandler cgi-script .py
44    AddHandler cgi-script .cgi
45    Options +ExecCGI
46</Directory>
