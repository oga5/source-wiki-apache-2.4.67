23    # test case: list two managed domains
24    def test_md_120_001(self, env):
25        domains = [ 
26            ["test120-001.com", "test120-001a.com", "test120-001b.com"],
27            ["greenbytes2.de", "www.greenbytes2.de", "mail.greenbytes2.de"]
28        ]
29        for dns in domains:
30            assert env.a2md(["add"] + dns).exit_code == 0
31        #
32        # list all store content
33        jout = env.a2md(["list"]).json
34        assert len(jout['output']) == len(domains)
35        domains.reverse()
36        for i in range(0, len(jout['output'])):
37            env.check_json_contains(jout['output'][i], {
38                "name": domains[i][0],
39                "domains": domains[i],
40                "contacts": [],
41                "ca": {
42                    "urls": [env.acme_url],
43                    "proto": "ACME"
44                },
45                "state": env.MD_S_INCOMPLETE
46            })
47        # list md by name
48        for dns in ["test120-001.com", "greenbytes2.de"]:
49            md = env.a2md(["list", dns]).json['output'][0]
50            assert md['name'] == dns

52    # test case: validate md state in store
53    def test_md_120_002(self, env):
54        # check: md without pkey/cert -> INCOMPLETE
55        domain = f"test1.{env.http_tld}"
56        assert env.a2md(["add", domain]).exit_code == 0
57        assert env.a2md(["update", domain, "contacts", "admin@" + domain]).exit_code == 0
58        assert env.a2md(["update", domain, "agreement", env.acme_tos]).exit_code == 0
59        assert env.a2md(["list", domain]).json['output'][0]['state'] == env.MD_S_INCOMPLETE
60        # check: valid pkey/cert -> COMPLETE
61        cred = env.get_credentials_for_name(domain)[0]
62        copyfile(cred.pkey_file, env.store_domain_file(domain, 'privkey.pem'))
63        copyfile(cred.cert_file, env.store_domain_file(domain, 'pubcert.pem'))
64        assert env.a2md(["list", domain]).json['output'][0]['state'] == env.MD_S_COMPLETE
65        # check: expired cert -> EXPIRED
66        cred = env.get_credentials_for_name(f"expired.{env.http_tld}")[0]
67        copyfile(cred.pkey_file, env.store_domain_file(domain, 'privkey.pem'))
68        copyfile(cred.cert_file, env.store_domain_file(domain, 'pubcert.pem'))
69        out = env.a2md(["list", domain]).json['output'][0]
70        assert out['state'] == env.MD_S_INCOMPLETE
71        assert out['renew'] is True
72
