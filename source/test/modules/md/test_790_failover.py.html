26    # set 2 ACME certificata authority, valid + invalid
27    def test_md_790_001(self, env):
28        domain = self.test_domain
29        # generate config with one MD
30        domains = [domain, "www." + domain]
31        conf = MDConf(env)
32        conf.add([
33            "MDRetryDelay 200ms",  # speed up failovers
34        ])
35        conf.start_md(domains)
36        conf.add([
37            f"MDCertificateAuthority {env.acme_url} https://does-not-exist/dir"
38        ])
39        conf.end_md()
40        conf.add_vhost(domains)
41        conf.install()
42        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
43        assert env.await_completion([domain])
44        env.check_md_complete(domain)
