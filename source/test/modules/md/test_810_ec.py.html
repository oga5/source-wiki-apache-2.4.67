28    def set_get_pkeys(self, env, domain, pkeys, conf=None):
29        domains = [domain]
30        if conf is None:
31            conf = MDConf(env)
32            conf.add("MDPrivateKeys {0}".format(" ".join([p['spec'] for p in pkeys])))
33            conf.add_md(domains)
34            conf.add_vhost(domains)
35        conf.install()
36        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
37        assert env.await_completion([domain])
38
39    def check_pkeys(self, env, domain, pkeys):
40        # check that files for all types have been created
41        for p in [p for p in pkeys if len(p['spec'])]:
42            env.check_md_complete(domain, p['spec'])
43        # check that openssl client sees the cert with given keylength for cipher
44        env.verify_cert_key_lenghts(domain, pkeys)
45    
46    def set_get_check_pkeys(self, env, domain, pkeys, conf=None):
47        self.set_get_pkeys(env, domain, pkeys, conf=conf)
48        self.check_pkeys(env, domain, pkeys)

130    def test_md_810_007(self, env):
131        domain = self.test_domain
132        domains = [domain]
133        conf = MDConf(env)
134        conf.add("MDPrivateKeys rsa3072")

150            {'spec': "rsa3072", 'ciphers': "ECDHE-RSA-CHACHA20-POLY1305", 'keylen': 3072},
