29    # test case: register a new account, vary length to check base64 encoding
30    @pytest.mark.parametrize("contact", [
31        "x@not-forbidden.org", "xx@not-forbidden.org", "xxx@not-forbidden.org"
32    ])
33    def test_md_202_000(self, env, contact):
34        r = env.a2md(["-t", "accepted", "acme", "newreg", contact], raw=True)
35        assert r.exit_code == 0, r
36        m = re.match("registered: (.*)$", r.stdout)
37        assert m, "did not match: {0}".format(r.stdout)
38        acct = m.group(1)
39        print("newreg: %s" % m.group(1))
40        self._check_account(env, acct, ["mailto:" + contact])
41
42    # test case: register a new account without accepting ToS, must fail
43    def test_md_202_000b(self, env):
44        r = env.a2md(["acme", "newreg", "x@not-forbidden.org"], raw=True)
45        assert r.exit_code == 1
46        m = re.match(".*must agree to terms of service.*", r.stderr)
47        if m is None:
48            # the pebble variant
49            m = re.match(".*account did not agree to the terms of service.*", r.stderr)
50        assert m, "did not match: {0}".format(r.stderr)

124    def _check_account(self, env, acct, contact):
125        with open(env.path_account(acct)) as f:
126            acctj = json.load(f)
127        assert acctj['registration']['contact'] == contact
128
129    def _prepare_account(self, env, contact):
130        r = env.a2md(["-t", "accepted", "acme", "newreg"] + contact, raw=True)
131        assert r.exit_code == 0
132        return re.match("registered: (.*)$", r.stdout).group(1)
