1import os

4import pytest

6from .md_conf import MDConf

8# set the environment variables
9#   ZEROSSL_TLD="<your registered dns name>"
10# these tests to become active

12
13DEMO_ACME = "https://acme.zerossl.com/v2/DV90"
14DEMO_EAB_URL = "http://api.zerossl.com/acme/eab-credentials-email"
15DEMO_TLD = None
16
17
18def missing_tld():
19    global DEMO_TLD
20    if 'ZEROSSL_TLD' in os.environ:
21        DEMO_TLD = os.environ['ZEROSSL_TLD']
22    return DEMO_TLD is None
23
24
25def get_new_eab(env):
26    r = env.curl_raw(DEMO_EAB_URL, options=[
27        "-d", f"email=admin@zerossl.{DEMO_TLD}"
28    ], force_resolve=False)
29    assert r.exit_code == 0
30    assert r.json
31    assert r.json['success'] is True
32    assert r.json['eab_kid']
33    assert r.json['eab_hmac_key']
34    return {'kid': r.json['eab_kid'], 'hmac': r.json['eab_hmac_key']}
35
36

52    def test_md_752_001(self, env):
53        # valid config, expect cert with correct chain
54        domain = f"test1.{DEMO_TLD}"
55        domains = [domain]
56        eab = get_new_eab(env)
57        conf = MDConf(env)
58        conf.start_md(domains)
59        conf.add(f"""
60            MDCertificateAuthority {DEMO_ACME}
61            MDCertificateAgreement accepted
62            MDContactEmail admin@zerossl.{DEMO_TLD}
63            MDCACertificateFile none
64            MDExternalAccountBinding {eab['kid']} {eab['hmac']}
65        """)
66        conf.end_md()
67        conf.add_vhost(domains=domains)
68        conf.install()
69        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
70        assert env.await_completion(domains)
71        r = env.curl_get(f"https://{domain}:{env.https_port}", options=[
72            "--cacert", f"{env.test_dir}/data/sectigo-demo-root.pem"
73        ])
74        assert r.response['status'] == 200

157            MDCACertificateFile none
158            MDExternalAccountBinding {eab['kid']} {eab['hmac']}
