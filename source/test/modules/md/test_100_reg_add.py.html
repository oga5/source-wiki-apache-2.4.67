17    # test case: add a single dns managed domain
18    def test_md_100_000(self, env):
19        dns = "greenbytes.de"
20        jout1 = env.a2md(["add", dns]).json
21        env.check_json_contains(jout1['output'][0], {
22            "name": dns,
23            "domains": [dns],
24            "contacts": [],
25            "ca": {
26                "urls": [env.acme_url],
27                "proto": "ACME"
28            },
29            "state": env.MD_S_INCOMPLETE
30        })
31        assert env.a2md(["list"]).json == jout1
