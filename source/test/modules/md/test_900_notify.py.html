12@pytest.mark.skipif(condition=not MDTestEnv.has_acme_server(),
13                    reason="no ACME test server configured")
14class TestNotify:
15    notify_cmd = None
16    notify_log = None

19    @pytest.fixture(autouse=True, scope='class')
20    def _class_scope(self, env, acme):
21        acme.start(config='default')
22        env.check_acme()
23        env.clear_store()

25    @pytest.fixture(autouse=True, scope='function')
26    def _method_scope(self, env, request):
27        self.domain = env.get_request_domain(request)
28        self.notify_cmd = os.path.join(env.test_dir, "../modules/md/notify.py")
29        self.notify_log = os.path.join(env.gen_dir, "notify.log")
30        if os.path.isfile(self.notify_log):
31            os.remove(self.notify_log)
32
33    def configure_httpd(self, env, domain, add_lines=""):
34        conf = MDConf(env)
35        conf.add(add_lines)
36        conf.add_md([domain])
37        conf.add_vhost(domain)
38        conf.install()
39        return domain

41    # test: invalid notify cmd, check error
42    def test_md_900_001(self, env):
43        command = "blablabla"
44        args = ""
45        self.configure_httpd(env, self.domain, f"""
46            MDNotifyCmd {command} {args}
47            """)
48        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
49        assert env.await_error(self.domain)
50        stat = env.get_md_status(self.domain)
51        assert stat["renewal"]["last"]["problem"] == "urn:org:apache:httpd:log:AH10108:"
52        #
53        env.httpd_error_log.ignore_recent(
54            matches = [
55                r'.*urn:org:apache:httpd:log:AH10108:.*'
56            ]
57        )

79    # test: valid notify that logs to file
80    def test_md_900_010(self, env):
81        command = self.notify_cmd
82        args = self.notify_log
83        self.configure_httpd(env, self.domain, f"""
84            MDNotifyCmd {command} {args}
85            """)
86        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
87        assert env.await_completion([self.domain], restart=False)
88        time.sleep(1)
89        stat = env.get_md_status(self.domain)
90        assert stat["renewal"]["last"]["status"] == 0
91        time.sleep(1)
92        nlines = open(self.notify_log).readlines()
93        assert 1 == len(nlines)
94        assert ("['%s', '%s', '%s']" % (command, args, self.domain)) == nlines[0].strip()

113    # test: signup with working notify cmd for 2 MD and expect it to be called twice
114    def test_md_900_012(self, env):
115        md1 = "a-" + self.domain
116        domains1 = [md1, "www." + md1]
117        md2 = "b-" + self.domain
118        domains2 = [md2, "www." + md2]
119        command = self.notify_cmd
120        args = self.notify_log
121        conf = MDConf(env)
122        conf.add(f"MDNotifyCmd {command} {args}")
123        conf.add_md(domains1)
124        conf.add_md(domains2)
125        conf.add_vhost(domains1)
126        conf.add_vhost(domains2)
127        conf.install()
128        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
129        assert env.await_completion([md1, md2], restart=False)
130        time.sleep(1)
131        stat = env.get_md_status(md1)
132        assert stat["renewal"]["last"]["status"] == 0
133        stat = env.get_md_status(md2)
134        assert stat["renewal"]["last"]["status"] == 0
135        nlines = open(args).readlines()
136        assert 2 == len(nlines)
