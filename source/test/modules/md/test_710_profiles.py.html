38    # create a MD with 'default' profile, get cert
39    def test_md_710_001(self, env):
40        domain = self.test_domain
41        # generate config with one MD
42        domains = [domain, "www." + domain]
43        conf = MDConf(env, admin="admin@" + domain)
44        conf.add_drive_mode("auto")
45        conf.start_md(domains)
46        conf.add(f'  MDProfile default')
47        conf.end_md()
48        conf.add_vhost(domains)
49        conf.install()
50        #
51        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
52        assert env.await_completion(domains)
53        stat = env.get_md_status(domain)
54        assert stat["watched"] == 1
55        assert stat["profile"] == "default", f'{stat}'
56        assert stat['cert']['rsa']['valid']['until'], f'{stat}'
57        ts = email.utils.parsedate_to_datetime(stat['cert']['rsa']['valid']['until'])
58        valid = ts - datetime.datetime.now(datetime.timezone.utc)
59        assert valid.days in [89, 90]
