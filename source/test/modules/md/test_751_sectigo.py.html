1import os
2import re
3import time
4
5import pytest
6
7from .md_conf import MDConf

9# set the environment variables
10#   SECTIGO_EAB="$kid $hmac" for
11#   SECTIGO_TLD="<your registered dns name>"
12# these tests to become active
13#
14
15DEMO_ACME = "https://acme.demo.sectigo.com/"
16DEMO_TLD = None
17
18EABS = [
19    {'kid': '0123', 'hmac': 'abcdef'},
20]
21
22
23def missing_eab():
24    global EABS
25    if len(EABS) == 1 and 'SECTIGO_EAB' in os.environ:
26        m = re.match(r'^\s*(\S+)\s+(\S+)\s*$', os.environ['SECTIGO_EAB'])
27        if m:
28            EABS.append({'kid': m.group(1), 'hmac': m.group(2)})
29    return len(EABS) == 1
30
31
32def missing_tld():
33    global DEMO_TLD
34    if 'SECTIGO_TLD' in os.environ:
35        DEMO_TLD = os.environ['SECTIGO_TLD']
36    return DEMO_TLD is None
37
38
39@pytest.mark.skipif(condition=missing_tld(), reason="env var SECTIGO_TLD not set")
40@pytest.mark.skipif(condition=missing_eab(), reason="env var SECTIGO_EAB not set")
41class TestSectigo:
42

56    def test_md_751_001(self, env):
57        # valid config, expect cert with correct chain
58        domain = f"test1.{DEMO_TLD}"
59        domains = [domain]
60        conf = MDConf(env)
61        conf.start_md(domains)
62        conf.add(f"MDCertificateAuthority {DEMO_ACME}")
63        conf.add("MDCACertificateFile none")
64        conf.add(f"MDExternalAccountBinding {EABS[1]['kid']} {EABS[1]['hmac']}")
65        conf.end_md()
66        conf.add_vhost(domains=domains)
67        conf.install()
68        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
69        assert env.await_completion(domains)
70        r = env.curl_get(f"https://{domain}:{env.https_port}", options=[
71            "--cacert", f"{env.test_dir}/data/sectigo-demo-root.pem"
72        ])
73        assert r.response['status'] == 200

110    def test_md_751_004(self, env):
111        # valid config, get cert, add dns name, renew cert
112        domain = f"test1.{DEMO_TLD}"
113        domain2 = f"test2.{DEMO_TLD}"
114        domains = [domain]
115        conf = MDConf(env)
116        conf.start_md(domains)
117        conf.add(f"MDCertificateAuthority {DEMO_ACME}")
118        conf.add("MDCACertificateFile none")
119        conf.add(f"MDExternalAccountBinding {EABS[1]['kid']} {EABS[1]['hmac']}")
120        conf.end_md()
121        conf.add_vhost(domains=domains)
122        conf.install()
123        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
124        assert env.await_completion(domains)
125        r = env.curl_get(f"https://{domain}:{env.https_port}", options=[
126            "--cacert", f"{env.test_dir}/data/sectigo-demo-root.pem"
127        ])
128        assert r.response['status'] == 200
129        r = env.curl_get(f"https://{domain2}:{env.https_port}", options=[
130            "--cacert", f"{env.test_dir}/data/sectigo-demo-root.pem"
131        ])
132        assert r.exit_code != 0
133        md1 = env.get_md_status(domain)
134        acct1 = md1['ca']['account']
135        # add the domain2 to the dns names
136        domains = [domain, domain2]
137        conf = MDConf(env)
138        conf.start_md(domains)
139        conf.add(f"MDCertificateAuthority {DEMO_ACME}")
140        conf.add("MDCACertificateFile none")
141        conf.add(f"MDExternalAccountBinding {EABS[1]['kid']} {EABS[1]['hmac']}")
142        conf.end_md()
143        conf.add_vhost(domains=domains)
144        conf.install()
145        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
146        assert env.await_completion(domains)
147        r = env.curl_get(f"https://{domain2}:{env.https_port}", options=[
148            "--cacert", f"{env.test_dir}/data/sectigo-demo-root.pem"
149        ])
150        assert r.response['status'] == 200
151        md2 = env.get_md_status(domain)
152        acct2 = md2['ca']['account']
153        assert acct2 == acct1, f"ACME account was not reused: {acct1} became {acct2}"

155    def test_md_751_020(self, env):
156        # valid config, get cert, check OCSP status
157        domain = f"test1.{DEMO_TLD}"
158        domains = [domain]
159        conf = MDConf(env)
160        conf.add("MDStapling on")
161        conf.start_md(domains)
162        conf.add(f"""
163            MDCertificateAuthority {DEMO_ACME}
164            MDCACertificateFile none
165            MDExternalAccountBinding {EABS[1]['kid']} {EABS[1]['hmac']}
166            """)
167        conf.end_md()
168        conf.add_vhost(domains=domains)
169        conf.install()
170        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
171        assert env.await_completion(domains)
172        r = env.curl_get(f"https://{domain}:{env.https_port}", options=[
173            "--cacert", f"{env.test_dir}/data/sectigo-demo-root.pem"
174        ])
175        assert r.response['status'] == 200
176        time.sleep(1)
177        for domain in domains:
178            stat = env.await_ocsp_status(domain,
179                                         ca_file=f"{env.test_dir}/data/sectigo-demo-root.pem")
180            assert stat['ocsp'] == "successful (0x0)"
181            assert stat['verify'] == "0 (ok)"
