414        # setup: prepare COMPLETE md
415        domain = self.test_domain
416        name = "www." + domain
417        conf = MDConf(env, admin="admin@" + domain)
418        conf.add_drive_mode("manual")

428        assert env.a2md(["list", name]).json['output'][0]['state'] == env.MD_S_COMPLETE

431        print("TRACE: start testing renew window: %s" % renew_window)
432        for tc in test_data_list:
433            print("TRACE: create self-signed cert: %s" % tc["valid"])
434            creds = env.create_self_signed_cert(CertificateSpec(domains=[name]),
435                                                valid_from=timedelta(days=tc["valid"]["notBefore"]),
436                                                valid_to=timedelta(days=tc["valid"]["notAfter"]))
437            assert creds.certificate.serial_number != cert1.get_serial_number()
438            # copy it over, assess status again
439            creds.save_cert_pem(env.store_domain_file(name, 'pubcert.pem'))
440            md = env.a2md(["list", name]).json['output'][0]
