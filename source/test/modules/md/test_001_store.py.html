112    # test case: list two managed domains
113    def test_md_001_201(self, env: MDTestEnv):
114        domains = [ 
115            ["test000-201.com", "test000-201a.com", "test000-201b.com"],
116            ["greenbytes2.de", "www.greenbytes2.de", "mail.greenbytes2.de"]
117        ]
118        for dns in domains:
119            assert env.a2md(["store", "add"] + dns).exit_code == 0
120        #
121        # list all store content
122        jout = env.a2md(["store", "list"]).json
123        assert len(jout['output']) == len(domains)
124        domains.reverse()
125        jout['output'] = sorted(jout['output'], key=md_name)
126        for i in range(0, len(jout['output'])):
127            env.check_json_contains(jout['output'][i], {
128                "name": domains[i][0],
129                "domains": domains[i],
130                "contacts": [],
131                "ca": {
132                    "urls": [env.acme_url],
133                    "proto": "ACME"
