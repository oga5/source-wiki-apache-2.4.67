25def _private_key(key_type):
26    if isinstance(key_type, str):
27        key_type = key_type.upper()
28        m = re.match(r'^(RSA)?(\d+)$', key_type)
29        if m:
30            key_type = int(m.group(2))
31
32    if isinstance(key_type, int):
33        return rsa.generate_private_key(
34            public_exponent=65537,
35            key_size=key_type,
36            backend=default_backend()
37        )
38    if not isinstance(key_type, ec.EllipticCurve) and key_type in EC_SUPPORTED:
39        key_type = EC_SUPPORTED[key_type]
40    return ec.generate_private_key(
41        curve=key_type,
42        backend=default_backend()
43    )

46class CertificateSpec:
47
48    def __init__(self, name: str = None, domains: List[str] = None,
49                 email: str = None,
50                 key_type: str = None, single_file: bool = False,
51                 valid_from: timedelta = timedelta(days=-1),
52                 valid_to: timedelta = timedelta(days=89),
53                 client: bool = False,
54                 sub_specs: List['CertificateSpec'] = None):
55        self._name = name
56        self.domains = domains
57        self.client = client
58        self.email = email
59        self.key_type = key_type
60        self.single_file = single_file
61        self.valid_from = valid_from
62        self.valid_to = valid_to
63        self.sub_specs = sub_specs
64
65    @property
66    def name(self) -> Optional[str]:
67        if self._name:
68            return self._name
69        elif self.domains:
70            return self.domains[0]
71        return None

74class Credentials:
75
76    def __init__(self, name: str, cert: Any, pkey: Any):
77        self._name = name
78        self._cert = cert
79        self._pkey = pkey
80        self._cert_file = None
81        self._pkey_file = None
82        self._store = None
83
84    @property
85    def name(self) -> str:
86        return self._name
87
88    @property
89    def subject(self) -> x509.Name:
90        return self._cert.subject
91
92    @property
93    def key_type(self):
94        if isinstance(self._pkey, RSAPrivateKey):
95            return f"rsa{self._pkey.key_size}"
96        elif isinstance(self._pkey, EllipticCurvePrivateKey):
97            return f"{self._pkey.curve.name}"
98        else:
99            raise Exception(f"unknown key type: {self._pkey}")
100
101    @property
102    def private_key(self) -> Any:
103        return self._pkey
104
105    @property
106    def certificate(self) -> Any:
107        return self._cert
108
109    @property
110    def cert_pem(self) -> bytes:
111        return self._cert.public_bytes(Encoding.PEM)
112
113    @property
114    def pkey_pem(self) -> bytes:
115        return self._pkey.private_bytes(
116            Encoding.PEM,
117            PrivateFormat.TraditionalOpenSSL if self.key_type.startswith('rsa') else PrivateFormat.PKCS8,
118            NoEncryption())
119
120    def set_store(self, store: 'CertStore'):
121        self._store = store
122
123    def set_files(self, cert_file: str, pkey_file: str = None):
124        self._cert_file = cert_file
125        self._pkey_file = pkey_file
126
127    @property
128    def cert_file(self) -> str:
129        return self._cert_file
130
131    @property
132    def pkey_file(self) -> Optional[str]:
133        return self._pkey_file
134
135    def get_first(self, name) -> Optional['Credentials']:
136        creds = self._store.get_credentials_for_name(name) if self._store else []
137        return creds[0] if len(creds) else None
138
139    def get_credentials_for_name(self, name) -> List['Credentials']:
140        return self._store.get_credentials_for_name(name) if self._store else []
141
142    def issue_certs(self, specs: List[CertificateSpec],
143                    chain: List['Credentials'] = None) -> List['Credentials']:
144        return [self.issue_cert(spec=spec, chain=chain) for spec in specs]
145
146    def issue_cert(self, spec: CertificateSpec, chain: List['Credentials'] = None) -> 'Credentials':
147        key_type = spec.key_type if spec.key_type else self.key_type
148        creds = self._store.load_credentials(name=spec.name, key_type=key_type, single_file=spec.single_file) \
149            if self._store else None
150        if creds is None:
151            creds = MDTestCA.create_credentials(spec=spec, issuer=self, key_type=key_type,
152                                                valid_from=spec.valid_from, valid_to=spec.valid_to)
153            if self._store:
154                self._store.save(creds, single_file=spec.single_file)
155
156        if spec.sub_specs:
157            if self._store:
158                sub_store = CertStore(fpath=os.path.join(self._store.path, creds.name))
159                creds.set_store(sub_store)
160            subchain = chain.copy() if chain else []
161            subchain.append(self)
162            creds.issue_certs(spec.sub_specs, chain=subchain)
163        return creds

166class CertStore:
167
168    def __init__(self, fpath: str):
169        self._store_dir = fpath
170        if not os.path.exists(self._store_dir):
171            os.makedirs(self._store_dir)
172        self._creds_by_name = {}
173
174    @property
175    def path(self) -> str:
176        return self._store_dir
177
178    def save(self, creds: Credentials, name: str = None,
179             chain: List[Credentials] = None,
180             single_file: bool = False) -> None:
181        name = name if name is not None else creds.name
182        cert_file = self.get_cert_file(name=name, key_type=creds.key_type)
183        pkey_file = self.get_pkey_file(name=name, key_type=creds.key_type)
184        if single_file:
185            pkey_file = None
186        with open(cert_file, "wb") as fd:
187            fd.write(creds.cert_pem)
188            if chain:
189                for c in chain:
190                    fd.write(c.cert_pem)
191            if pkey_file is None:
192                fd.write(creds.pkey_pem)
193        if pkey_file is not None:
194            with open(pkey_file, "wb") as fd:
195                fd.write(creds.pkey_pem)
196        creds.set_files(cert_file, pkey_file)
197        self._add_credentials(name, creds)
198
199    def _add_credentials(self, name: str, creds: Credentials):
200        if name not in self._creds_by_name:
201            self._creds_by_name[name] = []
202        self._creds_by_name[name].append(creds)
203
204    def get_credentials_for_name(self, name) -> List[Credentials]:
205        return self._creds_by_name[name] if name in self._creds_by_name else []
206
207    def get_cert_file(self, name: str, key_type=None) -> str:
208        key_infix = ".{0}".format(key_type) if key_type is not None else ""
209        return os.path.join(self._store_dir, f'{name}{key_infix}.cert.pem')
210
211    def get_pkey_file(self, name: str, key_type=None) -> str:
212        key_infix = ".{0}".format(key_type) if key_type is not None else ""
213        return os.path.join(self._store_dir, f'{name}{key_infix}.pkey.pem')
214
215    def load_pem_cert(self, fpath: str) -> x509.Certificate:
216        with open(fpath) as fd:
217            return x509.load_pem_x509_certificate("".join(fd.readlines()).encode())
218
219    def load_pem_pkey(self, fpath: str):
220        with open(fpath) as fd:
221            return load_pem_private_key("".join(fd.readlines()).encode(), password=None)
222
223    def load_credentials(self, name: str, key_type=None, single_file: bool = False):
224        cert_file = self.get_cert_file(name=name, key_type=key_type)
225        pkey_file = cert_file if single_file else self.get_pkey_file(name=name, key_type=key_type)
226        if os.path.isfile(cert_file) and os.path.isfile(pkey_file):
227            cert = self.load_pem_cert(cert_file)
228            pkey = self.load_pem_pkey(pkey_file)
229            creds = Credentials(name=name, cert=cert, pkey=pkey)
230            creds.set_store(self)
231            creds.set_files(cert_file, pkey_file)
232            self._add_credentials(name, creds)
233            return creds
234        return None

237class MDTestCA:
238
239    @classmethod
240    def create_root(cls, name: str, store_dir: str, key_type: str = "rsa2048") -> Credentials:
241        store = CertStore(fpath=store_dir)
242        creds = store.load_credentials(name="ca", key_type=key_type)
243        if creds is None:
244            creds = MDTestCA._make_ca_credentials(name=name, key_type=key_type)
245            store.save(creds, name="ca")
246            creds.set_store(store)
247        return creds

249    @staticmethod
250    def create_credentials(spec: CertificateSpec, issuer: Credentials, key_type: Any,
251                           valid_from: timedelta = timedelta(days=-1),
252                           valid_to: timedelta = timedelta(days=89),
253                           ) -> Credentials:
254        """Create a certificate signed by this CA for the given domains.
255        :returns: the certificate and private key PEM file paths
256        """
257        if spec.domains and len(spec.domains):
258            creds = MDTestCA._make_server_credentials(name=spec.name, domains=spec.domains,
259                                                      issuer=issuer, valid_from=valid_from,
260                                                      valid_to=valid_to, key_type=key_type)
261        elif spec.client:
262            creds = MDTestCA._make_client_credentials(name=spec.name, issuer=issuer,
263                                                      email=spec.email, valid_from=valid_from,
264                                                      valid_to=valid_to, key_type=key_type)
265        elif spec.name:
266            creds = MDTestCA._make_ca_credentials(name=spec.name, issuer=issuer,
267                                                  valid_from=valid_from, valid_to=valid_to,
268                                                  key_type=key_type)
269        else:
270            raise Exception(f"unrecognized certificate specification: {spec}")
271        return creds

273    @staticmethod
274    def _make_x509_name(org_name: str = None, common_name: str = None, parent: x509.Name = None) -> x509.Name:
275        name_pieces = []
276        if org_name:
277            oid = NameOID.ORGANIZATIONAL_UNIT_NAME if parent else NameOID.ORGANIZATION_NAME
278            name_pieces.append(x509.NameAttribute(oid, org_name))
279        elif common_name:
280            name_pieces.append(x509.NameAttribute(NameOID.COMMON_NAME, common_name))
281        if parent:
282            name_pieces.extend([rdn for rdn in parent])
283        return x509.Name(name_pieces)

285    @staticmethod
286    def _make_csr(
287            subject: x509.Name,
288            pkey: Any,
289            issuer_subject: Optional[Credentials],
290            valid_from_delta: timedelta = None,
291            valid_until_delta: timedelta = None
292    ):
293        pubkey = pkey.public_key()
294        issuer_subject = issuer_subject if issuer_subject is not None else subject
295
296        valid_from = datetime.now()
297        if valid_until_delta is not None:
298            valid_from += valid_from_delta
299        valid_until = datetime.now()
300        if valid_until_delta is not None:
301            valid_until += valid_until_delta
302
303        return (
304            x509.CertificateBuilder()
305            .subject_name(subject)
306            .issuer_name(issuer_subject)
307            .public_key(pubkey)
308            .not_valid_before(valid_from)
309            .not_valid_after(valid_until)
310            .serial_number(x509.random_serial_number())
311            .add_extension(
312                x509.SubjectKeyIdentifier.from_public_key(pubkey),
313                critical=False,
314            )
315        )

317    @staticmethod
318    def _add_ca_usages(csr: Any) -> Any:
319        return csr.add_extension(
320            x509.BasicConstraints(ca=True, path_length=9),
321            critical=True,
322        ).add_extension(
323            x509.KeyUsage(
324                digital_signature=True,
325                content_commitment=False,
326                key_encipherment=False,
327                data_encipherment=False,
328                key_agreement=False,
329                key_cert_sign=True,
330                crl_sign=True,
331                encipher_only=False,
332                decipher_only=False),
333            critical=True
334        ).add_extension(
335            x509.ExtendedKeyUsage([
336                ExtendedKeyUsageOID.CLIENT_AUTH,
337                ExtendedKeyUsageOID.SERVER_AUTH,
338                ExtendedKeyUsageOID.CODE_SIGNING,
339            ]),
340            critical=True
341        )
342
343    @staticmethod
344    def _add_leaf_usages(csr: Any, domains: List[str], issuer: Credentials) -> Any:
345        return csr.add_extension(
346            x509.BasicConstraints(ca=False, path_length=None),
347            critical=True,
348        ).add_extension(
349            x509.AuthorityKeyIdentifier.from_issuer_subject_key_identifier(
350                issuer.certificate.extensions.get_extension_for_class(
351                    x509.SubjectKeyIdentifier).value),
352            critical=False
353        ).add_extension(
354            x509.SubjectAlternativeName([x509.DNSName(domain) for domain in domains]),
355            critical=True,
356        ).add_extension(
357            x509.ExtendedKeyUsage([
358                ExtendedKeyUsageOID.SERVER_AUTH,
359            ]),
360            critical=True
361        )
362
363    @staticmethod
364    def _add_client_usages(csr: Any, issuer: Credentials, rfc82name: str = None) -> Any:
365        cert = csr.add_extension(
366            x509.BasicConstraints(ca=False, path_length=None),
367            critical=True,
368        ).add_extension(
369            x509.AuthorityKeyIdentifier.from_issuer_subject_key_identifier(
370                issuer.certificate.extensions.get_extension_for_class(
371                    x509.SubjectKeyIdentifier).value),
372            critical=False
373        )
374        if rfc82name:
375            cert.add_extension(
376                x509.SubjectAlternativeName([x509.RFC822Name(rfc82name)]),
377                critical=True,
378            )
379        cert.add_extension(
380            x509.ExtendedKeyUsage([
381                ExtendedKeyUsageOID.CLIENT_AUTH,
382            ]),
383            critical=True
384        )
385        return cert

387    @staticmethod
388    def _make_ca_credentials(name, key_type: Any,
389                             issuer: Credentials = None,
390                             valid_from: timedelta = timedelta(days=-1),
391                             valid_to: timedelta = timedelta(days=89),
392                             ) -> Credentials:
393        pkey = _private_key(key_type=key_type)
394        if issuer is not None:
395            issuer_subject = issuer.certificate.subject
396            issuer_key = issuer.private_key
397        else:
398            issuer_subject = None
399            issuer_key = pkey
400        subject = MDTestCA._make_x509_name(org_name=name, parent=issuer.subject if issuer else None)
401        csr = MDTestCA._make_csr(subject=subject,
402                                 issuer_subject=issuer_subject, pkey=pkey,
403                                 valid_from_delta=valid_from, valid_until_delta=valid_to)
404        csr = MDTestCA._add_ca_usages(csr)
405        cert = csr.sign(private_key=issuer_key,
406                        algorithm=hashes.SHA256(),
407                        backend=default_backend())
408        return Credentials(name=name, cert=cert, pkey=pkey)

414                                 valid_to: timedelta = timedelta(days=89),
415                                 ) -> Credentials:

421                                 valid_from_delta=valid_from, valid_until_delta=valid_to)

423        cert = csr.sign(private_key=issuer.private_key,
424                        algorithm=hashes.SHA256(),
425                        backend=default_backend())
426        return Credentials(name=name, cert=cert, pkey=pkey)
