222        domain_a = f"a{self.test_domain}"
223        domain_b = f"b{self.test_domain}"
224        conf = MDConf(env)
225        conf.start_md([domain_a])
226        conf.add("MDExternalAccountBinding kid-1 zWNDZM6eQGHWpSRTPal5eIUYFTu7EajVIoguysqZ9wG44nMEtx3MUAsUDkMTQ12W")
227        conf.end_md()
228        conf.add_vhost(domains=[domain_a])
229        conf.start_md([domain_b])

232        conf.add_vhost(domains=[domain_b])
