302    def test_md_702_009(self, env, acme):
303        domain = self.test_domain
304        domains = [domain]
305        #
306        # prepare md
307        conf = MDConf(env, admin="admin@" + domain)
308        conf.add_drive_mode("auto")
309        conf.add_renew_window("10d")
310        conf.add_md(domains)
311        conf.add_vhost(domain)
312        conf.install()
313        #
314        # restart (-> drive), check that md+cert is in store, TLS is up
315        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
316        assert env.await_completion([domain], restart=False)
317        # Overwrite stages cert with one which has little remaining lifetime
318        creds = env.create_self_signed_cert(CertificateSpec(domains=[domain]),
319                                            valid_from=timedelta(days=-120),
320                                            valid_to=timedelta(days=2),
321                                            serial=7029)
322        assert creds.certificate.serial_number == 7029
323        creds.save_cert_pem(env.store_staged_file(domain, 'pubcert.pem'))
324        creds.save_pkey_pem(env.store_staged_file(domain, 'privkey.pem'))
325        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
326        stat = env.get_certificate_status(domain)
327        assert creds.certificate.serial_number == int(stat['rsa']['serial'], 16)
328        # cert should renew and be different afterwards
329        assert env.await_completion([domain], must_renew=True)
330        stat = env.get_certificate_status(domain)
331        assert creds.certificate.serial_number != int(stat['rsa']['serial'], 16)

769    # Test a domain name longer than 64 chars, but components < 64, see #227
770    # Background: DNS has an official limit of 253 ASCII chars and components must be
771    # of length [1, 63].
772    # However the CN in a certificate is restricted too, see
773    # <https://github.com/letsencrypt/boulder/issues/2093>.
774    @pytest.mark.skipif(MDTestEnv.is_pebble(), reason="pebble differs here from boulder")
775    @pytest.mark.parametrize("challenge_type", [
776        "tls-alpn-01", "http-01"
777    ])
778    def test_md_702_060(self, env, challenge_type):
779        domain = self.test_domain
780        # use only too long names, this is expected to fail:
781        # see <https://github.com/jetstack/cert-manager/issues/1462>
782        long_domain = ("x" * (65 - len(domain))) + domain
783        domains = [long_domain, "www." + long_domain]
784        conf = MDConf(env, admin="admin@" + domain)

791        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
792        env.check_md(domains)
793        assert env.await_error(long_domain)
794        # add a short domain to the SAN list, the CA should now use that one
795        # and issue a cert.
796        long_domain = ("y" * (65 - len(domain))) + domain
797        domains = [long_domain, "www." + long_domain, "xxx." + domain]

809        # check SSL running OK
