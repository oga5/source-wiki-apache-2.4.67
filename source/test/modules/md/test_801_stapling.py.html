40    def configure_httpd(self, env, domains=None, add_lines="", ssl_stapling=False):
41        if not isinstance(domains, list):
42            domains = [domains] if domains else []
43        conf = MDConf(env)
44        conf.add("""
45        <IfModule tls_module>
46            LogLevel tls:trace4
47        </IfModule>
48        <IfModule ssl_module>
49            LogLevel ssl:trace4
50        </IfModule>
51            """)
52        if ssl_stapling:
53            conf.add("""
54            <IfModule ssl_module>
55                SSLUseStapling On
56                SSLStaplingCache shmcb:stapling_cache(128000)
57            </IfModule>
58                """)
59        conf.add(add_lines)
60        for domain in domains:
61            conf.add_md([domain])
62            conf.add_vhost(domain)
63        return conf

67    def test_md_801_001(self, env):
68        md = self.mdA
69        self.configure_httpd(env, md).install()
70        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
71        stat = env.get_ocsp_status(md)
72        if MDTestEnv.lacks_ocsp():
73            assert 'ocsp' not in stat or stat['ocsp'] == "no response sent", f'{stat}'
74        else:
75            assert stat['ocsp'] == "no response sent"
76        stat = env.get_md_status(md)
77        assert not stat["stapling"]
78        #
79        # turn stapling on, wait for it to appear in connections
80        self.configure_httpd(env, md, """
81            MDStapling on
82            LogLevel md:trace5
83            """).install()
84        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
85        if MDTestEnv.lacks_ocsp():
86            stat = env.get_md_status(md)
87            assert 'ocsp' not in stat or stat['ocsp'] == "no response sent", f'{stat}'
88        else:
89            stat = env.await_ocsp_status(md)
90            assert stat['ocsp'] == "successful (0x0)"
91            assert stat['verify'] == "0 (ok)"
92        stat = env.get_md_status(md)
93        if MDTestEnv.lacks_ocsp():
94            assert stat["stapling"]
95            pkey = 'rsa'
96            assert 'ocsp' not in stat["cert"][pkey], f'{stat}'
97        else:
98            assert stat["stapling"]
99            pkey = 'rsa'
100            assert stat["cert"][pkey]["ocsp"]["status"] == "good"
101            assert stat["cert"][pkey]["ocsp"]["valid"]
102        #
103        # turn stapling off (explicitly) again, should disappear
104        self.configure_httpd(env, md, "MDStapling off").install()
105        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
106        stat = env.get_ocsp_status(md)
107        if MDTestEnv.lacks_ocsp():
108            assert 'ocsp' not in stat or stat['ocsp'] == "no response sent", f'{stat}'
109        else:
110            assert stat['ocsp'] == "no response sent"
111        stat = env.get_md_status(md)
112        assert not stat["stapling"]

218    # MD, check that restart leaves response unchanged, reconfigure keep interval,

220    @pytest.mark.skipif(MDTestEnv.lacks_ocsp(), reason="no OCSP responder")
221    def test_md_801_005(self, env):
222        # TODO: mod_watchdog seems to have problems sometimes with fast restarts
223        # turn stapling on, wait for it to appear in connections
224        md = self.mdA
225        self.configure_httpd(env, md, "MDStapling on").install()
226        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
227        stat = env.await_ocsp_status(md)
228        assert stat['ocsp'] == "successful (0x0)" 
229        assert stat['verify'] == "0 (ok)"
230        # fine the file where the ocsp response is stored
231        dirpath = os.path.join(env.store_dir, 'ocsp', md)
232        files = os.listdir(dirpath)
233        ocsp_file = None
234        for name in files:
235            if name.startswith("ocsp-"):
236                ocsp_file = os.path.join(dirpath, name)
237        assert ocsp_file
238        mtime1 = os.path.getmtime(ocsp_file)
239        # wait a sec, restart and check that file does not change
240        time.sleep(1)
241        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
242        stat = env.await_ocsp_status(md)
243        assert stat['ocsp'] == "successful (0x0)" 
244        mtime2 = os.path.getmtime(ocsp_file)
245        assert mtime1 == mtime2
246        # configure a keep time of 1 second, restart, the file is gone
247        # (which is a side effec that we load it before the cleanup removes it.
248        #  since it was valid, no new one needed fetching
249        self.configure_httpd(env, md, """
250            MDStapling on
251            MDStaplingKeepResponse 1s
252            """).install()
253        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
254        stat = env.await_ocsp_status(md)
255        assert stat['ocsp'] == "successful (0x0)"
256        assert not os.path.exists(ocsp_file)
257        # if we restart again, a new file needs to appear
258        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
259        stat = env.await_ocsp_status(md)
260        assert stat['ocsp'] == "successful (0x0)"
261        mtime3 = os.path.getmtime(ocsp_file)
262        assert mtime1 != mtime3

362        md = self.mdA
363        domains = [md]
364        testpath = os.path.join(env.gen_dir, 'test_801_009')
365        env.mkpath(testpath)

367        creds = env.create_self_signed_cert(CertificateSpec(domains=domains),
368                                            valid_from=timedelta(days=-60),
369                                            valid_to=timedelta(days=30),
370                                            serial=801009)

372        pkey_file = os.path.join(testpath, 'privkey.pem')

378        conf.add("MDCertificateKeyFile %s" % pkey_file)
