31    # simple MD, drive it, check status before activation
32    def test_md_920_001(self, env):
33        domain = self.test_domain
34        domains = [domain]
35        conf = MDConf(env)
36        conf.add_md(domains)
37        conf.add_vhost(domain)
38        conf.install()
39        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
40        assert env.await_completion([domain], restart=False)
41        # we started without a valid certificate, so we expect /.httpd/certificate-status
42        # to not give information about one and - since we waited for the ACME signup
43        # to complete - to give information in 'renewal' about the new cert.
44        status = env.get_certificate_status(domain)
45        assert 'sha256-fingerprint' not in status
46        assert 'valid' not in status
47        assert 'renewal' in status
48        assert 'valid' in status['renewal']['cert']
49        assert 'sha256-fingerprint' in status['renewal']['cert']['rsa']
50        # restart and activate
51        # once activated, the staging must be gone and attributes exist for the active cert
52        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
53        status = env.get_certificate_status(domain)
54        assert 'renewal' not in status
55        assert 'sha256-fingerprint' in status['rsa']
56        assert 'valid' in status['rsa']
57        assert 'from' in status['rsa']['valid']

111    def test_md_920_010(self, env):
112        domain = self.test_domain
113        domains = [domain]
114        conf = MDConf(env, std_vhosts=False, std_ports=False, text=f"""
115MDBaseServer on
116MDPortMap http:- https:{env.https_port}
117MDStapling on
118
119ServerName {domain}
120<IfModule ssl_module>
121SSLEngine on
122</IfModule>
123<IfModule tls_module>
124TLSListen {env.https_port}
125TLSStrictSNI off
126</IfModule>
127Protocols h2 http/1.1 acme-tls/1
128
129<Location "/server-status">
130    SetHandler server-status
131</Location>
132<Location "/md-status">
133    SetHandler md-status
134</Location>
135<VirtualHost *:{env.http_port}>
136  SSLEngine off
137</VirtualHost>
138            """)
139        conf.add_md(domains)
140        conf.install()
141        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
142        assert env.await_completion([domain], restart=False,
143                                    via_domain=env.http_addr, use_https=False)
144        status = env.get_md_status("", via_domain=env.http_addr, use_https=False)
145        assert "version" in status
146        assert "managed-domains" in status
147        assert 1 == len(status["managed-domains"])
148        # get the html page
149        status = env.get_server_status(via_domain=env.http_addr, use_https=False)
150        assert re.search(r'<h3>Managed Certificates</h3>', status, re.MULTILINE)
151        # get the ascii summary
152        status = env.get_server_status(query="?auto", via_domain=env.http_addr, use_https=False)
153        m = re.search(r'ManagedCertificatesTotal: (\d+)', status, re.MULTILINE)
154        assert m, status
155        assert int(m.group(1)) == 1
156        m = re.search(r'ManagedCertificatesOK: (\d+)', status, re.MULTILINE)
157        assert int(m.group(1)) == 0
158        m = re.search(r'ManagedCertificatesRenew: (\d+)', status, re.MULTILINE)
159        assert int(m.group(1)) == 1
160        m = re.search(r'ManagedCertificatesErrored: (\d+)', status, re.MULTILINE)
161        assert int(m.group(1)) == 0
162        m = re.search(r'ManagedCertificatesReady: (\d+)', status, re.MULTILINE)
163        assert int(m.group(1)) == 1
164        m = re.search(r'ManagedDomain\[0]Stapling: (on)', status, re.MULTILINE)
165        assert m, f'{status}'

167    def test_md_920_011(self, env):
168        # MD with static cert files in base server, see issue #161
169        domain = self.test_domain
170        domains = [domain, 'www.%s' % domain]
171        testpath = os.path.join(env.gen_dir, 'test_920_011')
172        env.mkpath(testpath)
173        # cert that is only 20 more days valid
174        creds = env.create_self_signed_cert(CertificateSpec(domains=domains),
175                                            valid_from=timedelta(days=-70),
176                                            valid_to=timedelta(days=20),
177                                            serial=920011)
178        cert_file = os.path.join(testpath, 'pubcert.pem')
179        pkey_file = os.path.join(testpath, 'privkey.pem')
180        creds.save_cert_pem(cert_file)
181        creds.save_pkey_pem(pkey_file)
182        conf = MDConf(env, std_vhosts=False, std_ports=False, text=f"""
183        MDBaseServer on
184        MDPortMap http:- https:{env.https_port}
185
186        ServerName {domain}
187        <IfModule ssl_module>
188        SSLEngine on
189        </IfModule>
190        <IfModule tls_module>
191        TLSListen {env.https_port}
192        TLSStrictSNI off
193        </IfModule>
194        Protocols h2 http/1.1 acme-tls/1
195
196        <Location "/server-status">
197            SetHandler server-status
198        </Location>
199        <Location "/md-status">
200            SetHandler md-status
201        </Location>
202            """)
203        conf.start_md(domains)
204        conf.add(f"MDCertificateFile {cert_file}")
205        conf.add(f"MDCertificateKeyFile {pkey_file}")
206        conf.end_md()
207        conf.start_vhost([env.http_addr], port=env.http_port)
208        conf.add("SSLEngine off")
209        conf.end_vhost()
210        conf.install()
211        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
212        status = env.get_md_status(domain, via_domain=env.http_addr, use_https=False)
213        assert status
214        assert 'renewal' not in status
215        print(status)
216        assert status['state'] == env.MD_S_COMPLETE
217        assert status['renew-mode'] == 0  # manual
218        env.httpd_error_log.ignore_recent(
219            matches = [
220                r'.*cert has no authority key id extension.*'
221            ]
222        )

224    # MD with 2 certificates
225    def test_md_920_020(self, env):
226        domain = self.test_domain
227        domains = [domain]
228        conf = MDConf(env)
229        conf.add("MDStapling on")
230        conf.add("MDPrivateKeys secp256r1 RSA")
231        conf.add_md(domains)
232        conf.add_vhost(domain)
233        conf.install()
234        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
235        assert env.await_completion([domain], restart=False)
236        # In the stats JSON, we expect 2 certificates under 'renewal'
237        stat = env.get_md_status(domain)
238        assert 'renewal' in stat
239        assert 'cert' in stat['renewal']
240        assert 'rsa' in stat['renewal']['cert']
241        assert 'secp256r1' in stat['renewal']['cert']
242        # In /.httpd/certificate-status 'renewal' we expect 2 certificates
243        status = env.get_certificate_status(domain)
244        assert 'renewal' in status
245        assert 'cert' in status['renewal']
246        assert 'secp256r1' in status['renewal']['cert']
247        assert 'rsa' in status['renewal']['cert']
248        # restart and activate
249        # once activated, certs are listed in status
250        assert env.apache_restart() == 0, f'{env.apachectl_stderr}'
251        stat = env.get_md_status(domain)
252        assert 'cert' in stat
253        assert 'valid' in stat['cert']
254        for ktype in ['rsa', 'secp256r1']:
255            assert ktype in stat['cert']
256            if env.acme_server == 'boulder':
257                assert 'ocsp' in stat['cert'][ktype]
258        #
259        env.httpd_error_log.ignore_recent(
260            matches = [
261                r'.*certificate with serial \w+ has no OCSP responder URL.*'
262            ]
263        )
