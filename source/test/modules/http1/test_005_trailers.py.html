1import os
2import pytest
3
4from .env import H1Conf
5
6
7# The trailer tests depend on "nghttp" as no other client seems to be able to send those
8# rare things.
9class TestTrailers:
10
11    @pytest.fixture(autouse=True, scope='class')
12    def _class_scope(self, env):
13        H1Conf(env).add_vhost_cgi(proxy_self=True).install()
14        assert env.apache_restart() == 0
15
16    # check that we get a trailer out when telling the handler to add one
17    def test_h1_005_01(self, env):
18        if not env.httpd_is_at_least("2.5.0"):
19            pytest.skip(f'need at least httpd 2.5.0 for this')
20        url = env.mkurl("https", "cgi", "/h1test/echo")
21        host = f"cgi.{env.http_tld}"
22        fpath = os.path.join(env.gen_dir, "data-1k")
23        r = env.curl_upload(url, fpath, options=["--header", "Add-Trailer: 005_01"])
24        assert r.exit_code == 0, f"{r}"
25        assert 200 <= r.response["status"] < 300
26        assert r.response["trailer"], f"no trailers received: {r}"
27        assert "h1test-add-trailer" in r.response["trailer"]
28        assert r.response["trailer"]["h1test-add-trailer"] == "005_01"
29
30    # check that we get out trailers through the proxy
31    def test_h1_005_02(self, env):
32        if not env.httpd_is_at_least("2.5.0"):
33            pytest.skip(f'need at least httpd 2.5.0 for this')
34        url = env.mkurl("https", "cgi", "/proxy/h1test/echo")
35        host = f"cgi.{env.http_tld}"
36        fpath = os.path.join(env.gen_dir, "data-1k")
37        r = env.curl_upload(url, fpath, options=["--header", "Add-Trailer: 005_01"])
38        assert r.exit_code == 0, f"{r}"
39        assert 200 <= r.response["status"] < 300
40        assert r.response["trailer"], f"no trailers received: {r}"
41        assert "h1test-add-trailer" in r.response["trailer"]
42        assert r.response["trailer"]["h1test-add-trailer"] == "005_01"
