28        r = env.curl_upload(url, fpath, options=options)
29        assert r.exit_code == 0, f"{r}"
30        assert 200 <= r.response["status"] < 300
31
32        r2 = env.curl_get(r.response["header"]["location"])
33        assert r2.exit_code == 0
34        assert r2.response["status"] == 200
35        with open(self.local_src(fpath), mode='rb') as file:
36            src = file.read()
37        assert src == r2.response["body"]
38        return r
