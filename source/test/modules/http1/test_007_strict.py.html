21    # strict tests from t/apache/http_strict.t
22    # possible expected results:
23    #   0:       any HTTP error
24    #   1:       any HTTP success
25    #   200-500: specific HTTP status code
26    #   undef:   HTTPD should drop connection without error message
27    @pytest.mark.parametrize(["intext", "status"], [
28        ["GET / HTTP/1.0\n\n", 400],
29        ["G/T / HTTP/1.0\r\n\r\n", 400],
30        ["GET / HTTP/1.0  \r\nHost: localhost\r\n\r\n", 400],
31        ["GET / HTTP/1.0\r\nFoo: b\x01ar\r\n\r\n", 400],
32        ["GET / HTTP/1.0\r\nF\x01o: bar\r\n\r\n", 400],
33        ["GET / HTTP/1.0\r\r", None],
34        ["GET / HTTP/1.0\r\nHost: localhost\r\nHost: localhost\r\n\r\n", 400],
35        ["GET http://017700000001/ HTTP/1.0\r\n\r\n", 400],
36        ["GET http://0x7f.1/ HTTP/1.0\r\n\r\n", 400],
37        ["GET http://127.01.0.1/ HTTP/1.0\r\n\r\n", 400],
38        ["GET http://%3127.0.0.1/ HTTP/1.0\r\n\r\n", 400],
39        ["GET / HTTP/1.0\r\nHost: localhost:80\r\nHost: localhost:80\r\n\r\n", 400],
40        ["GET http://foo@localhost:80/ HTTP/1.0\r\n\r\n", 400],
41        ["GET / HTTP/1.0\r\nHost: 4714::abcd:8001\r\n\r\n", 400],
42        ["GET / HTTP/1.0\r\nHost: abc\xa0\r\n\r\n", 400],
43        ["GET / HTTP/1.0\r\nHost: foo_bar.example.com\r\n\r\n", 200],
44        ["GET http://foo_bar/ HTTP/1.0\r\n\r\n", 200],
45    ])
46    def test_h1_007_01(self, env, intext, status: Optional[int]):
47        with socket.create_connection(('localhost', int(env.http_port))) as sock:
48            # on some OS, the server does not see our connection until there is
49            # something incoming
50            sock.sendall(intext.encode())
51            sock.shutdown(socket.SHUT_WR)
52            buff = sock.recv(1024)
53            msg = buff.decode()
54            if status is None:
55                assert len(msg) == 0, f"unexpected answer: {msg}"
56            else:
57                assert len(msg) > 0, "no answer from server"
58                rlines = msg.splitlines()
59                response = rlines[0]
60                m = re.match(r'^HTTP/1.1 (\d+)\s+(\S+)', response)
61                assert m, f"unrecognized response: {rlines}"
62                if status == 1:
63                    assert int(m.group(1)) >= 200
64                elif status == 90:
65                    assert len(rlines) >= 1, f"{rlines}"
66                elif status > 0:
67                    assert int(m.group(1)) == status, f"{rlines}"
68                else:
69                    assert int(m.group(1)) >= 400, f"{rlines}"

71    @pytest.mark.parametrize(["hvalue", "expvalue", "status", "lognos"], [
72        ['"123"', '123', 200, None],
73        ['"123 "', '123 ', 200, None],       # trailing space stays
74        ['"123\t"', '123\t', 200, None],     # trailing tab stays
75        ['" 123"', '123', 200, None],        # leading space is stripped
76        ['"          123"', '123', 200, None],  # leading spaces are stripped
77        ['"\t123"', '123', 200, None],       # leading tab is stripped
78        ['"expr=%{unescape:123%0A 123}"', '', 500, ["AH02430"]],  # illegal char
79        ['" \t "', '', 200, None],           # just ws
80    ])
81    def test_h1_007_02(self, env, hvalue, expvalue, status, lognos: Optional[List[str]]):
82        hname = 'ap-test-007'
83        conf = H1Conf(env, extras={
84            f'test1.{env.http_tld}': [
85                '<Location /index.html>',
86                f'Header add {hname} {hvalue}',
87                '</Location>',
88            ]
89        })
90        conf.add_vhost_test1(proxy_self=True)
91        conf.install()
92        assert env.apache_restart() == 0
93        url = env.mkurl("https", "test1", "/index.html")
94        r = env.curl_get(url, options=['--http1.1'])
95        assert r.response["status"] == status
96        if int(status) < 400:
97            assert r.response["header"][hname] == expvalue
98        #
99        if lognos is not None:
100            env.httpd_error_log.ignore_recent(lognos = lognos)
