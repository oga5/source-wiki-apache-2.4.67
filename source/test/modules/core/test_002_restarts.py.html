13class Loader:
14
15    def __init__(self, env, url: str, clients: int, req_per_client: int = 10):
16        self.env = env
17        self.url = url
18        self.clients = clients
19        self.req_per_client = req_per_client
20        self.result = None
21        self.total_request = 0
22        self._thread = None
23
24    def run(self):
25        self.total_requests = self.clients * self.req_per_client
26        conn_per_client = 5
27        args = [self.env.h2load, f"--connect-to=localhost:{self.env.https_port}",
28                "--h1",                                 # use only http/1.1
29                "-n", str(self.total_requests),              # total # of requests to make
30                "-c", str(conn_per_client * self.clients),   # total # of connections to make
31                "-r", str(self.clients),                     # connections at a time
32                "--rate-period", "2",                   # create conns every 2 sec
33                self.url,
34                ]
35        self.result = self.env.run(args)
36
37    def start(self):
38        self._thread = Thread(target=self.run)
39        self._thread.start()
40
41    def join(self):
42        self._thread.join()

45class ChildDynamics:
46
47    RE_DATE_TIME = re.compile(r'\[(?P<date_time>[^\]]+)\] .*')
48    RE_TIME_FRAC = re.compile(r'(?P<dt>.* \d\d:\d\d:\d\d)(?P<frac>.(?P<micros>.\d+)) (?P<year>\d+)')
49    RE_CHILD_CHANGE = re.compile(r'\[(?P<date_time>[^\]]+)\] '
50                                 r'\[mpm_event:\w+\]'
51                                 r' \[pid (?P<main_pid>\d+):tid \w+\] '
52                                 r'.* Child (?P<child_no>\d+) (?P<action>\w+): '
53                                 r'pid (?P<pid>\d+), gen (?P<generation>\d+), .*')
54
55    def __init__(self, env: CoreTestEnv):
56        self.env = env
57        self.changes = list()
58        self._start = None
59        for l in open(env.httpd_error_log.path):
60            m = self.RE_CHILD_CHANGE.match(l)
61            if m:
62                self.changes.append({
63                    'pid': int(m.group('pid')),
64                    'child_no': int(m.group('child_no')),
65                    'gen': int(m.group('generation')),
66                    'action': m.group('action'),
67                    'rtime' : self._rtime(m.group('date_time'))
68                })
69                continue
70            if self._start is None:
71                m = self.RE_DATE_TIME.match(l)
72                if m:
73                    self._rtime(m.group('date_time'))
74
75    def _rtime(self, s: str) -> timedelta:
76        micros = 0
77        m = self.RE_TIME_FRAC.match(s)
78        if m:
79            micros = int(m.group('micros'))
80            s = f"{m.group('dt')} {m.group('year')}"
81        d = datetime.strptime(s, '%a %b %d %H:%M:%S %Y') + timedelta(microseconds=micros)
82        if self._start is None:
83            self._start = d
84        delta = d - self._start
85        return f"{delta.seconds:+02d}.{delta.microseconds:06d}"

89@pytest.mark.skipif(condition='STRESS_TEST' not in os.environ,
90                    reason="STRESS_TEST not set in env")
91@pytest.mark.skipif(condition=not CoreTestEnv().h2load_is_at_least('1.41.0'),
92                    reason="h2load unavailable or misses --connect-to option")
93class TestRestarts:
94
95    def test_core_002_01(self, env):
96        # Lets make a tight config that triggers dynamic child behaviour
97        conf = HttpdConf(env, extras={
98            'base': f"""
99        StartServers            1
100        ServerLimit             3
101        ThreadLimit             4
102        ThreadsPerChild         4
103        MinSpareThreads         4
104        MaxSpareThreads         6
105        MaxRequestWorkers       12
106        MaxConnectionsPerChild  0
107
108        LogLevel mpm_event:trace6
109                """,
110        })
111        conf.add_vhost_cgi()
112        conf.install()
113
114        # clear logs and start server, start load
115        env.httpd_error_log.clear_log()
116        assert env.apache_restart() == 0
117        # we should see a single child started
118        cd = ChildDynamics(env)
119        assert len(cd.changes) == 1, f"{cd.changes}"
120        assert cd.changes[0]['action'] == 'started'
121        # This loader simulates 6 clients, each making 10 requests.
122        # delay.py sleeps for 1sec, so this should run for about 10 seconds
123        loader = Loader(env=env, url=env.mkurl("https", "cgi", "/delay.py"),
124                        clients=6, req_per_client=10)
125        loader.start()
126        # Expect 2 more children to have been started after half time
127        time.sleep(5)
128        cd = ChildDynamics(env)
129        assert len(cd.changes) == 3, f"{cd.changes}"
130        assert len([x for x in cd.changes if x['action'] == 'started']) == 3, f"{cd.changes}"
131
132        # Trigger a server reload
133        assert env.apache_reload() == 0
134        # a graceful reload lets ongoing requests continue, but
135        # after a while all gen 0 children should have stopped
136        time.sleep(3)  # FIXME: this pbly depends on the runtime a lot, do we have expectations?
137        cd = ChildDynamics(env)
138        gen0 = [x for x in cd.changes if x['gen'] == 0]
139        assert len([x for x in gen0 if x['action'] == 'stopped']) == 3
140
141        # wait for the loader to finish and stop the server
142        loader.join()
143        env.apache_stop()
144
145        # Similar to before the reload, we expect 3 children to have
146        # been started and stopped again on server stop
147        cd = ChildDynamics(env)
148        gen1 = [x for x in cd.changes if x['gen'] == 1]
149        assert len([x for x in gen1 if x['action'] == 'started']) == 3
150        assert len([x for x in gen1 if x['action'] == 'stopped']) == 3
