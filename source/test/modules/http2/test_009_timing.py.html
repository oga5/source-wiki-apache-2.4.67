1import inspect
2import json
3import os
4import pytest
5
6from .env import H2Conf, H2TestEnv

8
9@pytest.mark.skipif(condition=H2TestEnv.is_unsupported, reason="mod_http2 not supported here")
10@pytest.mark.skipif(not H2TestEnv().h2load_is_at_least('1.41.0'), reason="h2load misses --connect-to option")
11class TestTiming:
12
13    LOGFILE = ""
14
15    @pytest.fixture(autouse=True, scope='class')
16    def _class_scope(self, env):
17        TestTiming.LOGFILE = os.path.join(env.server_logs_dir, "test_009")
18        if os.path.isfile(TestTiming.LOGFILE):
19            os.remove(TestTiming.LOGFILE)
20        conf = H2Conf(env=env)
21        conf.add([
22            "CustomLog logs/test_009 combined"
23        ])
24        conf.add_vhost_cgi()
25        conf.add_vhost_test1()
26        conf.install()
27        assert env.apache_restart() == 0
28
29    # check that we get a positive time_taken reported on a simple GET
30    def test_h2_009_01(self, env):
31        path = '/002.jpg'
32        url = env.mkurl("https", "test1", f'{path}?01')
33        args = [
34            env.h2load, "-n", "1", "-c", "1", "-m", "1",
35            f"--connect-to=localhost:{env.https_port}",
36            f"--base-uri={url}", url
37        ]
38        r = env.run(args)
39        # Restart for logs to be flushed out
40        assert env.apache_restart() == 0
41        found = False
42        for line in open(TestTiming.LOGFILE).readlines():
43            e = json.loads(line)
44            if e['request'] == f'GET {path}?01 HTTP/2.0':
45                assert e['time_taken'] > 0
46                found = True
47        assert found, f'request not found in {TestTiming.LOGFILE}'
48
49    # test issue #253, where time_taken in a keepalive situation is not
50    # reported until the next request arrives
51    def test_h2_009_02(self, env):
52        baseurl = env.mkurl("https", "test1", '/')
53        tscript = os.path.join(env.gen_dir, 'h2load-timing-009_02')
54        with open(tscript, 'w') as fd:
55            fd.write('\n'.join([
56                f'0.0\t/002.jpg?02a',        # 1st request right away
57                f'1000.0\t/002.jpg?02b',     # 2nd a second later
58            ]))
59        args = [
60            env.h2load,
61            f'--timing-script-file={tscript}',
62            f"--connect-to=localhost:{env.https_port}",
63            f"--base-uri={baseurl}"
64        ]
65        r = env.run(args)
66        # Restart for logs to be flushed out
67        assert env.apache_restart() == 0
68        found = False
69        for line in open(TestTiming.LOGFILE).readlines():
70            e = json.loads(line)
71            if e['request'] == f'GET /002.jpg?02a HTTP/2.0':
72                assert e['time_taken'] > 0
73                assert e['time_taken'] < 500 * 1000, f'time for 1st request not reported correctly'
74                found = True
75        assert found, f'request not found in {TestTiming.LOGFILE}'
