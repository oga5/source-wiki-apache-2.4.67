15    def check_h2load_ok(self, env, r, n):
16        assert 0 == r.exit_code
17        r = env.h2load_status(r)
18        assert n == r.results["h2load"]["requests"]["total"]
19        assert n == r.results["h2load"]["requests"]["started"]
20        assert n == r.results["h2load"]["requests"]["done"]
21        assert n == r.results["h2load"]["requests"]["succeeded"]
22        assert n == r.results["h2load"]["status"]["2xx"]
23        assert 0 == r.results["h2load"]["status"]["3xx"]
24        assert 0 == r.results["h2load"]["status"]["4xx"]
25        assert 0 == r.results["h2load"]["status"]["5xx"]

27    # test POST on cgi, where input is read
28    def test_h2_711_10(self, env, repeat):
29        assert env.is_live()
30        url = env.mkurl("https", "test1", "/echo.py")
31        n = 100
32        m = 5
33        conn = 1
34        fname = "data-100k"
35        args = [
36            env.h2load, "-n", str(n), "-c", str(conn), "-m", str(m),
37            f"--base-uri={env.https_base_url}",
38            "-d", os.path.join(env.gen_dir, fname), url
39        ]
40        r = env.run(args)
41        self.check_h2load_ok(env, r, n)
