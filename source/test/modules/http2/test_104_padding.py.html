6def frame_padding(payload, padbits):
7    mask = (1 << padbits) - 1
8    return ((payload + 9 + mask) & ~mask) - (payload + 9)

50    # default paddings settings: 0 bits
51    def test_h2_104_01(self, env, repeat):
52        url = env.mkurl("https", "ssl", "/h2test/echo")
53        # we get 2 frames back: one with data and an empty one with EOF
54        # check the number of padding bytes is as expected
55        for data in ["x", "xx", "xxx", "xxxx", "xxxxx", "xxxxxx", "xxxxxxx", "xxxxxxxx"]:
56            r = env.nghttp().post_data(url, data, 5)
57            assert r.response["status"] == 200
58            for i in r.results["paddings"]:
59                assert i == frame_padding(len(data)+1, 0)
