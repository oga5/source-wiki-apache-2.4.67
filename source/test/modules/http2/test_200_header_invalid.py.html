15    # let the hecho.py CGI echo chars < 0x20 in field name
16    # for almost all such characters, the stream returns a 500
17    # or in httpd >= 2.5.0 gets aborted with a h2 error
18    # cr is handled special
19    def test_h2_200_01(self, env):
20        url = env.mkurl("https", "cgi", "/hecho.py")
21        for x in range(1, 32):
22            data = f'name=x%{x:02x}x&value=yz'
23            r = env.curl_post_data(url, data)
24            if x in [13]:
25                assert 0 == r.exit_code, f'unexpected exit code for char 0x{x:02}'
26                assert 200 == r.response["status"], f'unexpected status for char 0x{x:02}'
27            elif x in [10] or env.httpd_is_at_least('2.5.0'):
28                assert 0 == r.exit_code, f'unexpected exit code for char 0x{x:02}'
29                assert 500 == r.response["status"], f'unexpected status for char 0x{x:02}'
30            else:
31                assert 0 != r.exit_code, f'unexpected exit code for char 0x{x:02}'
32        #
33        env.httpd_error_log.ignore_recent(
34            lognos = [
35                "AH02429"   # Response header name contains invalid characters
36            ],
37            matches = [
38                r'.*malformed header from script \'hecho.py\': Bad header: x.*'
39            ]
40        )

109    def test_h2_200_11(self, env):
110        conf = H2Conf(env)
111        conf.add("""
112            LimitRequestFieldSize 1024
113            """)
114        conf.add_vhost_cgi()
115        conf.install()
116        assert env.apache_restart() == 0
117        url = env.mkurl("https", "cgi", "/")
118        val = 200*"1234567890"
119        # two fields, concatenated with ', '
120        # LimitRequestFieldSize, one more char -> 400 in HTTP/1.1
121        r = env.curl_get(url, options=[
122            '-H', f'x: {val[:500]}', '-H', f'x: {val[:519]}'
123        ])
124        assert r.exit_code == 0, f'{r}'
125        assert r.response["status"] == 200, f'{r}'
126        r = env.curl_get(url, options=[
127            '--http1.1', '-H', f'x: {val[:500]}', '-H', f'x: {val[:523]}'
128        ])
129        assert 400 == r.response["status"]
130        r = env.curl_get(url, options=[
131            '-H', f'x: {val[:500]}', '-H', f'x: {val[:520]}'
132        ])
133        assert 431 == r.response["status"]
