11    @pytest.fixture(autouse=True, scope='class')
12    def _class_scope(self, env):
13        domain = f"ssl.{env.http_tld}"
14        conf = H2Conf(env, extras={
15            'base': [
16                "SSLCipherSuite ECDHE-RSA-AES256-GCM-SHA384",
17                f"<Directory \"{env.server_dir}/htdocs/ssl-client-verify\">",
18                "    Require all granted",
19                "    SSLVerifyClient require",
20                "    SSLVerifyDepth 0",
21                "</Directory>"
22            ],
23            domain: [
24                "Protocols h2 http/1.1",
25                "<Location /renegotiate/cipher>",
26                "    SSLCipherSuite ECDHE-RSA-CHACHA20-POLY1305",
27                "</Location>",
28                "<Location /renegotiate/err-doc-cipher>",
29                "    SSLCipherSuite ECDHE-RSA-CHACHA20-POLY1305",
30                "    ErrorDocument 403 /forbidden.html",
31                "</Location>",
32                "<Location /renegotiate/verify>",
33                "    SSLVerifyClient require",
34                "</Location>",
35                f"<Directory \"{env.server_dir}/htdocs/sslrequire\">",
36                "    SSLRequireSSL",
37                "</Directory>",
38                f"<Directory \"{env.server_dir}/htdocs/requiressl\">",
39                "    Require ssl",
40                "</Directory>",
41        ]})
42        conf.add_vhost(domains=[domain], port=env.https_port,
43                       doc_root=f"{env.server_dir}/htdocs")
44        conf.install()
45        # the dir needs to exists for the configuration to have effect
46        env.mkpath("%s/htdocs/ssl-client-verify" % env.server_dir)
47        env.mkpath("%s/htdocs/renegotiate/cipher" % env.server_dir)
48        env.mkpath("%s/htdocs/sslrequire" % env.server_dir)
49        env.mkpath("%s/htdocs/requiressl" % env.server_dir)
50        assert env.apache_restart() == 0

52    # access a resource with SSL renegotiation, using HTTP/1.1
53    def test_h2_101_01(self, env):
54        url = env.mkurl("https", "ssl", "/renegotiate/cipher/")
55        r = env.curl_get(url, options=["-v", "--http1.1", "--tlsv1.2", "--tls-max", "1.2"])
56        assert 0 == r.exit_code, f"{r}"
57        assert r.response
58        assert 403 == r.response["status"]
59        #
60        env.httpd_error_log.ignore_recent(
61            lognos = [
62                "AH01276" # No matching DirectoryIndex found
63            ]
64        )
65        
66    # try to renegotiate the cipher, should fail with correct code
67    def test_h2_101_02(self, env):
68        if not (env.curl_is_at_least('8.2.0') or env.curl_is_less_than('8.1.0')):
69            pytest.skip("need curl != 8.1.x version")
70        url = env.mkurl("https", "ssl", "/renegotiate/cipher/")
71        r = env.curl_get(url, options=[
72            "-vvv", "--tlsv1.2", "--tls-max", "1.2", "--ciphers", "ECDHE-RSA-AES256-GCM-SHA384"
73        ])
74        assert 0 != r.exit_code
75        assert not r.response
76        assert re.search(r'HTTP_1_1_REQUIRED \(err 13\)', r.stderr)
77        #
78        env.httpd_error_log.ignore_recent(
79            lognos = [
80                "AH02261"   # Re-negotiation handshake failed
81            ],
82            matches = [
83                r'.*:tls_post_process_client_hello:.*',
84                r'.*SSL Library Error:.*:SSL routines::no shared cipher.*'
85            ]
86        )
87

130    # make 10 requests on the same connection, none should produce a status code
131    # reported by erki@example.ee
132    def test_h2_101_05(self, env):
133        r = env.run([env.h2load, "-n", "10", "-c", "1", "-m", "1", "-vvvv",
134                     f"{env.https_base_url}/ssl-client-verify/index.html"])
135        assert 0 == r.exit_code
136        r = env.h2load_status(r)
137        assert 10 == r.results["h2load"]["requests"]["total"]
138        assert 10 == r.results["h2load"]["requests"]["started"]
139        assert 10 == r.results["h2load"]["requests"]["done"]
140        assert 0 == r.results["h2load"]["requests"]["succeeded"]
141        assert 0 == r.results["h2load"]["status"]["2xx"]
142        assert 0 == r.results["h2load"]["status"]["3xx"]
143        assert 0 == r.results["h2load"]["status"]["4xx"]
144        assert 0 == r.results["h2load"]["status"]["5xx"]
