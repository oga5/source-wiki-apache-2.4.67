16    # by default, we accept leading/trailing ws in request fields
17    def test_h2_203_01_ws_ignore(self, env):
18        url = env.mkurl("https", "test1", "/")
19        r = env.curl_get(url, options=['-H', 'trailing-space: must not  '])
20        assert r.exit_code == 0, f'curl output: {r.stderr}'
21        assert r.response["status"] == 200, f'curl output: {r.stdout}'
22        r = env.curl_get(url, options=['-H', 'trailing-space: must not\t'])
23        assert r.exit_code == 0, f'curl output: {r.stderr}'
24        assert r.response["status"] == 200, f'curl output: {r.stdout}'

26    # response header are also handled, but we strip ws before sending
27    @pytest.mark.parametrize(["hvalue", "expvalue", "status", "lognos"], [
28        ['"123"', '123', 200, None],
29        ['"123 "', '123', 200, None],       # trailing space stripped
30        ['"123\t"', '123', 200, None],     # trailing tab stripped
31        ['" 123"', '123', 200, None],        # leading space is stripped
32        ['"          123"', '123', 200, None],  # leading spaces are stripped
33        ['"\t123"', '123', 200, None],       # leading tab is stripped
34        ['"expr=%{unescape:123%0A 123}"', '', 500, ["AH02430"]],  # illegal char
35        ['" \t "', '', 200, None],          # just ws
36    ])
37    def test_h2_203_02(self, env, hvalue, expvalue, status, lognos: Optional[List[str]]):
38        hname = 'ap-test-007'
39        conf = H2Conf(env, extras={
40            f'test1.{env.http_tld}': [
41                '<Location /index.html>',
42                f'Header add {hname} {hvalue}',
43                '</Location>',
44            ]
45        })
46        conf.add_vhost_test1(proxy_self=True)
47        conf.install()
48        assert env.apache_restart() == 0
49        url = env.mkurl("https", "test1", "/index.html")
50        r = env.curl_get(url, options=['--http2'])
51        if status == 500 and r.exit_code != 0:
52            # in 2.4.x we fail late on control chars in a response
53            # and RST_STREAM. That's also ok
54            return
55        assert r.response["status"] == status
56        if int(status) < 400:
57            assert r.response["header"][hname] == expvalue
58        #
59        if lognos is not None:
60            env.httpd_error_log.ignore_recent(lognos = lognos)
