1import pytest
2
3from .env import H2Conf, H2TestEnv
4
5
6# The push tests depend on "nghttp"
7@pytest.mark.skipif(condition=H2TestEnv.is_unsupported, reason="mod_http2 not supported here")
8class TestEarlyHints:
9
10    @pytest.fixture(autouse=True, scope='class')
11    def _class_scope(self, env):
12        if not env.httpd_is_at_least('2.4.58'):
13            pytest.skip(f'needs httpd 2.4.58')
14        H2Conf(env).start_vhost(domains=[f"hints.{env.http_tld}"],
15                                port=env.https_port, doc_root="htdocs/test1"
16        ).add("""
17        H2EarlyHints on
18        RewriteEngine on
19        RewriteRule ^/006-(.*)?\\.html$ /006.html
20        <Location /006-hints.html>
21            H2PushResource "/006/006.css" critical
22        </Location>
23        <Location /006-nohints.html>
24            Header add Link "</006/006.css>;rel=preload"
25        </Location>
26        <Location /006-early.html>
27            H2EarlyHint Link "</006/006.css>;rel=preload;as=style"
28        </Location>
29        <Location /006-early-no-push.html>
30            H2Push off
31            H2EarlyHint Link "</006/006.css>;rel=preload;as=style"
32        </Location>
33        """).end_vhost(
34        ).install()
35        assert env.apache_restart() == 0
36
37    # H2EarlyHints enabled in general, check that it works for H2PushResource
38    def test_h2_401_31(self, env, repeat):
39        url = env.mkurl("https", "hints", "/006-hints.html")
40        r = env.nghttp().get(url)
41        assert r.response["status"] == 200
42        promises = r.results["streams"][r.response["id"]]["promises"]
43        assert 1 == len(promises)
44        early = r.response["previous"]
45        assert early
46        assert 103 == int(early["header"][":status"])
47        assert early["header"]["link"]
48
49    # H2EarlyHints enabled in general, but does not trigger on added response headers
50    def test_h2_401_32(self, env, repeat):
51        url = env.mkurl("https", "hints", "/006-nohints.html")
52        r = env.nghttp().get(url)
53        assert r.response["status"] == 200
54        promises = r.results["streams"][r.response["id"]]["promises"]
55        assert 1 == len(promises)
56        assert "previous" not in r.response
57
58    # H2EarlyHints enabled in general, check that it works for H2EarlyHint

71    def test_h2_401_34(self, env, repeat):
72        if not env.httpd_is_at_least('2.4.58'):
73            pytest.skip(f'needs httpd 2.4.58')
74        url = env.mkurl("https", "hints", "/006-early-no-push.html")

78        assert 0 == len(promises)
