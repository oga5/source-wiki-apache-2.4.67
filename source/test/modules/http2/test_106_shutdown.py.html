14@pytest.mark.skipif(condition=H2TestEnv.is_unsupported, reason="mod_http2 not supported here")
15class TestShutdown:

17    @pytest.fixture(autouse=True, scope='class')
18    def _class_scope(self, env):
19        conf = H2Conf(env)
20        conf.add_vhost_cgi()
21        conf.install()
22        assert env.apache_restart() == 0
23
24    def test_h2_106_01(self, env):
25        url = env.mkurl("https", "cgi", "/necho.py")
26        lines = 100000
27        text = "123456789"
28        wait2 = 1.0
29        self.r = None
30        def long_request():
31            args = ["-vvv",
32                    "-F", f"count={lines}",
33                    "-F", f"text={text}",
34                    "-F", f"wait2={wait2}",
35                    ]
36            self.r = env.curl_get(url, 5, options=args)
37
38        t = Thread(target=long_request)
39        t.start()
40        time.sleep(0.5)
41        assert env.apache_reload() == 0
42        t.join()
43        # noinspection PyTypeChecker
44        time.sleep(1)
45        r: ExecResult = self.r
46        assert r.exit_code == 0
47        assert r.response, f"no response via {r.args} in {r.stderr}\nstdout: {len(r.stdout)} bytes"
48        assert r.response["status"] == 200, f"{r}"
49        assert len(r.response["body"]) == (lines * (len(text)+1)), f"{r}"
50
51    def test_h2_106_02(self, env):
52        # PR65731: invalid GOAWAY frame at session start when
53        # MaxRequestsPerChild is reached
54        # Create a low limit and only 2 children, so we'll encounter this easily
55        conf = H2Conf(env, extras={
56            'base': [
57                "ServerLimit 2",
58                "MaxRequestsPerChild 3"
59            ]
60        })
61        conf.add_vhost_test1()
62        conf.install()
63        assert env.apache_restart() == 0
64        url = env.mkurl("https", "test1", "/index.html")
65        for i in range(7):
66            r = env.curl_get(url, options=['-v'])
67            # requests should succeed, but rarely connections get closed
68            # before the response is received
69            if r.exit_code in [16, 55]:
70                # curl send error
71                assert r.response is None
72            else:
73                assert r.exit_code == 0, f"failed on {i}. request: {r.stdout} {r.stderr}"
74                assert r.response["status"] == 200
75                assert "HTTP/2" == r.response["protocol"]
76        #
77        env.httpd_error_log.ignore_recent(
78            lognos = [
79                "AH03490"   # scoreboard is full, not at MaxRequestWorkers
80            ]
81        )
