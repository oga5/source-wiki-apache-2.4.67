64    # plain resource without configured pushes 
65    def test_h2_400_00(self, env):
66        url = env.mkurl("https", "push", "/006.html")
67        r = env.nghttp().get(url)
68        assert r.response["status"] == 200
69        promises = r.results["streams"][r.response["id"]]["promises"]
70        assert 0 == len(promises)
71
72    # 2 link headers configured, only 1 triggers push
73    def test_h2_400_01(self, env):
74        url = env.mkurl("https", "push", "/006-push.html")
75        r = env.nghttp().get(url, options=["-Haccept-encoding: none"])
76        assert r.response["status"] == 200
77        promises = r.results["streams"][r.response["id"]]["promises"]
78        assert 1 == len(promises)
79        assert '/006/006.css' == promises[0]["request"]["header"][":path"]
80        assert 216 == len(promises[0]["response"]["body"])
