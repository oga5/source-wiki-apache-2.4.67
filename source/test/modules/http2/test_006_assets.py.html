35    def test_h2_006_03(self, env):
36        # create the tiles files we originally had checked in
37        exp_assets = [
38            {"status": 200, "size": "10K", "path": "/004.html"},
39            {"status": 200, "size": "742", "path": "/004/gophertiles.jpg"},
40        ]
41        for i in range(2, 181):
42            with open(f"{env.server_docs_dir}/test1/004/gophertiles_{i:03d}.jpg", "w") as fd:
43                fd.write("0123456789\n")
44            exp_assets.append(
45                {"status": 200, "size": "11", "path": f"/004/gophertiles_{i:03d}.jpg"},
46            )
47
48        url = env.mkurl("https", "test1", "/004.html")
49        r = env.nghttp().assets(url, options=["-Haccept-encoding: none"])
50        assert 0 == r.exit_code
51        assert 181 == len(r.assets)
52        assert r.assets == exp_assets
