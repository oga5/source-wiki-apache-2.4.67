7def mk_text_file(fpath: str, lines: int):
8    t110 = ""
9    for _ in range(11):
10        t110 += "0123456789"
11    with open(fpath, "w") as fd:
12        for i in range(lines):
13            fd.write("{0:015d}: ".format(i))  # total 128 bytes per line
14            fd.write(t110)
15            fd.write("\n")

18@pytest.mark.skipif(condition=H2TestEnv.is_unsupported, reason="mod_http2 not supported here")
19class TestFrameLengths:
20
21    URI_PATHS = []
22
23    @pytest.fixture(autouse=True, scope='class')

31    @pytest.mark.parametrize("data_frame_len", [
32        99, 1024, 8192
33    ])
34    def test_h2_107_01(self, env, data_frame_len):
35        conf = H2Conf(env, extras={
36            f'cgi.{env.http_tld}': [
37                f'H2MaxDataFrameLen {data_frame_len}',
38            ]
39        })
40        conf.add_vhost_cgi()
41        conf.install()
42        assert env.apache_restart() == 0
43        for p in self.URI_PATHS:
44            url = env.mkurl("https", "cgi", p)
45            r = env.nghttp().get(url, options=[
46                '--header=Accept-Encoding: none',
47            ])
48            assert r.response["status"] == 200
49            assert len(r.results["data_lengths"]) > 0, f'{r}'
50            too_large = [ x for x in r.results["data_lengths"] if x > data_frame_len]
51            assert len(too_large) == 0, f'{p}: {r.results["data_lengths"]}'
