18class FileItem:
19
20    def __init__(self, mparse_item):
21        self.item = mparse_item
22
23    @property
24    def file_name(self):
25        return os.path.basename(self.item.file_name.decode())
26
27    def save_to(self, destpath: str):
28        fsrc = self.item.file_object
29        fsrc.seek(0)
30        with open(destpath, 'wb') as fd:
31            shutil.copyfileobj(fsrc, fd)

34def get_request_params():
35    oforms = {}
36    ofiles = {}
37    if "REQUEST_URI" in os.environ:
38        qforms = parse.parse_qs(parse.urlsplit(os.environ["REQUEST_URI"]).query)
39        for name, values in qforms.items():
40            oforms[name] = values[0]
41    if "CONTENT_TYPE" in os.environ:
42        ctype = os.environ["CONTENT_TYPE"]
43        if ctype == "application/x-www-form-urlencoded":
44            s = sys.stdin.read()
45            qforms = parse.parse_qs(s)
46            for name, values in qforms.items():
47                oforms[name] = values[0]
48        elif ctype.startswith("multipart/"):
49            def on_field(field):
50                oforms[field.field_name.decode()] = field.value.decode()
51            def on_file(file):
52                ofiles[file.field_name.decode()] = FileItem(file)
53            multipart.parse_form(headers={"Content-Type": ctype},
54                                 input_stream=sys.stdin.buffer,
55                                 on_field=on_field, on_file=on_file)
56    return oforms, ofiles
