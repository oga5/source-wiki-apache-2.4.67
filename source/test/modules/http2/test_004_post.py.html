93    # POST some data using nghttp and see it echo'ed properly back
94    def nghttp_post_and_verify(self, env, fname, options=None):
95        url = env.mkurl("https", "cgi", "/echo.py")
96        fpath = os.path.join(env.gen_dir, fname)
97
98        r = env.nghttp().upload(url, fpath, options=options)
99        assert r.exit_code == 0
100        assert r.response["status"] >= 200 and r.response["status"] < 300
101
102        with open(self.local_src(fpath), mode='rb') as file:
103            src = file.read()
104        assert 'request-length' in r.response["header"]
105        assert int(r.response["header"]['request-length']) == len(src)
106        if len(r.response["body"]) != len(src):
107            sys.stderr.writelines(difflib.unified_diff(
108                src.decode().splitlines(True),
109                r.response["body"].decode().splitlines(True),
110                fromfile='source',
111                tofile='response'
112            ))
113            assert len(r.response["body"]) == len(src)
114        assert r.response["body"] == src, f"expected '{src}', got '{r.response['body']}'"

164    def test_h2_004_40(self, env):
165        # echo content using h2test_module "echo" handler
166        def post_and_verify(fname, options=None):
167            url = env.mkurl("https", "cgi", "/h2test/echo")
168            fpath = os.path.join(env.gen_dir, fname)
169            r = env.curl_upload(url, fpath, options=options)
170            assert r.exit_code == 0
171            assert r.response["status"] >= 200 and r.response["status"] < 300
172            
173            ct = r.response["header"]["content-type"]
174            mail_hd = "Content-Type: " + ct + "\r\nMIME-Version: 1.0\r\n\r\n"
175            mime_msg = mail_hd.encode() + r.response["body"]
176            # this MIME API is from hell
177            body = email.parser.BytesParser().parsebytes(mime_msg)
178            assert body
179            assert body.is_multipart()
180            filepart = None
181            for part in body.walk():
182                if fname == part.get_filename():
183                    filepart = part
184            assert filepart
185            with open(self.local_src(fpath), mode='rb') as file:
186                src = file.read()
187            assert src == filepart.get_payload(decode=True)
188        
189        post_and_verify("data-1k", [])
