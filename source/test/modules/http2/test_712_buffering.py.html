18    @pytest.mark.skip(reason="this test shows unreliable jitter")
19    def test_h2_712_01(self, env):
20        # test gRPC like requests that do not end, but give answers, see #207
21        #
22        # this test works like this:
23        # - use curl to POST data to the server /h2test/echo
24        # - feed curl the data in chunks, wait a bit between chunks
25        # - since some buffering on curl's stdout to Python is involved,
26        #   we will see the response data only at the end.
27        # - therefore, we enable tracing with timestamps in curl on stderr
28        #   and see when the response chunks arrive
29        # - if the server sends the incoming data chunks back right away,
30        #   as it should, we see receiving timestamps separated roughly by the
31        #   wait time between sends.
32        #
33        url = env.mkurl("https", "cgi", "/h2test/echo")
34        base_chunk = "0123456789"
35        chunks = ["chunk-{0:03d}-{1}\n".format(i, base_chunk) for i in range(5)]
36        stutter = timedelta(seconds=0.2)
37        piper = CurlPiper(env=env, url=url)
38        piper.stutter_check(chunks, stutter)
39
