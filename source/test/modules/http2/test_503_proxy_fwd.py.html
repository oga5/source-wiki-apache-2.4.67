6@pytest.mark.skipif(condition=H2TestEnv.is_unsupported, reason="mod_http2 not supported here")
7class TestProxyFwd:
8
9    @classmethod
10    def config_fwd_proxy(cls, env, h2_enabled=False):
11        conf = H2Conf(env, extras={
12            'base': [
13                f'Listen {env.proxy_port}',
14                'Protocols h2c http/1.1',
15                'LogLevel proxy_http2:trace2 proxy:trace2',
16            ],
17        })
18        conf.add_vhost_cgi(proxy_self=False, h2proxy_self=False)
19        conf.start_vhost(domains=[f"test1.{env.http_tld}"],
20                         port=env.proxy_port, with_ssl=True)
21        conf.add([
22            'Protocols h2c http/1.1',
23            'ProxyRequests on',
24            f'H2ProxyRequests {"on" if h2_enabled else "off"}',
25        ])
26        conf.end_vhost()
27        conf.install()
28        assert env.apache_restart() == 0
29
30    @pytest.fixture(autouse=True, scope='class')
31    def _class_scope(cls, env):
32        cls.config_fwd_proxy(env)

34    # test the HTTP/1.1 setup working
35    def test_h2_503_01_proxy_fwd_h1(self, env):
36        url = f'http://localhost:{env.http_port}/hello.py'
37        proxy_host = f'test1.{env.http_tld}'
38        options = [
39            '--proxy', f'https://{proxy_host}:{env.proxy_port}',
40            '--resolve', f'{proxy_host}:{env.proxy_port}:127.0.0.1',
41            '--proxy-cacert', f'{env.get_ca_pem_file(proxy_host)}',
42        ]
43        r = env.curl_get(url, 5, options=options)
44        assert r.exit_code == 0, f'{r}'
45        assert r.response['status'] == 200
46        assert r.json['port'] == f'{env.http_port}'
