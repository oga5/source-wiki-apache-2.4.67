105    # github issue #133
106    def clean_header(self, s):
107        s = re.sub(r'\r\n', '\n', s, flags=re.MULTILINE)
108        s = re.sub(r'^date:.*\n', '', s, flags=re.MULTILINE)
109        s = re.sub(r'^server:.*\n', '', s, flags=re.MULTILINE)
110        s = re.sub(r'^last-modified:.*\n', '', s, flags=re.MULTILINE)
111        s = re.sub(r'^etag:.*\n', '', s, flags=re.MULTILINE)
112        s = re.sub(r'^vary:.*\n', '', s, flags=re.MULTILINE)
113        return re.sub(r'^accept-ranges:.*\n', '', s, flags=re.MULTILINE)

194    @pytest.mark.parametrize("path", [
195        "/004.html", "/proxy/004.html", "/h2proxy/004.html"
196    ])
197    def test_h2_003_50(self, env, path, repeat):
198        # check that the resource supports ranges and we see its raw content-length
199        url = env.mkurl("https", "test1", path)
200        r = env.curl_get(url, 5)
201        assert r.response["status"] == 200
202        assert "HTTP/2" == r.response["protocol"]
203        h = r.response["header"]
204        assert "accept-ranges" in h
205        assert "bytes" == h["accept-ranges"]
206        assert "content-length" in h
207        clen = h["content-length"]
208        # get the first 1024 bytes of the resource, 206 status, but content-length as original
209        r = env.curl_get(url, 5, options=["-H", "range: bytes=0-1023"])
210        assert 206 == r.response["status"]
211        assert "HTTP/2" == r.response["protocol"]
212        assert 1024 == len(r.response["body"])
213        assert "content-length" in h
214        assert clen == h["content-length"]
