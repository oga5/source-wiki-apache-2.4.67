49static apr_status_t duration_parse(apr_interval_time_t *ptimeout, const char *value,
50                                   const char *def_unit)
51{
52    char *endp;
53    apr_int64_t n;
54
55    n = apr_strtoi64(value, &endp, 10);
56    if (errno) {
57        return errno;
58    }
59    if (!endp || !*endp) {
60        if (!def_unit) def_unit = "s";
61    }
62    else if (endp == value) {
63        return APR_EINVAL;
64    }
65    else {
66        def_unit = endp;
67    }
68
69    switch (*def_unit) {
70    case 'D':
71    case 'd':
72        *ptimeout = apr_time_from_sec(n * SECS_PER_DAY);
73        break;
74    case 's':
75    case 'S':
76        *ptimeout = (apr_interval_time_t) apr_time_from_sec(n);
77        break;
78    case 'h':
79    case 'H':
80        /* Time is in hours */ view japanese
81        *ptimeout = (apr_interval_time_t) apr_time_from_sec(n * SECS_PER_HOUR);
82        break;
83    case 'm':
84    case 'M':
85        switch (*(++def_unit)) {
86        /* Time is in milliseconds */ view japanese
87        case 's':
88        case 'S':
89            *ptimeout = (apr_interval_time_t) n * 1000;
90            break;
91        /* Time is in minutes */ view japanese
92        case 'i':
93        case 'I':
94            *ptimeout = (apr_interval_time_t) apr_time_from_sec(n * 60);
95            break;
96        default:
97            return APR_EGENERAL;
98        }
99        break;
100    default:
101        return APR_EGENERAL;
102    }
103    return APR_SUCCESS;
104}

132static int h2test_echo_handler(request_rec *r)
133{
134    conn_rec *c = r->connection;
135    apr_bucket_brigade *bb;
136    apr_bucket *b;
137    apr_status_t rv;
138    char buffer[8192];
139    const char *ct;
140    long l;
141    int i;
142    apr_time_t chunk_delay = 0;
143    apr_array_header_t *args = NULL;
144    apr_size_t blen, fail_after = 0;
145    int fail_requested = 0, error_bucket = 1;
146
147    if (strcmp(r->handler, "h2test-echo")) {
148        return DECLINED;
149    }
150    if (r->method_number != M_GET && r->method_number != M_POST) {
151        return DECLINED;
152    }
153
154    if(r->args) {
155        args = apr_cstr_split(r->args, "&", 1, r->pool);
156        for(i = 0; i < args->nelts; ++i) {
157            char *s, *val, *arg = APR_ARRAY_IDX(args, i, char*);
158            s = strchr(arg, '=');
159            if(s) {
160                *s = '\0';
161                val = s + 1;
162                if(!strcmp("id", arg)) {
163                    /* accepted, but not processed */ view japanese
164                    continue;
165                }
166                else if(!strcmp("chunk_delay", arg)) {
167                    rv = duration_parse(&chunk_delay, val, "s");
168                    if(APR_SUCCESS == rv) {
169                        continue;
170                    }
171                }
172                else if(!strcmp("fail_after", arg)) {
173                    fail_after = (int)apr_atoi64(val);
174                    if(fail_after >= 0) {
175                      fail_requested = 1;
176                      continue;
177                    }
178                }
179            }
180            ap_log_rerror(APLOG_MARK, APLOG_ERR, 0, r, "query parameter not "
181                          "understood: '%s' in %s",
182                          arg, r->args);
183            ap_die(HTTP_BAD_REQUEST, r);
184            return OK;
185        }
186    }
187
188    ap_log_rerror(APLOG_MARK, APLOG_TRACE1, 0, r, "echo_handler: processing request");
189    r->status = 200;
190    r->clength = -1;
191    r->chunked = 1;
192    apr_table_unset(r->headers_out, "Content-Length");
193    /* Discourage content-encodings */
194    apr_table_unset(r->headers_out, "Content-Encoding");
195    apr_table_setn(r->subprocess_env, "no-brotli", "1");
196    apr_table_setn(r->subprocess_env, "no-gzip", "1");
197
198    ct = apr_table_get(r->headers_in, "content-type");
199    ap_set_content_type(r, ct? ct : "application/octet-stream");
200
201    bb = apr_brigade_create(r->pool, c->bucket_alloc);
202    /* copy any request body into the response */
203    if ((rv = ap_setup_client_block(r, REQUEST_CHUNKED_DECHUNK))) goto cleanup;
204    if (ap_should_client_block(r)) {
205        while (0 < (l = ap_get_client_block(r, &buffer[0], sizeof(buffer)))) {
206            ap_log_rerror(APLOG_MARK, APLOG_TRACE1, 0, r,
207                          "echo_handler: copying %ld bytes from request body", l);
208            blen = (apr_size_t)l;
209            if (fail_requested) {
210              if (blen > fail_after) {
211                blen = fail_after;
212              }
213              fail_after -= blen;
214            }
215            rv = apr_brigade_write(bb, NULL, NULL, buffer, blen);
216            if (APR_SUCCESS != rv) goto cleanup;
217            if (chunk_delay) {
218                apr_sleep(chunk_delay);
219            }
220            rv = ap_pass_brigade(r->output_filters, bb);
221            if (APR_SUCCESS != rv) goto cleanup;
222            ap_log_rerror(APLOG_MARK, APLOG_TRACE1, 0, r,
223                          "echo_handler: passed %ld bytes from request body", l);
224            if (fail_requested && fail_after == 0) {
225              rv = APR_EINVAL;
226              goto cleanup;
227            }
228        }
229    }
230    /* we are done */
231    b = apr_bucket_eos_create(c->bucket_alloc);
232    APR_BRIGADE_INSERT_TAIL(bb, b);
233    ap_log_rerror(APLOG_MARK, APLOG_TRACE1, 0, r, "echo_handler: request read");
234
235    if (r->trailers_in && !apr_is_empty_table(r->trailers_in)) {
236        ap_log_rerror(APLOG_MARK, APLOG_TRACE2, 0, r,
237                      "echo_handler: seeing incoming trailers");
238        apr_table_setn(r->trailers_out, "h2test-trailers-in", 
239                       apr_itoa(r->pool, 1));
240    }
241    
242    rv = ap_pass_brigade(r->output_filters, bb);
243    
244cleanup:
245    if (rv == APR_SUCCESS
246        || r->status != HTTP_OK
247        || c->aborted) {
248        ap_log_rerror(APLOG_MARK, APLOG_TRACE1, rv, r, "echo_handler: request handled");
249        return OK;
250    }
251    else if (error_bucket) {
252        int status = ap_map_http_request_error(rv, HTTP_BAD_REQUEST);
253        b = ap_bucket_error_create(status, NULL, r->pool, c->bucket_alloc);
254        APR_BRIGADE_INSERT_TAIL(bb, b);
255        ap_pass_brigade(r->output_filters, bb);
256    }
257    else {
258        /* no way to know what type of error occurred */ view japanese
259        ap_log_rerror(APLOG_MARK, APLOG_TRACE1, rv, r, "h2test_echo_handler failed");
260        return AP_FILTER_ERROR;
261    }
262    return DECLINED;
263}

265static int h2test_delay_handler(request_rec *r)
266{
267    conn_rec *c = r->connection;

276    if (strcmp(r->handler, "h2test-delay")) {
277        return DECLINED;
278    }
279    if (r->method_number != M_GET && r->method_number != M_POST) {
280        return DECLINED;
281    }
282
283    if (r->args) {
284        rv = duration_parse(&delay, r->args, "s");
285        if (APR_SUCCESS != rv) {
286            ap_die(HTTP_BAD_REQUEST, r);
287            return OK;
288        }
289    }
290
291    ap_log_rerror(APLOG_MARK, APLOG_TRACE1, 0, r, "delay_handler: processing request, %ds delay",
292                  (int)apr_time_sec(delay));
293    r->status = 200;

305    /* copy any request body into the response */ view japanese
306    if ((rv = ap_setup_client_block(r, REQUEST_CHUNKED_DECHUNK))) goto cleanup;
307    if (ap_should_client_block(r)) {
308        do {
309            l = ap_get_client_block(r, &buffer[0], sizeof(buffer));
310            if (l > 0) {
311                ap_log_rerror(APLOG_MARK, APLOG_TRACE1, 0, r,
312                              "delay_handler: reading %ld bytes from request body", l);
313            }
314        } while (l > 0);
315        if (l < 0) {
316            return AP_FILTER_ERROR;
317        }
318    }
319
320    memset(buffer, 0, sizeof(buffer));
321    l = sizeof(buffer);
322    for (i = 0; i < chunks; ++i) {
323        rv = apr_brigade_write(bb, NULL, NULL, buffer, l);
324        if (APR_SUCCESS != rv) goto cleanup;

326        if (APR_SUCCESS != rv) goto cleanup;
327        ap_log_rerror(APLOG_MARK, APLOG_TRACE1, 0, r,
328                      "delay_handler: passed %ld bytes as response body", l);
329        if (delay) {
330            apr_sleep(delay);
331        }
332    }
333    /* we are done */
334    b = apr_bucket_eos_create(c->bucket_alloc);
335    APR_BRIGADE_INSERT_TAIL(bb, b);
336    rv = ap_pass_brigade(r->output_filters, bb);
337    apr_brigade_cleanup(bb);
338    ap_log_rerror(APLOG_MARK, APLOG_TRACE1, rv, r, "delay_handler: response passed");
339
340cleanup:
341    ap_log_rerror(APLOG_MARK, APLOG_TRACE1, rv, r,
342                  "delay_handler: request cleanup, r->status=%d, aborte=%d",
343                  r->status, c->aborted);
344    if (rv == APR_SUCCESS
345        || r->status != HTTP_OK
346        || c->aborted) {
347        return OK;
348    }
349    return AP_FILTER_ERROR;

416static int status_from_str(const char *s, apr_status_t *pstatus)
417{
418    if (!strcmp("timeout", s)) {
419        *pstatus = APR_TIMEUP;
420        return 1;
421    }
422    else if (!strcmp("reset", s)) {
423        *pstatus = APR_ECONNRESET;
424        return 1;
425    }
426    return 0;

529        if (body_delay) {

531        }
532        rv = apr_brigade_write(bb, NULL, NULL, buffer, l);

541        }
542    }

555        return OK;
556    }
557    if (error_bucket) {
558        http_status = ap_map_http_request_error(rv, HTTP_BAD_REQUEST);
559        b = ap_bucket_error_create(http_status, NULL, r->pool, c->bucket_alloc);
560        APR_BRIGADE_INSERT_TAIL(bb, b);
561        ap_pass_brigade(r->output_filters, bb);
562    }
563    return AP_FILTER_ERROR;
564}
