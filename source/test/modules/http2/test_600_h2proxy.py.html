9    def test_h2_600_01(self, env):
10        conf = H2Conf(env, extras={
11            f'cgi.{env.http_tld}': [
12                "SetEnvIf Host (.+) X_HOST=$1",
13            ]
14        })
15        conf.add_vhost_cgi(h2proxy_self=True)
16        conf.install()
17        assert env.apache_restart() == 0
18        url = env.mkurl("https", "cgi", "/h2proxy/hello.py")
19        r = env.curl_get(url, 5)
20        assert r.response["status"] == 200
21        assert r.response["json"]["protocol"] == "HTTP/2.0"
22        assert r.response["json"]["https"] == "on"
23        assert r.response["json"]["ssl_protocol"] != ""
24        assert r.response["json"]["h2"] == "on"
25        assert r.response["json"]["h2push"] == "off"
26        assert r.response["json"]["host"] == f"cgi.{env.http_tld}:{env.https_port}"

68    # check that connection reuse actually happens as configured
69    @pytest.mark.parametrize("enable_reuse", [ "on", "off" ])
70    def test_h2_600_04(self, env, enable_reuse):
71        conf = H2Conf(env, extras={
72            f'cgi.{env.http_tld}': [
73                f"ProxyPassMatch ^/h2proxy/([0-9]+)/(.*)$ "
74                f"  h2c://127.0.0.1:$1/$2 enablereuse={enable_reuse} keepalive=on",
75            ]
76        })
77        conf.add_vhost_cgi()
78        conf.install()
79        assert env.apache_restart() == 0
80        url = env.mkurl("https", "cgi", f"/h2proxy/{env.http_port}/hello.py")
81        # httpd 2.5.0 disables reuse, not matter the config
82        if enable_reuse == "on" and not env.httpd_is_at_least("2.4.60"):
83            # reuse is not guaranteed for each request, but we expect some
84            # to do it and run on a h2 stream id > 1
85            reused = False
86            count = 10
87            r = env.curl_raw([url] * count, 5)
88            response = r.response
89            for n in range(count):
90                assert response["status"] == 200
91                if n == (count - 1):
92                    break
93                response = response["previous"]
94            assert r.json[0]["h2_stream_id"] == "1"
95            for n in range(1, count):
96                if int(r.json[n]["h2_stream_id"]) > 1:
97                    reused = True
98                    break
99            assert reused
100        else:
101            r = env.curl_raw([url, url], 5)
102            assert r.response["previous"]["status"] == 200
103            assert r.response["status"] == 200
104            assert r.json[0]["h2_stream_id"] == "1"
105            assert r.json[1]["h2_stream_id"] == "1"

107    # do some flexible setup from #235 to proper connection selection
108    @pytest.mark.parametrize("enable_reuse", [ "on", "off" ])
109    def test_h2_600_05(self, env, enable_reuse):
110        conf = H2Conf(env, extras={
111            f'cgi.{env.http_tld}': [
112                f"ProxyPassMatch ^/h2proxy/([0-9]+)/(.*)$ "
113                f"  h2c://127.0.0.1:$1/$2 enablereuse={enable_reuse} keepalive=on",
114            ]
115        })
116        conf.add_vhost_cgi()
117        conf.add([
118            f'Listen {env.http_port2}',
119            'UseCanonicalName On',
120            'UseCanonicalPhysicalPort On'
121        ])
122        conf.start_vhost(domains=[f'cgi.{env.http_tld}'],
123                         port=5004, doc_root="htdocs/cgi")
124        conf.add("AddHandler cgi-script .py")
125        conf.end_vhost()
126        conf.install()
127        assert env.apache_restart() == 0
128        url = env.mkurl("https", "cgi", f"/h2proxy/{env.http_port}/hello.py")
129        url2 = env.mkurl("https", "cgi", f"/h2proxy/{env.http_port2}/hello.py")
130        r = env.curl_raw([url, url2], 5)
131        assert r.response["previous"]["status"] == 200
132        assert int(r.json[0]["port"]) == env.http_port
133        assert r.response["status"] == 200
134        exp_port = env.http_port if enable_reuse == "on" \
135                                    and not env.httpd_is_at_least("2.4.60")\
136            else env.http_port2
137        assert int(r.json[1]["port"]) == exp_port

139    # test X-Forwarded-* headers
140    def test_h2_600_06(self, env):
141        conf = H2Conf(env, extras={
142            f'cgi.{env.http_tld}': [
143                "SetEnvIf Host (.+) X_HOST=$1",
144                f"ProxyPreserveHost on",
145                f"ProxyPass /h2c/ h2c://127.0.0.1:{env.http_port}/",
146                f"ProxyPass /h1c/ http://127.0.0.1:{env.http_port}/",
147            ]
148        })
149        conf.add_vhost_cgi(proxy_self=True)
150        conf.install()
151        assert env.apache_restart() == 0
152        url = env.mkurl("https", "cgi", "/h1c/hello.py")
153        r1 = env.curl_get(url, 5)
154        assert r1.response["status"] == 200
155        url = env.mkurl("https", "cgi", "/h2c/hello.py")
156        r2 = env.curl_get(url, 5)
157        assert r2.response["status"] == 200
158        for key in ['x-forwarded-for', 'x-forwarded-host','x-forwarded-server']:
159            assert r1.json[key] == r2.json[key], f'{key} differs proxy_http != proxy_http2'
