10    def _class_scope(self, env):
11        conf = H2Conf(env, extras={
12            'base': [
13                f'Listen {env.proxy_port}',
14                'Protocols h2c http/1.1',
15                'LogLevel proxy_http2:trace2 proxy:trace2',
16            ],
17            f'cgi.{env.http_tld}': [
18                "Header unset Server",
19                "Header always set Server cgi",
20            ]
21        })
22        conf.add_vhost_cgi(proxy_self=False, h2proxy_self=False)
23        conf.start_vhost(domains=[f"test1.{env.http_tld}"], port=env.proxy_port)
24        conf.add([
25            'Protocols h2c',
26            'RewriteEngine On',
27            'RewriteRule "^/(.*)$" "h2c://%{HTTP_HOST}/$1"[NC,P]',
28            'ProxyPassMatch / "h2c://$1/"',
29        ])
30        conf.end_vhost()
31        conf.install()
32        assert env.apache_restart() == 0
