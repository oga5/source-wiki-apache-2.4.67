20    @pytest.mark.parametrize("name", [
21        "data-1k", "data-10k", "data-100k", "data-1m",
22    ])
23    def test_h2_601_01_echo_uploads(self, env, name):
24        fpath = os.path.join(env.gen_dir, name)
25        url = env.mkurl("https", "cgi", "/h2proxy/h2test/echo")
26        r = env.curl_upload(url, fpath, options=[])
27        assert r.exit_code == 0
28        assert 200 <= r.response["status"] < 300
29        # we POST a form, so echoed input is larger than the file itself
30        assert len(r.response["body"]) > os.path.getsize(fpath)

44    @pytest.mark.parametrize("name", [

47    def test_h2_601_03_echo_fail_early(self, env, name):

56    @pytest.mark.parametrize("name", [
57        "data-1k", "data-10k", "data-100k", "data-1m",
58    ])
59    def test_h2_601_04_echo_fail_late(self, env, name):
60        if not env.httpd_is_at_least('2.4.58'):
61            pytest.skip(f'needs httpd 2.4.58')
62        fpath = os.path.join(env.gen_dir, name)
63        url = env.mkurl("https", "cgi", f"/h2proxy/h2test/echo?fail_after={os.path.getsize(fpath)}")
64        r = env.curl_upload(url, fpath, options=[])
65        # 92 is curl's CURLE_HTTP2_STREAM
66        if r.exit_code != 0:
67            # H2 stream or partial file error
68            assert r.exit_code == 92 or r.exit_code == 18, f'{r}'
69        else:
70            assert r.response["status"] == 502, f'{r}'

72    def test_h2_601_05_echo_fail_many(self, env):
73        if not env.httpd_is_at_least('2.4.58'):
74            pytest.skip(f'needs httpd 2.4.58')
75        if not env.curl_is_at_least('8.0.0'):
76            pytest.skip(f'need at least curl v8.0.0 for this')
77        count = 200
78        fpath = os.path.join(env.gen_dir, "data-100k")
79        args = [env.curl, '--parallel', '--parallel-max', '20']
80        for i in range(count):
81            if i > 0:
82                args.append('--next')
83            url = env.mkurl("https", "cgi", f"/h2proxy/h2test/echo?id={i}&fail_after={os.path.getsize(fpath)}")
84            args.extend(env.curl_resolve_args(url=url))
85            args.extend([
86                '-o', '/dev/null', '-w', '%{json}\\n', '--form', f'file=@{fpath}', url
87            ])
88        log.error(f'run: {args}')
89        r = env.run(args)
90        stats = []
91        for line in r.stdout.splitlines():
92            stats.append(json.loads(line))
93        assert len(stats) == count
94        for st in stats:
95            if st['exitcode'] != 0:
96                # H2 stream or partial file error
97                assert st['exitcode'] == 92 or st['exitcode'] == 18, f'{r}'
98            else:
99                assert st['http_code'] == 502, f'{r}'
