14    # make sure the protocol selection on the different hosts work as expected
15    def test_h2_100_01(self, env):
16        # this host defaults to h2, but we can request h1
17        url = env.mkurl("https", "cgi", "/hello.py")
18        assert "2" == env.curl_protocol_version( url )
19        assert "1.1" == env.curl_protocol_version( url, options=[ "--http1.1" ] )
20        
21        # this host does not enable h2, it always falls back to h1
22        url = env.mkurl("https", "noh2", "/hello.py")
23        assert "1.1" == env.curl_protocol_version( url )
24        assert "1.1" == env.curl_protocol_version( url, options=[ "--http2" ] )
25
26    # access a ServerAlias, after using ServerName in SNI
27    def test_h2_100_02(self, env):
28        url = env.mkurl("https", "cgi", "/hello.py")
29        hostname = ("cgi-alias.%s" % env.http_tld)
30        r = env.curl_get(url, 5, options=["-H", f"Host: {hostname}"])
31        assert r.response["status"] == 200
32        assert "HTTP/2" == r.response["protocol"]
33        assert hostname == r.response["json"]["host"]
34
35    # access another vhost, after using ServerName in SNI, that uses same SSL setup
36    def test_h2_100_03(self, env):
37        url = env.mkurl("https", "cgi", "/")
38        hostname = ("test1.%s" % env.http_tld)
39        r = env.curl_get(url, 5, options=[ "-H", "Host:%s" % hostname ])
40        assert r.response["status"] == 200
41        assert "HTTP/2" == r.response["protocol"]
42        assert "text/html" == r.response["header"]["content-type"]
43
44    # access another vhost, after using ServerName in SNI, 
45    # that has different SSL certificate. This triggers a 421 (misdirected request) response.
46    def test_h2_100_04(self, env):
47        url = env.mkurl("https", "cgi", "/hello.py")
48        hostname = ("noh2.%s" % env.http_tld)

50        assert 421 == r.response["status"]

52        env.httpd_error_log.ignore_recent(

55            ]
