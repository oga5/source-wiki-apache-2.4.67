19    # check handling of 'if-modified-since' header
20    def test_h2_201_01(self, env):
21        url = env.mkurl("https", "test1", "/006/006.css")
22        r = env.curl_get(url)
23        assert r.response["status"] == 200
24        lm = r.response["header"]["last-modified"]
25        assert lm
26        r = env.curl_get(url, options=["-H", "if-modified-since: %s" % lm])
27        assert 304 == r.response["status"]
28        r = env.curl_get(url, options=["-H", "if-modified-since: Tue, 04 Sep 2010 11:51:59 GMT"])
29        assert r.response["status"] == 200

43    @pytest.mark.skipif(True, reason="304 misses the Vary header in trunk and 2.4.x")
44    def test_h2_201_03(self, env):
45        url = env.mkurl("https", "test1", "/006.html")
46        r = env.curl_get(url, options=["-H", "Accept-Encoding: gzip"])
47        assert r.response["status"] == 200
48        for h in r.response["header"]:
49            print("%s: %s" % (h, r.response["header"][h]))
50        lm = r.response["header"]["last-modified"]
51        assert lm
52        assert "gzip" == r.response["header"]["content-encoding"]
53        assert "Accept-Encoding" in r.response["header"]["vary"]
54        
55        r = env.curl_get(url, options=["-H", "if-modified-since: %s" % lm,
56                                       "-H", "Accept-Encoding: gzip"])
57        assert 304 == r.response["status"]
58        for h in r.response["header"]:
59            print("%s: %s" % (h, r.response["header"][h]))
60        assert "vary" in r.response["header"]

62    # Check if "Keep-Alive" response header is removed in HTTP/2.
63    def test_h2_201_04(self, env):
64        url = env.mkurl("https", "test1", "/006.html")
65        r = env.curl_get(url, options=["--http1.1", "-H", "Connection: keep-alive"])
66        assert r.response["status"] == 200
67        assert "timeout=30, max=30" == r.response["header"]["keep-alive"]
68        r = env.curl_get(url, options=["-H", "Connection: keep-alive"])
69        assert r.response["status"] == 200
70        assert "keep-alive" not in r.response["header"]
