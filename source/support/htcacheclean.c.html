176/*
177 * print purge statistics
178 */ view japanese
179static void printstats(char *path, struct stats *s)
180{
181    char ttype, stype, mtype, utype;
182    apr_off_t tfrag, sfrag, ufrag;
183
184    if (!verbose) {
185        return;
186    }
187
188    ttype = 'K';
189    tfrag = ((s->total * 10) / KBYTE) % 10;
190    s->total /= KBYTE;
191    if (s->total >= KBYTE) {
192        ttype = 'M';
193        tfrag = ((s->total * 10) / KBYTE) % 10;
194        s->total /= KBYTE;
195    }
196
197    stype = 'K';
198    sfrag = ((s->sum * 10) / KBYTE) % 10;
199    s->sum /= KBYTE;
200    if (s->sum >= KBYTE) {
201        stype = 'M';
202        sfrag = ((s->sum * 10) / KBYTE) % 10;
203        s->sum /= KBYTE;
204    }
205
206    mtype = 'K';
207    s->max /= KBYTE;
208    if (s->max >= KBYTE) {
209        mtype = 'M';
210        s->max /= KBYTE;
211    }
212
213    apr_file_printf(errfile, "Cleaned %s. Statistics:" APR_EOL_STR, path);
214    if (unsolicited) {
215        utype = 'K';
216        ufrag = ((unsolicited * 10) / KBYTE) % 10;
217        unsolicited /= KBYTE;
218        if (unsolicited >= KBYTE) {
219            utype = 'M';
220            ufrag = ((unsolicited * 10) / KBYTE) % 10;
221            unsolicited /= KBYTE;
222        }
223        if (!unsolicited && !ufrag) {
224            ufrag = 1;
225        }
226        apr_file_printf(errfile, "unsolicited size %d.%d%c" APR_EOL_STR,
227                        (int)(unsolicited), (int)(ufrag), utype);
228    }
229    apr_file_printf(errfile, "size limit %" APR_OFF_T_FMT ".0%c" APR_EOL_STR,
230            s->max, mtype);
231    apr_file_printf(errfile, "inodes limit %" APR_OFF_T_FMT APR_EOL_STR,
232            s->inodes);
233    apr_file_printf(
234            errfile,
235            "total size was %" APR_OFF_T_FMT ".%" APR_OFF_T_FMT "%c, total size now "
236            "%" APR_OFF_T_FMT ".%" APR_OFF_T_FMT "%c" APR_EOL_STR, s->total,
237            tfrag, ttype, s->sum, sfrag, stype);
238    apr_file_printf(errfile, "total inodes was %" APR_OFF_T_FMT
239            ", total %sinodes now "
240            "%" APR_OFF_T_FMT APR_EOL_STR, s->ntotal, dryrun && deldirs ? "estimated "
241            : "", s->nodes);
242    apr_file_printf(
243            errfile,
244            "total entries was %" APR_OFF_T_FMT ", total entries now %" APR_OFF_T_FMT
245            APR_EOL_STR, s->etotal, s->entries);
246    apr_file_printf(
247            errfile,
248            "%" APR_OFF_T_FMT " entries deleted (%" APR_OFF_T_FMT " from future, %"
249            APR_OFF_T_FMT " expired, %" APR_OFF_T_FMT " fresh)" APR_EOL_STR,
250            (s->etotal - s->entries), s->dfuture, s->dexpired, s->dfresh);
251}

253/**
254 * Round the value up to the given threshold.
255 */ view japanese
256static apr_size_t round_up(apr_size_t val, apr_off_t round)
257{
258    if (round > 1) {
259        return (apr_size_t)(((val + round - 1) / round) * round);
260    }
261    return val;
262}

264/*
265 * delete parent directories
266 */ view japanese
267static void delete_parent(const char *path, const char *basename,
268        apr_off_t *nodes, apr_pool_t *pool)
269{
270    char *nextpath, *name;
271    apr_pool_t *p;
272
273    /* temp pool, otherwise lots of memory could be allocated */
274    apr_pool_create(&p, pool);
275    name = apr_pstrdup(p, basename);
276
277    /* If asked to delete dirs, do so now. We don't care if it fails.
278     * If it fails, it likely means there was something else there.
279     */
280    if (deldirs && !dryrun) {
281        const char *vary;
282        char *end = strrchr(name, '/');
283        while (end) {
284            *end = 0;
285
286            /* remove the directory */ view japanese
287            nextpath = apr_pstrcat(p, path, "/", name, NULL);
288            if (!apr_dir_remove(nextpath, p)) {
289                (*nodes)--;
290
291                /* vary directory found? */ view japanese
292                vary = strstr(name, CACHE_VDIR_SUFFIX);
293                if (vary && !vary[sizeof(CACHE_VDIR_SUFFIX) - 1]) {
294                    nextpath = apr_pstrcat(p, path, "/", apr_pstrndup(p, name, vary
295                            - name), NULL);
296                    if (!apr_file_remove(nextpath, p)) {
297                        (*nodes)--;
298                    }
299                }
300
301            }
302            else {
303                break;
304            }
305            end = strrchr(name, '/');
306        }
307    }
308
309    apr_pool_destroy(p);
310
311    if (benice) {
312        if (++delcount >= DELETE_NICE) {
313            apr_sleep(NICE_DELAY);
314            delcount = 0;
315        }
316    }
317
318}

404/*
405 * list the cache directory tree
406 */ view japanese
407static int list_urls(char *path, apr_pool_t *pool, apr_off_t round)
408{
409    apr_dir_t *dir;
410    apr_finfo_t info;
411    apr_size_t len;
412    apr_pool_t *p;
413    apr_file_t *fd;
414    const char *ext, *nextpath;
415    char *url;
416    apr_uint32_t format;
417    disk_cache_info_t disk_info;
418
419    apr_pool_create(&p, pool);
420
421    if (apr_dir_open(&dir, path, p) != APR_SUCCESS) {
422        return 1;
423    }
424
425    while (apr_dir_read(&info, APR_FINFO_TYPE, dir) == APR_SUCCESS && !interrupted) {
426
427        if (info.filetype == APR_DIR) {
428            if (!strcmp(info.name, ".") || !strcmp(info.name, "..")) {
429                continue;
430            }
431
432            if (list_urls(apr_pstrcat(p, path, "/", info.name, NULL), pool, round)) {
433                return 1;
434            }
435        }
436
437        else if (info.filetype == APR_REG) {
438
439            ext = strchr(info.name, '.');
440
441            if (ext && !strcasecmp(ext, CACHE_HEADER_SUFFIX)) {
442
443                nextpath = apr_pstrcat(p, path, "/", info.name, NULL);
444
445                if (apr_file_open(&fd, nextpath, APR_FOPEN_READ
446                        | APR_FOPEN_BINARY, APR_OS_DEFAULT, p) == APR_SUCCESS) {
447                    len = sizeof(format);
448                    if (apr_file_read_full(fd, &format, len, &len)
449                            == APR_SUCCESS) {
450                        if (format == DISK_FORMAT_VERSION) {
451                            apr_off_t offset = 0;
452
453                            apr_file_seek(fd, APR_SET, &offset);
454
455                            len = sizeof(disk_cache_info_t);
456
457                            if (apr_file_read_full(fd, &disk_info, len, &len)
458                                    == APR_SUCCESS) {
459                                len = disk_info.name_len;
460                                url = apr_palloc(p, len + 1);
461                                url[len] = 0;
462
463                                if (apr_file_read_full(fd, url, len, &len)
464                                        == APR_SUCCESS) {
465
466                                    if (listextended) {
467                                        apr_finfo_t hinfo, dinfo;
468
469                                        /* stat the header file */ view japanese
470                                        if (APR_SUCCESS != apr_file_info_get(
471                                                &hinfo, APR_FINFO_SIZE, fd)) {
472                                            /* ignore the file */
473                                        }
474                                        else if (disk_info.has_body && APR_SUCCESS
475                                                != apr_stat(
476                                                        &dinfo,
477                                                        apr_pstrcat(
478                                                                p,
479                                                                path,
480                                                                "/",
481                                                                apr_pstrndup(
482                                                                        p,
483                                                                        info.name,
484                                                                        ext
485                                                                                - info.name),
486                                                                CACHE_DATA_SUFFIX,
487                                                                NULL),
488                                                        APR_FINFO_SIZE
489                                                                | APR_FINFO_IDENT,
490                                                        p)) {
491                                            /* ignore the file */
492                                        }
493                                        else if (disk_info.has_body && (dinfo.device
494                                                != disk_info.device
495                                                || dinfo.inode
496                                                        != disk_info.inode)) {
497                                            /* ignore the file */
498                                        }
499                                        else {
500
501                                            apr_file_printf(
502                                                    outfile,
503                                                    "%s %" APR_SIZE_T_FMT
504                                                    " %" APR_SIZE_T_FMT
505                                                    " %d %" APR_SIZE_T_FMT
506                                                    " %" APR_TIME_T_FMT
507                                                    " %" APR_TIME_T_FMT
508                                                    " %" APR_TIME_T_FMT
509                                                    " %" APR_TIME_T_FMT
510                                                    " %d %d\n",
511                                                    url,
512                                                    round_up((apr_size_t)hinfo.size, round),
513                                                    round_up(
514                                                            disk_info.has_body ? (apr_size_t)dinfo.size
515                                                                    : 0, round),
516                                                    disk_info.status,
517                                                    disk_info.entity_version,
518                                                    disk_info.date,
519                                                    disk_info.expire,
520                                                    disk_info.request_time,
521                                                    disk_info.response_time,
522                                                    disk_info.has_body,
523                                                    disk_info.header_only);
524                                        }
525                                    }
526                                    else {
527                                        apr_finfo_t dinfo;
528
529                                        /* stat the data file */ view japanese
530                                        if (disk_info.has_body && APR_SUCCESS
531                                                != apr_stat(
532                                                        &dinfo,
533                                                        apr_pstrcat(
534                                                                p,
535                                                                path,
536                                                                "/",
537                                                                apr_pstrndup(
538                                                                        p,
539                                                                        info.name,
540                                                                        ext
541                                                                                - info.name),
542                                                                CACHE_DATA_SUFFIX,
543                                                                NULL),
544                                                        APR_FINFO_SIZE
545                                                                | APR_FINFO_IDENT,
546                                                        p)) {
547                                            /* ignore the file */
548                                        }
549                                        else if (disk_info.has_body && (dinfo.device
550                                                != disk_info.device
551                                                || dinfo.inode
552                                                        != disk_info.inode)) {
553                                            /* ignore the file */ view japanese
554                                        }
555                                        else {
556                                            apr_file_printf(outfile, "%s\n",
557                                                    url);
558                                        }
559                                    }
560                                }
561                            }
562                        }
563                    }
564                    apr_file_close(fd);
565
566                }
567            }
568        }
569
570    }
571
572    apr_dir_close(dir);
573
574    if (interrupted) {
575        return 1;
576    }
577
578    apr_pool_destroy(p);
579
580    if (benice) {
581        apr_sleep(NICE_DELAY);
582    }
583
584    if (interrupted) {
585        return 1;
586    }
587
588    return 0;
589}

591/*
592 * walk the cache directory tree
593 */ view japanese
594static int process_dir(char *path, apr_pool_t *pool, apr_off_t *nodes)
595{
596    apr_dir_t *dir;
597    apr_pool_t *p;
598    apr_hash_t *h;
599    apr_hash_index_t *i;
600    apr_file_t *fd;
601    apr_status_t status;
602    apr_finfo_t info;
603    apr_size_t len;
604    apr_time_t current, deviation;
605    char *nextpath, *base, *ext;
606    DIRENTRY *d, *t, *n, anchor;
607    ENTRY *e;
608    int skip, retries;
609    disk_cache_info_t disk_info;
610
611    APR_RING_INIT(&anchor.link, _direntry, link);
612    apr_pool_create(&p, pool);
613    h = apr_hash_make(p);
614    fd = NULL;
615    deviation = MAXDEVIATION * APR_USEC_PER_SEC;
616
617    if (apr_dir_open(&dir, path, p) != APR_SUCCESS) {
618        return 1;
619    }
620
621    while (apr_dir_read(&info, 0, dir) == APR_SUCCESS && !interrupted) {
622        if (!strcmp(info.name, ".") || !strcmp(info.name, "..")) {
623            continue;
624        }
625        d = apr_pcalloc(p, sizeof(DIRENTRY));
626        d->basename = apr_pstrcat(p, path, "/", info.name, NULL);
627        APR_RING_INSERT_TAIL(&anchor.link, d, _direntry, link);
628        (*nodes)++;
629    }
630
631    apr_dir_close(dir);
632
633    if (interrupted) {
634        return 1;
635    }
636
637    skip = baselen + 1;
638
639    for (d = APR_RING_FIRST(&anchor.link);
640         !interrupted && d != APR_RING_SENTINEL(&anchor.link, _direntry, link);
641         d=n) {
642        n = APR_RING_NEXT(d, link);
643        base = strrchr(d->basename, '/');
644        if (!base++) {
645            base = d->basename;
646        }
647        ext = strchr(base, '.');
648
649        /* there may be temporary files which may be gone before
650         * processing, always skip these if not in realclean mode
651         */ view japanese
652        if (!ext && !realclean) {
653            if (!strncasecmp(base, AP_TEMPFILE_BASE, AP_TEMPFILE_BASELEN)
654                && strlen(base) == AP_TEMPFILE_NAMELEN) {
655                continue;
656            }
657        }
658
659        /* this may look strange but apr_stat() may return an error which
660         * is system dependent and there may be transient failures,
661         * so just blindly retry for a short while
662         */ view japanese
663        retries = STAT_ATTEMPTS;
664        status = APR_SUCCESS;
665        do {
666            if (status != APR_SUCCESS) {
667                apr_sleep(STAT_DELAY);
668            }
669            status = apr_stat(&info, d->basename, DIRINFO, p);
670        } while (status != APR_SUCCESS && !interrupted && --retries);
671
672        /* what may happen here is that apache did create a file which
673         * we did detect but then does delete the file before we can
674         * get file information, so if we don't get any file information
675         * we will ignore the file in this case
676         */ view japanese
677        if (status != APR_SUCCESS) {
678            if (!realclean && !interrupted) {
679                continue;
680            }
681            return 1;
682        }
683
684        if (info.filetype == APR_DIR) {
685            char *dirpath = apr_pstrdup(p, d->basename);
686
687            if (process_dir(d->basename, pool, nodes)) {
688                return 1;
689            }
690            /* When given the -t option htcacheclean does not
691             * delete directories that are already empty, so we'll do that here
692             * since process_dir checks all the directories.
693             * If it fails, it likely means there was something else there.
694             */ view japanese
695            if (deldirs && !dryrun) {
696                apr_dir_remove(dirpath, p);
697            }
698            continue;
699        }
700
701        if (info.filetype != APR_REG) {
702            continue;
703        }
704
705        if (!ext) {
706            if (!strncasecmp(base, AP_TEMPFILE_BASE, AP_TEMPFILE_BASELEN)
707                && strlen(base) == AP_TEMPFILE_NAMELEN) {
708                d->basename += skip;
709                d->type = TEMP;
710                d->dsize = info.size;
711                apr_hash_set(h, d->basename, APR_HASH_KEY_STRING, d);
712            }
713            continue;
714        }
715
716        if (!strcasecmp(ext, CACHE_HEADER_SUFFIX)) {
717            *ext = '\0';
718            d->basename += skip;
719            /* if a user manually creates a '.header' file */ view japanese
720            if (d->basename[0] == '\0') {
721                continue;
722            }
723            t = apr_hash_get(h, d->basename, APR_HASH_KEY_STRING);
724            if (t) {
725                d = t;
726            }
727            d->type |= HEADER;
728            d->htime = info.mtime;
729            d->hsize = info.size;
730            apr_hash_set(h, d->basename, APR_HASH_KEY_STRING, d);
731            continue;
732        }
733
734        if (!strcasecmp(ext, CACHE_DATA_SUFFIX)) {
735            *ext = '\0';
736            d->basename += skip;
737            /* if a user manually creates a '.data' file */ view japanese
738            if (d->basename[0] == '\0') {
739                continue;
740            }
741            t = apr_hash_get(h, d->basename, APR_HASH_KEY_STRING);
742            if (t) {
743                d = t;
744            }
745            d->type |= DATA;
746            d->dtime = info.mtime;
747            d->dsize = info.size;
748            apr_hash_set(h, d->basename, APR_HASH_KEY_STRING, d);
749        }
750    }
751
752    if (interrupted) {
753        return 1;
754    }
755
756    path[baselen] = '\0';
757
758    for (i = apr_hash_first(p, h); i && !interrupted; i = apr_hash_next(i)) {
759        void *hvalue;
760        apr_uint32_t format;
761
762        apr_hash_this(i, NULL, NULL, &hvalue);
763        d = hvalue;
764
765        switch(d->type) {
766        case HEADERDATA:
767            nextpath = apr_pstrcat(p, path, "/", d->basename,
768                                   CACHE_HEADER_SUFFIX, NULL);
769            if (apr_file_open(&fd, nextpath, APR_FOPEN_READ | APR_FOPEN_BINARY,
770                              APR_OS_DEFAULT, p) == APR_SUCCESS) {
771                len = sizeof(format);
772                if (apr_file_read_full(fd, &format, len,
773                                       &len) == APR_SUCCESS) {
774                    if (format == DISK_FORMAT_VERSION) {
775                        apr_off_t offset = 0;
776
777                        apr_file_seek(fd, APR_SET, &offset);
778
779                        len = sizeof(disk_cache_info_t);
780
781                        if (apr_file_read_full(fd, &disk_info, len,
782                                               &len) == APR_SUCCESS) {
783                            apr_file_close(fd);
784                            e = apr_palloc(pool, sizeof(ENTRY));
785                            APR_RING_INSERT_TAIL(&root.link, e, _entry, link);
786                            e->expire = disk_info.expire;
787                            e->response_time = disk_info.response_time;
788                            e->htime = d->htime;
789                            e->dtime = d->dtime;
790                            e->hsize = d->hsize;
791                            e->dsize = d->dsize;
792                            e->basename = apr_pstrdup(pool, d->basename);
793                            if (!disk_info.has_body) {
794                                delete_file(path, apr_pstrcat(p, path, "/",
795                                        d->basename, CACHE_DATA_SUFFIX, NULL),
796                                        nodes, p);
797                            }
798                            break;
799                        }
800                        else {
801                            apr_file_close(fd);
802                        }
803                    }
804                    else if (format == VARY_FORMAT_VERSION) {
805                        apr_finfo_t finfo;
806
807                        /* This must be a URL that added Vary headers later,
808                         * so kill the orphaned .data file
809                         */ view japanese
810                        apr_file_close(fd);
811
812                        if (apr_stat(&finfo, apr_pstrcat(p, nextpath,
813                                CACHE_VDIR_SUFFIX, NULL), APR_FINFO_TYPE, p)
814                                || finfo.filetype != APR_DIR) {
815                            delete_entry(path, d->basename, nodes, p);
816                        }
817                        else {
818                            delete_file(path, apr_pstrcat(p, path, "/",
819                                    d->basename, CACHE_DATA_SUFFIX, NULL),
820                                    nodes, p);
821                        }
822                        break;
823                    }
824                    else {
825                        /* We didn't recognise the format, kill the files */ view japanese
826                        apr_file_close(fd);
827                        delete_entry(path, d->basename, nodes, p);
828                        break;
829                    }
830                }
831                else {
832                    apr_file_close(fd);
833                }
834
835            }
836            /* we have a somehow unreadable headers file which is associated
837             * with a data file. this may be caused by apache currently
838             * rewriting the headers file. thus we may delete the file set
839             * either in realclean mode or if the headers file modification
840             * timestamp is not within a specified positive or negative offset
841             * to the current time.
842             */ view japanese
843            current = apr_time_now();
844            if (realclean || d->htime < current - deviation
845                || d->htime > current + deviation) {
846                delete_entry(path, d->basename, nodes, p);
847                unsolicited += d->hsize;
848                unsolicited += d->dsize;
849            }
850            break;
851
852        /* single data and header files may be deleted either in realclean
853         * mode or if their modification timestamp is not within a
854         * specified positive or negative offset to the current time.
855         * this handling is necessary due to possible race conditions
856         * between apache and this process
857         */ view japanese
858        case HEADER:
859            current = apr_time_now();
860            nextpath = apr_pstrcat(p, path, "/", d->basename,
861                                   CACHE_HEADER_SUFFIX, NULL);
862            if (apr_file_open(&fd, nextpath, APR_FOPEN_READ | APR_FOPEN_BINARY,
863                              APR_OS_DEFAULT, p) == APR_SUCCESS) {
864                len = sizeof(format);
865                if (apr_file_read_full(fd, &format, len,
866                                       &len) == APR_SUCCESS) {
867                    if (format == VARY_FORMAT_VERSION) {
868                        apr_time_t expires;
869
870                        len = sizeof(expires);
871
872                        if (apr_file_read_full(fd, &expires, len,
873                                               &len) == APR_SUCCESS) {
874                            apr_finfo_t finfo;
875
876                            apr_file_close(fd);
877
878                            if (apr_stat(&finfo, apr_pstrcat(p, nextpath,
879                                    CACHE_VDIR_SUFFIX, NULL), APR_FINFO_TYPE, p)
880                                    || finfo.filetype != APR_DIR) {
881                                delete_entry(path, d->basename, nodes, p);
882                            }
883                            else if (expires < current) {
884                                delete_entry(path, d->basename, nodes, p);
885                            }
886
887                            break;
888                        }
889                    }
890                    else if (format == DISK_FORMAT_VERSION) {
891                        apr_off_t offset = 0;
892
893                        apr_file_seek(fd, APR_SET, &offset);
894
895                        len = sizeof(disk_cache_info_t);
896
897                        if (apr_file_read_full(fd, &disk_info, len,
898                                               &len) == APR_SUCCESS) {
899                            apr_file_close(fd);
900                            e = apr_palloc(pool, sizeof(ENTRY));
901                            APR_RING_INSERT_TAIL(&root.link, e, _entry, link);
902                            e->expire = disk_info.expire;
903                            e->response_time = disk_info.response_time;
904                            e->htime = d->htime;
905                            e->dtime = d->dtime;
906                            e->hsize = d->hsize;
907                            e->dsize = d->dsize;
908                            e->basename = apr_pstrdup(pool, d->basename);
909                            break;
910                        }
911                        else {
912                            apr_file_close(fd);
913                        }
914                    }
915                    else {
916                        apr_file_close(fd);
917                        delete_entry(path, d->basename, nodes, p);
918                        break;
919                    }
920                }
921                else {
922                    apr_file_close(fd);
923                }
924            }
925
926            if (realclean || d->htime < current - deviation
927                || d->htime > current + deviation) {
928                delete_entry(path, d->basename, nodes, p);
929                unsolicited += d->hsize;
930            }
931            break;
932
933        case DATA:
934            current = apr_time_now();
935            if (realclean || d->dtime < current - deviation
936                || d->dtime > current + deviation) {
937                delete_entry(path, d->basename, nodes, p);
938                unsolicited += d->dsize;
939            }
940            break;
941
942        /* temp files may only be deleted in realclean mode which
943         * is asserted above if a tempfile is in the hash array
944         */ view japanese
945        case TEMP:
946            delete_file(path, d->basename, nodes, p);
947            unsolicited += d->dsize;
948            break;
949        }
950    }
951
952    if (interrupted) {
953        return 1;
954    }
955
956    apr_pool_destroy(p);
957
958    if (benice) {
959        apr_sleep(NICE_DELAY);
960    }
961
962    if (interrupted) {
963        return 1;
964    }
965
966    return 0;
967}

969/*
970 * purge cache entries
971 */ view japanese
972static void purge(char *path, apr_pool_t *pool, apr_off_t max,
973        apr_off_t inodes, apr_off_t nodes, apr_off_t round)
974{
975    ENTRY *e, *n, *oldest;
976
977    struct stats s;
978    s.sum = 0;
979    s.entries = 0;
980    s.dfuture = 0;
981    s.dexpired = 0;
982    s.dfresh = 0;
983    s.max = max;
984    s.nodes = nodes;
985    s.inodes = inodes;
986    s.ntotal = nodes;
987
988    for (e = APR_RING_FIRST(&root.link);
989         e != APR_RING_SENTINEL(&root.link, _entry, link);
990         e = APR_RING_NEXT(e, link)) {
991        s.sum += round_up((apr_size_t)e->hsize, round);
992        s.sum += round_up((apr_size_t)e->dsize, round);
993        s.entries++;
994    }
995
996    s.total = s.sum;
997    s.etotal = s.entries;
998
999    if ((!s.max || s.sum <= s.max) && (!s.inodes || s.nodes <= s.inodes)) {
1000        printstats(path, &s);
1001        return;
1002    }
1003
1004    /* process all entries with a timestamp in the future, this may
1005     * happen if a wrong system time is corrected
1006     */ view japanese
1007
1008    for (e = APR_RING_FIRST(&root.link);
1009         e != APR_RING_SENTINEL(&root.link, _entry, link) && !interrupted;) {
1010        n = APR_RING_NEXT(e, link);
1011        if (e->response_time > now || e->htime > now || e->dtime > now) {
1012            delete_entry(path, e->basename, &s.nodes, pool);
1013            s.sum -= round_up((apr_size_t)e->hsize, round);
1014            s.sum -= round_up((apr_size_t)e->dsize, round);
1015            s.entries--;
1016            s.dfuture++;
1017            APR_RING_REMOVE(e, link);
1018            if ((!s.max || s.sum <= s.max) && (!s.inodes || s.nodes <= s.inodes)) {
1019                if (!interrupted) {
1020                    printstats(path, &s);
1021                }
1022                return;
1023            }
1024        }
1025        e = n;
1026    }
1027
1028    if (interrupted) {
1029        return;
1030    }
1031
1032    /* process all entries which are expired */ view japanese
1033    for (e = APR_RING_FIRST(&root.link);
1034         e != APR_RING_SENTINEL(&root.link, _entry, link) && !interrupted;) {
1035        n = APR_RING_NEXT(e, link);
1036        if (e->expire != APR_DATE_BAD && e->expire < now) {
1037            delete_entry(path, e->basename, &s.nodes, pool);
1038            s.sum -= round_up((apr_size_t)e->hsize, round);
1039            s.sum -= round_up((apr_size_t)e->dsize, round);
1040            s.entries--;
1041            s.dexpired++;
1042            APR_RING_REMOVE(e, link);
1043            if ((!s.max || s.sum <= s.max) && (!s.inodes || s.nodes <= s.inodes)) {
1044                if (!interrupted) {
1045                    printstats(path, &s);
1046                }
1047                return;
1048            }
1049        }
1050        e = n;
1051    }
1052
1053    if (interrupted) {
1054         return;
1055    }
1056
1057    /* process remaining entries oldest to newest, the check for an empty
1058     * ring actually isn't necessary except when the compiler does
1059     * corrupt 64bit arithmetics which happened to me once, so better safe
1060     * than sorry
1061     */ view japanese
1062    while (!((!s.max || s.sum <= s.max) && (!s.inodes || s.nodes <= s.inodes))
1063            && !interrupted && !APR_RING_EMPTY(&root.link, _entry, link)) {
1064        oldest = APR_RING_FIRST(&root.link);
1065
1066        for (e = APR_RING_NEXT(oldest, link);
1067             e != APR_RING_SENTINEL(&root.link, _entry, link);
1068             e = APR_RING_NEXT(e, link)) {
1069            if (e->dtime < oldest->dtime) {
1070                oldest = e;
1071            }
1072        }
1073
1074        delete_entry(path, oldest->basename, &s.nodes, pool);
1075        s.sum -= round_up((apr_size_t)oldest->hsize, round);
1076        s.sum -= round_up((apr_size_t)oldest->dsize, round);
1077        s.entries--;
1078        s.dfresh++;
1079        APR_RING_REMOVE(oldest, link);
1080    }
1081
1082    if (!interrupted) {
1083        printstats(path, &s);
1084    }
1085}

1087static apr_status_t remove_directory(apr_pool_t *pool, const char *dir)
1088{
1089    apr_status_t rv;
1090    apr_dir_t *dirp;
1091    apr_finfo_t dirent;
1092
1093    rv = apr_dir_open(&dirp, dir, pool);
1094    if (APR_STATUS_IS_ENOENT(rv)) {
1095        return rv;
1096    }
1097    if (rv != APR_SUCCESS) {
1098        apr_file_printf(errfile, "Could not open directory %s: %pm" APR_EOL_STR,
1099                dir, &rv);
1100        return rv;
1101    }
1102
1103    while (apr_dir_read(&dirent, APR_FINFO_DIRENT | APR_FINFO_TYPE, dirp)
1104            == APR_SUCCESS) {
1105        if (dirent.filetype == APR_DIR) {
1106            if (strcmp(dirent.name, ".") && strcmp(dirent.name, "..")) {
1107                rv = remove_directory(pool, apr_pstrcat(pool, dir, "/",
1108                        dirent.name, NULL));
1109                /* tolerate the directory not being empty, the cache may have
1110                 * attempted to recreate the directory in the mean time.
1111                 */ view japanese
1112                if (APR_SUCCESS != rv && APR_ENOTEMPTY != rv) {
1113                    break;
1114                }
1115            }
1116        } else {
1117            const char *file = apr_pstrcat(pool, dir, "/", dirent.name, NULL);
1118            rv = apr_file_remove(file, pool);
1119            if (APR_SUCCESS != rv) {
1120                apr_file_printf(errfile,
1121                        "Could not remove file '%s': %pm" APR_EOL_STR, file,
1122                        &rv);
1123                break;
1124            }
1125        }
1126    }
1127
1128    apr_dir_close(dirp);
1129
1130    if (rv == APR_SUCCESS) {
1131        rv = apr_dir_remove(dir, pool);
1132        if (APR_ENOTEMPTY == rv) {
1133            rv = APR_SUCCESS;
1134        }
1135        if (rv != APR_SUCCESS) {
1136            apr_file_printf(errfile, "Could not remove directory %s: %pm" APR_EOL_STR,
1137                    dir, &rv);
1138        }
1139    }
1140
1141    return rv;
1142}

1144static apr_status_t find_directory(apr_pool_t *pool, const char *base,
1145        const char *rest)
1146{

1151    const char *header = apr_pstrcat(pool, rest, CACHE_HEADER_SUFFIX, NULL);
1152    const char *data = apr_pstrcat(pool, rest, CACHE_DATA_SUFFIX, NULL);
1153    const char *vdir = apr_pstrcat(pool, rest, CACHE_HEADER_SUFFIX,
1154            CACHE_VDIR_SUFFIX, NULL);

1156
1157    rv = apr_dir_open(&dirp, base, pool);

1164    rv = APR_ENOENT;
1165
1166    while (apr_dir_read(&dirent, APR_FINFO_DIRENT | APR_FINFO_TYPE, dirp)

1169        int restlen = strlen(rest);
1170        if (dirent.filetype == APR_DIR && !strncmp(rest, dirent.name, len)) {
1171            dirname = apr_pstrcat(pool, base, "/", dirent.name, NULL);
1172            rv = find_directory(pool, dirname, rest + (len < restlen ? len
1173                    : restlen));
1174            if (APR_SUCCESS == rv) {
1175                found = 1;
1176            }
1177        }
1178        if (dirent.filetype == APR_DIR) {
1179            if (!strcmp(dirent.name, vdir)) {
1180                files = 1;
1181            }
1182        }
1183        if (dirent.filetype == APR_REG) {
1184            if (!strcmp(dirent.name, header) || !strcmp(dirent.name, data)) {
1185                files = 1;
1186            }
1187        }
1188    }
1189
1190    apr_dir_close(dirp);
1191
1192    if (files) {
1193        rv = APR_SUCCESS;
1194        if (!dryrun) {
1195            const char *remove;
1196            apr_status_t status;
1197
1198            remove = apr_pstrcat(pool, base, "/", header, NULL);
1199            status = apr_file_remove(remove, pool);
1200            if (status != APR_SUCCESS && !APR_STATUS_IS_ENOENT(status)) {
1201                apr_file_printf(errfile, "Could not remove file %s: %pm" APR_EOL_STR,
1202                        remove, &status);
1203                rv = status;
1204            }
1205
1206            remove = apr_pstrcat(pool, base, "/", data, NULL);
1207            status = apr_file_remove(remove, pool);
1208            if (status != APR_SUCCESS && !APR_STATUS_IS_ENOENT(status)) {
1209                apr_file_printf(errfile, "Could not remove file %s: %pm" APR_EOL_STR,
1210                        remove, &status);
1211                rv = status;
1212            }
1213
1214            status = remove_directory(pool, apr_pstrcat(pool, base, "/", vdir, NULL));
1215            if (status != APR_SUCCESS && !APR_STATUS_IS_ENOENT(status)) {
1216                rv = status;
1217            }
1218        }
1219    }
1220
1221    /* If asked to delete dirs, do so now. We don't care if it fails.
1222     * If it fails, it likely means there was something else there.
1223     */ view japanese
1224    if (dirname && deldirs && !dryrun) {
1225        apr_dir_remove(dirname, pool);
1226    }
1227
1228    if (found) {
1229        return APR_SUCCESS;
1230    }
1231
1232    return rv;
1233}

1235/**
1236 * Delete a specific URL from the cache.
1237 */ view japanese
1238static apr_status_t delete_url(apr_pool_t *pool, const char *proxypath, const char *url)
1239{
1240    apr_md5_ctx_t context;
1241    unsigned char digest[16];
1242    char tmp[23];
1243    int i, k;
1244    unsigned int x;
1245    static const char enc_table[64] =
1246            "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789_@";
1247
1248    apr_md5_init(&context);
1249    apr_md5_update(&context, (const unsigned char *) url, strlen(url));
1250    apr_md5_final(digest, &context);
1251
1252    /* encode 128 bits as 22 characters, using a modified uuencoding
1253     * the encoding is 3 bytes -> 4 characters* i.e. 128 bits is
1254     * 5 x 3 bytes + 1 byte -> 5 * 4 characters + 2 characters
1255     */ view japanese
1256    for (i = 0, k = 0; i < 15; i += 3) {
1257        x = (digest[i] << 16) | (digest[i + 1] << 8) | digest[i + 2];
1258        tmp[k++] = enc_table[x >> 18];
1259        tmp[k++] = enc_table[(x >> 12) & 0x3f];
1260        tmp[k++] = enc_table[(x >> 6) & 0x3f];
1261        tmp[k++] = enc_table[x & 0x3f];
1262    }
1263
1264    /* one byte left */ view japanese
1265    x = digest[15];
1266    tmp[k++] = enc_table[x >> 2]; /* use up 6 bits */ view japanese
1267    tmp[k++] = enc_table[(x << 4) & 0x3f];
1268    tmp[k] = 0;
1269
1270    /* automatically find the directory levels */ view japanese
1271    return find_directory(pool, proxypath, tmp);
1272}
