92#   this subroutine is derived from Perl's getopts.pl with the enhancement of
93#   the "+" metacharacter at the format string to allow a list to be built by
94#   subsequent occurrences of the same option.
95sub Getopts {
96    my ($argumentative, @ARGV) = @_;
97    my $errs = 0;
98    local $_;
99
100    my @args = split / */, $argumentative;
101    while (@ARGV && ($_ = $ARGV[0]) =~ /^-(.)(.*)/) {
102        my ($first, $rest) = ($1,$2);
103        if ($_ =~ m|^--$|) {
104            shift @ARGV;
105            last;
106        }
107        my $pos = index($argumentative,$first);
108        if ($pos >= 0) {
109            if ($pos < $#args && $args[$pos+1] eq ':') {
110                shift @ARGV;
111                if ($rest eq '') {
112                    unless (@ARGV) {
113                        error("Incomplete option: $first (needs an argument)");
114                        $errs++;
115                    }
116                    $rest = shift(@ARGV);
117                }
118                eval "\$opt_$first = \$rest;";
119            }
120            elsif ($pos < $#args && $args[$pos+1] eq '+') {
121                shift @ARGV;
122                if ($rest eq '') {
123                    unless (@ARGV) {
124                        error("Incomplete option: $first (needs an argument)");
125                        $errs++;
126                    }
127                    $rest = shift(@ARGV);
128                }
129                eval "push(\@opt_$first, \$rest);";
130            }
131            else {
132                eval "\$opt_$first = 1";
133                if ($rest eq '') {
134                    shift(@ARGV);
135                }
136                else {
137                    $ARGV[0] = "-$rest";
138                }
139            }
140        }
141        else {
142            error("Unknown option: $first");
143            $errs++;
144            if ($rest ne '') {
145                $ARGV[0] = "-$rest";
146            }
147            else {
148                shift(@ARGV);
149            }
150        }
151    }
152    return ($errs == 0, @ARGV);
153}

155sub usage {
156    print STDERR "Usage: apxs -g [-S <var>=<val>] -n <modname>\n";
157    print STDERR "       apxs -q [-v] [-S <var>=<val>] [<query> ...]\n";
158    print STDERR "       apxs -c [-S <var>=<val>] [-o <dsofile>] [-D <name>[=<value>]]\n";
159    print STDERR "               [-I <incdir>] [-L <libdir>] [-l <libname>] [-Wc,<flags>]\n";
160    print STDERR "               [-Wl,<flags>] [-p] <files> ...\n";
161    print STDERR "       apxs -i [-S <var>=<val>] [-a] [-A] [-n <modname>] <dsofile> ...\n";
162    print STDERR "       apxs -e [-S <var>=<val>] [-a] [-A] [-n <modname>] <dsofile> ...\n";
163    exit(1);
164}

210sub get_config_vars{
211    my ($file, $rh_config) = @_;
212
213    open IN, $file or die "cannot open $file: $!";
214    while (<IN>){
215        if (/^\s*(.*?)\s*=\s*(.*)$/){
216            $rh_config->{$1} = $2;
217        }
218    }
219    close IN;
220}

222sub get_vars {
223    my $result = '';
224    my $ok = 0;
225    my $arg;
226    foreach $arg (@_) {
227        if (exists $config_vars{$arg} or exists $config_vars{lc $arg}) {
228            my $val = exists $config_vars{$arg}
229                ? $config_vars{$arg}
230                : $config_vars{lc $arg};
231            $val =~ s/[()]//g;
232            $result .= eval "qq($val)" if defined $val;
233            $result .= ";;";
234            $ok = 1;
235        }
236        if (not $ok) {
237            if (exists $internal_vars{$arg} or exists $internal_vars{lc $arg}) {
238                my $val = exists $internal_vars{$arg} ? $arg : lc $arg;
239                $val = eval "\$CFG_$val";
240                $result .= eval "qq($val)" if defined $val;
241                $result .= ";;";
242                $ok = 1;
243            }
244            if (not $ok) {
245                error("Invalid query string `$arg'");
246                exit(1);
247            }
248        }
249    }
250    $result =~ s|;;$||;
251    return $result;
252}

498    #   determine installation commands
499    #   and corresponding LoadModule directive
500    my @lmd = ();
501    my @cmds = ();

507            exit(1);
508        }

510        $t =~ s|^.+/([^/]+)$|$1|;

534            }

537                    $name = "$1";
538                    $filename = $base;
539                    $filename =~ s|^[^/]+/||;
540                }
541            }
542            if ($name eq '') {
543                error("Sorry, cannot determine bootstrap symbol name");
544                error("Please specify one with option `-n'");
545                exit(1);
546            }
547        }
548        if ($filename eq '') {
549            $filename = "mod_${name}.c";
550        }
551        my $dir = $CFG_LIBEXECDIR;
552        $dir =~ s|^$CFG_PREFIX/?||;
553        $dir =~ s|(.)$|$1/|;
554	$t =~ s|\.la$|.so|;
555        push(@lmd, sprintf("LoadModule %-18s %s", "${name}_module", "$dir$t"));

564            error("Config file $CFG_SYSCONFDIR/$CFG_TARGET.conf not found");
565            exit(1);
566        }
567
568        open(FP, "<$CFG_SYSCONFDIR/$CFG_TARGET.conf") || die;
569        my $content = join('', <FP>);
570        close(FP);
571
572        if ($content !~ m|\n#?\s*LoadModule\s+|) {
573            error("Activation failed for custom $CFG_SYSCONFDIR/$CFG_TARGET.conf file.");
574            error("At least one `LoadModule' directive already has to exist.");
575            exit(1);
576        }
577
578        my $lmd;
579        my $c = '';
580        $c = '#' if ($opt_A);
581        foreach $lmd (@lmd) {
582            my $what = $opt_A ? "preparing" : "activating";
583            my $lmd_re = $lmd;
584            $lmd_re =~ s/\s+/\\s+/g;
585
586            if ($content !~ m|\n#?\s*$lmd_re|) {
587                # check for open <containers>, so that the new LoadModule
588                # directive always appears *outside* of an <container>.
589
590                my $before = ($content =~ m|^(.*\n)#?\s*LoadModule\s+[^\n]+\n|s)[0];
591
592                # the '()=' trick forces list context and the scalar
593                # assignment counts the number of list members (aka number
594                # of matches) then
595                my $cntopen = () = ($before =~ m|^\s*<[^/].*$|mg);
596                my $cntclose = () = ($before =~ m|^\s*</.*$|mg);
597
598                if ($cntopen == $cntclose) {
599                    # fine. Last LoadModule is contextless.
600                    $content =~ s|^(.*\n#?\s*LoadModule\s+[^\n]+\n)|$1$c$lmd\n|s;
601                }
602                elsif ($cntopen < $cntclose) {
603                    error('Configuration file is not valid. There are sections'
604                          . ' closed before opened.');
605                    exit(1);
606                }
607                else {
608                    # put our cmd after the section containing the last
609                    # LoadModule.
610                    my $found =
611                    $content =~ s!\A (               # string and capture start
612                                  (?:(?:
613                                    ^\s*             # start of conf line with a
614                                    (?:[^<]|<[^/])   # directive which does not
615                                                     # start with '</'
616
617                                    .*(?:$)\n        # rest of the line.
618                                                     # the '$' is in parentheses
619                                                     # to avoid misinterpreting
620                                                     # the string "$\" as
621                                                     # perl variable.
622
623                                    )*               # catch as much as possible
624                                                     # of such lines. (including
625                                                     # zero)
626
627                                    ^\s*</.*(?:$)\n? # after the above, we
628                                                     # expect a config line with
629                                                     # a closing container (</)
630
631                                  ) {$cntopen}       # the whole pattern (bunch
632                                                     # of lines that end up with
633                                                     # a closing directive) must
634                                                     # be repeated $cntopen
635                                                     # times. That's it.
636                                                     # Simple, eh? ;-)
637
638                                  )                  # capture end
639                                 !$1$c$lmd\n!mx;
640
641                    unless ($found) {
642                        error('Configuration file is not valid. There are '
643                              . 'sections opened and not closed.');
644                        exit(1);
645                    }
646                }
647            } else {
648                # replace already existing LoadModule line
649                $content =~ s|^(.*\n)#?\s*$lmd_re[^\n]*\n|$1$c$lmd\n|s;
650            }
651            $lmd =~ m|LoadModule\s+(.+?)_module.*|;
652            notice("[$what module `$1' in $CFG_SYSCONFDIR/$CFG_TARGET.conf]");
653        }
654        if (@lmd) {
655            if (open(FP, ">$CFG_SYSCONFDIR/$CFG_TARGET.conf.new")) {
656                print FP $content;
657                close(FP);
658                system("cp $CFG_SYSCONFDIR/$CFG_TARGET.conf $CFG_SYSCONFDIR/$CFG_TARGET.conf.bak && " .
659                       "cp $CFG_SYSCONFDIR/$CFG_TARGET.conf.new $CFG_SYSCONFDIR/$CFG_TARGET.conf && " .
660                       "rm $CFG_SYSCONFDIR/$CFG_TARGET.conf.new");
661            } else {
662                notice("unable to open configuration file");
663            }
664	}
665    }
666}
