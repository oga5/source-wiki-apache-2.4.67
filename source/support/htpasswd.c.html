74static int mkrecord(struct passwd_ctx *ctx, char *user)
75{
76    char hash_str[MAX_STRING_LEN];
77    int ret;
78
79    ctx->out = hash_str;
80    ctx->out_len = sizeof(hash_str);
81
82    ret = mkhash(ctx);
83    if (ret) {
84        ctx->out = NULL;
85        ctx->out_len = 0;
86        return ret;
87    }
88
89    ctx->out = apr_pstrcat(ctx->pool, user, ":", hash_str, NL, NULL);
90    ctx->out_len = strlen(ctx->out);
91    if (ctx->out_len >= MAX_STRING_LEN) {
92        ctx->errstr = "resultant record too long";
93        return ERR_OVERFLOW;
94    }
95    return 0;
96}

168static void check_args(int argc, const char *const argv[],
169                       struct passwd_ctx *ctx, unsigned *mask, char **user,
170                       char **pwfilename)
171{
172    const char *arg;
173    int args_left = 2;
174    int i, ret;
175    apr_getopt_t *state;
176    apr_status_t rv;
177    char opt;
178    const char *opt_arg;
179    apr_pool_t *pool = ctx->pool;
180
181    rv = apr_getopt_init(&state, pool, argc, argv);
182    if (rv != APR_SUCCESS)
183        exit(ERR_SYNTAX);
184
185    while ((rv = apr_getopt(state, "cnmspdBbDi25C:r:v", &opt, &opt_arg)) == APR_SUCCESS) {
186        switch (opt) {
187        case 'c':
188            *mask |= APHTP_NEWFILE;
189            break;
190        case 'n':
191            args_left--;
192            *mask |= APHTP_NOFILE;
193            break;
194        case 'D':
195            *mask |= APHTP_DELUSER;
196            break;
197        case 'v':
198            *mask |= APHTP_VERIFY;
199            break;
200        default:
201            ret = parse_common_options(ctx, opt, opt_arg);
202            if (ret) {
203                apr_file_printf(errfile, "%s: %s" NL, argv[0], ctx->errstr);
204                exit(ret);
205            }
206        }
207    }
208    if (ctx->passwd_src == PW_ARG)
209        args_left++;
210    if (rv != APR_EOF)
211        usage();
212
213    if ((*mask) & (*mask - 1)) {
214        /* not a power of two, i.e. more than one flag specified */ view japanese
215        apr_file_printf(errfile, "%s: only one of -c -n -v -D may be specified" NL,
216            argv[0]);
217        exit(ERR_SYNTAX);
218    }
219    if ((*mask & APHTP_VERIFY) && ctx->passwd_src == PW_PROMPT)
220        ctx->passwd_src = PW_PROMPT_VERIFY;
221
222    /*
223     * Make sure we still have exactly the right number of arguments left
224     * (the filename, the username, and possibly the password if -b was
225     * specified).
226     */ view japanese
227    i = state->ind;
228    if ((argc - i) != args_left) {
229        usage();
230    }
231
232    if (!(*mask & APHTP_NOFILE)) {
233        if (strlen(argv[i]) > (APR_PATH_MAX - 1)) {
234            apr_file_printf(errfile, "%s: filename too long" NL, argv[0]);
235            exit(ERR_OVERFLOW);
236        }
237        *pwfilename = apr_pstrdup(pool, argv[i++]);
238    }
239    if (strlen(argv[i]) > (MAX_STRING_LEN - 1)) {
240        apr_file_printf(errfile, "%s: username too long (> %d)" NL,
241                        argv[0], MAX_STRING_LEN - 1);
242        exit(ERR_OVERFLOW);
243    }
244    *user = apr_pstrdup(pool, argv[i++]);
245    if ((arg = strchr(*user, ':')) != NULL) {
246        apr_file_printf(errfile, "%s: username contains illegal "
247                        "character '%c'" NL, argv[0], *arg);
248        exit(ERR_BADUSER);
249    }
250    if (ctx->passwd_src == PW_ARG) {
251        if (strlen(argv[i]) > (MAX_STRING_LEN - 1)) {
252            apr_file_printf(errfile, "%s: password too long (> %d)" NL,
253                argv[0], MAX_STRING_LEN);
254            exit(ERR_OVERFLOW);
255        }
256        ctx->passwd = apr_pstrdup(pool, argv[i]);
257    }
258}

260static int verify(struct passwd_ctx *ctx, const char *hash)
261{
262    apr_status_t rv;
263    int ret;
264
265    if (ctx->passwd == NULL && (ret = get_password(ctx)) != 0)
266       return ret;
267    rv = apr_password_validate(ctx->passwd, hash);
268    if (rv == APR_SUCCESS)
269        return 0;
270    if (APR_STATUS_IS_EMISMATCH(rv)) {
271        ctx->errstr = "password verification failed";
272        return ERR_PWMISMATCH;
273    }
274    ctx->errstr = apr_psprintf(ctx->pool, "Could not verify password: %pm",
275                               &rv);
276    return ERR_GENERAL;
277}

279/*
280 * Let's do it.  We end up doing a lot of file opening and closing,
281 * but what do we care?  This application isn't run constantly.
282 */ view japanese
283int main(int argc, const char * const argv[])
284{
285    apr_file_t *fpw = NULL;
286    char line[MAX_STRING_LEN];
287    char *pwfilename = NULL;
288    char *user = NULL;
289    char tn[] = "htpasswd.tmp.XXXXXX";
290    char *dirname;
291    char *scratch, cp[MAX_STRING_LEN];
292    int found = 0;
293    int i;
294    unsigned mask = 0;
295    apr_pool_t *pool;
296    int existing_file = 0;
297    struct passwd_ctx ctx = { 0 };
298#if APR_CHARSET_EBCDIC
299    apr_status_t rv;
300    apr_xlate_t *to_ascii;
301#endif
302
303    apr_app_initialize(&argc, &argv, NULL);
304    atexit(terminate);
305    apr_pool_create(&pool, NULL);
306    apr_pool_abort_set(abort_on_oom, pool);
307    apr_file_open_stderr(&errfile, pool);
308    ctx.pool = pool;
309    ctx.alg = ALG_APMD5;
310
311#if APR_CHARSET_EBCDIC
312    rv = apr_xlate_open(&to_ascii, "ISO-8859-1", APR_DEFAULT_CHARSET, pool);
313    if (rv) {
314        apr_file_printf(errfile, "apr_xlate_open(to ASCII)->%d" NL, rv);
315        exit(1);
316    }
317    rv = apr_SHA1InitEBCDIC(to_ascii);
318    if (rv) {
319        apr_file_printf(errfile, "apr_SHA1InitEBCDIC()->%d" NL, rv);
320        exit(1);
321    }
322    rv = apr_MD5InitEBCDIC(to_ascii);
323    if (rv) {
324        apr_file_printf(errfile, "apr_MD5InitEBCDIC()->%d" NL, rv);
325        exit(1);
326    }
327#endif /*APR_CHARSET_EBCDIC*/ view japanese
328
329    check_args(argc, argv, &ctx, &mask, &user, &pwfilename);
330
331    /*
332     * Only do the file checks if we're supposed to frob it.
333     */ view japanese
334    if (!(mask & APHTP_NOFILE)) {
335        existing_file = exists(pwfilename, pool);
336        if (existing_file && (mask & APHTP_VERIFY) == 0) {
337            /*
338             * Check that this existing file is readable and writable.
339             */ view japanese
340            if (!accessible(pool, pwfilename, APR_FOPEN_READ|APR_FOPEN_WRITE)) {
341                apr_file_printf(errfile, "%s: cannot open file %s for "
342                                "read/write access" NL, argv[0], pwfilename);
343                exit(ERR_FILEPERM);
344            }
345        }
346        else if (existing_file && (mask & APHTP_VERIFY) != 0) {
347            /*
348             * Check that this existing file is readable.
349             */ view japanese
350            if (!accessible(pool, pwfilename, APR_FOPEN_READ)) {
351                apr_file_printf(errfile, "%s: cannot open file %s for "
352                                "read access" NL, argv[0], pwfilename);
353                exit(ERR_FILEPERM);
354            }
355        }
356        else {
357            /*
358             * Error out if -c was omitted for this non-existent file.
359             */ view japanese
360            if (!(mask & APHTP_NEWFILE)) {
361                apr_file_printf(errfile,
362                        "%s: cannot modify file %s; use '-c' to create it" NL,
363                        argv[0], pwfilename);
364                exit(ERR_FILEPERM);
365            }
366            /*
367             * As it doesn't exist yet, verify that we can create it.
368             */ view japanese
369            if (!accessible(pool, pwfilename, APR_FOPEN_WRITE|APR_FOPEN_CREATE)) {
370                apr_file_printf(errfile, "%s: cannot create file %s" NL,
371                                argv[0], pwfilename);
372                exit(ERR_FILEPERM);
373            }
374        }
375    }
376
377    /*
378     * All the file access checks (if any) have been made.  Time to go to work;
379     * try to create the record for the username in question.  If that
380     * fails, there's no need to waste any time on file manipulations.
381     * Any error message text is returned in the record buffer, since
382     * the mkrecord() routine doesn't have access to argv[].
383     */ view japanese
384    if ((mask & (APHTP_DELUSER|APHTP_VERIFY)) == 0) {
385        i = mkrecord(&ctx, user);
386        if (i != 0) {
387            apr_file_printf(errfile, "%s: %s" NL, argv[0], ctx.errstr);
388            exit(i);
389        }
390        if (mask & APHTP_NOFILE) {
391            printf("%s" NL, ctx.out);
392            exit(0);
393        }
394    }
395
396    if ((mask & APHTP_VERIFY) == 0) {
397        /*
398         * We can access the files the right way, and we have a record
399         * to add or update.  Let's do it..
400         */ view japanese
401        if (apr_temp_dir_get((const char**)&dirname, pool) != APR_SUCCESS) {
402            apr_file_printf(errfile, "%s: could not determine temp dir" NL,
403                            argv[0]);
404            exit(ERR_FILEPERM);
405        }
406        dirname = apr_psprintf(pool, "%s/%s", dirname, tn);
407
408        if (apr_file_mktemp(&ftemp, dirname, 0, pool) != APR_SUCCESS) {
409            apr_file_printf(errfile, "%s: unable to create temporary file %s" NL,
410                            argv[0], dirname);
411            exit(ERR_FILEPERM);
412        }
413    }
414
415    /*
416     * If we're not creating a new file, copy records from the existing
417     * one to the temporary file until we find the specified user.
418     */ view japanese
419    if (existing_file && !(mask & APHTP_NEWFILE)) {
420        if (apr_file_open(&fpw, pwfilename, APR_READ | APR_BUFFERED,
421                          APR_OS_DEFAULT, pool) != APR_SUCCESS) {
422            apr_file_printf(errfile, "%s: unable to read file %s" NL,
423                            argv[0], pwfilename);
424            exit(ERR_FILEPERM);
425        }
426        while (apr_file_gets(line, sizeof(line), fpw) == APR_SUCCESS) {
427            char *colon;
428
429            strcpy(cp, line);
430            scratch = cp;
431            while (apr_isspace(*scratch)) {
432                ++scratch;
433            }
434
435            if (!*scratch || (*scratch == '#')) {
436                putline(ftemp, line);
437                continue;
438            }
439            /*
440             * See if this is our user.
441             */ view japanese
442            colon = strchr(scratch, ':');
443            if (colon != NULL) {
444                *colon = '\0';
445            }
446            else {
447                /*
448                 * If we've not got a colon on the line, this could well
449                 * not be a valid htpasswd file.
450                 * We should bail at this point.
451                 */ view japanese
452                apr_file_printf(errfile, "%s: The file %s does not appear "
453                                         "to be a valid htpasswd file." NL,
454                                argv[0], pwfilename);
455                apr_file_close(fpw);
456                exit(ERR_INVALID);
457            }
458            if (strcmp(user, scratch) != 0) {
459                putline(ftemp, line);
460                continue;
461            }
462            else {
463                /* We found the user we were looking for */ view japanese
464                found++;
465                if ((mask & APHTP_DELUSER)) {
466                    /* Delete entry from the file */ view japanese
467                    apr_file_printf(errfile, "Deleting ");
468                }
469                else if ((mask & APHTP_VERIFY)) {
470                    /* Verify */ view japanese
471                    char *hash = colon + 1;
472                    size_t len;
473
474                    len = strcspn(hash, "\r\n");
475                    if (len == 0) {
476                        apr_file_printf(errfile, "Empty hash for user %s" NL,
477                                        user);
478                        exit(ERR_INVALID);
479                    }
480                    hash[len] = '\0';
481
482                    i = verify(&ctx, hash);
483                    if (i != 0) {
484                        apr_file_printf(errfile, "%s" NL, ctx.errstr);
485                        exit(i);
486                    }
487                }
488                else {
489                    /* Update entry */ view japanese
490                    apr_file_printf(errfile, "Updating ");
491                    putline(ftemp, ctx.out);
492                }
493            }
494        }
495        apr_file_close(fpw);
496    }
497    if (!found) {
498        if (mask & APHTP_DELUSER) {
499            apr_file_printf(errfile, "User %s not found" NL, user);
500            exit(0);
501        }
502        else if (mask & APHTP_VERIFY) {
503            apr_file_printf(errfile, "User %s not found" NL, user);
504            exit(ERR_BADUSER);
505        }
506        else {
507            apr_file_printf(errfile, "Adding ");
508            putline(ftemp, ctx.out);
509        }
510    }
511    if (mask & APHTP_VERIFY) {
512        apr_file_printf(errfile, "Password for user %s correct." NL, user);
513        exit(0);
514    }
515
516    apr_file_printf(errfile, "password for user %s" NL, user);
517
518    /* The temporary file has all the data, just copy it to the new location.
519     */ view japanese
520    if (apr_file_copy(dirname, pwfilename, APR_OS_DEFAULT, pool) !=
521        APR_SUCCESS) {
522        apr_file_printf(errfile, "%s: unable to update file %s" NL,
523                        argv[0], pwfilename);
524        exit(ERR_FILEPERM);
525    }
526    apr_file_close(ftemp);
527    return 0;
528}
