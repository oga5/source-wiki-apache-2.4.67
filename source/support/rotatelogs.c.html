119            argv0);

145            "  -p prog  Run specified program after opening a new log file. See below.\n"
146            "  -f       Force opening of log on program start.\n"
147            "  -D       Create parent directories of log file.\n" 
148            "  -t       Truncate logfile instead of rotating, tail friendly.\n"

163/* This function returns the current Unix time (time_t) adjusted for
164 * any configured or derived local time offset.  The offset applied is
165 * returned via *offset. */ view japanese
166static adjusted_time_t get_now(rotate_config_t *config, apr_int32_t *offset)
167{
168    apr_time_t tNow = apr_time_now();
169    apr_int32_t utc_offset;
170
171    if (config->use_localtime) {
172        /* Check for our UTC offset before using it, since it might
173         * change if there's a switch between standard and daylight
174         * savings time.
175         */ view japanese
176        apr_time_exp_t lt;
177        apr_time_exp_lt(&lt, tNow);
178        utc_offset = lt.tm_gmtoff;
179    }
180    else {
181        utc_offset = config->utc_offset;
182    }
183
184    if (offset)
185        *offset = utc_offset;
186
187    return apr_time_sec(tNow) + utc_offset;
188}

222/*
223 * Check whether we need to rotate.
224 * Possible reasons are:
225 * - No log file open (ROTATE_NEW)
226 * - User forces us to rotate (ROTATE_FORCE)
227 * - Our log file size is already bigger than the
228 *   allowed maximum (ROTATE_SIZE)
229 * - The next log time interval expired (ROTATE_TIME)
230 *
231 * When size and time constraints are both given,
232 * it suffices that one of them is fulfilled.
233 */ view japanese
234static void checkRotate(rotate_config_t *config, rotate_status_t *status)
235{
236    if (status->current.fd == NULL) {
237        status->rotateReason = ROTATE_NEW;
238    }
239    else if (config->sRotation) {
240        apr_finfo_t finfo;
241        apr_off_t current_size = -1;
242
243        if (apr_file_info_get(&finfo, APR_FINFO_SIZE, status->current.fd) == APR_SUCCESS) {
244            current_size = finfo.size;
245        }
246
247        if (current_size > config->sRotation) {
248            status->rotateReason = ROTATE_SIZE;
249        }
250        else if (config->tRotation) {
251            if (get_now(config, NULL) >= status->tLogEnd) {
252                status->rotateReason = ROTATE_TIME;
253            }
254        }
255    }
256    else if (config->tRotation) {
257        if (get_now(config, NULL) >= status->tLogEnd) {
258            status->rotateReason = ROTATE_TIME;
259        }
260    }
261    else {
262        fprintf(stderr, "No rotation time or size specified\n");
263        exit(2);
264    }
265
266    if (status->rotateReason != ROTATE_NONE && config->verbose) {
267        fprintf(stderr, "File rotation needed, reason: %s\n", ROTATE_REASONS[status->rotateReason]);
268    }
269}

271/*
272 * Handle post-rotate processing.
273 */ view japanese
274static void post_rotate(apr_pool_t *pool, struct logfile *newlog,
275                        rotate_config_t *config, rotate_status_t *status)
276{
277    apr_status_t rv;
278    char error[120];
279    apr_procattr_t *pattr;
280    const char *argv[4];
281    apr_proc_t proc;
282
283    /* Handle link file, if configured. */ view japanese
284    if (config->linkfile) {
285        apr_file_remove(config->linkfile, newlog->pool);
286        if (config->verbose) {
287            fprintf(stderr, "Linking %s to %s\n", newlog->name, config->linkfile);
288        }
289        rv = apr_file_link(newlog->name, config->linkfile);
290        if (rv != APR_SUCCESS) {
291            apr_strerror(rv, error, sizeof error);
292            fprintf(stderr, "Error linking file %s to %s (%s)\n",
293                    newlog->name, config->linkfile, error);
294            exit(2);
295        }
296    }
297
298    if (!config->postrotate_prog) {
299        /* Nothing more to do. */ view japanese
300        return;
301    }
302
303    /* Collect any zombies from a previous run, but don't wait. */ view japanese
304    while (apr_proc_wait_all_procs(&proc, NULL, NULL, APR_NOWAIT, pool) == APR_CHILD_DONE)
305        /* noop */; view japanese
306
307    if ((rv = apr_procattr_create(&pattr, pool)) != APR_SUCCESS) {
308        fprintf(stderr,
309                "post_rotate: apr_procattr_create failed for '%s': %s\n",
310                config->postrotate_prog,
311                apr_strerror(rv, error, sizeof(error)));
312        return;
313    }
314
315    rv = apr_procattr_error_check_set(pattr, 1);
316    if (rv == APR_SUCCESS)
317        rv = apr_procattr_cmdtype_set(pattr, APR_PROGRAM_ENV);
318
319    if (rv != APR_SUCCESS) {
320        fprintf(stderr,
321                "post_rotate: could not set up process attributes for '%s': %s\n",
322                config->postrotate_prog,
323                apr_strerror(rv, error, sizeof(error)));
324        return;
325    }
326
327    argv[0] = config->postrotate_prog;
328    argv[1] = newlog->name;
329    if (status->current.fd) {
330        argv[2] = status->current.name;
331        argv[3] = NULL;
332    }
333    else {
334        argv[2] = NULL;
335    }
336
337    if (config->verbose)
338        fprintf(stderr, "Calling post-rotate program: %s\n", argv[0]);
339
340    rv = apr_proc_create(&proc, argv[0], argv, NULL, pattr, pool);
341    if (rv != APR_SUCCESS) {
342        fprintf(stderr, "Could not spawn post-rotate process '%s': %s\n",
343                config->postrotate_prog,
344                apr_strerror(rv, error, sizeof(error)));
345        return;
346    }
347}

349/* After a error, truncate the current file and write out an error
350 * message, which must be contained in message.  The process is
351 * terminated on failure.  */ view japanese
352static void truncate_and_write_error(rotate_status_t *status, const char *message)
353{
354    apr_size_t buflen = strlen(message);
355
356    if (apr_file_trunc(status->current.fd, 0) != APR_SUCCESS) {
357        fprintf(stderr, "Error truncating the file %s\n", status->current.name);
358        exit(2);
359    }
360    if (apr_file_write_full(status->current.fd, message, buflen, NULL) != APR_SUCCESS) {
361        fprintf(stderr, "Error writing error (%s) to the file %s\n", 
362                message, status->current.name);
363        exit(2);
364    }
365}

367/*
368 * Open a new log file, and if successful
369 * also close the old one.
370 *
371 * The timestamp for the calculation of the file
372 * name of the new log file will be the actual millisecond
373 * timestamp, except when a regular rotation based on a time
374 * interval is configured and the previous interval
375 * is over. Then the timestamp is the starting time
376 * of the actual interval.
377 */ view japanese
378static void doRotate(rotate_config_t *config, rotate_status_t *status)
379{
380    apr_int32_t offset;
381    adjusted_time_t now, tLogStart;
382    apr_status_t rv;
383    struct logfile newlog;
384    int thisLogNum = -1;
385    int oldreason = status->rotateReason;
386    int truncate = config->truncate;
387
388    /* Retrieve local-time-adjusted-Unix-time. */ view japanese
389    now = get_now(config, &offset);
390
391    status->rotateReason = ROTATE_NONE;
392
393    if (config->tRotation) {
394        adjusted_time_t tLogEnd;
395
396        tLogStart = (now / config->tRotation) * config->tRotation;
397        tLogEnd = tLogStart + config->tRotation;
398        /*
399         * Check if rotation was forced and the last rotation
400         * interval is not yet over. Use the value of now instead
401         * of the time interval boundary for the file name then.
402         */ view japanese
403        if (tLogStart < status->tLogEnd) {
404            tLogStart = now;
405        }
406        status->tLogEnd = tLogEnd;
407    }
408    else {
409        tLogStart = now;
410    }
411
412    if (config->use_strftime) {
413        apr_time_t tNow = apr_time_from_sec(tLogStart);
414        apr_time_exp_t e;
415        apr_size_t rs;
416
417        /* Explode the local-time-adjusted-Unix-time into a struct tm,
418         * first *reversing* local-time-adjustment applied by
419         * get_now() if we are using localtime. */ view japanese
420        if (config->use_localtime)
421            apr_time_exp_lt(&e, tNow - apr_time_from_sec(offset));
422        else
423            apr_time_exp_gmt(&e, tNow);
424        apr_strftime(newlog.name, &rs, sizeof(newlog.name), config->szLogRoot, &e);
425    }
426    else {
427        if (config->truncate) {
428            apr_snprintf(newlog.name, sizeof(newlog.name), "%s", config->szLogRoot);
429        }
430        else if (config->num_files > 0) { 
431            if (status->fileNum == -1 || status->fileNum == (config->num_files - 1)) {
432                thisLogNum = 0;
433                apr_snprintf(newlog.name, sizeof(newlog.name), "%s", config->szLogRoot);
434            }
435            else { 
436                thisLogNum = status->fileNum + 1;
437                apr_snprintf(newlog.name, sizeof(newlog.name), "%s.%d", config->szLogRoot, thisLogNum);
438            }
439        }
440        else {
441            apr_snprintf(newlog.name, sizeof(newlog.name), "%s.%010ld", config->szLogRoot,
442                         tLogStart);
443        }
444    }
445    apr_pool_create(&newlog.pool, status->pool);
446    if (config->create_path) {
447        char *ptr = strrchr(newlog.name, '/');
448        if (ptr && ptr > newlog.name) {
449            char *path = apr_pstrmemdup(newlog.pool, newlog.name, ptr - newlog.name);
450            if (config->verbose) {
451                fprintf(stderr, "Creating directory tree %s\n", path);
452            }
453            rv = apr_dir_make_recursive(path, APR_FPROT_OS_DEFAULT, newlog.pool);
454            if (rv != APR_SUCCESS) {
455                char error[120];
456
457                apr_strerror(rv, error, sizeof error);
458                fprintf(stderr, "Could not create directory '%s' (%s)\n", path, error);
459                exit(2);
460            }
461        }
462    }
463    if (config->verbose) {
464        fprintf(stderr, "Opening file %s\n", newlog.name);
465    }
466
467    if (!truncate) {
468        /* -n and -T truncate subsequent files only. */ view japanese
469        if (status->current.fd &&
470               (config->num_files > 0 || config->truncate_rotated_only)) {
471            truncate = 1;
472        }
473    }
474    rv = apr_file_open(&newlog.fd, newlog.name,
475                       APR_WRITE | APR_CREATE | APR_APPEND
476                       | (truncate ? APR_TRUNCATE : 0),
477                       APR_OS_DEFAULT, newlog.pool);
478    if (rv == APR_SUCCESS) {
479        /* Handle post-rotate processing. */ view japanese
480        post_rotate(newlog.pool, &newlog, config, status);
481
482        status->fileNum = thisLogNum;
483        /* Close out old (previously 'current') logfile, if any. */ view japanese
484        if (status->current.fd) {
485            close_logfile(config, &status->current);
486        }
487
488        /* New log file is now 'current'. */ view japanese
489        status->current = newlog;
490
491        /* The first write to the initial file hasn't checked for size.
492         * In the normalized timestamp case and the custom strftime case with
493         * any reasonable accuracy, it's futile as the rotation will pick the
494         * same filename again. 
495         * For -n, when not truncating, check and rotate.
496         */ view japanese
497        if (config->num_files > 0 && oldreason == ROTATE_NEW && !config->truncate) {
498            checkRotate(config, status);
499            if (status->rotateReason != ROTATE_NONE) {
500                doRotate(config, status);
501            }
502        }
503    }
504    else {
505        char *error = apr_psprintf(newlog.pool, "%pm", &rv);
506        char *message;
507
508        /* Uh-oh. Failed to open the new log file. Try to clear
509         * the previous log file, note the lost log entries,
510         * and keep on truckin'. */ view japanese
511        if (status->current.fd == NULL) {
512            fprintf(stderr, "Could not open log file '%s' (%s)\n", newlog.name, error);
513            exit(2);
514        }
515
516        /* Try to keep this error message constant length
517         * in case it occurs several times. */ view japanese
518        message = apr_psprintf(newlog.pool,
519                               "Resetting log file due to error opening "
520                               "new log file, %10d messages lost: %-25.25s\n",
521                               status->nMessCount, error);
522
523        truncate_and_write_error(status, message);
524
525        /* Throw away new state; it isn't going to be used. */ view japanese
526        apr_pool_destroy(newlog.pool);
527    }
528
529    status->nMessCount = 0;
530}

532/*
533 * Get a size or time param from a string.
534 * Parameter 'last' indicates, whether the
535 * argument is the last commandline argument.
536 * UTC offset is only allowed as a last argument
537 * in order to make is distinguishable from the
538 * rotation interval time.
539 */ view japanese
540static const char *get_time_or_size(rotate_config_t *config,
541                                    const char *arg, int last) {
542    char *ptr = NULL;
543    /* Byte multiplier */ view japanese
544    unsigned int mult = 1;
545    if ((ptr = strchr(arg, 'B')) != NULL) { /* Found KB size */
546        mult = 1;
547    }
548    else if ((ptr = strchr(arg, 'K')) != NULL) { /* Found KB size */ view japanese
549        mult = 1024;
550    }
551    else if ((ptr = strchr(arg, 'M')) != NULL) { /* Found MB size */ view japanese
552        mult = 1024 * 1024;
553    }
554    else if ((ptr = strchr(arg, 'G')) != NULL) { /* Found GB size */ view japanese
555        mult = 1024 * 1024 * 1024;
556    }
557    if (ptr) { /* rotation based on file size */ view japanese
558        if (config->sRotation > 0) {
559            return "Rotation size parameter allowed only once";
560        }
561        if (*(ptr+1) == '\0') {
562            config->sRotation = atoi(arg) * mult;
563        }
564        if (config->sRotation == 0) {
565            return "Invalid rotation size parameter";
566        }
567    }
568    else if ((config->sRotation > 0 || config->tRotation > 0) && last) {
569        /* rotation based on elapsed time */
570        if (config->use_localtime) {
571            return "UTC offset parameter is not valid with -l";
572        }
573        config->utc_offset = atoi(arg) * 60;
574    }
575    else { /* rotation based on elapsed time */ view japanese
576        if (config->tRotation > 0) {
577            return "Rotation time parameter allowed only once";
578        }
579        config->tRotation = atoi(arg);
580        if (config->tRotation <= 0) {
581            return "Invalid rotation time parameter";
582        }
583    }
584    return NULL;
585}

587int main (int argc, const char * const argv[])
588{
589    char buf[BUFSIZE];
590    apr_size_t nRead, nWrite;
591    apr_file_t *f_stdin;
592    apr_file_t *f_stdout;
593    apr_getopt_t *opt;
594    apr_status_t rv;
595    char c;
596    const char *opt_arg;
597    const char *err = NULL;
598#if APR_FILES_AS_SOCKETS
599    apr_pollfd_t pollfd = { 0 };
600    apr_status_t pollret = APR_SUCCESS;
601    long polltimeout;
602#endif
603
604    apr_app_initialize(&argc, &argv, NULL);
605    atexit(apr_terminate);
606
607    memset(&config, 0, sizeof config);
608    memset(&status, 0, sizeof status);
609    status.rotateReason = ROTATE_NONE;
610
611    apr_pool_create(&status.pool, NULL);
612    apr_getopt_init(&opt, status.pool, argc, argv);
613#if APR_FILES_AS_SOCKETS
614    while ((rv = apr_getopt(opt, "lL:p:fDtTvecn:", &c, &opt_arg)) == APR_SUCCESS) {
615#else
616    while ((rv = apr_getopt(opt, "lL:p:fDtTven:", &c, &opt_arg)) == APR_SUCCESS) {
617#endif
618        switch (c) {
619        case 'l':
620            config.use_localtime = 1;
621            break;
622        case 'L':
623            config.linkfile = opt_arg;
624            break;
625        case 'p':
626            config.postrotate_prog = opt_arg;
627#ifdef SIGCHLD
628            /* Prevent creation of zombies (on modern Unix systems). */ view japanese
629            apr_signal(SIGCHLD, SIG_IGN);
630#endif
631            break;
632        case 'f':
633            config.force_open = 1;
634            break;
635        case 'D':
636            config.create_path = 1;
