474/*
475 * Similar to standard strstr() but we ignore case in this version.
476 * Copied from ap_strcasestr().
477 */ view japanese
478static char *xstrcasestr(const char *s1, const char *s2)
479{
480    char *p1, *p2;
481    if (*s2 == '\0') {
482        /* an empty s2 */ view japanese
483        return((char *)s1);
484    }
485    while(1) {
486        for ( ; (*s1 != '\0') && (apr_tolower(*s1) != apr_tolower(*s2)); s1++);
487        if (*s1 == '\0') {
488            return(NULL);
489        }
490        /* found first character of s2, see if the rest matches */ view japanese
491        p1 = (char *)s1;
492        p2 = (char *)s2;
493        for (++p1, ++p2; apr_tolower(*p1) == apr_tolower(*p2); ++p1, ++p2) {
494            if (*p1 == '\0') {
495                /* both strings ended together */ view japanese
496                return((char *)s1);
497            }
498        }
499        if (*p2 == '\0') {
500            /* second string ended, a match */ view japanese
501            break;
502        }
503        /* didn't find a match here, try starting at next character in s1 */ view japanese
504        s1++;
505    }
506    return((char *)s1);
507}

518static void set_polled_events(struct connection *c, apr_int16_t new_reqevents)
519{
520    apr_status_t rv;
521
522    if (c->pollfd.reqevents != new_reqevents) {
523        if (c->pollfd.reqevents != 0) {
524            rv = apr_pollset_remove(readbits, &c->pollfd);
525            if (rv != APR_SUCCESS) {
526                apr_err("apr_pollset_remove()", rv);
527            }
528        }
529
530        if (new_reqevents != 0) {
531            c->pollfd.reqevents = new_reqevents;
532            rv = apr_pollset_add(readbits, &c->pollfd);
533            if (rv != APR_SUCCESS) {
534                apr_err("apr_pollset_add()", rv);
535            }
536        }
537    }
538}

559 * (small) request out in one go into our new socket buffer

562#ifdef USE_SSL
563#if OPENSSL_VERSION_NUMBER >= 0x30000000L
564static long ssl_print_cb(BIO *bio, int cmd, const char *argp,
565                         size_t len, int argi, long argl, int ret,
566                         size_t *processed)
567#else
568static long ssl_print_cb(BIO *bio, int cmd, const char *argp,
569                         int argi, long argl, long ret)
570#endif
571{
572    BIO *out;
573#if OPENSSL_VERSION_NUMBER >= 0x30000000L
574    (void)len;
575    (void)processed;
576#endif
577
578    out=(BIO *)BIO_get_callback_arg(bio);
579    if (out == NULL) return(ret);
580
581    if (cmd == (BIO_CB_READ|BIO_CB_RETURN)) {
582        BIO_printf(out,"read from %p [%p] (%d bytes => %ld (0x%lX))\n",
583                   bio, argp, argi, (long)ret, (long)ret);
584        BIO_dump(out,(char *)argp,(int)ret);
585        return(ret);
586    }
587    else if (cmd == (BIO_CB_WRITE|BIO_CB_RETURN)) {
588        BIO_printf(out,"write to %p [%p] (%d bytes => %ld (0x%lX))\n",
589                   bio, argp, argi, (long)ret, (long)ret);
590        BIO_dump(out,(char *)argp,(int)ret);
591    }
592    return ret;
593}
