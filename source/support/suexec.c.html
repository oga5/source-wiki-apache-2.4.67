149static void log_err(const char *fmt,...) 
150    __attribute__((format(printf,1,2)));
151static void log_no_err(const char *fmt,...)  
152    __attribute__((format(printf,1,2)));
153static void err_output(int is_error, const char *fmt, va_list ap) 
154    __attribute__((format(printf,2,0)));
155
156static void err_output(int is_error, const char *fmt, va_list ap)
157{
158#if defined(AP_LOG_SYSLOG)
159    if (!log_open) {
160        openlog("suexec", LOG_PID, AP_LOG_FACILITY);
161        log_open = 1;
162    }
163
164    vsyslog(is_error ? LOG_ERR : LOG_INFO, fmt, ap);
165#elif defined(AP_LOG_EXEC)
166    time_t timevar;
167    struct tm *lt;
168
169    if (!log) {
170#if defined(_LARGEFILE64_SOURCE) && HAVE_FOPEN64
171        if ((log = fopen64(AP_LOG_EXEC, "a")) == NULL) {
172#else
173        if ((log = fopen(AP_LOG_EXEC, "a")) == NULL) {
174#endif
175            fprintf(stderr, "suexec failure: could not open log file\n");
176            perror("fopen");
177            exit(1);
178        }
179    }
180
181    if (is_error) {
182        fprintf(stderr, "suexec policy violation: see suexec log for more "
183                        "details\n");
184    }
185
186    time(&timevar);
187    lt = localtime(&timevar);
188
189    fprintf(log, "[%d-%.2d-%.2d %.2d:%.2d:%.2d]: ",
190            lt->tm_year + 1900, lt->tm_mon + 1, lt->tm_mday,
191            lt->tm_hour, lt->tm_min, lt->tm_sec);
192
193    vfprintf(log, fmt, ap);
194
195    fflush(log);
196#endif /* AP_LOG_EXEC */
197    return;
198}

200static void log_err(const char *fmt,...)
201{
202#ifdef AP_LOG_EXEC
203    va_list ap;
204
205    va_start(ap, fmt);
206    err_output(1, fmt, ap); /* 1 == is_error */ view japanese
207    va_end(ap);
208#endif /* AP_LOG_EXEC */
209    return;
210}
211
212static void log_no_err(const char *fmt,...)
213{
214#ifdef AP_LOG_EXEC
215    va_list ap;
216
217    va_start(ap, fmt);
218    err_output(0, fmt, ap); /* 0 == !is_error */ view japanese
219    va_end(ap);
220#endif /* AP_LOG_EXEC */ view japanese
221    return;
222}

224static void clean_env(void)
225{
226    char **cleanenv;
227    char **ep;
228    int cidx = 0;
229    int idx;
230
231    /* While cleaning the environment, the environment should be clean.
232     * (e.g. malloc() may get the name of a file for writing debugging info.
233     * Bad news if MALLOC_DEBUG_FILE is set to /etc/passwd.  Sprintf() may be
234     * susceptible to bad locale settings....)
235     * (from PR 2790)
236     */ view japanese
237    char **envp = environ;
238    char *empty_ptr = NULL;
239
240    environ = &empty_ptr; /* VERY safe environment */ view japanese
241
242    if ((cleanenv = (char **) calloc(AP_ENVBUF, sizeof(char *))) == NULL) {
243        log_err("failed to malloc memory for environment\n");
244        exit(123);
245    }
246
247    cleanenv[cidx] = strdup("PATH=" AP_SAFE_PATH);
248    if (cleanenv[cidx] == NULL) {
249        log_err("failed to malloc memory for environment\n");
250        exit(124);
251    }
252    cidx++;
253
254    for (ep = envp; *ep && cidx < AP_ENVBUF-1; ep++) {
255        for (idx = 0; safe_env_lst[idx]; idx++) {
256            if (!strncmp(*ep, safe_env_lst[idx],
257                         strlen(safe_env_lst[idx]))) {
258                cleanenv[cidx] = *ep;
259                cidx++;
260                break;
261            }
262        }
263    }
264
265    cleanenv[cidx] = NULL;
266
267    environ = cleanenv;
268}

270int main(int argc, char *argv[])
271{
272    int userdir = 0;        /* ~userdir flag             */ view japanese
273    uid_t uid;              /* user information          */ view japanese
274    gid_t gid;              /* target group placeholder  */ view japanese
275    char *target_uname;     /* target user name          */ view japanese
276    char *target_gname;     /* target group name         */ view japanese
277    char *target_homedir;   /* target home directory     */ view japanese
278    char *actual_uname;     /* actual user name          */ view japanese
279    char *actual_gname;     /* actual group name         */ view japanese
280    char *cmd;              /* command to be executed    */ view japanese
281    char cwd[AP_MAXPATH];   /* current working directory */ view japanese
282    char dwd[AP_MAXPATH];   /* docroot working directory */ view japanese
283    struct passwd *pw;      /* password entry holder     */ view japanese
284    struct group *gr;       /* group entry holder        */ view japanese
285    struct stat dir_info;   /* directory info holder     */ view japanese
286    struct stat prg_info;   /* program info holder       */ view japanese
287
288    /*
289     * Start with a "clean" environment
290     */ view japanese
291    clean_env();
292
293    /*
294     * Check existence/validity of the UID of the user
295     * running this program.  Error out if invalid.
296     */ view japanese
297    uid = getuid();
298    if ((pw = getpwuid(uid)) == NULL) {
299        log_err("crit: invalid uid: (%lu)\n", (unsigned long)uid);
300        exit(102);
301    }
302    /*
303     * See if this is a 'how were you compiled' request, and
304     * comply if so.
305     */ view japanese
306    if ((argc > 1)
307        && (! strcmp(argv[1], "-V"))
308        && ((uid == 0)
309#ifdef _OSD_POSIX
310        /* User name comparisons are case insensitive on BS2000/OSD */
311            || (! strcasecmp(AP_HTTPD_USER, pw->pw_name)))
312#else  /* _OSD_POSIX */
313            || (! strcmp(AP_HTTPD_USER, pw->pw_name)))
314#endif /* _OSD_POSIX */ view japanese
315        ) {
316#ifdef AP_DOC_ROOT
317        fprintf(stderr, " -D AP_DOC_ROOT=\"%s\"\n", AP_DOC_ROOT);
318#endif
319#ifdef AP_GID_MIN
320        fprintf(stderr, " -D AP_GID_MIN=%d\n", AP_GID_MIN);
321#endif
322#ifdef AP_HTTPD_USER
323        fprintf(stderr, " -D AP_HTTPD_USER=\"%s\"\n", AP_HTTPD_USER);
324#endif
325#if defined(AP_LOG_SYSLOG)
326        fprintf(stderr, " -D AP_LOG_SYSLOG\n");
327#elif defined(AP_LOG_EXEC)
328        fprintf(stderr, " -D AP_LOG_EXEC=\"%s\"\n", AP_LOG_EXEC);
329#endif
330#ifdef AP_SAFE_PATH
331        fprintf(stderr, " -D AP_SAFE_PATH=\"%s\"\n", AP_SAFE_PATH);
332#endif
333#ifdef AP_SUEXEC_UMASK
334        fprintf(stderr, " -D AP_SUEXEC_UMASK=%03o\n", AP_SUEXEC_UMASK);
335#endif
336#ifdef AP_UID_MIN
337        fprintf(stderr, " -D AP_UID_MIN=%d\n", AP_UID_MIN);
338#endif
339#ifdef AP_USERDIR_SUFFIX
340        fprintf(stderr, " -D AP_USERDIR_SUFFIX=\"%s\"\n", AP_USERDIR_SUFFIX);
341#endif
342        exit(0);
343    }
344    /*
345     * If there are a proper number of arguments, set
346     * all of them to variables.  Otherwise, error out.
347     */ view japanese
348    if (argc < 4) {
349        log_err("too few arguments\n");
350        exit(101);
351    }
352    target_uname = argv[1];
353    target_gname = argv[2];
354    cmd = argv[3];
355
356    /*
357     * Check to see if the user running this program
358     * is the user allowed to do so as defined in
359     * suexec.h.  If not the allowed user, error out.
360     */ view japanese
361#ifdef _OSD_POSIX
362    /* User name comparisons are case insensitive on BS2000/OSD */ view japanese
363    if (strcasecmp(AP_HTTPD_USER, pw->pw_name)) {
364        log_err("user mismatch (%s instead of %s)\n", pw->pw_name, AP_HTTPD_USER);
365        exit(103);
366    }
367#else  /*_OSD_POSIX*/
368    if (strcmp(AP_HTTPD_USER, pw->pw_name)) {
369        log_err("user mismatch (%s instead of %s)\n", pw->pw_name, AP_HTTPD_USER);
370        exit(103);
371    }
372#endif /*_OSD_POSIX*/
373
374    /*
375     * Check for a leading '/' (absolute path) in the command to be executed,
376     * or attempts to back up out of the current directory,
377     * to protect against attacks.  If any are
378     * found, error out.  Naughty naughty crackers.
379     */ view japanese
380    if ((cmd[0] == '/') || (!strncmp(cmd, "../", 3))
381        || (strstr(cmd, "/../") != NULL)) {
382        log_err("invalid command (%s)\n", cmd);
383        exit(104);
384    }
385
386    /*
387     * Check to see if this is a ~userdir request.  If
388     * so, set the flag, and remove the '~' from the
389     * target username.
390     */ view japanese
391    if (!strncmp("~", target_uname, 1)) {
392        target_uname++;
393        userdir = 1;
394    }
395
396    /*
397     * Error out if the target username is invalid.
398     */ view japanese
399    if (strspn(target_uname, "1234567890") != strlen(target_uname)) {
400        if ((pw = getpwnam(target_uname)) == NULL) {
401            log_err("invalid target user name: (%s)\n", target_uname);
402            exit(105);
403        }
404    }
405    else {
406        if ((pw = getpwuid(atoi(target_uname))) == NULL) {
407            log_err("invalid target user id: (%s)\n", target_uname);
408            exit(121);
409        }
410    }
411
412    /*
413     * Error out if the target group name is invalid.
414     */ view japanese
415    if (strspn(target_gname, "1234567890") != strlen(target_gname)) {
416        if ((gr = getgrnam(target_gname)) == NULL) {
417            log_err("invalid target group name: (%s)\n", target_gname);
418            exit(106);
419        }
420    }
421    else {
422        if ((gr = getgrgid(atoi(target_gname))) == NULL) {
423            log_err("invalid target group id: (%s)\n", target_gname);
424            exit(106);
425        }
426    }
427    gid = gr->gr_gid;
428    if ((actual_gname = strdup(gr->gr_name)) == NULL) {
429        log_err("failed to alloc memory\n");
430        exit(125);
431    }
432
433#ifdef _OSD_POSIX
434    /*
435     * Initialize BS2000 user environment
436     */ view japanese
437    {
438        pid_t pid;
439        int status;
440
441        switch (pid = ufork(target_uname)) {
442        case -1:    /* Error */ view japanese
443            log_err("failed to setup bs2000 environment for user %s: %s\n",
444                    target_uname, strerror(errno));
445            exit(150);
446        case 0:     /* Child */ view japanese
447            break;
448        default:    /* Father */ view japanese
449            while (pid != waitpid(pid, &status, 0))
450                ;
451            /* @@@ FIXME: should we deal with STOP signals as well? */ view japanese
452            if (WIFSIGNALED(status)) {
453                kill (getpid(), WTERMSIG(status));
454            }
455            exit(WEXITSTATUS(status));
456        }
457    }
458#endif /*_OSD_POSIX*/ view japanese
459
460    /*
461     * Save these for later since initgroups will hose the struct
462     */ view japanese
463    uid = pw->pw_uid;
464    actual_uname = strdup(pw->pw_name);
465    target_homedir = strdup(pw->pw_dir);
466    if (actual_uname == NULL || target_homedir == NULL) {
467        log_err("failed to alloc memory\n");
468        exit(126);
469    }
470
471    /*
472     * Log the transaction here to be sure we have an open log
473     * before we setuid().
474     */ view japanese
475    log_no_err("uid: (%s/%s) gid: (%s/%s) cmd: %s\n",
476               target_uname, actual_uname,
477               target_gname, actual_gname,
478               cmd);
479
480    /*
481     * Error out if attempt is made to execute as root or as
482     * a UID less than AP_UID_MIN.  Tsk tsk.
483     */ view japanese
484    if ((uid == 0) || (uid < AP_UID_MIN)) {
485        log_err("cannot run as forbidden uid (%lu/%s)\n", (unsigned long)uid, cmd);
486        exit(107);
487    }
488
489    /*
490     * Error out if attempt is made to execute as root group
491     * or as a GID less than AP_GID_MIN.  Tsk tsk.
492     */ view japanese
493    if ((gid == 0) || (gid < AP_GID_MIN)) {
494        log_err("cannot run as forbidden gid (%lu/%s)\n", (unsigned long)gid, cmd);
495        exit(108);
496    }
497
498    /*
499     * Change UID/GID here so that the following tests work over NFS.
500     *
501     * Initialize the group access list for the target user,
502     * and setgid() to the target group. If unsuccessful, error out.
503     */ view japanese
504    if (((setgid(gid)) != 0) || (initgroups(actual_uname, gid) != 0)) {
505        log_err("failed to setgid/initgroups (%lu: %s): %s\n",
506                (unsigned long)gid, cmd, strerror(errno));
507        exit(109);
508    }
509
510    /*
511     * setuid() to the target user.  Error out on fail.
512     */ view japanese
513    if ((setuid(uid)) != 0) {
514        log_err("failed to setuid (%lu: %s): %s\n",
515                (unsigned long)uid, cmd, strerror(errno));
516        exit(110);
517    }
518
519    /*
520     * Get the current working directory, as well as the proper
521     * document root (dependent upon whether or not it is a
522     * ~userdir request).  Error out if we cannot get either one,
523     * or if the current working directory is not in the docroot.
524     * Use chdir()s and getcwd()s to avoid problems with symlinked
525     * directories.  Yuck.
526     */ view japanese
527    if (getcwd(cwd, AP_MAXPATH) == NULL) {
528        log_err("cannot get current working directory\n");
529        exit(111);
530    }
531
532    if (userdir) {
533        if (((chdir(target_homedir)) != 0) ||
534            ((chdir(AP_USERDIR_SUFFIX)) != 0) ||
535            ((getcwd(dwd, AP_MAXPATH)) == NULL) ||
536            ((chdir(cwd)) != 0)) {
537            log_err("cannot get docroot information (%s)\n", target_homedir);
538            exit(112);
539        }
540    }
541    else {
542        if (((chdir(AP_DOC_ROOT)) != 0) ||
543            ((getcwd(dwd, AP_MAXPATH)) == NULL) ||
544            ((chdir(cwd)) != 0)) {
545            log_err("cannot get docroot information (%s)\n", AP_DOC_ROOT);
546            exit(113);
547        }
548    }
549
550    if ((strncmp(cwd, dwd, strlen(dwd))) != 0) {
551        log_err("command not in docroot (%s/%s)\n", cwd, cmd);
552        exit(114);
553    }
554
555    /*
556     * Stat the cwd and verify it is a directory, or error out.
557     */ view japanese
558    if (((lstat(cwd, &dir_info)) != 0) || !(S_ISDIR(dir_info.st_mode))) {
559        log_err("cannot stat directory: (%s)\n", cwd);
560        exit(115);
561    }
562
563    /*
564     * Error out if cwd is writable by others.
565     */ view japanese
566    if ((dir_info.st_mode & S_IWOTH) || (dir_info.st_mode & S_IWGRP)) {
567        log_err("directory is writable by others: (%s)\n", cwd);
568        exit(116);
569    }
570
571    /*
572     * Error out if we cannot stat the program.
573     */ view japanese
574    if (((lstat(cmd, &prg_info)) != 0) || (S_ISLNK(prg_info.st_mode))) {
575        log_err("cannot stat program: (%s)\n", cmd);
576        exit(117);
577    }
578
579    /*
580     * Error out if the program is writable by others.
581     */ view japanese
582    if ((prg_info.st_mode & S_IWOTH) || (prg_info.st_mode & S_IWGRP)) {
583        log_err("file is writable by others: (%s/%s)\n", cwd, cmd);
584        exit(118);
585    }
586
587    /*
588     * Error out if the file is setuid or setgid.
589     */ view japanese
590    if ((prg_info.st_mode & S_ISUID) || (prg_info.st_mode & S_ISGID)) {
591        log_err("file is either setuid or setgid: (%s/%s)\n", cwd, cmd);
592        exit(119);
593    }
594
595    /*
596     * Error out if the target name/group is different from
597     * the name/group of the cwd or the program.
598     */ view japanese
599    if ((uid != dir_info.st_uid) ||
600        (gid != dir_info.st_gid) ||
601        (uid != prg_info.st_uid) ||
602        (gid != prg_info.st_gid)) {
603        log_err("target uid/gid (%lu/%lu) mismatch "
604                "with directory (%lu/%lu) or program (%lu/%lu)\n",
605                (unsigned long)uid, (unsigned long)gid,
606                (unsigned long)dir_info.st_uid, (unsigned long)dir_info.st_gid,
607                (unsigned long)prg_info.st_uid, (unsigned long)prg_info.st_gid);
608        exit(120);
609    }
610    /*
611     * Error out if the program is not executable for the user.
612     * Otherwise, she won't find any error in the logs except for
613     * "[error] Premature end of script headers: ..."
614     */ view japanese
615    if (!(prg_info.st_mode & S_IXUSR)) {
616        log_err("file has no execute permission: (%s/%s)\n", cwd, cmd);
617        exit(121);
618    }
619
620#ifdef AP_SUEXEC_UMASK
621    /*
622     * umask() uses inverse logic; bits are CLEAR for allowed access.
623     */ view japanese
624    if ((~AP_SUEXEC_UMASK) & 0022) {
625        log_err("notice: AP_SUEXEC_UMASK of %03o allows "
626                "write permission to group and/or other\n", AP_SUEXEC_UMASK);
627    }
628    umask(AP_SUEXEC_UMASK);
629#endif /* AP_SUEXEC_UMASK */ view japanese
630
631    /* Be sure to close the log file so the CGI can't mess with it. */ view japanese
632#ifdef AP_LOG_SYSLOG
633    if (log_open) {
634        closelog();
635        log_open = 0;
636    }
637#else
638    if (log != NULL) {
639#if APR_HAVE_FCNTL_H
640        /*
641         * ask fcntl(2) to set the FD_CLOEXEC flag on the log file,
642         * so it'll be automagically closed if the exec() call succeeds.
643         */ view japanese
644        fflush(log);
645        setbuf(log, NULL);
646        if ((fcntl(fileno(log), F_SETFD, FD_CLOEXEC) == -1)) {
647            log_err("error: can't set close-on-exec flag");
648            exit(122);
649        }
650#else
651        /*
652         * In this case, exec() errors won't be logged because we have already
653         * dropped privileges and won't be able to reopen the log file.
654         */ view japanese
655        fclose(log);
656        log = NULL;
657#endif
658    }
659#endif
660
661    /*
662     * Execute the command, replacing our image with its own.
663     */ view japanese
664#ifdef NEED_HASHBANG_EMUL
665    /* We need the #! emulation when we want to execute scripts */ view japanese
666    {
667        extern char **environ;
668
