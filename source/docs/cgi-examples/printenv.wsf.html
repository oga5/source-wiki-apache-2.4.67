14''  printenv -- demo CGI program which just prints its environment
15''
16<job>
17<script language="JScript">
18  WScript.Echo("Content-type: text/plain; charset=iso-8859-1\n");
19  var objShell = new ActiveXObject("WScript.Shell");
20  var objArray = new Array();
21  var e = new Enumerator(objShell.Environment("PROCESS"));
22  for (;!e.atEnd();e.moveNext()) {
23    var i = e.item().indexOf("=");
24    var envvar = e.item().substring(0, i);
25    var envval = e.item().substring(i + 1, e.item().length);
26    envval = envval.replace("\n", "\\n");
27    objArray.push(envvar + "=\"" + envval + "\"");
28  }
29  objArray.sort();
30  WScript.Echo(objArray.join("\n"));
31</script>
32</job>
