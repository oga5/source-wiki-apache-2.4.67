14##
15##  printenv -- demo CGI program which just prints its environment
16##
17use strict;
18use warnings;
19
20print "Content-type: text/plain; charset=iso-8859-1\n\n";
21foreach my $var (sort(keys(%ENV))) {
22    my $val = $ENV{$var};
23    $val =~ s|\n|\\n|g;
24    $val =~ s|"|\\"|g;
25    print "${var}=\"${val}\"\n";
26}
